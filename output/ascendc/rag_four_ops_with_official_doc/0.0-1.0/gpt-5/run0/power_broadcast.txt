kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "power_broadcast_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // single-buffering per row

class KernelPowerBroadcast {
public:
    __aicore__ inline KernelPowerBroadcast() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR exponent, GM_ADDR y, uint32_t totalLength, uint32_t alignedLength, uint32_t innerSize, uint32_t alignedInnerSize)
    {
        this->totalLength = totalLength;
        this->innerSize = innerSize;
        this->alignedInnerSize = alignedInnerSize;
        
        // Single core processing
        this->tileLength = 2048; // Must be multiple of 8
        this->tileNum = (alignedLength + this->tileLength - 1) / this->tileLength;

        uint32_t alignedBlockLength = (this->totalLength + 7) / 8 * 8;
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, alignedBlockLength);
        eGm.SetGlobalBuffer((__gm__ DTYPE_EXPONENT *)exponent, alignedInnerSize);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, alignedBlockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueE, 1, this->alignedInnerSize * sizeof(DTYPE_EXPONENT));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        // Load e to UB first (broadcast target)
        AscendC::LocalTensor<DTYPE_EXPONENT> eLocal = inQueueE.AllocTensor<DTYPE_EXPONENT>();
        AscendC::DataCopy(eLocal, eGm[0], this->alignedInnerSize);
        inQueueE.EnQue(eLocal);

        AscendC::LocalTensor<DTYPE_EXPONENT> eLocalCompute = inQueueE.DeQue<DTYPE_EXPONENT>();

        for (int32_t i = 0; i < this->tileNum; i++) {
            CopyIn(i);
            Compute(i, eLocalCompute);
            CopyOut(i);
        }
        
        // Free e
        inQueueE.FreeTensor(eLocalCompute);
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        uint32_t currentLength = this->tileLength;
        if (progress == this->tileNum - 1) {
            currentLength = this->totalLength - progress * this->tileLength;
        }
        uint32_t alignedLength = (currentLength + 7) / 8 * 8; // align for vector ops
        
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], alignedLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t progress, AscendC::LocalTensor<DTYPE_EXPONENT>& eLocal)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
        
        uint32_t currentLength = this->tileLength;
        if (progress == this->tileNum - 1) {
            currentLength = this->totalLength - progress * this->tileLength;
        }
        uint32_t alignedLength = (currentLength + 7) / 8 * 8;

        uint32_t innerOffset = (progress * this->tileLength) % this->innerSize;
        
        AscendC::Ln(yLocal, xLocal, alignedLength);
        AscendC::Mul(yLocal, yLocal, eLocal[innerOffset], alignedLength);
        AscendC::Exp(yLocal, yLocal, alignedLength);
        
        outQueueY.EnQue<DTYPE_Y>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        uint32_t currentLength = this->tileLength;
        if (progress == this->tileNum - 1) {
            currentLength = this->totalLength - progress * this->tileLength;
        }
        
        uint32_t alignedLength = (currentLength + 7) / 8 * 8; // align for vector ops
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, alignedLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueE;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_EXPONENT> eGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t totalLength;
    uint32_t tileNum;
    uint32_t tileLength;
    uint32_t innerSize;
    uint32_t alignedInnerSize;
};

extern "C" __global__ __aicore__ void power_broadcast_custom(GM_ADDR x, GM_ADDR exponent, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelPowerBroadcast op;
    op.Init(x, exponent, y, tiling_data.totalLength, tiling_data.alignedLength, tiling_data.innerSize, tiling_data.alignedInnerSize);
    op.Process();
}
