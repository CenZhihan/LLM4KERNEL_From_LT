#include "clamp_broadcast_custom_tiling.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t BUFFER_NUM = 2;
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    ClampBroadcastCustomTilingData tiling;
    uint32_t totalLength = context->GetInputShape(0)->GetOriginShape().GetShapeSize();
    uint32_t rowLength = context->GetInputShape(1)->GetOriginShape().GetShapeSize(); // min/max length
    uint32_t rows = totalLength / rowLength;

    context->SetBlockDim(BLOCK_DIM);

    uint32_t rowsPerBlock = rows / BLOCK_DIM;
    uint32_t tileNum = rowsPerBlock / BUFFER_NUM;

    tiling.set_totalLength(totalLength);
    tiling.set_tileNum(tileNum);
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)

    const auto inputDataType = context->GetInputDataType(0);

    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "clamp_broadcast_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // double buffer, one row per buffer

class KernelClampBroadcast {
public:
    __aicore__ inline KernelClampBroadcast() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR min_val, GM_ADDR max_val, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum(); // elements per block
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM; // equals row length

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        // min/max are broadcast per row; keep full row length
        minGm.SetGlobalBuffer((__gm__ DTYPE_MIN_VAL *)min_val, this->tileLength);
        maxGm.SetGlobalBuffer((__gm__ DTYPE_MAX_VAL *)max_val, this->tileLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueMin, BUFFER_NUM, this->tileLength * sizeof(DTYPE_MIN_VAL));
        pipe.InitBuffer(inQueueMax, BUFFER_NUM, this->tileLength * sizeof(DTYPE_MAX_VAL));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM; // rows per block
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_MIN_VAL> minLocal = inQueueMin.AllocTensor<DTYPE_MIN_VAL>();
        AscendC::LocalTensor<DTYPE_MAX_VAL> maxLocal = inQueueMax.AllocTensor<DTYPE_MAX_VAL>();

        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(minLocal, minGm[0], this->tileLength);
        AscendC::DataCopy(maxLocal, maxGm[0], this->tileLength);

        inQueueX.EnQue(xLocal);
        inQueueMin.EnQue(minLocal);
        inQueueMax.EnQue(maxLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_MIN_VAL> minLocal = inQueueMin.DeQue<DTYPE_MIN_VAL>();
        AscendC::LocalTensor<DTYPE_MAX_VAL> maxLocal = inQueueMax.DeQue<DTYPE_MAX_VAL>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

        AscendC::Max(yLocal, xLocal, minLocal, this->tileLength);
        AscendC::Min(yLocal, yLocal, maxLocal, this->tileLength);

        outQueueY.EnQue<DTYPE_Y>(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueMin.FreeTensor(minLocal);
        inQueueMax.FreeTensor(maxLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueMin, inQueueMax;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_MIN_VAL> minGm;
    AscendC::GlobalTensor<DTYPE_MAX_VAL> maxGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void clamp_broadcast_custom(GM_ADDR x, GM_ADDR min_val, GM_ADDR max_val, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelClampBroadcast op;
    op.Init(x, min_val, max_val, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}
