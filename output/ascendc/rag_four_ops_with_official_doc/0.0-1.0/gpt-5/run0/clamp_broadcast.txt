#include "clamp_broadcast_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    ClampBroadcastCustomTilingData tiling;
    uint32_t totalLength = context->GetInputShape(0)->GetOriginShape().GetShapeSize();
    uint32_t rowLength = context->GetInputShape(1)->GetOriginShape().GetShapeSize(); // min/max length
    
    // align to 32 bytes (8 floats) for float32
    uint32_t alignedLength = (totalLength + 7) / 8 * 8;
    uint32_t alignedInnerSize = (rowLength + 7) / 8 * 8;
    
    context->SetBlockDim(1);

    tiling.set_totalLength(totalLength);
    tiling.set_alignedLength(alignedLength);
    tiling.set_innerSize(rowLength);
    // ensure innerSize is not 0 for alignedInnerSize
    if (alignedInnerSize == 0) {
        alignedInnerSize = 8;
    }
    tiling.set_alignedInnerSize(alignedInnerSize);
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "clamp_broadcast_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // double buffer, one row per buffer

class KernelClampBroadcast {
public:
    __aicore__ inline KernelClampBroadcast() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR min_val, GM_ADDR max_val, GM_ADDR y,
                                uint32_t totalLength, uint32_t alignedLength, uint32_t innerSize, uint32_t alignedInnerSize)
    {
        this->totalLength = totalLength;
        this->innerSize = innerSize;
        this->alignedInnerSize = alignedInnerSize;
        
        // Single core processing
        this->tileLength = 2048; // Must be multiple of 8
        this->tileNum = (alignedLength + this->tileLength - 1) / this->tileLength;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, alignedLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, alignedLength);

        // min/max are broadcast per row; we load them to UB once
        minGm.SetGlobalBuffer((__gm__ DTYPE_MIN_VAL *)min_val, alignedInnerSize);
        maxGm.SetGlobalBuffer((__gm__ DTYPE_MAX_VAL *)max_val, alignedInnerSize);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueMin, 1, this->alignedInnerSize * sizeof(DTYPE_MIN_VAL));
        pipe.InitBuffer(inQueueMax, 1, this->alignedInnerSize * sizeof(DTYPE_MAX_VAL));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        // Load min and max to UB first (broadcast target)
        AscendC::LocalTensor<DTYPE_MIN_VAL> minLocal = inQueueMin.AllocTensor<DTYPE_MIN_VAL>();
        AscendC::LocalTensor<DTYPE_MAX_VAL> maxLocal = inQueueMax.AllocTensor<DTYPE_MAX_VAL>();
        
        AscendC::DataCopy(minLocal, minGm[0], this->alignedInnerSize);
        AscendC::DataCopy(maxLocal, maxGm[0], this->alignedInnerSize);
        
        inQueueMin.EnQue(minLocal);
        inQueueMax.EnQue(maxLocal);

        AscendC::LocalTensor<DTYPE_MIN_VAL> minLocalCompute = inQueueMin.DeQue<DTYPE_MIN_VAL>();
        AscendC::LocalTensor<DTYPE_MAX_VAL> maxLocalCompute = inQueueMax.DeQue<DTYPE_MAX_VAL>();

        for (int32_t i = 0; i < this->tileNum; i++) {
            CopyIn(i);
            Compute(i, minLocalCompute, maxLocalCompute);
            CopyOut(i);
        }
        
        // Free min and max
        inQueueMin.FreeTensor(minLocalCompute);
        inQueueMax.FreeTensor(maxLocalCompute);
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        uint32_t currentLength = this->tileLength;
        if (progress == this->tileNum - 1) {
            currentLength = this->totalLength - progress * this->tileLength;
        }
        uint32_t alignedLength = (currentLength + 7) / 8 * 8; // align for vector ops
        
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], alignedLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t progress, AscendC::LocalTensor<DTYPE_MIN_VAL>& minLocal, AscendC::LocalTensor<DTYPE_MAX_VAL>& maxLocal)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
        
        uint32_t currentLength = this->tileLength;
        if (progress == this->tileNum - 1) {
            currentLength = this->totalLength - progress * this->tileLength;
        }
        uint32_t alignedLength = (currentLength + 7) / 8 * 8;

        uint32_t innerOffset = (progress * this->tileLength) % this->innerSize;
        
        AscendC::Max(yLocal, xLocal, minLocal[innerOffset], alignedLength);
        AscendC::Min(yLocal, yLocal, maxLocal[innerOffset], alignedLength);

        outQueueY.EnQue<DTYPE_Y>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        uint32_t currentLength = this->tileLength;
        if (progress == this->tileNum - 1) {
            currentLength = this->totalLength - progress * this->tileLength;
        }
        
        uint32_t alignedLength = (currentLength + 7) / 8 * 8;
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, alignedLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueMin, inQueueMax;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_MIN_VAL> minGm;
    AscendC::GlobalTensor<DTYPE_MAX_VAL> maxGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t totalLength;
    uint32_t tileNum;
    uint32_t tileLength;
    uint32_t innerSize;
    uint32_t alignedInnerSize;
};

extern "C" __global__ __aicore__ void clamp_broadcast_custom(GM_ADDR x, GM_ADDR min_val, GM_ADDR max_val, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelClampBroadcast op;
    op.Init(x, min_val, max_val, y, tiling_data.totalLength, tiling_data.alignedLength, tiling_data.innerSize, tiling_data.alignedInnerSize);
    op.Process();
}
"""
