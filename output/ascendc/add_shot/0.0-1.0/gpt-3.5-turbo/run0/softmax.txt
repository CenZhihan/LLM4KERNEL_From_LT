kernel_src="""
#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelSoftmax {
public:
    __aicore__ inline KernelSoftmax() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t batchSize, uint32_t featureSize, uint32_t tileNum)
    {
        this->batchSize = batchSize;
        this->featureSize = featureSize;
        this->tileNum = tileNum;
        this->blockBatch = batchSize / AscendC::GetBlockNum();
        this->tileLength = featureSize / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockBatch * AscendC::GetBlockIdx() * featureSize, this->blockBatch * featureSize);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->blockBatch * AscendC::GetBlockIdx() * featureSize, this->blockBatch * featureSize);
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }
    __aicore__ inline void Process()
    {
        // For each batch in this block
        for (uint32_t b = 0; b < blockBatch; ++b) {
            // Compute max for the batch row
            float maxVal = -FLT_MAX;
            for (uint32_t i = 0; i < featureSize; ++i) {
                float val = xGm[b * featureSize + i];
                if (val > maxVal) maxVal = val;
            }
            // Compute exp and sum
            float sumExp = 0.0f;
            for (uint32_t i = 0; i < featureSize; ++i) {
                float val = xGm[b * featureSize + i];
                float e = expf(val - maxVal);
                yGm[b * featureSize + i] = e;
                sumExp += e;
            }
            // Normalize
            for (uint32_t i = 0; i < featureSize; ++i) {
                yGm[b * featureSize + i] /= sumExp;
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;
    uint32_t batchSize;
    uint32_t featureSize;
    uint32_t tileNum;
    uint32_t blockBatch;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmax op;
    op.Init(x, y, tiling_data.batchSize, tiling_data.featureSize, tiling_data.tileNum);
    op.Process();
}
"""
