kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv2d_gelu_global_avg_pool_custom_tiling_data.h"

__aicore__ inline float gelu_approx(float x) {
    // Approximation: 0.5 * x * (1 + tanh( sqrt(2/pi) * (x + 0.044715 x^3) ))
    const float kAlpha = 0.7978845608f; // sqrt(2/pi)
    float x3 = x * x * x;
    float t  = kAlpha * (x + 0.044715f * x3);
    float th = tanhf(t);
    return 0.5f * x * (1.0f + th);
}

class KernelConv2dGeluGAP {
public:
    __aicore__ inline KernelConv2dGeluGAP() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t Ci, uint32_t Co,
                                uint32_t H, uint32_t W, uint32_t Kh, uint32_t Kw,
                                uint32_t Ho, uint32_t Wo, uint32_t tileCo) {
        xGm.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N * Ci * H * W);
        wGm.SetGlobalBuffer((__gm__ float*)w, (uint64_t)Co * Ci * Kh * Kw);
        bGm.SetGlobalBuffer((__gm__ float*)b, (uint64_t)Co);
        yGm.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N * Co);

        this->N = N; this->Ci = Ci; this->Co = Co;
        this->H = H; this->W = W; this->Kh = Kh; this->Kw = Kw;
        this->Ho = Ho; this->Wo = Wo; this->tileCo = tileCo;

        // No queues needed for this simple fused kernel
    }

    __aicore__ inline void Process() {
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t bid    = AscendC::GetBlockIdx();

        uint32_t nPerBlock = (N + blocks - 1) / blocks;
        uint32_t nStart    = bid * nPerBlock;
        uint32_t nEnd      = AscendC::Min(nStart + nPerBlock, N);

        if (Ho == 0 || Wo == 0) {
            // Degenerate, set output zeros
            for (uint32_t n = nStart; n < nEnd; ++n) {
                uint64_t yBase = (uint64_t)n * Co;
                for (uint32_t co = 0; co < Co; ++co) {
                    yGm.SetValue(yBase + co, 0.0f);
                }
            }
            return;
        }

        const float denom = 1.0f / (float)(Ho * Wo);

        for (uint32_t n = nStart; n < nEnd; ++n) {
            for (uint32_t coBase = 0; coBase < Co; coBase += tileCo) {
                uint32_t coUpper = AscendC::Min(coBase + tileCo, Co);

                // Running sums per output channel
                // Allocate in registers/local
                for (uint32_t co = coBase; co < coUpper; ++co) {
                    float sumSpatial = 0.0f;

                    for (uint32_t ho = 0; ho < Ho; ++ho) {
                        for (uint32_t wo = 0; wo < Wo; ++wo) {
                            // Convolution accumulate over Ci, Kh, Kw
                            float acc = 0.0f;
                            for (uint32_t ci = 0; ci < Ci; ++ci) {
                                for (uint32_t kh = 0; kh < Kh; ++kh) {
                                    uint32_t hIn = ho + kh;
                                    for (uint32_t kw = 0; kw < Kw; ++kw) {
                                        uint32_t wIn = wo + kw;

                                        uint64_t xIndex = (((uint64_t)n * Ci + ci) * H + hIn) * W + wIn;
                                        uint64_t wIndex = (((uint64_t)co * Ci + ci) * Kh + kh) * Kw + kw;

                                        float xv = xGm.GetValue(xIndex);
                                        float wv = wGm.GetValue(wIndex);
                                        acc += xv * wv;
                                    }
                                }
                            }
                            // add bias and GELU
                            float accBias = acc + bGm.GetValue((uint64_t)co);
                            float act = gelu_approx(accBias);
                            sumSpatial += act;
                        }
                    }
                    // Global average pooling
                    float outv = sumSpatial * denom;
                    uint64_t yIndex = (uint64_t)n * Co + co;
                    yGm.SetValue(yIndex, outv);
                }
            }
        }
    }

private:
    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> wGm;
    AscendC::GlobalTensor<float> bGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t N, Ci, Co, H, W, Kh, Kw, Ho, Wo, tileCo;
};

extern "C" __global__ __aicore__ void conv2d_gelu_global_avg_pool_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dGeluGAP op;
    op.Init(x, w, b, y,
            tiling_data.N, tiling_data.Ci, tiling_data.Co,
            tiling_data.H, tiling_data.W, tiling_data.Kh, tiling_data.Kw,
            tiling_data.Ho, tiling_data.Wo, tiling_data.tileCo);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused model: Conv2d -> GELU -> GlobalAvgPool(1x1) implemented by a custom AscendC kernel.
    Produces output of shape (batch_size, out_channels).
    """
    def __init__(self, in_channels, out_channels, kernel_size):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size, bias=True)

    def forward(self, x):
        # Delegate the whole fused computation to the custom operator.
        # Expects NCHW input and weights shaped [Co, Ci, Kh, Kw], bias [Co]
        return custom_ops_lib.conv2d_gelu_global_avg_pool_custom(x, self.conv.weight, self.conv.bias)
