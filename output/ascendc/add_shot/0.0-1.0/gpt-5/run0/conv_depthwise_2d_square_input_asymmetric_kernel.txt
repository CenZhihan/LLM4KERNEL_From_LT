kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_depthwise_2d_square_input_asymmetric_kernel_custom_tiling_data.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelDWConvH {
public:
    __aicore__ inline KernelDWConvH() {}
    __aicore__ inline void Init(GM_ADDR x,
                                GM_ADDR w,
                                GM_ADDR b,
                                GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w_in,
                                uint32_t kh, uint32_t kw,
                                uint32_t stride_h, uint32_t stride_w,
                                uint32_t pad_h, uint32_t pad_w,
                                uint32_t dil_h, uint32_t dil_w,
                                uint32_t ho, uint32_t wo,
                                uint32_t tileNum)
    {
        // Save shapes/params
        this->n = n; this->c = c; this->h = h; this->w_in = w_in;
        this->kh = kh; this->kw = kw;
        this->stride_h = stride_h; this->stride_w = stride_w;
        this->pad_h = pad_h; this->pad_w = pad_w;
        this->dil_h = dil_h; this->dil_w = dil_w;
        this->ho = ho; this->wo = wo;
        this->tileNum = tileNum;

        xGm.SetGlobalBuffer((__gm__ float *)x, static_cast<uint32_t>(n) * c * h * w_in);
        wGm.SetGlobalBuffer((__gm__ float *)w, static_cast<uint32_t>(c) * kh); // layout: c,1,kh,1
        bGm.SetGlobalBuffer((__gm__ float *)b, static_cast<uint32_t>(c));
        yGm.SetGlobalBuffer((__gm__ float *)y, static_cast<uint32_t>(n) * c * ho * wo);

        // Tile along width dimension
        this->vecTile = (wo + (tileNum * BUFFER_NUM) - 1) / (tileNum * BUFFER_NUM);
        if (this->vecTile < 1) this->vecTile = 1;
        uint32_t tileLen = this->vecTile;

        pipe.InitBuffer(xQue, BUFFER_NUM, tileLen * sizeof(float));
        pipe.InitBuffer(yQue, BUFFER_NUM, tileLen * sizeof(float));
        pipe.InitBuffer(tmpQue, BUFFER_NUM, tileLen * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // Partition work across blocks by (n,c)
        uint32_t ncTotal = n * c;
        uint32_t blocks  = GetBlockNum();
        uint32_t blockId = GetBlockIdx();
        uint32_t ncPerBlock = (ncTotal + blocks - 1) / blocks;
        uint32_t ncStart = blockId * ncPerBlock;
        uint32_t ncEnd   = ncStart + ncPerBlock;
        if (ncEnd > ncTotal) ncEnd = ncTotal;

        for (uint32_t ncIdx = ncStart; ncIdx < ncEnd; ++ncIdx) {
            uint32_t nIdx = ncIdx / c;
            uint32_t cIdx = ncIdx % c;

            // Load bias for this channel
            float biasVal = bGm.GetValue(cIdx);

            // For each output row
            for (uint32_t hoIdx = 0; hoIdx < ho; ++hoIdx) {
                uint32_t loopCount = Ceil(wo, this->vecTile);
                for (uint32_t t = 0; t < loopCount; ++t) {
                    uint32_t wStart = t * this->vecTile;
                    uint32_t curLen = Min(this->vecTile, wo - wStart);

                    LocalTensor<float> yLocal = yQue.AllocTensor<float>();
                    pipe.Barrier();

                    // Initialize output tile with bias
                    Scalar<float> bScalar(biasVal);
                    SetValue(yLocal, bScalar, curLen);

                    // Accumulate over kh
                    for (uint32_t k = 0; k < kh; ++k) {
                        int32_t in_h = static_cast<int32_t>(hoIdx) * static_cast<int32_t>(stride_h)
                                     - static_cast<int32_t>(pad_h)
                                     + static_cast<int32_t>(k) * static_cast<int32_t>(dil_h);
                        if (in_h < 0 || in_h >= static_cast<int32_t>(h)) {
                            continue;
                        }

                        // Load a tile from input row
                        LocalTensor<float> xLocal = xQue.AllocTensor<float>();
                        // compute base offsets
                        uint64_t xBase = static_cast<uint64_t>(nIdx) * c * h * w_in
                                       + static_cast<uint64_t>(cIdx) * h * w_in
                                       + static_cast<uint64_t>(in_h) * w_in
                                       + static_cast<uint64_t>(wStart) * stride_w; // assume stride_w=1 typically
                        DataCopy(xLocal, xGm[xBase], curLen);
                        xQue.EnQue(xLocal);

                        // tmp = xLocal * wVal
                        LocalTensor<float> tmpLocal = tmpQue.AllocTensor<float>();
                        float wVal = wGm.GetValue(static_cast<uint64_t>(cIdx) * kh + k);
                        Muls(tmpLocal, xLocal, wVal, curLen);

                        // yLocal += tmp
                        Add(yLocal, yLocal, tmpLocal, curLen);

                        tmpQue.FreeTensor(tmpLocal);
                        xQue.DeQue<float>();
                        xQue.FreeTensor(xLocal);
                    }

                    yQue.EnQue(yLocal);

                    // Store output tile
                    LocalTensor<float> yLocalOut = yQue.DeQue<float>();
                    uint64_t yBase = static_cast<uint64_t>(nIdx) * c * ho * wo
                                   + static_cast<uint64_t>(cIdx) * ho * wo
                                   + static_cast<uint64_t>(hoIdx) * wo
                                   + static_cast<uint64_t>(wStart);
                    DataCopy(yGm[yBase], yLocalOut, curLen);
                    yQue.FreeTensor(yLocalOut);
                }
            }
        }
    }

private:
    __aicore__ inline uint32_t Ceil(uint32_t a, uint32_t b) { return (a + b - 1) / b; }
    __aicore__ inline uint32_t Min(uint32_t a, uint32_t b) { return a < b ? a : b; }

private:
    TPipe pipe;
    TQue<TPosition::VECIN,  BUFFER_NUM> xQue;
    TQue<TPosition::VECOUT, BUFFER_NUM> yQue;
    TQue<TPosition::VECIN,  BUFFER_NUM> tmpQue;

    GlobalTensor<float> xGm;
    GlobalTensor<float> wGm;
    GlobalTensor<float> bGm;
    GlobalTensor<float> yGm;

    uint32_t n, c, h, w_in;
    uint32_t kh, kw;
    uint32_t stride_h, stride_w;
    uint32_t pad_h, pad_w;
    uint32_t dil_h, dil_w;
    uint32_t ho, wo;
    uint32_t tileNum;
    uint32_t vecTile;
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_square_input_asymmetric_kernel_custom(GM_ADDR x,
                                                                                               GM_ADDR w,
                                                                                               GM_ADDR b,
                                                                                               GM_ADDR y,
                                                                                               GM_ADDR workspace,
                                                                                               GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelDWConvH op;
    op.Init(x, w, b, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.kh, tiling_data.kw,
            tiling_data.stride_h, tiling_data.stride_w,
            tiling_data.pad_h, tiling_data.pad_w,
            tiling_data.dil_h, tiling_data.dil_w,
            tiling_data.ho, tiling_data.wo,
            tiling_data.tileNum);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, dilation: int = 1, bias: bool = False):
        super().__init__()
        # Keep a Conv2d module to own parameters (depthwise)
        self.conv2d = torch.nn.Conv2d(in_channels, in_channels, kernel_size=(kernel_size, 1),
                                      stride=stride, padding=padding, dilation=dilation,
                                      groups=in_channels, bias=bias)
        self.stride = stride
        self.padding = padding
        self.dilation = dilation

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        weight = self.conv2d.weight
        if self.conv2d.bias is not None:
            bias = self.conv2d.bias
        else:
            # Provide zero bias for required input slot
            bias = torch.zeros(x.size(1), dtype=x.dtype, device=x.device)
        return custom_ops_lib.conv_depthwise_2d_square_input_asymmetric_kernel_custom(
            x, weight, bias, self.stride, self.padding, self.dilation
