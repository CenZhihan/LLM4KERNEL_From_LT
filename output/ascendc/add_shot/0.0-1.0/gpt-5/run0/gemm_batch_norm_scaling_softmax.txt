inline uint32_t GetDimSafe(const gert::Shape* s, uint32_t idx) {
    if (s == nullptr) return 0;
    auto os = s->GetOriginShape();
    if (idx >= os.GetDimNum()) return 0;
    return static_cast<uint32_t>(os.GetDim(idx));
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    GemmBatchNormScalingSoftmaxCustomTilingData tiling;

    const gert::Shape* x_shape = context->GetInputShape(0);     // [M, K]
    const gert::Shape* w_shape = context->GetInputShape(1);     // [N, K] (row-major, like PyTorch Linear weight)

    uint32_t m = GetDimSafe(x_shape, 0);
    uint32_t kx = GetDimSafe(x_shape, 1);
    uint32_t n = GetDimSafe(w_shape, 0);
    uint32_t kw = GetDimSafe(w_shape, 1);

    // Basic validation (best-effort)
    if (kx != kw) {
        // let runtime fail gracefully if mismatched
    }

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_m(m);
    tiling.set_n(n);
    tiling.set_k(kx);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* workspace = context->GetWorkspaceSizes(1);
    // workspace to store logits before softmax: M * N * sizeof(float)
    workspace[0] = static_cast<size_t>(m) * static_cast<size_t>(n) * sizeof(float);
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0); // [M, K]
    const gert::Shape* w_shape = context->GetInputShape(1); // [N, K]
    if (x_shape == nullptr || w_shape == nullptr) {
        return GRAPH_FAILED;
    }
    auto xo = x_shape->GetOriginShape();
    auto wo = w_shape->GetOriginShape();
    if (xo.GetDimNum() < 2 || wo.GetDimNum() < 2) {
        return GRAPH_FAILED;
    }
    int64_t m = xo.GetDim(0);
    int64_t n = wo.GetDim(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    gert::Shape::OriginShape yOrigin;
    yOrigin.SetDimNum(2);
    yOrigin.SetDim(0, m);
    yOrigin.SetDim(1, n);
    y_shape->SetOriginShape(yOrigin);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "gemm_batch_norm_scaling_softmax_custom_tiling_data.h"
#include <cfloat>
#include <cmath>

extern "C" __global__ __aicore__ void gemm_batch_norm_scaling_softmax_custom(
    GM_ADDR x,
    GM_ADDR weight,
    GM_ADDR bias,
    GM_ADDR running_mean,
    GM_ADDR running_var,
    GM_ADDR bn_weight,
    GM_ADDR bn_bias,
    GM_ADDR scale,
    GM_ADDR eps,
    GM_ADDR y,
    GM_ADDR workspace,
    GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    const int32_t M = static_cast<int32_t>(tiling_data.m);
    const int32_t N = static_cast<int32_t>(tiling_data.n);
    const int32_t K = static_cast<int32_t>(tiling_data.k);

    __gm__ float* x_gm  = (__gm__ float*)x;           // [M, K]
    __gm__ float* w_gm  = (__gm__ float*)weight;      // [N, K] (row-major)
    __gm__ float* b_gm  = (__gm__ float*)bias;        // [N]
    __gm__ float* rm_gm = (__gm__ float*)running_mean;// [N]
    __gm__ float* rv_gm = (__gm__ float*)running_var; // [N]
    __gm__ float* gw_gm = (__gm__ float*)bn_weight;   // [N]
    __gm__ float* gb_gm = (__gm__ float*)bn_bias;     // [N]
    __gm__ float* y_gm  = (__gm__ float*)y;           // [M, N]
    __gm__ float* ws_gm = (__gm__ float*)workspace;   // [M, N]
    float eps_val       = ((__gm__ float*)eps)[0];    // scalar
    float scale_val     = ((__gm__ float*)scale)[0];  // scalar

    const int32_t blockNum = AscendC::GetBlockNum();
    const int32_t blockIdx = AscendC::GetBlockIdx();
    const int32_t rowsPerBlock = (M + blockNum - 1) / blockNum;
    const int32_t rowStart = blockIdx * rowsPerBlock;
    const int32_t rowEnd   = rowStart + rowsPerBlock > M ? M : rowStart + rowsPerBlock;

    // Pass 1: GEMM + bias + BatchNorm (inference) + scaling -> store logits to workspace
    for (int32_t i = rowStart; i < rowEnd; ++i) {
        const int32_t x_base = i * K;
        const int32_t ws_base = i * N;
        for (int32_t j = 0; j < N; ++j) {
            // Dot product: row x[i,:] with weight row w[j,:]
            float acc = b_gm[j];
            const int32_t w_base = j * K;
            for (int32_t t = 0; t < K; ++t) {
                acc += x_gm[x_base + t] * w_gm[w_base + t];
            }
            // BatchNorm (inference with running stats)
            float mean  = rm_gm[j];
            float var   = rv_gm[j];
            float gamma = gw_gm[j];
            float beta  = gb_gm[j];
            float norm = (acc - mean) / sqrtf(var + eps_val);
            float bn_out = norm * gamma + beta;
            float scaled = bn_out * scale_val;
            ws_gm[ws_base + j] = scaled;
        }
    }

    // Pass 2: Row-wise softmax on logits in workspace -> write to y
    for (int32_t i = rowStart; i < rowEnd; ++i) {
        const int32_t base = i * N;
        // max
        float maxv = -FLT_MAX;
        for (int32_t j = 0; j < N; ++j) {
            float v = ws_gm[base + j];
            maxv = v > maxv ? v : maxv;
        }
        // exp and sum
        float sumv = 0.f;
        for (int32_t j = 0; j < N; ++j) {
            float e = expf(ws_gm[base + j] - maxv);
            ws_gm[base + j] = e;
            sumv += e;
        }
        float inv_sum = 1.f / sumv;
        for (int32_t j = 0; j < N; ++j) {
            y_gm[base + j] = ws_gm[base + j] * inv_sum;
        }
    }
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused: GEMM (Linear) + BatchNorm (inference) + scaling + Softmax(dim=1)
    """
    def __init__(self, in_features, out_features, bn_eps=1e-5, bn_momentum=0.1, scale_shape=(1,)):
        super(ModelNew, self).__init__()
        self.gemm = torch.nn.Linear(in_features, out_features)
        self.bn = torch.nn.BatchNorm1d(out_features, eps=bn_eps, momentum=bn_momentum)
        self.scale = torch.nn.Parameter(torch.ones(scale_shape))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Use BN running stats (inference-style) in the fused kernel.
        eps_tensor = torch.tensor([self.bn.eps], dtype=x.dtype, device=x.device)
        return custom_ops_lib.gemm_batch_norm_scaling_softmax_custom(
            x,
            self.gemm.weight,
            self.gemm.bias if self.gemm.bias is not None else torch.zeros(self.gemm.out_features, dtype=x.dtype, device=x.device),
            self.bn.running_mean,
            self.bn.running_var,
            self.bn.weight,
            self.bn.bias,
            self.scale,
            eps_tensor
        )
'''
