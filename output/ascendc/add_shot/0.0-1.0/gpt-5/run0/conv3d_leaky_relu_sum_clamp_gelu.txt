kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv3d_leaky_relu_sum_clamp_gelu_custom_tiling_data.h"

class KernelConv3dLeakyReluSumClampGelu {
public:
    __aicore__ inline KernelConv3dLeakyReluSumClampGelu() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR s, GM_ADDR z,
                                const Conv3dLeakyReluSumClampGeluCustomTilingData& t) {
        xGm = (__gm__ float*)x;
        wGm = (__gm__ float*)w;
        bGm = (__gm__ float*)b;
        sGm = (__gm__ float*)s;
        zGm = (__gm__ float*)z;

        n  = t.n();   ci = t.ci(); co = t.co();
        di = t.di();  hi = t.hi(); wi = t.wi();
        kd = t.kd();  kh = t.kh(); kw = t.kw();
        sd = t.sd();  sh = t.sh(); sw = t.sw();
        pd = t.pd();  ph = t.ph(); pw = t.pw();
        dout = t.dout(); hout = t.hout(); wout = t.wout();

        totalLength = t.totalLength();
        blockLength = totalLength / AscendC::GetBlockNum();
        blockStart  = blockLength * AscendC::GetBlockIdx();

        negSlope = 0.2f;
        clampMin = -1.0f;
        clampMax =  1.0f;
    }

    __aicore__ inline void Process() {
        uint64_t end = blockStart + blockLength;
        for (uint64_t idx = blockStart; idx < end; ++idx) {
            uint64_t t = idx;
            uint32_t w_o = (uint32_t)(t % wout); t /= wout;
            uint32_t h_o = (uint32_t)(t % hout); t /= hout;
            uint32_t d_o = (uint32_t)(t % dout); t /= dout;
            uint32_t c_o = (uint32_t)(t % co);   t /= co;
            uint32_t n_o = (uint32_t)(t);

            float acc = 0.0f;

            int32_t in_d_base = (int32_t)d_o * (int32_t)sd - (int32_t)pd;
            int32_t in_h_base = (int32_t)h_o * (int32_t)sh - (int32_t)ph;
            int32_t in_w_base = (int32_t)w_o * (int32_t)sw - (int32_t)pw;

            // Strides for input [N, Ci, Di, Hi, Wi]
            uint64_t in_stride_w = 1;
            uint64_t in_stride_h = (uint64_t)wi;
            uint64_t in_stride_d = (uint64_t)hi * (uint64_t)wi;
            uint64_t in_stride_c = (uint64_t)di * (uint64_t)hi * (uint64_t)wi;
            uint64_t in_stride_n = (uint64_t)ci * (uint64_t)di * (uint64_t)hi * (uint64_t)wi;

            // Strides for weight [Co, Ci, Kd, Kh, Kw]
            uint64_t w_stride_w = 1;
            uint64_t w_stride_h = (uint64_t)kw;
            uint64_t w_stride_d = (uint64_t)kh * (uint64_t)kw;
            uint64_t w_stride_c = (uint64_t)kd * (uint64_t)kh * (uint64_t)kw;
            uint64_t w_stride_o = (uint64_t)ci * (uint64_t)kd * (uint64_t)kh * (uint64_t)kw;

            for (uint32_t c_i = 0; c_i < ci; ++c_i) {
                for (uint32_t kd_i = 0; kd_i < kd; ++kd_i) {
                    int32_t in_d = in_d_base + (int32_t)kd_i;
                    if (in_d < 0 || in_d >= (int32_t)di) continue;
                    for (uint32_t kh_i = 0; kh_i < kh; ++kh_i) {
                        int32_t in_h = in_h_base + (int32_t)kh_i;
                        if (in_h < 0 || in_h >= (int32_t)hi) continue;
                        for (uint32_t kw_i = 0; kw_i < kw; ++kw_i) {
                            int32_t in_w = in_w_base + (int32_t)kw_i;
                            if (in_w < 0 || in_w >= (int32_t)wi) continue;

                            uint64_t in_off =
                                (uint64_t)n_o * in_stride_n +
                                (uint64_t)c_i * in_stride_c +
                                (uint64_t)in_d * in_stride_d +
                                (uint64_t)in_h * in_stride_h +
                                (uint64_t)in_w * in_stride_w;

                            uint64_t w_off =
                                (uint64_t)c_o * w_stride_o +
                                (uint64_t)c_i * w_stride_c +
                                (uint64_t)kd_i * w_stride_d +
                                (uint64_t)kh_i * w_stride_h +
                                (uint64_t)kw_i * w_stride_w;

                            float xv = xGm[in_off];
                            float wv = wGm[w_off];
                            acc += xv * wv;
                        }
                    }
                }
            }

            // bias add
            acc += bGm[c_o];

            // LeakyReLU
            float v = (acc >= 0.0f) ? acc : (acc * negSlope);

            // Add broadcast sum tensor s shape [Co, 1, 1, 1]
            v += sGm[c_o];

            // Clamp
            if (v < clampMin) v = clampMin;
            if (v > clampMax) v = clampMax;

            // GELU approximate using tanh rational approximation
            float x = v;
            float x3 = x * x * x;
            float t0 = 0.79788456f * (x + 0.044715f * x3); // sqrt(2/pi)*(x + 0.044715x^3)
            float t02 = t0 * t0;
            // tanh(t0) approx: t0*(27 + t0^2)/(27 + 9*t0^2)
            float tanh_t0 = t0 * (27.0f + t02) / (27.0f + 9.0f * t02);
            float gelu = 0.5f * x * (1.0f + tanh_t0);

            // write output in [N, Co, Do, Ho, Wo] contiguous
            zGm[idx] = gelu;
        }
    }

private:
    __gm__ float* xGm;
    __gm__ float* wGm;
    __gm__ float* bGm;
    __gm__ float* sGm;
    __gm__ float* zGm;

    uint32_t n, ci, co;
    uint32_t di, hi, wi;
    uint32_t kd, kh, kw;
    uint32_t sd, sh, sw;
    uint32_t pd, ph, pw;
    uint32_t dout, hout, wout;

    uint64_t totalLength;
    uint64_t blockLength;
    uint64_t blockStart;

    float negSlope;
    float clampMin;
    float clampMax;
};

extern "C" __global__ __aicore__ void conv3d_leaky_relu_sum_clamp_gelu_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR s, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {

    GET_TILING_DATA(tiling_data, tiling);

    KernelConv3dLeakyReluSumClampGelu op;
    op.Init(x, w, b, s, z, tiling_data);
    op.Process();
}
