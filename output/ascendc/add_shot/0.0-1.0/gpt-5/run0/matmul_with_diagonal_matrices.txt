    MatmulWithDiagonalMatricesCustomTilingData tiling;

    const auto* b_shape = context->GetInputShape(1);

    uint32_t n = a_shape->GetOriginShape().GetShapeSize(); // A is 1D of length N
    // B is 2D: (N, M)
    uint32_t m = 1;
    if (b_shape->GetOriginShape().GetDimNum() >= 2) {

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "matmul_with_diagonal_matrices_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelRowScale {
public:
    __aicore__ inline KernelRowScale() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c, uint32_t n, uint32_t m, uint32_t tileNum)
    {
        this->n = n;
        this->m = m;
        this->tileNum = tileNum;

        this->rowsPerBlock = n / AscendC::GetBlockNum();  // assume divisible
        this->rowStart = this->rowsPerBlock * AscendC::GetBlockIdx();
        this->tileLength = (m / tileNum) / BUFFER_NUM;    // assume divisible

        aGm.SetGlobalBuffer((__gm__ DTYPE_X *)a + this->rowStart, this->rowsPerBlock);
        bGm.SetGlobalBuffer((__gm__ DTYPE_Y *)b + (uint64_t)this->rowStart * m, (uint64_t)this->rowsPerBlock * m);
        cGm.SetGlobalBuffer((__gm__ DTYPE_Z *)c + (uint64_t)this->rowStart * m, (uint64_t)this->rowsPerBlock * m);

        aPtr = (__gm__ DTYPE_X *)a + this->rowStart;

        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueC, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t innerLoop = this->tileNum * BUFFER_NUM; // tiles per row
        for (uint32_t r = 0; r < this->rowsPerBlock; ++r) {
            DTYPE_X aVal = aPtr[r]; // scalar for this row
            for (int32_t i = 0; i < innerLoop; ++i) {
                CopyIn(r, i);
                Compute(aVal);
                CopyOut(r, i);
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t row, int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> bLocal = inQueueB.AllocTensor<DTYPE_Y>();
        uint64_t offset = (uint64_t)row * m + (uint64_t)progress * this->tileLength;
        AscendC::DataCopy(bLocal, bGm[offset], this->tileLength);
        inQueueB.EnQue(bLocal);
    }

    __aicore__ inline void Compute(DTYPE_X aVal)
    {
        AscendC::LocalTensor<DTYPE_Y> bLocal = inQueueB.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> cLocal = outQueueC.AllocTensor<DTYPE_Z>();
        AscendC::Muls(cLocal, bLocal, aVal, this->tileLength);
        outQueueC.EnQue<DTYPE_Z>(cLocal);
        inQueueB.FreeTensor(bLocal);
    }

    __aicore__ inline void CopyOut(uint32_t row, int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> cLocal = outQueueC.DeQue<DTYPE_Z>();
        uint64_t offset = (uint64_t)row * m + (uint64_t)progress * this->tileLength;
        AscendC::DataCopy(cGm[offset], cLocal, this->tileLength);
        outQueueC.FreeTensor(cLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC;

    AscendC::GlobalTensor<DTYPE_X> aGm;
    AscendC::GlobalTensor<DTYPE_Y> bGm;
    AscendC::GlobalTensor<DTYPE_Z> cGm;

    __gm__ DTYPE_X* aPtr;

    uint32_t n;
    uint32_t m;
    uint32_t tileNum;

    uint32_t rowsPerBlock;
    uint32_t rowStart;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void matmul_with_diagonal_matrices_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelRowScale op;
    op.Init(a, b, c, tiling_data.n, tiling_data.m, tiling_data.tileNum);
    op.Process();
}
