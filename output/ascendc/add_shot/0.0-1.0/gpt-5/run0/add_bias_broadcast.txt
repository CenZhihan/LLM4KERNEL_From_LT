kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "add_bias_broadcast_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelAddBiasBroadcast {
public:
    __aicore__ inline KernelAddBiasBroadcast() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        bGm.SetGlobalBuffer((__gm__ DTYPE_Y *)bias, 1); // bias expected to be scalar or already expanded by host
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }
    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> bLocal = inQueueB.AllocTensor<DTYPE_Y>();

        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);

        // If bias is expanded by host to full size, read tile from GM.
        // If bias is scalar, duplicate scalar value across tile.
        if (biasIsExpanded) {
            AscendC::DataCopy(bLocal, bGm[progress * this->tileLength], this->tileLength);
        } else {
            // load scalar bias from GM and duplicate
            AscendC::LocalTensor<DTYPE_Y> tmp = pipe.AllocTmpBuffer<DTYPE_Y>(1);
            AscendC::DataCopy(tmp, bGm, 1);
            AscendC::Duplicate(bLocal, tmp.GetValue(0), this->tileLength);
            pipe.FreeTmpBuffer(tmp);
        }

        inQueueX.EnQue(xLocal);
        inQueueB.EnQue(bLocal);
    }
    __aicore__ inline void Compute(int32_t)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> bLocal = inQueueB.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        AscendC::Add(yLocal, xLocal, bLocal, this->tileLength);

        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueB.FreeTensor(bLocal);
    }
    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> bGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
    bool biasIsExpanded = true;
};

extern "C" __global__ __aicore__ void add_bias_broadcast_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelAddBiasBroadcast op;
    op.Init(x, bias, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor add_bias_broadcast_custom_impl_npu(const at::Tensor& self, const at::Tensor& bias) {
    at::Tensor bias_exp = bias;
    if (bias.numel() == 1 && bias.dim() == 0) {
        bias_exp = bias.expand_as(self).contiguous();
    } else {
        TORCH_CHECK(bias.sizes().equals(self.sizes()), "bias must be a scalar or have the same shape as x");
        if (!bias.is_contiguous()) {
            bias_exp = bias.contiguous();
        }
    }
    at::Tensor result = at::empty_like(self);
    EXEC_NPU_CMD(aclnnAddBiasBroadcastCustom, self, bias_exp, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("add_bias_broadcast_custom", &add_bias_broadcast_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("add_bias_broadcast_custom", &add_bias_broadcast_custom_impl_npu, "x + bias (with scalar broadcast)");
}
