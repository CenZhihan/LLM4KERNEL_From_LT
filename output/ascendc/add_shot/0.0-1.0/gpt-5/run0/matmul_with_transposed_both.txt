kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "matmul_with_transposed_both_custom_tiling_data.h"

// This kernel computes: C = A^T @ B^T
// A shape: (K, M)
// B shape: (N, K)
// C shape: (M, N)

constexpr int32_t INBUF_NUM = 2;
constexpr int32_t OUTBUF_NUM = 2;

class KernelMatmulTT {
public:
    __aicore__ inline KernelMatmulTT() {}
    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK) {
        this->M = M;
        this->N = N;
        this->K = K;
        this->tileM = tileM;
        this->tileN = tileN;
        this->tileK = tileK;

        aGm.SetGlobalBuffer((__gm__ DTYPE_X*)a, (uint64_t)K * (uint64_t)M); // A is (K, M)
        bGm.SetGlobalBuffer((__gm__ DTYPE_Y*)b, (uint64_t)N * (uint64_t)K); // B is (N, K)
        cGm.SetGlobalBuffer((__gm__ DTYPE_Z*)c, (uint64_t)M * (uint64_t)N); // C is (M, N)

        // One queue each for A tiles, B tiles and C tiles
        pipe.InitBuffer(inQueueA, INBUF_NUM, this->tileM * this->tileK * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueB, INBUF_NUM, this->tileN * this->tileK * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueC, OUTBUF_NUM, this->tileM * this->tileN * sizeof(DTYPE_Z));

        // Split rows of C across blocks
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        rowsPerBlock = (M + blockNum - 1) / blockNum;
        mStart = blockIdx * rowsPerBlock;
        mEnd = AscendC::Min(mStart + rowsPerBlock, M);
    }

    __aicore__ inline void Process() {
        for (uint32_t m0 = mStart; m0 < mEnd; m0 += tileM) {
            uint32_t mEff = AscendC::Min(tileM, mEnd - m0);
            for (uint32_t n0 = 0; n0 < N; n0 += tileN) {
                uint32_t nEff = AscendC::Min(tileN, N - n0);

                // Prepare output tile buffer
                AscendC::LocalTensor<DTYPE_Z> cTile = outQueueC.AllocTensor<DTYPE_Z>();
                AscendC::SetZero(cTile, mEff * nEff);

                // Accumulate over K
                for (uint32_t k0 = 0; k0 < K; k0 += tileK) {
                    uint32_t kEff = AscendC::Min(tileK, K - k0);

                    // Load A tile: shape (mEff, kEff) but A is stored as (K, M), we will load block (kEff, mEff)
                    AscendC::LocalTensor<DTYPE_X> aTile = inQueueA.AllocTensor<DTYPE_X>();
                    // Load B tile: shape (nEff, kEff) but B is stored as (N, K), we will load block (nEff, kEff)
                    AscendC::LocalTensor<DTYPE_Y> bTile = inQueueB.AllocTensor<DTYPE_Y>();

                    // Copy A block (kEff x mEff) from GM with stride
                    // Each row in GM has length M for A
                    for (uint32_t kk = 0; kk < kEff; ++kk) {
                        AscendC::DataCopy(aTile[kk * mEff],
                                          aGm[(k0 + kk) * M + m0],
                                          mEff);
                    }
                    // Copy B block (nEff x kEff) from GM with stride
                    // Each row in GM has length K for B
                    for (uint32_t nn = 0; nn < nEff; ++nn) {
                        AscendC::DataCopy(bTile[nn * kEff],
                                          bGm[(n0 + nn) * K + k0],
                                          kEff);
                    }

                    inQueueA.EnQue(aTile);
                    inQueueB.EnQue(bTile);

                    // Compute C += A^T (mEff x kEff) @ B^T (kEff x nEff)
                    AscendC::LocalTensor<DTYPE_X> aBlk = inQueueA.DeQue<DTYPE_X>();
                    AscendC::LocalTensor<DTYPE_Y> bBlk = inQueueB.DeQue<DTYPE_Y>();

                    // Perform naive GEMM on tiles using vector FMA style loops
                    for (uint32_t mm = 0; mm < mEff; ++mm) {
                        for (uint32_t nn = 0; nn < nEff; ++nn) {
                            // Accumulate dot over kEff
                            DTYPE_Z acc = static_cast<DTYPE_Z>(0);
                            for (uint32_t kk = 0; kk < kEff; ++kk) {
                                // aBlk is (kEff x mEff) laid out row-major; element a(kk, mm)
                                DTYPE_X a_val = aBlk[kk * mEff + mm];
                                // bBlk is (nEff x kEff) laid out row-major; element b(nn, kk)
                                DTYPE_Y b_val = bBlk[nn * kEff + kk];
                                acc = acc + static_cast<DTYPE_Z>(a_val) * static_cast<DTYPE_Z>(b_val);
                            }
                            cTile[mm * nEff + nn] = cTile[mm * nEff + nn] + acc;
                        }
                    }

                    inQueueA.FreeTensor(aBlk);
                    inQueueB.FreeTensor(bBlk);
                }

                // Write C tile back to GM at (m0, n0)
                // C is stored as (M, N)
                for (uint32_t mm = 0; mm < mEff; ++mm) {
                    AscendC::DataCopy(cGm[(m0 + mm) * N + n0], cTile[mm * nEff], nEff);
                }
                outQueueC.FreeTensor(cTile);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, INBUF_NUM>  inQueueA;
    AscendC::TQue<AscendC::TPosition::VECIN, INBUF_NUM>  inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUTBUF_NUM> outQueueC;

    AscendC::GlobalTensor<DTYPE_X> aGm;
    AscendC::GlobalTensor<DTYPE_Y> bGm;
    AscendC::GlobalTensor<DTYPE_Z> cGm;

    uint32_t M, N, K;
    uint32_t tileM, tileN, tileK;

    uint32_t rowsPerBlock;
    uint32_t mStart;
    uint32_t mEnd;
};

extern "C" __global__ __aicore__ void matmul_with_transposed_both_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulTT op;
    op.Init(a, b, c,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK);
    op.Process();
}
