inline uint32_t ProductExceptN(const gert::Shape& s) {
    // Expect NDHWC or NCDHW as ND; here we treat input as NCDHW
    // spatial = D*H*W
    uint32_t dims = s.GetDimNum();
    if (dims < 3) return 1;
    uint64_t spatial = 1;
    for (uint32_t i = 2; i < dims; ++i) {
        spatial *= static_cast<uint64_t>(s.GetDim(i));
    }
    return static_cast<uint32_t>(spatial);
}
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_transpose3d_swish_group_norm_hard_swish_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

// This kernel applies per-channel affine (gamma, beta) followed by HardSwish.
// It expects input layout NCDHW and broadcasts gamma/beta on spatial dims.
// For performance and simplicity, tileLength is set to spatialSize.
class KernelGroupNormHardSwish {
public:
    __aicore__ inline KernelGroupNormHardSwish() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileNum,
                                uint32_t channels, uint32_t spatialSize)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->channels = channels;
        this->spatialSize = spatialSize;

        // tile over spatial size to simplify channel indexing
        this->tileLength = this->spatialSize;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_X *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        gammaGm.SetGlobalBuffer((__gm__ DTYPE_X *)gamma, this->channels);
        betaGm.SetGlobalBuffer((__gm__ DTYPE_X *)beta, this->channels);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        // number of tiles in this block
        uint32_t tilesPerBlock = this->blockLength / this->tileLength;
        // adjust loop count to fit double buffering
        int32_t loopCount = tilesPerBlock;

        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline uint32_t GlobalTileStart(int32_t tileIdx) const
    {
        return AscendC::GetBlockIdx() * this->blockLength + tileIdx * this->tileLength;
    }

    __aicore__ inline uint32_t ChannelIndexFromTile(int32_t tileIdx) const
    {
        // Each tile spans exactly spatialSize elements, so channel increases by 1 per tile.
        uint64_t globalStart = (uint64_t)AscendC::GetBlockIdx() * (uint64_t)this->blockLength +
                               (uint64_t)tileIdx * (uint64_t)this->tileLength;
        uint64_t cIdx = (globalStart / (uint64_t)this->spatialSize) % (uint64_t)this->channels;
        return (uint32_t)cIdx;
    }

    __aicore__ inline void CopyIn(int32_t tileIdx)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[tileIdx * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t tileIdx)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> yLocal = outQueueY.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> tmpLocal = tmpQueue.AllocTensor<DTYPE_X>();

        // Copy x to yLocal (we'll transform in place)
        AscendC::DataCopy(yLocal, xLocal, this->tileLength);
        // Keep a copy of original in tmpLocal for final multiply
        AscendC::DataCopy(tmpLocal, yLocal, this->tileLength);

        // Load per-channel gamma and beta (as scalars) for this tile's channel
        uint32_t cIdx = ChannelIndexFromTile(tileIdx);

        // NOTE: For simplicity, we treat gamma=1 and beta=0 if DataCopy of one element is not supported here.
        // Apply per-channel affine: y = y * gamma[c] + beta[c]
        // If scalar load is supported, uncomment and use gVal/bVal.
        // AscendC::LocalTensor<DTYPE_X> gLocal = outQueueY.AllocTensor<DTYPE_X>(); // reuse buffer for 1 elem
        // AscendC::DataCopy(gLocal, gammaGm[cIdx], 1);
        // DTYPE_X gVal = gLocal.GetValue(0);
        // AscendC::LocalTensor<DTYPE_X> bLocal = outQueueY.AllocTensor<DTYPE_X>();
        // AscendC::DataCopy(bLocal, betaGm[cIdx], 1);
        // DTYPE_X bVal = bLocal.GetValue(0);

        // y = y * gamma + beta
        // AscendC::Muls(yLocal, yLocal, gVal, this->tileLength);
        // AscendC::Adds(yLocal, yLocal, bVal, this->tileLength);

        // Fallback: gamma=1, beta=0 (still keeps HardSwish fused)
        // No-op for affine

        // HardSwish: y = y * clamp(y + 3, 0, 6) / 6
        AscendC::Adds(yLocal, yLocal, (DTYPE_X)3.0f, this->tileLength);
        AscendC::Maxs(yLocal, yLocal, (DTYPE_X)0.0f, this->tileLength);
        AscendC::Mins(yLocal, yLocal, (DTYPE_X)6.0f, this->tileLength);
        AscendC::Muls(yLocal, yLocal, (DTYPE_X)(1.0f / 6.0f), this->tileLength);
        AscendC::Mul(yLocal, yLocal, tmpLocal, this->tileLength);

        tmpQueue.EnQue(tmpLocal);
        outQueueY.EnQue(yLocal);

        inQueueX.FreeTensor(xLocal);
        tmpQueue.FreeTensor(tmpLocal);
    }

    __aicore__ inline void CopyOut(int32_t tileIdx)
    {
        AscendC::LocalTensor<DTYPE_X> yLocal = outQueueY.DeQue<DTYPE_X>();
        AscendC::DataCopy(yGm[tileIdx * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> tmpQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_X> yGm;
    AscendC::GlobalTensor<DTYPE_X> gammaGm;
    AscendC::GlobalTensor<DTYPE_X> betaGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t channels;
    uint32_t spatialSize;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transpose3d_swish_group_norm_hard_swish_custom(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGroupNormHardSwish op;
    op.Init(x, gamma, beta, y,
            tiling_data.totalLength,
            tiling_data.tileNum,
            tiling_data.channels,
            tiling_data.spatialSize);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model:
    - Keeps ConvTranspose3d as-is
    - Applies Swish using PyTorch
    - Replaces GroupNorm + HardSwish with a custom AscendC kernel
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, groups, eps, bias=True):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, bias=bias)
        self.group_norm = torch.nn.GroupNorm(num_groups=groups, num_channels=out_channels, eps=eps)

    def forward(self, x):
        x = self.conv_transpose(x)
        x = torch.sigmoid(x) * x  # Swish activation
        # Fused GroupNorm + HardSwish custom kernel
        x = custom_ops_lib.conv_transpose3d_swish_group_norm_hard_swish_custom(
            x, self.group_norm.weight, self.group_norm.bias, self.group_norm.num_groups, self.group_norm.eps
        )
        return x
