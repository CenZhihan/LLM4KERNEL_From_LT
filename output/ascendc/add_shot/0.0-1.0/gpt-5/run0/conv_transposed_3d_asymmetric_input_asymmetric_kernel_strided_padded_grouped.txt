kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_transposed_3d_asymmetric_input_asymmetric_kernel_strided_padded_grouped_custom_tiling_data.h"

extern "C" __global__ __aicore__ void conv_transposed_3d_asymmetric_input_asymmetric_kernel_strided_padded_grouped_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {

    GET_TILING_DATA(tiling_data, tiling);

    // Extract tiling parameters
    const uint32_t N     = tiling_data.N;
    const uint32_t Cin   = tiling_data.Cin;
    const uint32_t Cout  = tiling_data.Cout;
    const uint32_t groups = tiling_data.groups;

    const uint32_t inD = tiling_data.inD;
    const uint32_t inH = tiling_data.inH;
    const uint32_t inW = tiling_data.inW;

    const uint32_t kD = tiling_data.kD;
    const uint32_t kH = tiling_data.kH;
    const uint32_t kW = tiling_data.kW;

    const int32_t strideD = tiling_data.strideD;
    const int32_t strideH = tiling_data.strideH;
    const int32_t strideW = tiling_data.strideW;

    const int32_t padD = tiling_data.padD;
    const int32_t padH = tiling_data.padH;
    const int32_t padW = tiling_data.padW;

    const uint32_t outD = tiling_data.outD;
    const uint32_t outH = tiling_data.outH;
    const uint32_t outW = tiling_data.outW;

    const uint64_t totalOutElems = tiling_data.totalOutElems;

    // Cast global memory pointers
    __gm__ float* xGm = reinterpret_cast<__gm__ float*>(x);
    __gm__ float* wGm = reinterpret_cast<__gm__ float*>(w);
    __gm__ float* bGm = reinterpret_cast<__gm__ float*>(b);
    __gm__ float* yGm = reinterpret_cast<__gm__ float*>(y);

    // Work partitioning
    uint32_t blockNum = AscendC::GetBlockNum();
    uint32_t blockIdx = AscendC::GetBlockIdx();
    uint64_t base = (totalOutElems / blockNum) * blockIdx;
    uint64_t tail = (blockIdx == (blockNum - 1)) ? (totalOutElems - (totalOutElems / blockNum) * (blockNum - 1)) : (totalOutElems / blockNum);
    uint64_t start = base;
    uint64_t end = base + tail;

    uint64_t outStrideW = 1;
    uint64_t outStrideH = outW;
    uint64_t outStrideD = outH * outW;
    uint64_t outStrideC = outD * outH * outW;
    uint64_t outStrideN = Cout * outStrideC;

    uint64_t inStrideW = 1;
    uint64_t inStrideH = inW;
    uint64_t inStrideD = inH * inW;
    uint64_t inStrideC = inD * inH * inW;
    uint64_t inStrideN = Cin * inStrideC;

    uint64_t wStrideW = 1;
    uint64_t wStrideH = kW;
    uint64_t wStrideD = kH * kW;
    // weight layout: [Cin, CoutPerGroup, kD, kH, kW]
    uint32_t CoutPerGroup = Cout / groups;
    uint64_t wStrideCoutPerGroup = kD * kH * kW;
    uint64_t wStrideCin = CoutPerGroup * wStrideCoutPerGroup;

    for (uint64_t idx = start; idx < end; ++idx) {
        // Decode output tensor indices (N, Cout, outD, outH, outW) in NCDHW order
        uint64_t tmp = idx;
        uint32_t ow = static_cast<uint32_t>(tmp % outW); tmp /= outW;
        uint32_t oh = static_cast<uint32_t>(tmp % outH); tmp /= outH;
        uint32_t od = static_cast<uint32_t>(tmp % outD); tmp /= outD;
        uint32_t oc = static_cast<uint32_t>(tmp % Cout); tmp /= Cout;
        uint32_t n  = static_cast<uint32_t>(tmp);

        float acc = 0.0f;

        uint32_t g = oc / CoutPerGroup;
        uint32_t ocInGroup = oc % CoutPerGroup;

        uint32_t cinPerGroup = Cin / groups;
        uint32_t cinStart = g * cinPerGroup;
        uint32_t cinEnd = cinStart + cinPerGroup;

        for (uint32_t ci = cinStart; ci < cinEnd; ++ci) {
            for (uint32_t kd = 0; kd < kD; ++kd) {
                int32_t inDnum = static_cast<int32_t>(od) + padD - static_cast<int32_t>(kd);
                if (inDnum % strideD != 0) continue;
                int32_t id = inDnum / strideD;
                if (id < 0 || id >= static_cast<int32_t>(inD)) continue;

                for (uint32_t kh = 0; kh < kH; ++kh) {
                    int32_t inHnum = static_cast<int32_t>(oh) + padH - static_cast<int32_t>(kh);
                    if (inHnum % strideH != 0) continue;
                    int32_t ih = inHnum / strideH;
                    if (ih < 0 || ih >= static_cast<int32_t>(inH)) continue;

                    for (uint32_t kw = 0; kw < kW; ++kw) {
                        int32_t inWnum = static_cast<int32_t>(ow) + padW - static_cast<int32_t>(kw);
                        if (inWnum % strideW != 0) continue;
                        int32_t iw = inWnum / strideW;
                        if (iw < 0 || iw >= static_cast<int32_t>(inW)) continue;

                        uint64_t xIndex = static_cast<uint64_t>(n) * inStrideN
                                        + static_cast<uint64_t>(ci) * inStrideC
                                        + static_cast<uint64_t>(id) * inStrideD
                                        + static_cast<uint64_t>(ih) * inStrideH
                                        + static_cast<uint64_t>(iw) * inStrideW;

                        uint64_t wIndex = static_cast<uint64_t>(ci) * wStrideCin
                                        + static_cast<uint64_t>(ocInGroup) * wStrideCoutPerGroup
                                        + static_cast<uint64_t>(kd) * wStrideD
                                        + static_cast<uint64_t>(kh) * wStrideH
                                        + static_cast<uint64_t>(kw) * wStrideW;

                        acc += xGm[xIndex] * wGm[wIndex];
                    }
                }
            }
        }

        if (bGm != nullptr) {
            acc += bGm[oc];
        }

        uint64_t yIndex = static_cast<uint64_t>(n) * outStrideN
                        + static_cast<uint64_t>(oc) * outStrideC
                        + static_cast<uint64_t>(od) * outStrideD
                        + static_cast<uint64_t>(oh) * outStrideH
                        + static_cast<uint64_t>(ow) * outStrideW;

        yGm[yIndex] = acc;
    }
}

model_src='''
import torch
import torch.nn as nn
import torch.nn.init as init
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    """
    Custom 3D transposed convolution using an AscendC kernel:
    ConvTransposed3dAsymmetricInputAsymmetricKernelStridedPaddedGroupedCustom
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: tuple, stride: tuple = (1, 1, 1),
                 padding: tuple = (0, 0, 0), output_padding: tuple = (0, 0, 0), groups: int = 1, bias: bool = False):
        super().__init__()
        assert out_channels % groups == 0, "out_channels must be divisible by groups"
        assert in_channels % groups == 0, "in_channels must be divisible by groups"
        kd, kh, kw = kernel_size
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = (kd, kh, kw)
        self.stride = stride
        self.padding = padding
        self.output_padding = output_padding
        self.groups = groups
        # Weight layout: [Cin, CoutPerGroup, kD, kH, kW]
        cout_per_group = out_channels // groups
        self.weight = nn.Parameter(torch.empty(in_channels, cout_per_group, kd, kh, kw))
        if bias:
            self.bias = nn.Parameter(torch.empty(out_channels))
        else:
            self.register_parameter("bias", None)
        self.reset_parameters()

    def reset_parameters(self):
        # Kaiming uniform similar style
        init.kaiming_uniform_(self.weight, a=math.sqrt(5))
        if self.bias is not None:
            fan_in = self.in_channels * self.kernel_size[0] * self.kernel_size[1] * self.kernel_size[2] // self.groups
            bound = 1 / math.sqrt(fan_in) if fan_in > 0 else 0
            init.uniform_(self.bias, -bound, bound)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return custom_ops_lib.conv_transposed_3d_asymmetric_input_asymmetric_kernel_strided_padded_grouped_custom(
            x, self.weight, self.bias, self.stride, self.padding, self.output_padding, self.groups
        )
