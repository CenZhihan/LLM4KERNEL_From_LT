kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "rmsprop_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelRmsprop {
public:
    __aicore__ inline KernelRmsprop() {}
    __aicore__ inline void Init(GM_ADDR param, GM_ADDR grad, GM_ADDR v,
                                GM_ADDR lr, GM_ADDR alpha, GM_ADDR eps,
                                GM_ADDR out,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        paramGm.SetGlobalBuffer((__gm__ float *)param + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        gradGm.SetGlobalBuffer((__gm__ float *)grad + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        vGm.SetGlobalBuffer((__gm__ float *)v + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        lrGm.SetGlobalBuffer((__gm__ float *)lr + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        alphaGm.SetGlobalBuffer((__gm__ float *)alpha + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        epsGm.SetGlobalBuffer((__gm__ float *)eps + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        outGm.SetGlobalBuffer((__gm__ float *)out + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueParam, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueGrad,  BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueV,     BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueLr,    BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueAlpha, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueEps,   BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(outQueueOut,  BUFFER_NUM, this->tileLength * sizeof(float));

        pipe.InitBuffer(tmpBuf1, this->tileLength * sizeof(float));
        pipe.InitBuffer(tmpBuf2, this->tileLength * sizeof(float));
        pipe.InitBuffer(tmpBuf3, this->tileLength * sizeof(float));
    }
    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<float> pLocal = inQueueParam.AllocTensor<float>();
        AscendC::LocalTensor<float> gLocal = inQueueGrad.AllocTensor<float>();
        AscendC::LocalTensor<float> vLocal = inQueueV.AllocTensor<float>();
        AscendC::LocalTensor<float> lrLocal = inQueueLr.AllocTensor<float>();
        AscendC::LocalTensor<float> alphaLocal = inQueueAlpha.AllocTensor<float>();
        AscendC::LocalTensor<float> epsLocal = inQueueEps.AllocTensor<float>();

        AscendC::DataCopy(pLocal,    paramGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(gLocal,    gradGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(vLocal,    vGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(lrLocal,   lrGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(alphaLocal,alphaGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(epsLocal,  epsGm[progress * this->tileLength], this->tileLength);

        inQueueParam.EnQue(pLocal);
        inQueueGrad.EnQue(gLocal);
        inQueueV.EnQue(vLocal);
        inQueueLr.EnQue(lrLocal);
        inQueueAlpha.EnQue(alphaLocal);
        inQueueEps.EnQue(epsLocal);
    }
    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<float> pLocal    = inQueueParam.DeQue<float>();
        AscendC::LocalTensor<float> gLocal    = inQueueGrad.DeQue<float>();
        AscendC::LocalTensor<float> vLocal    = inQueueV.DeQue<float>();
        AscendC::LocalTensor<float> lrLocal   = inQueueLr.DeQue<float>();
        AscendC::LocalTensor<float> alphaLocal= inQueueAlpha.DeQue<float>();
        AscendC::LocalTensor<float> epsLocal  = inQueueEps.DeQue<float>();
        AscendC::LocalTensor<float> outLocal  = outQueueOut.AllocTensor<float>();

        AscendC::LocalTensor<float> t1 = tmpBuf1.Get<float>(); // grad^2 then intermediates
        AscendC::LocalTensor<float> t2 = tmpBuf2.Get<float>(); // (1-alpha) or alpha*v
        AscendC::LocalTensor<float> t3 = tmpBuf3.Get<float>(); // v_new and denom path

        // t1 = grad * grad
        AscendC::Mul(t1, gLocal, gLocal, this->tileLength);
        // t2 = 1.0 - alpha
        AscendC::Duplicate(t2, 1.0f, this->tileLength);
        AscendC::Sub(t2, t2, alphaLocal, this->tileLength);
        // t1 = (1 - alpha) * grad^2
        AscendC::Mul(t1, t1, t2, this->tileLength);
        // t2 = alpha * v
        AscendC::Mul(t2, alphaLocal, vLocal, this->tileLength);
        // t3 = v_new = t1 + t2
        AscendC::Add(t3, t1, t2, this->tileLength);
        // t3 = sqrt(v_new)
        AscendC::Sqrt(t3, t3, this->tileLength);
        // t3 = sqrt(v_new) + eps
        AscendC::Add(t3, t3, epsLocal, this->tileLength);
        // t1 = grad / t3
        AscendC::Div(t1, gLocal, t3, this->tileLength);
        // t1 = lr * (grad / (sqrt(v_new) + eps))
        AscendC::Mul(t1, lrLocal, t1, this->tileLength);
        // out = param - t1
        AscendC::Sub(outLocal, pLocal, t1, this->tileLength);

        outQueueOut.EnQue<float>(outLocal);

        inQueueParam.FreeTensor(pLocal);
        inQueueGrad.FreeTensor(gLocal);
        inQueueV.FreeTensor(vLocal);
        inQueueLr.FreeTensor(lrLocal);
        inQueueAlpha.FreeTensor(alphaLocal);
        inQueueEps.FreeTensor(epsLocal);
    }
    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<float> outLocal = outQueueOut.DeQue<float>();
        AscendC::DataCopy(outGm[progress * this->tileLength], outLocal, this->tileLength);
        outQueueOut.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueParam, inQueueGrad, inQueueV, inQueueLr, inQueueAlpha, inQueueEps;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueOut;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpBuf1, tmpBuf2, tmpBuf3;

    AscendC::GlobalTensor<float> paramGm;
    AscendC::GlobalTensor<float> gradGm;
    AscendC::GlobalTensor<float> vGm;
    AscendC::GlobalTensor<float> lrGm;
    AscendC::GlobalTensor<float> alphaGm;
    AscendC::GlobalTensor<float> epsGm;
    AscendC::GlobalTensor<float> outGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void rmsprop_custom(GM_ADDR param, GM_ADDR grad, GM_ADDR v,
                                                     GM_ADDR lr, GM_ADDR alpha, GM_ADDR eps,
                                                     GM_ADDR out, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelRmsprop op;
    op.Init(param, grad, v, lr, alpha, eps, out, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, lr=1e-3, alpha=0.99, eps=1e-8):
        super().__init__()
        self.lr = lr
        self.alpha = alpha
        self.eps = eps

    def forward(self, param, grad, v):
        # Prepare per-element tensors for scalar hyperparameters
        lr_t = torch.full_like(param, self.lr)
        alpha_t = torch.full_like(param, self.alpha)
        eps_t = torch.full_like(param, self.eps)
        return custom_ops_lib.rmsprop_custom(param, grad, v, lr_t, alpha_t, eps_t)
