kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "nearest_neighbor_upsample_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // queue depth for double-buffering

class KernelNearestNeighborUpsample {
public:
    __aicore__ inline KernelNearestNeighborUpsample() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c,
                                uint32_t inH, uint32_t inW,
                                uint32_t outH, uint32_t outW,
                                uint32_t totalOutLength, uint32_t tileNum, uint32_t scale)
    {
        this->n = n;
        this->c = c;
        this->inH = inH;
        this->inW = inW;
        this->outH = outH;
        this->outW = outW;
        this->scale = scale;

        this->totalOutLength = totalOutLength;
        this->blockOutLength = totalOutLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileOutLength = this->blockOutLength / tileNum / BUFFER_NUM;

        uint32_t totalInLength = (inH * inW) * c * n;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, totalInLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->blockOutLength * AscendC::GetBlockIdx(), this->blockOutLength);

        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileOutLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void Compute(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
        // Fill yLocal by mapping each output index back to its nearest neighbor input index
        uint64_t blockBase = static_cast<uint64_t>(this->blockOutLength) * AscendC::GetBlockIdx();
        uint64_t tileBase = blockBase + static_cast<uint64_t>(progress) * this->tileOutLength;

        uint64_t outNCStride = static_cast<uint64_t>(outH) * outW;
        uint64_t outNStride = static_cast<uint64_t>(c) * outNCStride;
        uint64_t inNCStride = static_cast<uint64_t>(inH) * inW;
        uint64_t inNStride = static_cast<uint64_t>(c) * inNCStride;

        for (uint32_t j = 0; j < this->tileOutLength; ++j) {
            uint64_t gOut = tileBase + j;

            uint64_t nIdx = gOut / outNStride;
            uint64_t rem1 = gOut % outNStride;
            uint64_t cIdx = rem1 / outNCStride;
            uint64_t rem2 = rem1 % outNCStride;
            uint64_t oh = rem2 / outW;
            uint64_t ow = rem2 % outW;

            uint64_t ih = oh / scale;
            uint64_t iw = ow / scale;

            uint64_t gIn = nIdx * inNStride + cIdx * inNCStride + ih * inW + iw;

            // Pseudo element-wise read/write on GM/local tensor
            // These element-wise accesses represent scalar copy; on real hardware, vectorization is recommended.
            yLocal.SetValue(j, xGm.GetValue(gIn));
        }
        outQueueY.EnQue<DTYPE_Y>(yLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[progress * this->tileOutLength], yLocal, this->tileOutLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n;
    uint32_t c;
    uint32_t inH;
    uint32_t inW;
    uint32_t outH;
    uint32_t outW;
    uint32_t scale;

    uint32_t totalOutLength;
    uint32_t blockOutLength;
    uint32_t tileNum;
    uint32_t tileOutLength;
};

extern "C" __global__ __aicore__ void nearest_neighbor_upsample_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelNearestNeighborUpsample op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c,
            tiling_data.inH, tiling_data.inW,
            tiling_data.outH, tiling_data.outW,
            tiling_data.totalOutLength, tiling_data.tileNum, tiling_data.scale);
    op.Process();
}
