host_operator_src="""
#include "sum_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
inline int64_t NormalizeDim(int64_t dim, int64_t rank) {
    if (dim < 0) dim += rank;
    return dim;
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    // Get input shape
    const gert::Shape* in_shape = context->GetInputShape(0);
    int64_t rank = static_cast<int64_t>(in_shape->GetDimNum());
    // Get attr dim
    int64_t dim_attr = 0;
    (void)context->GetAttrs()->GetInt("dim", dim_attr);
    int64_t axis = NormalizeDim(dim_attr, rank);

    // Compute outer, reduce, inner
    uint64_t outer = 1, reduce = 1, inner = 1;
    for (int64_t i = 0; i < rank; ++i) {
        uint64_t d = static_cast<uint64_t>(in_shape->GetOriginShape().GetDim(i));
        if (i < axis) outer *= d;
        else if (i == axis) reduce = d;
        else inner *= d;
    }

    SumReductionOverADimensionCustomTilingData tiling;
    tiling.set_outerSize(static_cast<uint32_t>(outer));
    tiling.set_innerSize(static_cast<uint32_t>(inner));
    tiling.set_reduceSize(static_cast<uint32_t>(reduce));

    context->SetBlockDim(BLOCK_DIM);
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);

    int64_t rank = static_cast<int64_t>(x_shape->GetDimNum());
    int64_t dim_attr = 0;
    (void)context->GetAttrs()->GetInt("dim", dim_attr);
    int64_t axis = dim_attr < 0 ? dim_attr + rank : dim_attr;

    gert::Shape out = *x_shape;
    out.SetDim(axis, 1);
    *y_shape = out;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "sum_reduction_over_a_dimension_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 1; // number of local buffers per queue

class KernelReduceSum {
public:
    __aicore__ inline KernelReduceSum() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outerSize, uint32_t innerSize, uint32_t reduceSize)
    {
        this->outerSize = outerSize;
        this->innerSize = innerSize;
        this->reduceSize = reduceSize;

        // Work partition across blocks on 'outer' dimension
        this->blockOuter = this->outerSize / AscendC::GetBlockNum();
        this->outerStart = this->blockOuter * AscendC::GetBlockIdx();

        // Bind whole buffers; we address with absolute offsets
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, this->outerSize * this->reduceSize * this->innerSize);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, this->outerSize * this->innerSize);

        // Allocate queues sized to one 'inner' slice
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->innerSize * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->innerSize * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        // Each iteration handles one 'outer' slice of size 'innerSize'
        for (uint32_t o = 0; o < this->blockOuter; ++o) {
            uint32_t curOuter = this->outerStart + o;

            // Load first reduction slice into accumulator
            AscendC::LocalTensor<DTYPE_X> accLocal = outQueueY.AllocTensor<DTYPE_X>();
            uint64_t inOffset0 = (static_cast<uint64_t>(curOuter) * this->reduceSize + 0) * this->innerSize;
            AscendC::DataCopy(accLocal, xGm[inOffset0], this->innerSize);

            // Accumulate remaining slices
            for (uint32_t r = 1; r < this->reduceSize; ++r) {
                AscendC::LocalTensor<DTYPE_X> tmpLocal = inQueueX.AllocTensor<DTYPE_X>();
                uint64_t inOffset = (static_cast<uint64_t>(curOuter) * this->reduceSize + r) * this->innerSize;
                AscendC::DataCopy(tmpLocal, xGm[inOffset], this->innerSize);
                AscendC::Add(accLocal, accLocal, tmpLocal, this->innerSize);
                inQueueX.FreeTensor(tmpLocal);
            }

            // Store result (keepdim=True => one slice per 'outer')
            uint64_t outOffset = static_cast<uint64_t>(curOuter) * this->innerSize;
            AscendC::DataCopy(yGm[outOffset], accLocal, this->innerSize);
            outQueueY.FreeTensor(accLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t outerSize;
    uint32_t innerSize;
    uint32_t reduceSize;

    uint32_t blockOuter;
    uint32_t outerStart;
};

extern "C" __global__ __aicore__ void sum_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelReduceSum op;
    op.Init(x, y, tiling_data.outerSize, tiling_data.innerSize, tiling_data.reduceSize);
    op.Process();
}
