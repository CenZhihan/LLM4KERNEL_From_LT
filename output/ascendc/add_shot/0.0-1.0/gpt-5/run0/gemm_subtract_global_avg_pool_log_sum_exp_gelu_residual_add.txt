#include "kernel_operator.h"
#define __NPU_TILING__
#include "gemm_subtract_global_avg_pool_log_sum_exp_gelu_residual_add_custom_tiling_data.h"

// Define default DTYPE_* if not provided by build system
#ifndef DTYPE_X
#define DTYPE_X float
#endif
#ifndef DTYPE_W
#define DTYPE_W float
#endif
#ifndef DTYPE_B
#define DTYPE_B float
#endif
#ifndef DTYPE_S
#define DTYPE_S float
#endif
#ifndef DTYPE_Z
#define DTYPE_Z float
#endif

// Simple GELU approximation using tanh formulation
__aicore__ inline float gelu_approx(float x) {
    // 0.5 * x * (1 + tanh( sqrt(2/pi) * (x + 0.044715*x^3) ))
    const float k0 = 0.7978845608f; // sqrt(2/pi)
    const float k1 = 0.044715f;
    float x3 = x * x * x;
    float t = k0 * (x + k1 * x3);
    // use device-side tanh approximation through C math if available
    // fall back to linear if not supported by toolchain
#ifdef __CCE_KT_TEST__ 
    // placeholder if compile env defines special macro
    return 0.5f * x;
#else
    // AscendC provides vector math; scalar tanhf is acceptable in simple illustration
    float th = (float)tanh((double)t);
    return 0.5f * x * (1.0f + th);
#endif
}

extern "C" __global__ __aicore__ void gemm_subtract_global_avg_pool_log_sum_exp_gelu_residual_add_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR sub, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N = tiling_data.batch;
    const uint32_t K = tiling_data.inFeatures;
    const uint32_t M = tiling_data.outFeatures;

    const uint32_t bid = AscendC::GetBlockIdx();
    const uint32_t bnum = AscendC::GetBlockNum();

    const uint32_t rowStart = (uint32_t)((uint64_t)N * bid / bnum);
    const uint32_t rowEnd   = (uint32_t)((uint64_t)N * (bid + 1) / bnum);

    __gm__ DTYPE_X* xptr = (__gm__ DTYPE_X*)x;
    __gm__ DTYPE_W* wptr = (__gm__ DTYPE_W*)weight;   // [M, K]
    __gm__ DTYPE_B* bptr = (__gm__ DTYPE_B*)bias;     // [M]
    __gm__ DTYPE_S* sptr = (__gm__ DTYPE_S*)sub;      // [M]
    __gm__ DTYPE_Z* zptr = (__gm__ DTYPE_Z*)z;        // [N, K]

    // For each row assigned to this block
    for (uint32_t i = rowStart; i < rowEnd; ++i) {
        float sumOut = 0.0f;
        // Compute GEMM row (x_i @ W^T) + bias - sub, then accumulate for mean across M
        for (uint32_t j = 0; j < M; ++j) {
            float acc = 0.0f;
            // dot product over K
            __gm__ DTYPE_X* xrow = xptr + (uint64_t)i * K;
            __gm__ DTYPE_W* wrow = wptr + (uint64_t)j * K;
            for (uint32_t k = 0; k < K; ++k) {
                acc += (float)xrow[k] * (float)wrow[k];
            }
            if (bptr != nullptr) {
                acc += (float)bptr[j];
            }
            acc -= (float)sptr[j];
            sumOut += acc;
        }

        // GlobalAvgPool over features (M)
        float meanVal = (M > 0) ? sumOut / (float)M : 0.0f;

        // LogSumExp over single element keepdim=True => identity
        float lseVal = meanVal;

        // GELU
        float g = gelu_approx(lseVal);

        // Residual add with original x (broadcast g over feature dimension K)
        __gm__ DTYPE_X* xrow = xptr + (uint64_t)i * K;
        __gm__ DTYPE_Z* zrow = zptr + (uint64_t)i * K;
        for (uint32_t k = 0; k < K; ++k) {
            float outv = (float)xrow[k] + g;
            zrow[k] = (DTYPE_Z)outv;
        }
    }
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_features, out_features, bias=True):
        super(ModelNew, self).__init__()
        self.gemm = torch.nn.Linear(in_features, out_features, bias=bias)
        self.subtract = torch.nn.Parameter(torch.randn(out_features))

    def forward(self, x):
        # x: [N, in_features]
        weight = self.gemm.weight
        bias = self.gemm.bias
        sub = self.subtract
        # Ensure tensors are on same device/dtype
        weight = weight.to(device=x.device, dtype=x.dtype)
        sub = sub.to(device=x.device, dtype=x.dtype)
        if bias is not None:
            bias = bias.to(device=x.device, dtype=x.dtype)
        # Fused custom op
        return custom_ops_lib.gemm_subtract_global_avg_pool_log_sum_exp_gelu_residual_add_custom(x, weight, bias, sub)
