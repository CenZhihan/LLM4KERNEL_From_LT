kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_transpose2d_add_min_gelu_multiply_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelEpilogue {
public:
    __aicore__ inline KernelEpilogue() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR add, GM_ADDR mul, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        addGm.SetGlobalBuffer((__gm__ DTYPE_X *)add + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        mulGm.SetGlobalBuffer((__gm__ DTYPE_X *)mul + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_X *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueAdd, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueMul, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> addLocal = inQueueAdd.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> mulLocal = inQueueMul.AllocTensor<DTYPE_X>();

        AscendC::DataCopy(xLocal,  xGm[progress * this->tileLength],  this->tileLength);
        AscendC::DataCopy(addLocal, addGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(mulLocal, mulGm[progress * this->tileLength], this->tileLength);

        inQueueX.EnQue(xLocal);
        inQueueAdd.EnQue(addLocal);
        inQueueMul.EnQue(mulLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal  = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> addLocal = inQueueAdd.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> mulLocal = inQueueMul.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> yLocal  = outQueueY.AllocTensor<DTYPE_X>();

        // y = gelu(min(x + add, 0.0)) * mul
        AscendC::Add(yLocal, xLocal, addLocal, this->tileLength);
        AscendC::Minimum(yLocal, yLocal, static_cast<DTYPE_X>(0), this->tileLength);
        AscendC::Gelu(yLocal, yLocal, this->tileLength);
        AscendC::Mul(yLocal, yLocal, mulLocal, this->tileLength);

        outQueueY.EnQue<DTYPE_X>(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueAdd.FreeTensor(addLocal);
        inQueueMul.FreeTensor(mulLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> yLocal = outQueueY.DeQue<DTYPE_X>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueX, inQueueAdd, inQueueMul;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_X> addGm;
    AscendC::GlobalTensor<DTYPE_X> mulGm;
    AscendC::GlobalTensor<DTYPE_X> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transpose2d_add_min_gelu_multiply_custom(GM_ADDR x, GM_ADDR add, GM_ADDR mul, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelEpilogue op;
    op.Init(x, add, mul, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model with custom AscendC epilogue operator:
    y = gelu(min(conv_transpose2d(x) + add_value, 0.0)) * multiply_value
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, add_value, multiply_value):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose2d(in_channels, out_channels, kernel_size, stride=stride)
        self.add_value = float(add_value)
        self.multiply_value = float(multiply_value)

    def forward(self, x):
        x = self.conv_transpose(x)
        # Fused custom kernel for add + min(.,0) + gelu + multiply
        x = custom_ops_lib.conv_transpose2d_add_min_gelu_multiply_custom(x, self.add_value, self.multiply_value)
