kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "grid_sample_random_warp_custom_tiling_data.h"
#include <cmath>

class KernelGridSample {
public:
    __aicore__ inline KernelGridSample() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR grid, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t inH, uint32_t inW,
                                uint32_t outH, uint32_t outW, uint32_t totalLength, uint32_t tileNum)
    {
        xPtr = (__gm__ float*)x;
        gridPtr = (__gm__ float*)grid;
        yPtr = (__gm__ float*)y;

        this->N = n; this->C = c; this->inH = inH; this->inW = inW;
        this->outH = outH; this->outW = outW;
        this->totalLength = totalLength;
        this->tileNum = tileNum;

        this->blockNum = AscendC::GetBlockNum();
        this->blockIdx = AscendC::GetBlockIdx();

        // simple split, assume totalLength divisible by blockNum (common for typical sizes)
        this->blockLength = totalLength / blockNum;
        this->startIndex = this->blockLength * this->blockIdx;
    }

    __aicore__ inline float read_input(int n, int c, int h, int w) {
        if (h < 0 || h >= (int)inH || w < 0 || w >= (int)inW) {
            return 0.0f;
        }
        int idx = (((n * (int)C + c) * (int)inH + h) * (int)inW + w);
        return xPtr[idx];
    }

    __aicore__ inline void Process()
    {
        // For each output element in this block
        for (uint32_t i = 0; i < blockLength; ++i) {
            uint32_t outIndex = startIndex + i;

            // Decode n, c, oh, ow from linear index
            uint32_t perN = C * outH * outW;
            uint32_t n = outIndex / perN;
            uint32_t rem1 = outIndex % perN;
            uint32_t c = rem1 / (outH * outW);
            uint32_t rem2 = rem1 % (outH * outW);
            uint32_t oh = rem2 / outW;
            uint32_t ow = rem2 % outW;

            // grid index (N, outH, outW, 2)
            uint32_t gbase = ((n * outH + oh) * outW + ow) * 2;
            float gx = gridPtr[gbase + 0];
            float gy = gridPtr[gbase + 1];

            // align_corners = false mapping
            float ix = (gx + 1.0f) * 0.5f * (float)inW - 0.5f;
            float iy = (gy + 1.0f) * 0.5f * (float)inH - 0.5f;

            int x0 = (int)floorf(ix);
            int y0 = (int)floorf(iy);
            int x1 = x0 + 1;
            int y1 = y0 + 1;

            float wx1 = ix - (float)x0;
            float wy1 = iy - (float)y0;
            float wx0 = 1.0f - wx1;
            float wy0 = 1.0f - wy1;

            float v00 = read_input((int)n, (int)c, y0, x0);
            float v01 = read_input((int)n, (int)c, y0, x1);
            float v10 = read_input((int)n, (int)c, y1, x0);
            float v11 = read_input((int)n, (int)c, y1, x1);

            float val0 = v00 * wx0 + v01 * wx1;
            float val1 = v10 * wx0 + v11 * wx1;
            float val = val0 * wy0 + val1 * wy1;

            yPtr[outIndex] = val;
        }
    }

private:
    __gm__ float* xPtr;
    __gm__ float* gridPtr;
    __gm__ float* yPtr;

    uint32_t N, C, inH, inW, outH, outW;
    uint32_t totalLength;
    uint32_t tileNum;
    uint32_t blockNum;
    uint32_t blockIdx;
    uint32_t blockLength;
    uint32_t startIndex;
};

extern "C" __global__ __aicore__ void grid_sample_random_warp_custom(GM_ADDR x, GM_ADDR grid, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGridSample op;
    op.Init(x, grid, y,
            tiling_data.n, tiling_data.c, tiling_data.inH, tiling_data.inW,
            tiling_data.outH, tiling_data.outW, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}
