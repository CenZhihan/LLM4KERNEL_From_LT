namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)

    // Input 0: (K, M), Input 1: (K, N) => Output: (M, N)
    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);

    uint64_t M = a_shape->GetOriginShape().GetDim(1);

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "matmul_with_transposed_a_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulTransA {
public:
    __aicore__ inline KernelMatmulTransA() {}
    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c, uint32_t M, uint32_t N, uint32_t K, uint32_t tileN)
    {
        this->M = M;
        this->N = N;
        this->K = K;
        this->tileN = tileN;

        aGm.SetGlobalBuffer((__gm__ DTYPE_A*)a, (uint64_t)K * (uint64_t)M);
        bGm.SetGlobalBuffer((__gm__ DTYPE_B*)b, (uint64_t)K * (uint64_t)N);
        cGm.SetGlobalBuffer((__gm__ DTYPE_C*)c, (uint64_t)M * (uint64_t)N);

        pipe.InitBuffer(bQueue, BUFFER_NUM, this->tileN * sizeof(DTYPE_B));
        pipe.InitBuffer(cQueue, 1, this->tileN * sizeof(DTYPE_C));
        pipe.InitBuffer(aScalarQueue, 1, sizeof(DTYPE_A));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        uint32_t baseRows = M / blockNum;
        uint32_t remRows = M % blockNum;
        uint32_t startRow = blockIdx * baseRows + (blockIdx < remRows ? blockIdx : remRows);
        uint32_t rowCount = baseRows + (blockIdx < remRows ? 1 : 0);

        for (uint32_t m = 0; m < rowCount; ++m) {
            uint32_t rowIdx = startRow + m;
            for (uint32_t nStart = 0; nStart < N; nStart += tileN) {
                uint32_t curTileN = AscendC::Min(tileN, N - nStart);

                AscendC::LocalTensor<DTYPE_C> cLocal = cQueue.AllocTensor<DTYPE_C>();
                // First k iteration initializes cLocal
                {
                    AscendC::LocalTensor<DTYPE_B> bLocal = bQueue.AllocTensor<DTYPE_B>();
                    AscendC::DataCopy(bLocal, bGm[0 * N + nStart], curTileN);

                    AscendC::LocalTensor<DTYPE_A> aLocal = aScalarQueue.AllocTensor<DTYPE_A>();
                    AscendC::DataCopy(aLocal, aGm[0 * M + rowIdx], 1);
                    DTYPE_A alpha0 = aLocal.GetValue(0);

                    AscendC::Muls(cLocal, bLocal, alpha0, curTileN);

                    bQueue.FreeTensor(bLocal);
                    aScalarQueue.FreeTensor(aLocal);
                }

                for (uint32_t k = 1; k < K; ++k) {
                    AscendC::LocalTensor<DTYPE_B> bLocal = bQueue.AllocTensor<DTYPE_B>();
                    AscendC::DataCopy(bLocal, bGm[k * N + nStart], curTileN);

                    AscendC::LocalTensor<DTYPE_A> aLocal = aScalarQueue.AllocTensor<DTYPE_A>();
                    AscendC::DataCopy(aLocal, aGm[k * M + rowIdx], 1);
                    DTYPE_A alpha = aLocal.GetValue(0);

                    AscendC::Muls(bLocal, bLocal, alpha, curTileN);
                    AscendC::Add(cLocal, cLocal, bLocal, curTileN);

                    bQueue.FreeTensor(bLocal);
                    aScalarQueue.FreeTensor(aLocal);
                }

                AscendC::DataCopy(cGm[rowIdx * N + nStart], cLocal, curTileN);
                cQueue.FreeTensor(cLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> bQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> cQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> aScalarQueue;

    AscendC::GlobalTensor<DTYPE_A> aGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_C> cGm;

    uint32_t M;
    uint32_t N;
    uint32_t K;
    uint32_t tileN;
};

extern "C" __global__ __aicore__ void matmul_with_transposed_a_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulTransA op;
    op.Init(a, b, c, tiling_data.M, tiling_data.N, tiling_data.K, tiling_data.tileN);
    op.Process();
}
