{
    const gert::Shape* in_shape = context->GetInputShape(0);
    gert::Shape* out_shape = context->GetOutputShape(0);

    std::vector<int64_t> out_dims;
    if (in_shape->GetDimNum() >= 1) {
        out_dims.push_back(in_shape->GetDim(0));

        out_dims.push_back(1);

    out_shape->SetDimNum(out_dims.size());
    for (size_t i = 0; i < out_dims.size(); ++i) {
        out_shape->SetDim(i, out_dims[i]);
    }
    return GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "matmul_sigmoid_sum_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // double buffer for input tiles

class KernelSigmoidSum {
public:
    __aicore__ inline KernelSigmoidSum() {}
    __aicore__ inline void Init(GM_ADDR h, GM_ADDR y, uint32_t batch, uint32_t hidden, uint32_t tileNum)
    {
        this->batch = batch;
        this->hidden = hidden;
        this->tileNum = tileNum;

        hGm.SetGlobalBuffer((__gm__ float *)h, static_cast<uint32_t>(batch * hidden));
        yGm.SetGlobalBuffer((__gm__ float *)y, static_cast<uint32_t>(batch));

        // decide tile length along hidden dimension
        this->tileLen = (hidden / tileNum / BUFFER_NUM);
        if (this->tileLen == 0) {
            this->tileLen = hidden; // fallback to full row
            this->tileNum = 1;
        }

        pipe.InitBuffer(inQueueH, BUFFER_NUM, this->tileLen * sizeof(float));
        pipe.InitBuffer(tmpQueueVec, BUFFER_NUM, this->tileLen * sizeof(float));
        pipe.InitBuffer(tmpQueueSum, BUFFER_NUM, sizeof(float)); // for partial sums
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t rowsPerBlock = (batch + blockNum - 1) / blockNum;
        uint32_t rowStart = blockIdx * rowsPerBlock;
        uint32_t rowEnd = AscendC::min(rowStart + rowsPerBlock, batch);

        for (uint32_t row = rowStart; row < rowEnd; ++row) {
            float acc = 0.0f;
            uint32_t tilesPerRow = (hidden + (this->tileLen - 1)) / this->tileLen;
            // we implement a double-buffered pipeline over tiles
            for (uint32_t t = 0; t < tilesPerRow; ++t) {
                uint32_t offset = t * this->tileLen;
                uint32_t effLen = AscendC::min(this->tileLen, hidden - offset);

                // CopyIn
                AscendC::LocalTensor<float> hLocal = inQueueH.AllocTensor<float>();
                AscendC::DataCopy(hLocal, hGm[row * hidden + offset], effLen);
                inQueueH.EnQue(hLocal);

                // Compute
                AscendC::LocalTensor<float> hTile = inQueueH.DeQue<float>();
                AscendC::LocalTensor<float> sigTile = tmpQueueVec.AllocTensor<float>();
                AscendC::Sigmoid(sigTile, hTile, effLen);

                // Reduce partial sum
                AscendC::LocalTensor<float> sumBuf = tmpQueueSum.AllocTensor<float>();
                AscendC::ReduceSum(sumBuf, sigTile, effLen);
                tmpQueueVec.EnQue(sigTile);
                inQueueH.FreeTensor(hTile);

                AscendC::LocalTensor<float> sigTileOut = tmpQueueVec.DeQue<float>();
                tmpQueueVec.FreeTensor(sigTileOut);

                float part = 0.0f;
                // read back the reduced sum
                // assuming the first element stores the sum
                // (typical ReduceSum behavior in AscendC samples)
                part = sumBuf.GetValue(0);
                tmpQueueSum.FreeTensor(sumBuf);

                acc += part;
            }
            // write result
            yGm[row] = acc;
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueH;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> tmpQueueVec;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> tmpQueueSum;

    AscendC::GlobalTensor<float> hGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t batch;
    uint32_t hidden;
    uint32_t tileNum;
    uint32_t tileLen;
};

extern "C" __global__ __aicore__ void matmul_sigmoid_sum_custom(GM_ADDR h, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSigmoidSum op;
    op.Init(h, y, tiling_data.batch, tiling_data.hidden, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: keep Linear on framework side, fuse sigmoid + sum into a custom AscendC op.
    """
    def __init__(self, input_size, hidden_size):
        super(ModelNew, self).__init__()
        self.linear = torch.nn.Linear(input_size, hidden_size)

    def forward(self, x):
        h = self.linear(x)  # [B, H]
        # Fused sigmoid + sum over hidden dimension into a single custom kernel
        y = custom_ops_lib.matmul_sigmoid_sum_custom(h)
        return y
