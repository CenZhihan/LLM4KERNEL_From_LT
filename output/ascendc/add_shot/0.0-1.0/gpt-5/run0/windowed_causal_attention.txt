kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "windowed_causal_attention_custom_tiling_data.h"

typedef float DTYPE_Q;
typedef float DTYPE_K;
typedef float DTYPE_V;
typedef float DTYPE_O;

class KernelWindowedCausalAttention {
public:
    __aicore__ inline KernelWindowedCausalAttention() {}
    __aicore__ inline void Init(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR o,
                                uint32_t B, uint32_t N, uint32_t H, uint32_t Dh, uint32_t window)
    {
        this->B = B;
        this->N = N;
        this->H = H;
        this->Dh = Dh;
        this->window = window;

        qGm.SetGlobalBuffer((__gm__ DTYPE_Q*)q, (uint32_t)(B * N * H * Dh));
        kGm.SetGlobalBuffer((__gm__ DTYPE_K*)k, (uint32_t)(B * N * H * Dh));
        vGm.SetGlobalBuffer((__gm__ DTYPE_V*)v, (uint32_t)(B * N * H * Dh));
        oGm.SetGlobalBuffer((__gm__ DTYPE_O*)o, (uint32_t)(B * N * H * Dh));

        // local buffers
        uint32_t qSize = Dh;
        uint32_t kvSize = window * Dh;
        uint32_t sSize = window;

        pipe.InitBuffer(qQueue, 1, qSize * sizeof(DTYPE_Q));
        pipe.InitBuffer(kQueue, 1, kvSize * sizeof(DTYPE_K));
        pipe.InitBuffer(vQueue, 1, kvSize * sizeof(DTYPE_V));
        pipe.InitBuffer(oQueue, 1, qSize * sizeof(DTYPE_O));
        pipe.InitBuffer(sQueue, 1, sSize * sizeof(float));
    }

    __aicore__ inline void Process(uint32_t totalTokens)
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t tokensPerBlock = totalTokens / blockNum + (blockIdx < (totalTokens % blockNum) ? 1 : 0);
        uint32_t startToken = (totalTokens / blockNum) * blockIdx + AscendC::Min(blockIdx, totalTokens % blockNum);

        uint32_t pitchDh = Dh;
        uint32_t pitchH = Dh;
        uint32_t pitchN = H * Dh;
        uint32_t pitchB = N * H * Dh;

        float scale = 1.0f / AscendC::Sqrt((float)Dh);

        for (uint32_t t = 0; t < tokensPerBlock; ++t) {
            uint32_t g = startToken + t;
            if (g >= B * N * H) break;
            // decode (b, n, h)
            uint32_t tmp = g;
            uint32_t b = tmp / (N * H);
            tmp = tmp % (N * H);
            uint32_t n = tmp / H;
            uint32_t h = tmp % H;

            uint32_t start = (n >= window) ? (n - window + 1) : 0; // inclusive start
            uint32_t len = n - start + 1;

            // Allocate locals
            AscendC::LocalTensor<DTYPE_Q> qLocal = qQueue.AllocTensor<DTYPE_Q>();
            AscendC::LocalTensor<DTYPE_K> kLocal = kQueue.AllocTensor<DTYPE_K>();
            AscendC::LocalTensor<DTYPE_V> vLocal = vQueue.AllocTensor<DTYPE_V>();
            AscendC::LocalTensor<DTYPE_O> oLocal = oQueue.AllocTensor<DTYPE_O>();
            AscendC::LocalTensor<float> sLocal = sQueue.AllocTensor<float>();

            // copy q
            uint32_t qOff = b * pitchB + n * pitchN + h * pitchH;
            AscendC::DataCopy(qLocal, qGm[qOff], Dh);

            // copy K window
            uint32_t kOffBase = b * pitchB + start * pitchN + h * pitchH;
            AscendC::DataCopy(kLocal, kGm[kOffBase], len * Dh);

            // copy V window
            uint32_t vOffBase = b * pitchB + start * pitchN + h * pitchH;
            AscendC::DataCopy(vLocal, vGm[vOffBase], len * Dh);

            // scores = (q · k_j) * scale
            float sMax = -3.402823e38f;
            for (uint32_t j = 0; j < len; ++j) {
                float acc = 0.f;
                uint32_t base = j * Dh;
                for (uint32_t d = 0; d < Dh; ++d) {
                    acc += (float)qLocal.GetValue(d) * (float)kLocal.GetValue(base + d);
                }
                acc *= scale;
                sLocal.SetValue(j, acc);
                sMax = acc > sMax ? acc : sMax;
            }

            // softmax
            float sumExp = 0.f;
            for (uint32_t j = 0; j < len; ++j) {
                float e = AscendC::Exp(sLocal.GetValue(j) - sMax);
                sLocal.SetValue(j, e);
                sumExp += e;
            }
            float invSum = 1.0f / (sumExp + 1e-6f);
            for (uint32_t j = 0; j < len; ++j) {
                float w = sLocal.GetValue(j) * invSum;
                sLocal.SetValue(j, w);
            }

            // out = sum_j w_j * v_j
            for (uint32_t d = 0; d < Dh; ++d) {
                oLocal.SetValue(d, (DTYPE_O)0);
            }
            for (uint32_t j = 0; j < len; ++j) {
                float w = sLocal.GetValue(j);
                uint32_t base = j * Dh;
                for (uint32_t d = 0; d < Dh; ++d) {
                    float cur = (float)oLocal.GetValue(d) + w * (float)vLocal.GetValue(base + d);
                    oLocal.SetValue(d, (DTYPE_O)cur);
                }
            }

            // write out
            uint32_t oOff = qOff;
            AscendC::DataCopy(oGm[oOff], oLocal, Dh);

            // free
            qQueue.FreeTensor(qLocal);
            kQueue.FreeTensor(kLocal);
            vQueue.FreeTensor(vLocal);
            oQueue.FreeTensor(oLocal);
            sQueue.FreeTensor(sLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> qQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> kQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> vQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> oQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> sQueue;

    AscendC::GlobalTensor<DTYPE_Q> qGm;
    AscendC::GlobalTensor<DTYPE_K> kGm;
    AscendC::GlobalTensor<DTYPE_V> vGm;
    AscendC::GlobalTensor<DTYPE_O> oGm;

    uint32_t B, N, H, Dh, window;
};

extern "C" __global__ __aicore__ void windowed_causal_attention_custom(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR windowTensor, GM_ADDR o, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelWindowedCausalAttention op;
    uint32_t B = tiling_data.B;
    uint32_t N = tiling_data.N;
    uint32_t H = tiling_data.H;
    uint32_t Dh = tiling_data.Dh;
    uint32_t win = tiling_data.window;

    // In case the host didn't fill tiling window, read from tensor
    if (win == 0 && windowTensor != nullptr) {
        __gm__ int32_t* wptr = (__gm__ int32_t*)windowTensor;
        win = (uint32_t)(*wptr);
    }
    if (win == 0) win = 1;

    op.Init(q, k, v, o, B, N, H, Dh, win);
    op.Process(tiling_data.totalTokens);
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, d_model=1024, num_heads=16, window_size=256):
        super().__init__()
        self.q_proj = torch.nn.Linear(d_model, d_model)
        self.k_proj = torch.nn.Linear(d_model, d_model)
        self.v_proj = torch.nn.Linear(d_model, d_model)
        self.out_proj = torch.nn.Linear(d_model, d_model)
        self.num_heads = num_heads
        self.window_size = window_size

    def forward(self, x):
        B, N, D = x.shape
        H = self.num_heads
        d_h = D // H

        Q = self.q_proj(x).view(B, N, H, d_h)
        K = self.k_proj(x).view(B, N, H, d_h)
        V = self.v_proj(x).view(B, N, H, d_h)

        attn_out = custom_ops_lib.windowed_causal_attention_custom(Q, K, V, self.window_size)
        return self.out_proj(attn_out.reshape(B, N, D))
