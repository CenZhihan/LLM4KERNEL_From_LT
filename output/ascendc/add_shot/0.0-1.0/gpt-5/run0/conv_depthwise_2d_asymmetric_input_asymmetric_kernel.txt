kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_depthwise_2d_asymmetric_input_asymmetric_kernel_custom_tiling_data.h"
#include <cstdint>

typedef float DTYPE_X;
typedef float DTYPE_W;
typedef float DTYPE_Y;

__aicore__ inline uint32_t umin32(uint32_t a, uint32_t b) { return a < b ? a : b; }

extern "C" __global__ __aicore__ void conv_depthwise_2d_asymmetric_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N  = tiling_data.n;
    const uint32_t C  = tiling_data.c;
    const uint32_t H  = tiling_data.h;
    const uint32_t W  = tiling_data.w;
    const uint32_t KH = tiling_data.kh;
    const uint32_t KW = tiling_data.kw;
    const uint32_t OH = tiling_data.oh;
    const uint32_t OW = tiling_data.ow;

    // Assume stride=1, pad=0, dilation=1

    __gm__ DTYPE_X* x_gm = reinterpret_cast<__gm__ DTYPE_X*>(x);
    __gm__ DTYPE_W* w_gm = reinterpret_cast<__gm__ DTYPE_W*>(w);
    __gm__ DTYPE_Y* y_gm = reinterpret_cast<__gm__ DTYPE_Y*>(y);

    const uint32_t blockNum = AscendC::GetBlockNum();
    const uint32_t blockIdx = AscendC::GetBlockIdx();

    const uint32_t NC = N * C;
    const uint32_t perBlock = (NC + blockNum - 1) / blockNum;
    const uint32_t startIdx = perBlock * blockIdx;
    const uint32_t endIdx = umin32(NC, startIdx + perBlock);

    for (uint32_t idx = startIdx; idx < endIdx; ++idx) {
        uint32_t n = idx / C;
        uint32_t c = idx % C;

        // Base offsets
        uint64_t x_base = ((uint64_t)n * C + c) * H * W;
        uint64_t y_base = ((uint64_t)n * C + c) * OH * OW;
        uint64_t w_base = ((uint64_t)c) * 1 * KH * KW; // depthwise: weight shape [C,1,KH,KW]

        for (uint32_t oh = 0; oh < OH; ++oh) {
            for (uint32_t ow_ = 0; ow_ < OW; ++ow_) {
                float acc = 0.f;
                for (uint32_t kh = 0; kh < KH; ++kh) {
                    for (uint32_t kw = 0; kw < KW; ++kw) {
                        uint64_t in_h = oh + kh; // stride=1, pad=0
                        uint64_t in_w = ow_ + kw;
                        uint64_t x_idx = x_base + in_h * W + in_w;
                        uint64_t w_idx = w_base + kh * KW + kw;
                        acc += static_cast<float>(x_gm[x_idx]) * static_cast<float>(w_gm[w_idx]);
                    }
                }
                uint64_t y_idx = y_base + oh * OW + ow_;
                y_gm[y_idx] = static_cast<DTYPE_Y>(acc);
            }
        }
    }
}

    TORCH_CHECK(input.dim() == 4, \"input must be NCHW\");
    TORCH_CHECK(weight.dim() == 4, \"weight must be [C,1,KH,KW] for depthwise\");

    auto N = input.size(0);
    auto C = input.size(1);
    auto H = input.size(2);
    auto W = input.size(3);

    auto KH = weight.size(2);
    auto KW = weight.size(3);

    TORCH_CHECK(weight.size(0) == C && weight.size(1) == 1, \"weight shape mismatch for depthwise conv\");

    // Assume stride=1, pad=0, dilation=1
    auto OH = H - KH + 1;
    auto OW = W - KW + 1;
    TORCH_CHECK(OH > 0 && OW > 0, \"Invalid output size. Ensure H>=KH and W>=KW for stride=1, pad=0\");

    at::Tensor result = at::empty({N, C, OH, OW}, input.options());
    EXEC_NPU_CMD(aclnnConvDepthwise2dAsymmetricInputAsymmetricKernelCustom, input, weight, result);
    return result;
