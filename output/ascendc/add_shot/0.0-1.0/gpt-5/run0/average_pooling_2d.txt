inline uint32_t CalcOutDim(uint32_t in, uint32_t k, uint32_t s, uint32_t p) {
    return (in + 2 * p - k) / s + 1;
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    AveragePooling2dCustomTilingData tiling;

    // Input shape NCHW
    const gert::Shape* x_shape = context->GetInputShape(0);
    uint32_t n = static_cast<uint32_t>(x_shape->GetStorageShape().GetDim(0));
    uint32_t c = static_cast<uint32_t>(x_shape->GetStorageShape().GetDim(1));
    uint32_t h = static_cast<uint32_t>(x_shape->GetStorageShape().GetDim(2));
    uint32_t w = static_cast<uint32_t>(x_shape->GetStorageShape().GetDim(3));

    // Get attributes
    int64_t kH_i = 1, kW_i = 1, sH_i = 1, sW_i = 1, pH_i = 0, pW_i = 0;
    (void)context->GetAttrs()->GetInt("kernel_h", kH_i);
    (void)context->GetAttrs()->GetInt("kernel_w", kW_i);
    (void)context->GetAttrs()->GetInt("stride_h", sH_i);
    (void)context->GetAttrs()->GetInt("stride_w", sW_i);
    (void)context->GetAttrs()->GetInt("pad_h", pH_i);
    (void)context->GetAttrs()->GetInt("pad_w", pW_i);

    uint32_t kH = static_cast<uint32_t>(kH_i);
    uint32_t kW = static_cast<uint32_t>(kW_i);
    uint32_t sH = static_cast<uint32_t>(sH_i);
    uint32_t sW = static_cast<uint32_t>(sW_i);
    uint32_t pH = static_cast<uint32_t>(pH_i);
    uint32_t pW = static_cast<uint32_t>(pW_i);

    uint32_t outH = CalcOutDim(h, kH, sH, pH);
    uint32_t outW = CalcOutDim(w, kW, sW, pW);

    uint64_t totalOut = static_cast<uint64_t>(n) * c * outH * outW;

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_totalLengthOut(static_cast<uint32_t>(totalOut));
    tiling.set_tileNum(TILE_NUM);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_h(h);
    tiling.set_w(w);

    tiling.set_outH(outH);
    tiling.set_outW(outW);

    tiling.set_kH(kH);
    tiling.set_kW(kW);
    tiling.set_sH(sH);
    tiling.set_sW(sW);
    tiling.set_pH(pH);
    tiling.set_pW(pW);

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "average_pooling_2d_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2;

__aicore__ inline uint32_t ceil_div_u32(uint32_t a, uint32_t b) {
    return (a + b - 1) / b;
}

class KernelAvgPool2D {
public:
    __aicore__ inline KernelAvgPool2D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                const optiling::AveragePooling2dCustomTilingData &t)
    {
        n = t.n;
        c = t.c;
        h = t.h;
        w = t.w;
        outH = t.outH;
        outW = t.outW;
        kH = t.kH;
        kW = t.kW;
        sH = t.sH;
        sW = t.sW;
        pH = t.pH;
        pW = t.pW;
        totalOut = t.totalLengthOut;
        tileNum = t.tileNum;

        // Work partition among blocks
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t blk = AscendC::GetBlockIdx();
        uint32_t base = (totalOut / blocks) * blk + AscendC::Min(blk, totalOut % blocks);
        blockLen = (totalOut / blocks) + (blk < (totalOut % blocks) ? 1 : 0);
        startIdx = base;

        // Setup GM buffers
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint64_t)n * c * h * w);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + startIdx, blockLen);

        // Tile config
        tileLen = blockLen / (tileNum * BUFFER_NUM);
        if (tileLen == 0) { tileLen = 1; tileNum = 1; }

        pipe.InitBuffer(outQueueY, BUFFER_NUM, tileLen * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = ceil_div_u32(blockLen, tileLen);
        for (int32_t i = 0; i < loopCount; ++i) {
            ComputeAndStore(i);
        }
    }

private:
    __aicore__ inline void ComputeAndStore(int32_t progress)
    {
        uint32_t globalOffset = startIdx + progress * tileLen;
        if (globalOffset >= startIdx + blockLen) {
            return;
        }
        uint32_t currLen = AscendC::Min(tileLen, startIdx + blockLen - globalOffset);

        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
        // For safety, set to zero
        AscendC::Duplicate(yLocal, static_cast<DTYPE_Y>(0), currLen);

        for (uint32_t tIdx = 0; tIdx < currLen; ++tIdx) {
            uint32_t lin = globalOffset + tIdx;

            // Map linear index back to (n, c, oh, ow)
            uint32_t ow = lin % outW;
            lin /= outW;
            uint32_t oh = lin % outH;
            lin /= outH;
            uint32_t cc = lin % c;
            uint32_t nn = lin / c;

            int32_t ihBase = static_cast<int32_t>(oh) * static_cast<int32_t>(sH) - static_cast<int32_t>(pH);
            int32_t iwBase = static_cast<int32_t>(ow) * static_cast<int32_t>(sW) - static_cast<int32_t>(pW);

            // Compute base offsets
            uint64_t inBase = ((uint64_t)nn * c + cc) * h * w;

            // Accumulate sum over kernel window with padding
            float sum = 0.0f;
            uint32_t cnt = 0;
            for (uint32_t kh = 0; kh < kH; ++kh) {
                int32_t ih = ihBase + static_cast<int32_t>(kh);
                if (ih < 0 || ih >= static_cast<int32_t>(h)) continue;

                for (uint32_t kw = 0; kw < kW; ++kw) {
                    int32_t iw = iwBase + static_cast<int32_t>(kw);
                    if (iw < 0 || iw >= static_cast<int32_t>(w)) continue;

                    uint64_t off = inBase + (uint64_t)ih * w + (uint64_t)iw;

                    // load one element from GM to local and accumulate
                    AscendC::LocalTensor<DTYPE_X> tmpLocal = pipe.AllocLocalTensor<DTYPE_X>(1);
                    AscendC::DataCopy(tmpLocal, xGm[off], 1);
                    pipe.FreeLocalTensor(tmpLocal); // free immediately after copy

                    // Read value from local buffer index 0
                    // Note: LocalTensor supports operator[] in AscendC
                    sum += static_cast<float>(tmpLocal.GetValue(0));
                    cnt += 1;
                }
            }
            float avg = cnt > 0 ? (sum / static_cast<float>(cnt)) : 0.0f;
            yLocal.SetValue(tIdx, static_cast<DTYPE_Y>(avg));
        }

        outQueueY.EnQue<DTYPE_Y>(yLocal);
        AscendC::LocalTensor<DTYPE_Y> yOut = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[progress * tileLen], yOut, currLen);
        outQueueY.FreeTensor(yOut);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n, c, h, w;
    uint32_t outH, outW;
    uint32_t kH, kW, sH, sW, pH, pW;

    uint32_t totalOut;
    uint32_t tileNum;
    uint32_t tileLen;
    uint32_t startIdx;
    uint32_t blockLen;
};

extern "C" __global__ __aicore__ void average_pooling_2d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelAvgPool2D op;
    op.Init(x, y, tiling_data);
    op.Process();
}
