kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv2d_group_norm_tanh_hard_swish_residual_add_log_sum_exp_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelResidualAdd {
public:
    __aicore__ inline KernelResidualAdd() {}
    __aicore__ inline void Init(GM_ADDR x_conv, GM_ADDR x_hs, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xConvGm.SetGlobalBuffer((__gm__ DTYPE_X *)x_conv + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        xHsGm.SetGlobalBuffer((__gm__ DTYPE_Y *)x_hs + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        pipe.InitBuffer(inQueueXConv, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueXHs, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xConvLocal = inQueueXConv.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> xHsLocal = inQueueXHs.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(xConvLocal, xConvGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(xHsLocal, xHsGm[progress * this->tileLength], this->tileLength);
        inQueueXConv.EnQue(xConvLocal);
        inQueueXHs.EnQue(xHsLocal);
    }

    __aicore__ inline void Compute(int32_t)
    {
        AscendC::LocalTensor<DTYPE_X> xConvLocal = inQueueXConv.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> xHsLocal = inQueueXHs.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        // Fused residual add: y = x_conv + x_hs
        AscendC::Add(yLocal, xConvLocal, xHsLocal, this->tileLength);

        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueXConv.FreeTensor(xConvLocal);
        inQueueXHs.FreeTensor(xHsLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueXConv, inQueueXHs;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xConvGm;
    AscendC::GlobalTensor<DTYPE_Y> xHsGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv2d_group_norm_tanh_hard_swish_residual_add_log_sum_exp_custom(GM_ADDR x_conv, GM_ADDR x_hs, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelResidualAdd op;
    op.Init(x_conv, x_hs, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: keeps Conv2d, GroupNorm, Tanh, Hardswish on framework,
    replaces the residual addition with a custom AscendC kernel and then applies LogSumExp.
    """
    def __init__(self, in_channels, out_channels, kernel_size, groups, eps=1e-5):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)
        self.group_norm = torch.nn.GroupNorm(groups, out_channels, eps=eps)
        self.tanh = torch.nn.Tanh()
        self.hard_swish = torch.nn.Hardswish()

    def forward(self, x):
        # Convolution
        x_conv = self.conv(x)
        # Group Normalization
        x_norm = self.group_norm(x_conv)
        # Tanh
        x_tanh = self.tanh(x_norm)
        # HardSwish
        x_hard_swish = self.hard_swish(x_tanh)
        # Residual Addition via custom AscendC op
        x_res = custom_ops_lib.conv2d_group_norm_tanh_hard_swish_residual_add_log_sum_exp_custom(x_conv, x_hard_swish)
        # LogSumExp
        x_logsumexp = torch.logsumexp(x_res, dim=1, keepdim=True)
        return x_logsumexp
'''
