kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "triplet_margin_loss_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // buffers per queue

class KernelTripletMarginLoss {
public:
    __aicore__ inline KernelTripletMarginLoss() {}
    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR margin, GM_ADDR loss,
                                uint32_t batchSize, uint32_t featureSize, uint32_t tileNum)
    {
        this->batchSize = batchSize;
        this->featureSize = featureSize;
        this->tileNum = tileNum;

        // Split samples across blocks
        this->samplesPerBlock = (this->batchSize + AscendC::GetBlockNum() - 1) / AscendC::GetBlockNum();
        uint32_t startSample = this->samplesPerBlock * AscendC::GetBlockIdx();
        // Clamp for last block
        if (startSample + this->samplesPerBlock > this->batchSize) {
            this->samplesPerBlock = this->batchSize - startSample;
        }
        this->anchorGm.SetGlobalBuffer((__gm__ float *)anchor + startSample * this->featureSize,
                                       this->samplesPerBlock * this->featureSize);
        this->positiveGm.SetGlobalBuffer((__gm__ float *)positive + startSample * this->featureSize,
                                         this->samplesPerBlock * this->featureSize);
        this->negativeGm.SetGlobalBuffer((__gm__ float *)negative + startSample * this->featureSize,
                                         this->samplesPerBlock * this->featureSize);
        this->marginGm.SetGlobalBuffer((__gm__ float *)margin, 1);
        this->lossGm.SetGlobalBuffer((__gm__ float *)loss + startSample, this->samplesPerBlock);

        pipe.InitBuffer(inQueueAnchor, BUFFER_NUM, this->featureSize * sizeof(float));
        pipe.InitBuffer(inQueuePositive, BUFFER_NUM, this->featureSize * sizeof(float));
        pipe.InitBuffer(inQueueNegative, BUFFER_NUM, this->featureSize * sizeof(float));
        pipe.InitBuffer(tmpVecQueue, BUFFER_NUM, this->featureSize * sizeof(float));
        pipe.InitBuffer(outQueueLoss, BUFFER_NUM, sizeof(float));
        pipe.InitBuffer(tmpScalarQueue, BUFFER_NUM, sizeof(float));
        pipe.InitBuffer(marginQueue, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // Load margin scalar into a 1-element local once
        AscendC::LocalTensor<float> mLocal = marginQueue.AllocTensor<float>();
        AscendC::DataCopy(mLocal, marginGm, 1);
        marginQueue.EnQue(mLocal);

        for (uint32_t i = 0; i < this->samplesPerBlock; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
        // Free margin local
        AscendC::LocalTensor<float> mLocalFree = marginQueue.DeQue<float>();
        marginQueue.FreeTensor(mLocalFree);
    }

private:
    __aicore__ inline void CopyIn(uint32_t sampleIdx)
    {
        AscendC::LocalTensor<float> aLocal = inQueueAnchor.AllocTensor<float>();
        AscendC::LocalTensor<float> pLocal = inQueuePositive.AllocTensor<float>();
        AscendC::LocalTensor<float> nLocal = inQueueNegative.AllocTensor<float>();

        AscendC::DataCopy(aLocal, anchorGm[sampleIdx * this->featureSize], this->featureSize);
        AscendC::DataCopy(pLocal, positiveGm[sampleIdx * this->featureSize], this->featureSize);
        AscendC::DataCopy(nLocal, negativeGm[sampleIdx * this->featureSize], this->featureSize);

        inQueueAnchor.EnQue(aLocal);
        inQueuePositive.EnQue(pLocal);
        inQueueNegative.EnQue(nLocal);
    }

    __aicore__ inline void Compute(uint32_t /*sampleIdx*/)
    {
        AscendC::LocalTensor<float> aLocal = inQueueAnchor.DeQue<float>();
        AscendC::LocalTensor<float> pLocal = inQueuePositive.DeQue<float>();
        AscendC::LocalTensor<float> nLocal = inQueueNegative.DeQue<float>();

        AscendC::LocalTensor<float> workVec = tmpVecQueue.AllocTensor<float>();   // length = featureSize

        // d_ap = sqrt(sum((a - p)^2))
        AscendC::Sub(workVec, aLocal, pLocal, this->featureSize);
        AscendC::Mul(workVec, workVec, workVec, this->featureSize);
        AscendC::LocalTensor<float> sumApLocal = tmpScalarQueue.AllocTensor<float>(); // length = 1
        AscendC::ReduceSum(sumApLocal, workVec, this->featureSize);
        AscendC::LocalTensor<float> dapLocal = outQueueLoss.AllocTensor<float>(); // reuse final output buffer temporarily
        AscendC::Sqrt(dapLocal, sumApLocal, 1);

        // d_an = sqrt(sum((a - n)^2))
        AscendC::Sub(workVec, aLocal, nLocal, this->featureSize);
        AscendC::Mul(workVec, workVec, workVec, this->featureSize);
        AscendC::LocalTensor<float> danLocal = tmpScalarQueue.AllocTensor<float>(); // length = 1
        AscendC::ReduceSum(danLocal, workVec, this->featureSize);
        AscendC::Sqrt(danLocal, danLocal, 1);

        // loss = max(d_ap - d_an + margin, 0)
        AscendC::Sub(dapLocal, dapLocal, danLocal, 1);
        AscendC::LocalTensor<float> mLocal = marginQueue.DeQue<float>(); // 1 element margin
        AscendC::Add(dapLocal, dapLocal, mLocal, 1); // add margin
        marginQueue.EnQue(mLocal);

        // ReLU as max(x, 0)
        AscendC::Relu(dapLocal, dapLocal, 1);

        // enqueue output and free temps
        outQueueLoss.EnQue(dapLocal);
        tmpScalarQueue.FreeTensor(sumApLocal);
        tmpScalarQueue.FreeTensor(danLocal);
        tmpVecQueue.FreeTensor(workVec);
        inQueueAnchor.FreeTensor(aLocal);
        inQueuePositive.FreeTensor(pLocal);
        inQueueNegative.FreeTensor(nLocal);
    }

    __aicore__ inline void CopyOut(uint32_t sampleIdx)
    {
        AscendC::LocalTensor<float> lossLocal = outQueueLoss.DeQue<float>();
        AscendC::DataCopy(lossGm[sampleIdx], lossLocal, 1);
        outQueueLoss.FreeTensor(lossLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueAnchor, inQueuePositive, inQueueNegative;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpVecQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueLoss;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> tmpScalarQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> marginQueue;

    AscendC::GlobalTensor<float> anchorGm;
    AscendC::GlobalTensor<float> positiveGm;
    AscendC::GlobalTensor<float> negativeGm;
    AscendC::GlobalTensor<float> marginGm;
    AscendC::GlobalTensor<float> lossGm;

    uint32_t batchSize;
    uint32_t featureSize;
    uint32_t tileNum;
    uint32_t samplesPerBlock;
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR margin, GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLoss op;
    op.Init(anchor, positive, negative, margin, loss,
            tiling_data.batchSize, tiling_data.featureSize, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    A model that computes Triplet Margin Loss using a custom AscendC kernel for the heavy pairwise distance computation.
    The custom op returns per-sample losses; we apply mean reduction to match PyTorch semantics.
    """
    def __init__(self, margin=1.0):
        super(ModelNew, self).__init__()
        self.margin = float(margin)

    def forward(self, anchor, positive, negative):
        margin_tensor = torch.tensor(self.margin, dtype=anchor.dtype, device=anchor.device)
        per_sample = custom_ops_lib.triplet_margin_loss_custom(anchor, positive, negative, margin_tensor)
        return per_sample.mean()
