kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "matmul_for_upper_triangular_matrices_custom_tiling_data.h"

#ifndef DTYPE_X
#define DTYPE_X float
#endif
#ifndef DTYPE_Y
#define DTYPE_Y float
#endif
#ifndef DTYPE_Z
#define DTYPE_Z float
#endif

class KernelUpperTriMatMul {
public:
    __aicore__ inline KernelUpperTriMatMul() {}
    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c, uint32_t n, uint32_t /*tileNum*/) {
        aGm = reinterpret_cast<__gm__ DTYPE_X *>(a);
        bGm = reinterpret_cast<__gm__ DTYPE_Y *>(b);
        cGm = reinterpret_cast<__gm__ DTYPE_Z *>(c);
        this->N = n;

        uint32_t totalLength = n * n;
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t bid = AscendC::GetBlockIdx();

        uint32_t base = (totalLength / blocks) * bid;
        uint32_t len = (totalLength / blocks);
        if (bid == blocks - 1) {
            len = totalLength - base;
        }
        startIdx = base;
        endIdx = base + len;
    }

    __aicore__ inline void Process() {
        for (uint32_t idx = startIdx; idx < endIdx; ++idx) {
            uint32_t i = idx / N;
            uint32_t j = idx % N;

            if (j < i) {
                // Lower triangular part is zero for product of upper triangular matrices
                cGm[idx] = static_cast<DTYPE_Z>(0);
                continue;
            }

            DTYPE_Z sum = static_cast<DTYPE_Z>(0);
            // For upper triangular matrices, k ranges from i..j
            for (uint32_t k = i; k <= j; ++k) {
                DTYPE_X aval = aGm[i * N + k];  // A[i, k]
                DTYPE_Y bval = bGm[k * N + j];  // B[k, j]
                sum = sum + static_cast<DTYPE_Z>(aval * bval);
            }
            cGm[idx] = sum; // C[i, j]
        }
    }

private:
    __gm__ DTYPE_X* aGm;
    __gm__ DTYPE_Y* bGm;
    __gm__ DTYPE_Z* cGm;
    uint32_t N;
    uint32_t startIdx;
    uint32_t endIdx;
};

extern "C" __global__ __aicore__ void matmul_for_upper_triangular_matrices_custom(GM_ADDR x, GM_ADDR y, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelUpperTriMatMul op;
    op.Init(x, y, z, tiling_data.n, tiling_data.tileNum);
    op.Process();
}
