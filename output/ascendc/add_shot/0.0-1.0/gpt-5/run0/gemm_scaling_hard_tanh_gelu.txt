kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "gemm_scaling_hard_tanh_gelu_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2;

using DTYPE = float;

class KernelGemmScaleHTGelu {
public:
    __aicore__ inline KernelGemmScaleHTGelu() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                GM_ADDR scale, GM_ADDR hmin, GM_ADDR hmax,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK)
    {
        this->M = M;
        this->N = N;
        this->K = K;
        this->tileM = tileM;
        this->tileN = tileN;
        this->tileK = tileK;

        // Distribute rows by block
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t bid = AscendC::GetBlockIdx();
        uint32_t rowsPerBlock = (M + blocks - 1) / blocks;
        rowStart = rowsPerBlock * bid;
        rowCount = (rowStart + rowsPerBlock <= M) ? rowsPerBlock : (M - rowStart);

        // Set GM buffers (row major)
        xGm.SetGlobalBuffer((__gm__ DTYPE*)x + rowStart * K, rowCount * K);
        wGm.SetGlobalBuffer((__gm__ DTYPE*)w, N * K);        // weight: [N, K] (out, in)
        bGm.SetGlobalBuffer((__gm__ DTYPE*)b, N);
        yGm.SetGlobalBuffer((__gm__ DTYPE*)y + rowStart * N, rowCount * N);

        scaleGm = (__gm__ DTYPE*)scale;
        hminGm  = (__gm__ DTYPE*)hmin;
        hmaxGm  = (__gm__ DTYPE*)hmax;

        // Local buffers for tiles
        pipe.InitBuffer(inQueueA, BUFFER_NUM, tileM * tileK * sizeof(DTYPE));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, tileK * tileN * sizeof(DTYPE));
        pipe.InitBuffer(inQueueBias, BUFFER_NUM, tileN * sizeof(DTYPE));
        pipe.InitBuffer(outQueueC, BUFFER_NUM, tileM * tileN * sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        // Load scalar params once into registers (best effort)
        DTYPE s = (DTYPE)scaleGm[0];
        DTYPE hmin = (DTYPE)hminGm[0];
        DTYPE hmax = (DTYPE)hmaxGm[0];

        for (uint32_t m0 = 0; m0 < rowCount; m0 += tileM) {
            uint32_t curM = AscendC::min(tileM, rowCount - m0);
            for (uint32_t n0 = 0; n0 < N; n0 += tileN) {
                uint32_t curN = AscendC::min(tileN, N - n0);

                // Allocate output tile
                AscendC::LocalTensor<DTYPE> cTile = outQueueC.AllocTensor<DTYPE>();
                // Zero init output tile
                AscendC::SetVector(cTile, (DTYPE)0, curM * curN);

                // GEMM accumulation over K
                for (uint32_t k0 = 0; k0 < K; k0 += tileK) {
                    uint32_t curK = AscendC::min(tileK, K - k0);

                    // Load A tile [curM, curK] contiguous by rows
                    AscendC::LocalTensor<DTYPE> aTile = inQueueA.AllocTensor<DTYPE>();
                    for (uint32_t rm = 0; rm < curM; ++rm) {
                        AscendC::DataCopy(
                            aTile[rm * curK],
                            xGm[(m0 + rm) * K + k0],
                            curK);
                    }
                    inQueueA.EnQue(aTile);

                    // Load B tile [curK, curN] from weight [N, K] => we take rows n0..n0+curN-1, cols k0..k0+curK-1
                    // Store as transposed [curN, curK] to get coalesced access while multiplying
                    AscendC::LocalTensor<DTYPE> bTile = inQueueB.AllocTensor<DTYPE>();
                    for (uint32_t rn = 0; rn < curN; ++rn) {
                        AscendC::DataCopy(
                            bTile[rn * curK],
                            wGm[(n0 + rn) * K + k0],
                            curK);
                    }
                    inQueueB.EnQue(bTile);

                    // DeQue for compute
                    AscendC::LocalTensor<DTYPE> aT = inQueueA.DeQue<DTYPE>();
                    AscendC::LocalTensor<DTYPE> bT = inQueueB.DeQue<DTYPE>();

                    // Compute partial C += A * B^T (since bT is [curN, curK])
                    // Naive matmul in local buffer
                    for (uint32_t rm = 0; rm < curM; ++rm) {
                        for (uint32_t rn = 0; rn < curN; ++rn) {
                            DTYPE acc = (DTYPE)0;
                            AscendC::LocalTensor<DTYPE> aRow = aT[rm * curK];
                            AscendC::LocalTensor<DTYPE> bRow = bT[rn * curK];
                            for (uint32_t rk = 0; rk < curK; ++rk) {
                                acc += aRow[rk] * bRow[rk];
                            }
                            cTile[rm * curN + rn] = cTile[rm * curN + rn] + acc;
                        }
                    }

                    inQueueA.FreeTensor(aT);
                    inQueueB.FreeTensor(bT);
                }

                // Add bias over N
                AscendC::LocalTensor<DTYPE> biasTile = inQueueBias.AllocTensor<DTYPE>();
                AscendC::DataCopy(biasTile, bGm[n0], curN);
                inQueueBias.EnQue(biasTile);
                AscendC::LocalTensor<DTYPE> bT = inQueueBias.DeQue<DTYPE>();
                for (uint32_t rm = 0; rm < curM; ++rm) {
                    for (uint32_t rn = 0; rn < curN; ++rn) {
                        cTile[rm * curN + rn] = cTile[rm * curN + rn] + bT[rn];
                    }
                }
                inQueueBias.FreeTensor(bT);

                // Scale
                for (uint32_t i = 0; i < curM * curN; ++i) {
                    cTile[i] = cTile[i] * s;
                }

                // HardTanh clamp
                for (uint32_t i = 0; i < curM * curN; ++i) {
                    DTYPE v = cTile[i];
                    v = v < hmin ? hmin : v;
                    v = v > hmax ? hmax : v;
                    cTile[i] = v;
                }

                // GELU: 0.5 * x * (1 + erf(x / sqrt(2)))
                const DTYPE inv_sqrt2 = (DTYPE)0.7071067811865475f;
                for (uint32_t i = 0; i < curM * curN; ++i) {
                    DTYPE v = cTile[i];
                    DTYPE erfv = AscendC::Erf(v * inv_sqrt2);
                    cTile[i] = (DTYPE)0.5f * v * ((DTYPE)1.0f + erfv);
                }

                // Write back
                for (uint32_t rm = 0; rm < curM; ++rm) {
                    AscendC::DataCopy(
                        yGm[(m0 + rm) * N + n0],
                        cTile[rm * curN],
                        curN);
                }
                outQueueC.FreeTensor(cTile);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  inQueueA, inQueueB, inQueueBias;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> wGm;
    AscendC::GlobalTensor<DTYPE> bGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    __gm__ DTYPE* scaleGm;
    __gm__ DTYPE* hminGm;
    __gm__ DTYPE* hmaxGm;

    uint32_t M, N, K;
    uint32_t tileM, tileN, tileK;
    uint32_t rowStart;
    uint32_t rowCount;
};

extern "C" __global__ __aicore__ void gemm_scaling_hard_tanh_gelu_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
    GM_ADDR scale, GM_ADDR hmin, GM_ADDR hmax,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmScaleHTGelu op;
    op.Init(x, weight, bias, y, scale, hmin, hmax,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model using custom AscendC kernel: GEMM + scaling + HardTanh + GELU
    """
    def __init__(self, in_features, out_features, scaling_factor, hardtanh_min, hardtanh_max):
        super(ModelNew, self).__init__()
        self.gemm = torch.nn.Linear(in_features, out_features)
        self.scaling_factor = float(scaling_factor)
        self.hardtanh_min = float(hardtanh_min)
        self.hardtanh_max = float(hardtanh_max)

    def forward(self, x):
        # weight: [out_features, in_features]; bias: [out_features]
        scale_t = torch.tensor(self.scaling_factor, dtype=x.dtype, device=x.device)
        hmin_t = torch.tensor(self.hardtanh_min, dtype=x.dtype, device=x.device)
        hmax_t = torch.tensor(self.hardtanh_max, dtype=x.dtype, device=x.device)
        return custom_ops_lib.gemm_scaling_hard_tanh_gelu_custom(
            x, self.gemm.weight, self.gemm.bias, scale_t, hmin_t, hmax_t
        )
