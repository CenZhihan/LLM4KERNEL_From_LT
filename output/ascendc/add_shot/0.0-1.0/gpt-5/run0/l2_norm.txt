host_operator_src="""
#include "l2_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_COLS = 1024;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    L2NormCustomTilingData tiling;

    auto originShape = context->GetInputShape(0)->GetOriginShape();
    uint64_t totalLength = originShape.GetShapeSize();
    uint64_t rows = originShape.GetDimNum() > 0 ? static_cast<uint64_t>(originShape.GetDim(0)) : 1;
    uint64_t cols = (rows == 0) ? 0 : (totalLength / rows);

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_rows(static_cast<uint32_t>(rows));
    tiling.set_cols(static_cast<uint32_t>(cols));
    tiling.set_tileCols(TILE_COLS);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* currentWorkspace = context->GetWorkspaceSizes(1);
    // optional workspace for per-row scalars (not strictly needed by kernel, but reserved)
    currentWorkspace[0] = rows * sizeof(float);
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

namespace ops {
class L2NormCustom : public OpDef {
public:
    explicit L2NormCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore()

        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(L2NormCustom);
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "l2_norm_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr float EPS = 1e-12f;

class KernelL2Norm {
public:
    __aicore__ inline KernelL2Norm() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, uint32_t rows, uint32_t cols, uint32_t tileCols)
    {
        this->rows = rows;
        this->cols = cols;
        this->tileCols = tileCols;

        // rows per block
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t rowsPerBlock = (rows + blockNum - 1) / blockNum;
        this->rowStart = blockIdx * rowsPerBlock;
        this->rowEnd = AscendC::Min(this->rowStart + rowsPerBlock, rows);

        // Base pointers for this block
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + (uint64_t)this->rowStart * (uint64_t)this->cols, (uint64_t)(this->rowEnd - this->rowStart) * (uint64_t)this->cols);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + (uint64_t)this->rowStart * (uint64_t)this->cols, (uint64_t)(this->rowEnd - this->rowStart) * (uint64_t)this->cols);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileCols * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileCols * sizeof(DTYPE_Y));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileCols * sizeof(DTYPE_X));
        pipe.InitBuffer(scalarQueue, BUFFER_NUM, 1 * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        if (this->rowStart >= this->rowEnd || this->cols == 0) {
            return;
        }

        for (uint32_t r = this->rowStart; r < this->rowEnd; ++r) {
            float sumSquares = 0.0f;
            uint32_t localRowIdx = r - this->rowStart;
            uint64_t rowBase = (uint64_t)localRowIdx * (uint64_t)this->cols;

            // Pass 1: compute L2 norm for row r
            for (uint32_t c = 0; c < this->cols; c += this->tileCols) {
                uint32_t len = AscendC::Min(this->tileCols, this->cols - c);

                AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(xLocal, xGm[rowBase + c], len);
                inQueueX.EnQue(xLocal);

                AscendC::LocalTensor<DTYPE_X> xLocalIn = inQueueX.DeQue<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_X> sqLocal = tmpQueue.AllocTensor<DTYPE_X>();
                AscendC::Mul(sqLocal, xLocalIn, xLocalIn, len); // square

                AscendC::LocalTensor<DTYPE_X> reduceLocal = scalarQueue.AllocTensor<DTYPE_X>();
                AscendC::ReduceSum(reduceLocal, sqLocal, len); // 1-element result

                // Read scalar partial sum and accumulate
                float partial = reduceLocal.GetValue(0);
                sumSquares += partial;

                tmpQueue.FreeTensor(sqLocal);
                scalarQueue.FreeTensor(reduceLocal);
                inQueueX.FreeTensor(xLocalIn);
            }

            float invNorm = 1.0f / AscendC::Sqrt(sumSquares + EPS);

            // Pass 2: write normalized row
            for (uint32_t c = 0; c < this->cols; c += this->tileCols) {
                uint32_t len = AscendC::Min(this->tileCols, this->cols - c);

                AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(xLocal, xGm[rowBase + c], len);
                inQueueX.EnQue(xLocal);

                AscendC::LocalTensor<DTYPE_X> xLocalIn = inQueueX.DeQue<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
                AscendC::Muls(yLocal, xLocalIn, invNorm, len);
                outQueueY.EnQue(yLocal);
                inQueueX.FreeTensor(xLocalIn);

                AscendC::LocalTensor<DTYPE_Y> yLocalOut = outQueueY.DeQue<DTYPE_Y>();
                AscendC::DataCopy(yGm[rowBase + c], yLocalOut, len);
                outQueueY.FreeTensor(yLocalOut);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> scalarQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t rows;
    uint32_t cols;
    uint32_t tileCols;
    uint32_t rowStart;
    uint32_t rowEnd;
};

extern "C" __global__ __aicore__ void l2_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelL2Norm op;
    op.Init(x, y, workspace, tiling_data.rows, tiling_data.cols, tiling_data.tileCols);
    op.Process();
}
"""
