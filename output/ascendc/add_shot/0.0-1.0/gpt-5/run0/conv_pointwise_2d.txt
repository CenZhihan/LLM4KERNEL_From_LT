namespace {
inline uint32_t GetDimSafe(const gert::Shape &shape, size_t idx, uint32_t def) {
    if (idx < shape.GetDimNum()) {
        return static_cast<uint32_t>(shape.GetDim(idx));
    }
    return def;
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t DEFAULT_TILE_W = 256;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    ConvPointwise2dCustomTilingData tiling;

    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);

    // Expect x: [N, C_in, H, W]
    uint32_t N = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    uint32_t C_in = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    uint32_t H = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));
    uint32_t W = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(3));

    // Weight could be [C_out, C_in] or [C_out, C_in, 1, 1]
    uint32_t w_dims = static_cast<uint32_t>(w_shape->GetOriginShape().GetDimNum());
    uint32_t C_out = 0;
    uint32_t C_in_w = 0;
    if (w_dims == 2) {
        C_out = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));
        C_in_w = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(1));
    } else {
        C_out = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));
        C_in_w = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(1));
    }

    if (C_in != C_in_w) {
        return ge::GRAPH_FAILED;
    }

    uint32_t tileW = DEFAULT_TILE_W;
    if (tileW == 0) tileW = 1;
    if (tileW > W) tileW = W;

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_N(N);
    tiling.set_C_in(C_in);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_C_out(C_out);
    tiling.set_tileW(tileW);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_pointwise_2d_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelPointwiseConv2D {
public:
    __aicore__ inline KernelPointwiseConv2D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t C_in, uint32_t H, uint32_t W, uint32_t C_out,
                                uint32_t tileW)
    {
        this->N = N;
        this->C_in = C_in;
        this->H = H;
        this->W = W;
        this->C_out = C_out;
        this->tileW = tileW;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint64_t)N * C_in * H * W);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W *)w, (uint64_t)C_out * C_in);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y, (uint64_t)N * C_out * H * W);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileW * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileW * sizeof(DTYPE_Z));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileW * sizeof(DTYPE_Z));
        pipe.InitBuffer(wRowQueue, BUFFER_NUM, this->C_in * sizeof(DTYPE_W));
    }

    __aicore__ inline void Process()
    {
        uint32_t totalNH = N * H;
        uint32_t blkNum = AscendC::GetBlockNum();
        uint32_t blkIdx = AscendC::GetBlockIdx();

        uint32_t chunk = (totalNH + blkNum - 1) / blkNum;
        uint32_t startNH = blkIdx * chunk;
        uint32_t endNH = AscendC::Min(startNH + chunk, totalNH);

        for (uint32_t nh = startNH; nh < endNH; ++nh) {
            uint32_t n = nh / H;
            uint32_t h = nh % H;

            for (uint32_t co = 0; co < C_out; ++co) {
                // Load weight row for this out channel
                AscendC::LocalTensor<DTYPE_W> wRowLocal = wRowQueue.AllocTensor<DTYPE_W>();
                AscendC::DataCopy(wRowLocal, wGm[co * C_in], C_in);
                wRowQueue.EnQue(wRowLocal);
                wRowLocal = wRowQueue.DeQue<DTYPE_W>();

                for (uint32_t wStart = 0; wStart < W; wStart += tileW) {
                    uint32_t currTile = AscendC::Min(tileW, W - wStart);

                    AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();

                    // ci = 0 initialize
                    {
                        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                        uint64_t xOffset = (((uint64_t)n * C_in + 0) * H + h) * W + wStart;
                        AscendC::DataCopy(xLocal, xGm[xOffset], currTile);
                        inQueueX.EnQue(xLocal);
                        xLocal = inQueueX.DeQue<DTYPE_X>();

                        DTYPE_W w0 = wRowLocal.GetValue(0);
                        AscendC::Muls(zLocal, xLocal, w0, currTile);

                        inQueueX.FreeTensor(xLocal);
                    }

                    // accumulate for remaining input channels
                    for (uint32_t ci = 1; ci < C_in; ++ci) {
                        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                        uint64_t xOffset = (((uint64_t)n * C_in + ci) * H + h) * W + wStart;
                        AscendC::DataCopy(xLocal, xGm[xOffset], currTile);
                        inQueueX.EnQue(xLocal);
                        xLocal = inQueueX.DeQue<DTYPE_X>();

                        DTYPE_W wsc = wRowLocal.GetValue(ci);

                        AscendC::LocalTensor<DTYPE_Z> tmpLocal = tmpQueue.AllocTensor<DTYPE_Z>();
                        AscendC::Muls(tmpLocal, xLocal, wsc, currTile);
                        AscendC::Add(zLocal, zLocal, tmpLocal, currTile);

                        tmpQueue.FreeTensor(tmpLocal);
                        inQueueX.FreeTensor(xLocal);
                    }

                    // store
                    uint64_t yOffset = (((uint64_t)n * C_out + co) * H + h) * W + wStart;
                    AscendC::DataCopy(yGm[yOffset], zLocal, currTile);
                    outQueueZ.FreeTensor(zLocal);
                }

                wRowQueue.FreeTensor(wRowLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> wRowQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> tmpQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t N;
    uint32_t C_in;
    uint32_t H;
    uint32_t W;
    uint32_t C_out;
    uint32_t tileW;
};

extern "C" __global__ __aicore__ void conv_pointwise_2d_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelPointwiseConv2D op;
    op.Init(x, w, y,
            tiling_data.N, tiling_data.C_in, tiling_data.H, tiling_data.W, tiling_data.C_out,
            tiling_data.tileW);
    op.Process();
}

model_src='''
import math
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels: int, out_channels: int):
        super().__init__()
        # Store weights as [C_out, C_in] for the custom kernel
        self.weight = torch.nn.Parameter(torch.empty(out_channels, in_channels))
        torch.nn.init.kaiming_uniform_(self.weight, a=math.sqrt(5))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return custom_ops_lib.conv_pointwise_2d_custom(x, self.weight)
