inline uint32_t ucast(int64_t x) { return static_cast<uint32_t>(x); }

namespace optiling {
const uint32_t DEFAULT_BLOCKS = 128;
const uint32_t DEFAULT_TILE_K = 256;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    FourDimTensorMatrixMultiplicationCustomTilingData tiling;

    const auto *shapeA = context->GetInputShape(0);
    const auto *shapeB = context->GetInputShape(1);
    auto A = shapeA->GetOriginShape();
    auto B = shapeB->GetOriginShape();

    if (A.GetDimNum() != 4 || B.GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    uint32_t b = ucast(A.GetDim(0));
    uint32_t i = ucast(A.GetDim(1));
    uint32_t j = ucast(A.GetDim(2));
    uint32_t l = ucast(A.GetDim(3));
    uint32_t k = ucast(B.GetDim(1));

    uint64_t flat_n64 = static_cast<uint64_t>(b) * i * j;
    uint32_t flat_n = static_cast<uint32_t>(flat_n64);

    uint32_t blocks = DEFAULT_BLOCKS;
    if (flat_n < blocks) blocks = flat_n == 0 ? 1 : flat_n;

    context->SetBlockDim(blocks);

    tiling.set_b(b);
    tiling.set_i(i);
    tiling.set_j(j);
    tiling.set_l(l);
    tiling.set_k(k);
    tiling.set_flat_n(flat_n);
    tiling.set_tile_k(DEFAULT_TILE_K);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "four_dim_tensor_matrix_multiplication_custom_tiling_data.h"

// Simple GEMM-like accumulation per (b,i,j) row:
//   For each flattened n in [0, b*i*j):
//     C[n, :] = sum_{l} A[n, l] * B[l, :]
//
// We tile along K to control local buffer size.

constexpr int32_t BUFFER_NUM = 2;

class KernelFDTMM {
public:
    __aicore__ inline KernelFDTMM() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t b, uint32_t i, uint32_t j, uint32_t l, uint32_t k, uint32_t flat_n, uint32_t tile_k)
    {
        this->b = b; this->i = i; this->j = j; this->l = l; this->k = k; this->flat_n = flat_n;
        this->tileK = tile_k;

        // Assign GM buffers
        Agm.SetGlobalBuffer((__gm__ float*)A, (uint64_t)flat_n * l);
        Bgm.SetGlobalBuffer((__gm__ float*)B, (uint64_t)l * k);
        Cgm.SetGlobalBuffer((__gm__ float*)C, (uint64_t)flat_n * k);

        // Init local queues for double buffering on B tiles and output tiles
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->tileK * sizeof(float));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileK * sizeof(float));
        pipe.InitBuffer(outQueueC, BUFFER_NUM, this->tileK * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // Distribute flat_n rows across blocks almost evenly
        uint32_t rowsPerBlock = (flat_n + blockNum - 1) / blockNum;
        uint32_t startRow = blockIdx * rowsPerBlock;
        uint32_t endRow = AscendC::Min(flat_n, startRow + rowsPerBlock);

        for (uint32_t n = startRow; n < endRow; ++n) {
            // For each row n, compute C[n, 0:k] in tiles of tileK
            for (uint32_t kOff = 0; kOff < k; kOff += tileK) {
                uint32_t curK = AscendC::Min(tileK, k - kOff);
                // Allocate output tile and zero-initialize
                AscendC::LocalTensor<float> cLocal = outQueueC.AllocTensor<float>();
                AscendC::Duplicate(cLocal, 0.0f, curK);

                // Accumulate across L
                for (uint32_t lIdx = 0; lIdx < l; ++lIdx) {
                    // Load B[lIdx, kOff: kOff+curK]
                    AscendC::LocalTensor<float> bLocal = inQueueB.AllocTensor<float>();
                    AscendC::DataCopy(bLocal, Bgm[lIdx * k + kOff], curK);

                    // Load scalar a = A[n, lIdx]
                    // Copy 1 element into a tiny local tensor
                    float aScalar;
                    // Use a dedicated 1-length local tensor to fetch the scalar
                    AscendC::LocalTensor<float> aLocal = tmpQueue.AllocTensor<float>();
                    AscendC::DataCopy(aLocal, Agm[n * l + lIdx], 1);
                    // Move scalar out (assume we can read first element)
                    aScalar = aLocal.GetValue(0);

                    // temp = bLocal * aScalar
                    AscendC::LocalTensor<float> tmpLocal = tmpQueue.AllocTensor<float>();
                    AscendC::Muls(tmpLocal, bLocal, aScalar, curK);

                    // cLocal += temp
                    AscendC::Add(cLocal, cLocal, tmpLocal, curK);

                    tmpQueue.FreeTensor(tmpLocal);
                    tmpQueue.FreeTensor(aLocal);
                    inQueueB.FreeTensor(bLocal);
                }

                // Write back C[n, kOff: kOff+curK]
                AscendC::DataCopy(Cgm[n * k + kOff], cLocal, curK);
                outQueueC.FreeTensor(cLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC;

    AscendC::GlobalTensor<float> Agm;
    AscendC::GlobalTensor<float> Bgm;
    AscendC::GlobalTensor<float> Cgm;

    uint32_t b, i, j, l, k, flat_n;
    uint32_t tileK;
};

extern "C" __global__ __aicore__ void four_dim_tensor_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelFDTMM op;
    op.Init(A, B, C,
            tiling_data.b, tiling_data.i, tiling_data.j, tiling_data.l, tiling_data.k,
            tiling_data.flat_n, tiling_data.tile_k);
    op.Process();
}
