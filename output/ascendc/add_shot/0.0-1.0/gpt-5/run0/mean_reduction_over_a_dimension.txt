static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    MeanReductionOverADimensionCustomTilingData tiling;

    const gert::Shape* in_shape = context->GetInputShape(0);
    auto origin = in_shape->GetOriginShape();
    int64_t rank = origin.GetDimNum();

    int64_t dim = 0;
    if (context->GetAttrs() != nullptr) {
        dim = context->GetAttrs()->GetAttrInt("dim");

    if (dim < 0) {
        dim += rank;
    }

    uint64_t outer64 = 1, inner64 = 1;
    for (int64_t i = 0; i < dim; ++i) {
        outer64 *= static_cast<uint64_t>(origin.GetDim(i));

        inner64 *= static_cast<uint64_t>(origin.GetDim(i));

    uint64_t reduce64 = static_cast<uint64_t>(origin.GetDim(dim));

    uint32_t outer = static_cast<uint32_t>(outer64);
    uint32_t inner = static_cast<uint32_t>(inner64);
    uint32_t reduce = static_cast<uint32_t>(reduce64);

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_outer(outer);
    tiling.set_inner(inner);
    tiling.set_reduce(reduce);
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);

    auto origin = x_shape->GetOriginShape();
    int64_t rank = origin.GetDimNum();

    int64_t dim = 0;
    if (context->GetAttrs() != nullptr) {
        dim = context->GetAttrs()->GetAttrInt("dim");

    if (dim < 0) {
        dim += rank;
    }

    std::vector<int64_t> out_dims;
    out_dims.reserve(rank > 0 ? rank - 1 : 0);
    for (int64_t i = 0; i < rank; ++i) {

        out_dims.push_back(origin.GetDim(i));

    for (size_t i = 0; i < out_dims.size(); ++i) {

    }
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "mean_reduction_over_a_dimension_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMeanReduce {
public:
    __aicore__ inline KernelMeanReduce() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t outer, uint32_t reduce, uint32_t inner, uint32_t tileNum)
    {
        this->outer = outer;
        this->reduce = reduce;
        this->inner = inner;
        this->tileNum = tileNum;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blkIdx = AscendC::GetBlockIdx();

        this->outerPerBlock = (outer + blockNum - 1) / blockNum;
        this->outerStart = blkIdx * this->outerPerBlock;
        this->outerEnd = this->outerStart + this->outerPerBlock;
        if (this->outerEnd > outer) this->outerEnd = outer;

        // choose tile length along inner dimension
        uint32_t denom = tileNum * BUFFER_NUM;
        this->tileLength = (inner + denom - 1) / denom;
        if (this->tileLength == 0) this->tileLength = inner;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint64_t)outer * reduce * inner);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y, (uint64_t)outer * inner);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t o = outerStart; o < outerEnd; ++o) {
            for (uint32_t t = 0; t < inner; t += tileLength) {
                uint32_t currLen = (t + tileLength <= inner) ? tileLength : (inner - t);

                // Initialize accumulator with first r slice
                // Copy first r=0
                {
                    AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                    uint64_t srcIdx = ((uint64_t)o * reduce + 0) * inner + t;
                    AscendC::DataCopy(xLocal, xGm[srcIdx], currLen);
                    inQueueX.EnQue(xLocal);
                }
                AscendC::LocalTensor<DTYPE_X> firstSlice = inQueueX.DeQue<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_Z> acc = outQueueY.AllocTensor<DTYPE_Z>();
                // Cast copy: DTYPE_X -> DTYPE_Z via DataCopy when same type; assume types are same float
                AscendC::DataCopy(acc, firstSlice, currLen);
                inQueueX.FreeTensor(firstSlice);

                for (uint32_t r = 1; r < reduce; ++r) {
                    AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                    uint64_t srcIdx = ((uint64_t)o * reduce + r) * inner + t;
                    AscendC::DataCopy(xLocal, xGm[srcIdx], currLen);
                    inQueueX.EnQue(xLocal);

                    AscendC::LocalTensor<DTYPE_X> xLocalDeq = inQueueX.DeQue<DTYPE_X>();
                    AscendC::Add(acc, acc, xLocalDeq, currLen);
                    inQueueX.FreeTensor(xLocalDeq);
                }

                // Multiply by 1/reduce to get mean
                DTYPE_Z scale = static_cast<DTYPE_Z>(1.0f) / static_cast<DTYPE_Z>(reduce);
                AscendC::Muls(acc, acc, scale, currLen);
                outQueueY.EnQue(acc);

                AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
                uint64_t dstIdx = (uint64_t)o * inner + t;
                AscendC::DataCopy(yGm[dstIdx], yLocal, currLen);
                outQueueY.FreeTensor(yLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t outer;
    uint32_t reduce;
    uint32_t inner;

    uint32_t tileNum;
    uint32_t tileLength;

    uint32_t outerPerBlock;
    uint32_t outerStart;
    uint32_t outerEnd;
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReduce op;
    op.Init(x, y, tiling_data.outer, tiling_data.reduce, tiling_data.inner, tiling_data.tileNum);
    op.Process();
}

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor mean_reduction_over_a_dimension_custom_impl_npu(const at::Tensor& self, int64_t dim) {
    int64_t d = dim;
    if (d < 0) d += self.dim();
    std::vector<int64_t> out_sizes;
    out_sizes.reserve(self.dim() > 0 ? self.dim() - 1 : 0);
    for (int64_t i = 0; i < self.dim(); ++i) {
        if (i == d) continue;
        out_sizes.push_back(self.size(i));
    }
    at::Tensor result = at::empty(out_sizes, self.options());
    EXEC_NPU_CMD(aclnnMeanReductionOverADimensionCustom, self, d, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("mean_reduction_over_a_dimension_custom", &mean_reduction_over_a_dimension_custom_impl_npu);
