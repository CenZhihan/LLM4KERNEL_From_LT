kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_transpose3d_scaling_avg_pool_bias_add_scaling_custom_tiling_data.h"

extern "C" __global__ __aicore__ void conv_transpose3d_scaling_avg_pool_bias_add_scaling_custom(
    GM_ADDR x, GM_ADDR scale1, GM_ADDR bias, GM_ADDR scale2, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {

    GET_TILING_DATA(tiling_data, tiling);

    // Cast GM addresses
    __gm__ float* xGm = reinterpret_cast<__gm__ float*>(x);
    __gm__ float* yGm = reinterpret_cast<__gm__ float*>(y);
    __gm__ float* biasGm = reinterpret_cast<__gm__ float*>(bias);
    __gm__ float* s1Gm = reinterpret_cast<__gm__ float*>(scale1);
    __gm__ float* s2Gm = reinterpret_cast<__gm__ float*>(scale2);

    const int64_t N  = static_cast<int64_t>(tiling_data.n);
    const int64_t C  = static_cast<int64_t>(tiling_data.c);
    const int64_t D  = static_cast<int64_t>(tiling_data.d);
    const int64_t H  = static_cast<int64_t>(tiling_data.h);
    const int64_t W  = static_cast<int64_t>(tiling_data.w);
    const int64_t OD = static_cast<int64_t>(tiling_data.od);
    const int64_t OH = static_cast<int64_t>(tiling_data.oh);
    const int64_t OW = static_cast<int64_t>(tiling_data.ow);

    const int64_t totalOut = static_cast<int64_t>(tiling_data.totalOutLength);

    // Load scalar scales
    float s1 = s1Gm[0];
    float s2 = s2Gm[0];

    // Work partition
    int64_t blockNum = static_cast<int64_t>(AscendC::GetBlockNum());
    int64_t blockIdx = static_cast<int64_t>(AscendC::GetBlockIdx());
    int64_t blockLen = totalOut / blockNum;
    int64_t start = blockLen * blockIdx;
    int64_t end = (blockIdx == blockNum - 1) ? totalOut : start + blockLen;

    // Strides for input x in NCDHW
    const int64_t strideW = 1;
    const int64_t strideH = W;
    const int64_t strideD = H * W;
    const int64_t strideC = D * H * W;
    const int64_t strideN = C * D * H * W;

    for (int64_t idx = start; idx < end; ++idx) {
        // Unravel idx -> (n, c, od, oh, ow) for output
        int64_t tmp = idx;
        int64_t ow = tmp % OW; tmp /= OW;
        int64_t oh = tmp % OH; tmp /= OH;
        int64_t od = tmp % OD; tmp /= OD;
        int64_t c  = tmp % C;  tmp /= C;
        int64_t n  = tmp;

        // Map to input coordinates (stride=2, k=2)
        int64_t iw = ow * 2;
        int64_t ih = oh * 2;
        int64_t id = od * 2;

        // Base input linear index for (n,c,id,ih,iw)
        int64_t in_base = n * strideN + c * strideC + id * strideD + ih * strideH + iw * strideW;

        // Sum 2x2x2 cube
        float v000 = xGm[in_base + 0];
        float v001 = xGm[in_base + 1];
        float v010 = xGm[in_base + strideH + 0];
        float v011 = xGm[in_base + strideH + 1];
        float v100 = xGm[in_base + strideD + 0];
        float v101 = xGm[in_base + strideD + 1];
        float v110 = xGm[in_base + strideD + strideH + 0];
        float v111 = xGm[in_base + strideD + strideH + 1];

        float sum8 = v000 + v001 + v010 + v011 + v100 + v101 + v110 + v111;
        float avg  = sum8 * 0.125f; // divide by 8

        float outv = (avg * s1 + biasGm[c]) * s2;
        yGm[idx] = outv;
    }
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model using a custom fused AscendC operator for:
      scaling -> AvgPool3d(k=2,s=2) -> bias add -> scaling.
    ConvTranspose3d remains as a standard nn module.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, scale1, scale2, bias_shape):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding)
        self.scale1 = torch.nn.Parameter(torch.tensor(scale1, dtype=torch.float32))
        self.bias = torch.nn.Parameter(torch.randn(bias_shape, dtype=torch.float32))
        self.scale2 = torch.nn.Parameter(torch.tensor(scale2, dtype=torch.float32))

    def forward(self, x):
        x = self.conv_transpose(x)
        x = custom_ops_lib.conv_transpose3d_scaling_avg_pool_bias_add_scaling_custom(x, self.scale1, self.bias, self.scale2)
        return x
