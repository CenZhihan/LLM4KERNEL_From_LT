namespace {
inline void GetSizeFromTensor(const gert::Tensor* t, int32_t &h, int32_t &w) {
    h = 0; w = 0;
    if (t == nullptr) return;
    const void* data = t->GetData();
    if (data == nullptr) return;
    const int32_t* sz = reinterpret_cast<const int32_t*>(data);
    h = sz[0];
    w = sz[1];
}
}

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    InterpolateDynamicCustomTilingData tiling;

    // Input x: [N, C, H, W]
    const gert::Shape* x_shape = context->GetInputShape(0);
    uint32_t n = (x_shape->GetDimNum() >= 1) ? x_shape->GetOriginShape().GetDim(0) : 1;
    uint32_t c = (x_shape->GetDimNum() >= 2) ? x_shape->GetOriginShape().GetDim(1) : 1;
    uint32_t inH = (x_shape->GetDimNum() >= 3) ? x_shape->GetOriginShape().GetDim(2) : 1;
    uint32_t inW = (x_shape->GetDimNum() >= 4) ? x_shape->GetOriginShape().GetDim(3) : 1;

    int32_t outH_i = 0, outW_i = 0;
    auto* sizeTensor = context->GetOptionalInputTensor(1);
    GetSizeFromTensor(sizeTensor, outH_i, outW_i);
    uint32_t outH = (outH_i > 0) ? static_cast<uint32_t>(outH_i) : inH;
    uint32_t outW = (outW_i > 0) ? static_cast<uint32_t>(outW_i) : inW;

    uint32_t nc = n * c;
    uint32_t inPlane = inH * inW;
    uint32_t outPlane = outH * outW;

    // Use up to 32 blocks or nc, whichever is smaller
    const uint32_t BLOCK_DIM = (nc < 32U) ? nc : 32U;
    context->SetBlockDim(BLOCK_DIM);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_inH(inH);
    tiling.set_inW(inW);
    tiling.set_outH(outH);
    tiling.set_outW(outW);
    tiling.set_nc(nc);
    tiling.set_inPlane(inPlane);
    tiling.set_outPlane(outPlane);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);

    uint64_t n = (x_shape->GetDimNum() >= 1) ? x_shape->GetDim(0) : 1;
    uint64_t c = (x_shape->GetDimNum() >= 2) ? x_shape->GetDim(1) : 1;
    uint64_t inH = (x_shape->GetDimNum() >= 3) ? x_shape->GetDim(2) : 1;
    uint64_t inW = (x_shape->GetDimNum() >= 4) ? x_shape->GetDim(3) : 1;

    uint64_t outH = inH;
    uint64_t outW = inW;
    auto* sizeTensor = context->GetOptionalInputTensor(1);
    if (sizeTensor != nullptr && sizeTensor->GetData() != nullptr) {
        const int32_t* sz = reinterpret_cast<const int32_t*>(sizeTensor->GetData());
        if (sz[0] > 0) outH = static_cast<uint64_t>(sz[0]);
        if (sz[1] > 0) outW = static_cast<uint64_t>(sz[1]);
    }

    y_shape->SetDimNum(4);
    y_shape->SetDim(0, n);
    y_shape->SetDim(1, c);
    y_shape->SetDim(2, outH);
    y_shape->SetDim(3, outW);
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "interpolate_dynamic_custom_tiling_data.h"

typedef float DTYPE_X;
typedef float DTYPE_Z;
typedef int32_t DTYPE_SIZE;

constexpr int32_t BUFFER_NUM = 1;

class KernelInterpolateDynamic {
public:
    __aicore__ inline KernelInterpolateDynamic() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR size, GM_ADDR y,
                                uint32_t n, uint32_t c,
                                uint32_t inH, uint32_t inW,
                                uint32_t outH, uint32_t outW,
                                uint32_t nc, uint32_t inPlane, uint32_t outPlane)
    {
        (void)size; // size is carried via tiling, not read in device kernel
        this->n = n;
        this->c = c;
        this->inH = inH;
        this->inW = inW;
        this->outH = outH;
        this->outW = outW;
        this->nc = nc;
        this->inPlane = inPlane;
        this->outPlane = outPlane;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint32_t)(inPlane * nc));
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y, (uint32_t)(outPlane * nc));

        pipe.InitBuffer(inQueue, BUFFER_NUM, this->inPlane * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueue, BUFFER_NUM, this->outPlane * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        // each block processes multiple (n,c) planes in a strided fashion
        for (uint32_t plane = AscendC::GetBlockIdx(); plane < nc; plane += AscendC::GetBlockNum()) {
            CopyIn(plane);
            Compute(plane);
            CopyOut(plane);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t plane)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueue.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[plane * this->inPlane], this->inPlane);
        inQueue.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t plane)
    {
        (void)plane;
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueue.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueue.AllocTensor<DTYPE_Z>();

        // Bilinear interpolate with align_corners = false
        float scale_h = (this->outH > 1) ? static_cast<float>(this->inH) / static_cast<float>(this->outH) : 0.f;
        float scale_w = (this->outW > 1) ? static_cast<float>(this->inW) / static_cast<float>(this->outW) : 0.f;

        for (uint32_t oh = 0; oh < this->outH; ++oh) {
            float in_y = (static_cast<float>(oh) + 0.5f) * scale_h - 0.5f;
            int32_t y0 = static_cast<int32_t>(AscendC::Floor(in_y));
            int32_t y1 = y0 + 1;
            float ly = in_y - static_cast<float>(y0);
            float hy = 1.f - ly;
            if (y0 < 0) { y0 = 0; }
            if (y1 < 0) { y1 = 0; }
            if (y0 >= static_cast<int32_t>(this->inH)) { y0 = this->inH - 1; }
            if (y1 >= static_cast<int32_t>(this->inH)) { y1 = this->inH - 1; }

            for (uint32_t ow = 0; ow < this->outW; ++ow) {
                float in_x = (static_cast<float>(ow) + 0.5f) * scale_w - 0.5f;
                int32_t x0 = static_cast<int32_t>(AscendC::Floor(in_x));
                int32_t x1 = x0 + 1;
                float lx = in_x - static_cast<float>(x0);
                float hx = 1.f - lx;
                if (x0 < 0) { x0 = 0; }
                if (x1 < 0) { x1 = 0; }
                if (x0 >= static_cast<int32_t>(this->inW)) { x0 = this->inW - 1; }
                if (x1 >= static_cast<int32_t>(this->inW)) { x1 = this->inW - 1; }

                uint32_t i00 = static_cast<uint32_t>(y0) * this->inW + static_cast<uint32_t>(x0);
                uint32_t i01 = static_cast<uint32_t>(y0) * this->inW + static_cast<uint32_t>(x1);
                uint32_t i10 = static_cast<uint32_t>(y1) * this->inW + static_cast<uint32_t>(x0);
                uint32_t i11 = static_cast<uint32_t>(y1) * this->inW + static_cast<uint32_t>(x1);

                float v00 = static_cast<float>(xLocal.GetValue(i00));
                float v01 = static_cast<float>(xLocal.GetValue(i01));
                float v10 = static_cast<float>(xLocal.GetValue(i10));
                float v11 = static_cast<float>(xLocal.GetValue(i11));

                float top = hx * v00 + lx * v01;
                float bottom = hx * v10 + lx * v11;
                float v = hy * top + ly * bottom;

                uint32_t oidx = oh * this->outW + ow;
                zLocal.SetValue(oidx, static_cast<DTYPE_Z>(v));
            }
        }

        outQueue.EnQue<DTYPE_Z>(zLocal);
        inQueue.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t plane)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueue.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[plane * this->outPlane], zLocal, this->outPlane);
        outQueue.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t n;
    uint32_t c;
    uint32_t inH, inW;
    uint32_t outH, outW;
    uint32_t nc;
    uint32_t inPlane, outPlane;
};

extern "C" __global__ __aicore__ void interpolate_dynamic_custom(GM_ADDR x, GM_ADDR size, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelInterpolateDynamic op;
    op.Init(x, size, y,
            tiling_data.n, tiling_data.c,
            tiling_data.inH, tiling_data.inW,
            tiling_data.outH, tiling_data.outW,
            tiling_data.nc, tiling_data.inPlane, tiling_data.outPlane);
    op.Process();
}

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor interpolate_dynamic_custom_impl_npu(const at::Tensor& self, const at::Tensor& size_tensor) {
    TORCH_CHECK(size_tensor.numel() == 2, "size_tensor must have 2 elements (H, W)");
    auto out_h = size_tensor.cpu().to(at::kInt).contiguous().view({2});
    int h = out_h[0].item<int>();
    int w = out_h[1].item<int>();
    at::Tensor result = at::empty({self.size(0), self.size(1), h, w}, self.options());
    EXEC_NPU_CMD(aclnnInterpolateDynamicCustom, self, size_tensor.to(self.device(), /*dtype=*/at::kInt), result);
    return result;
}

at::Tensor interpolate_dynamic_custom_impl_npu_tuple(const at::Tensor& self, std::vector<int64_t> size) {
    TORCH_CHECK(size.size() == 2, "size must be (H, W)");
    int h = static_cast<int>(size[0]);
    int w = static_cast<int>(size[1]);
    at::Tensor size_tensor = at::tensor({h, w}, at::TensorOptions().dtype(at::kInt).device(self.device()));
    at::Tensor result = at::empty({self.size(0), self.size(1), h, w}, self.options());
    EXEC_NPU_CMD(aclnnInterpolateDynamicCustom, self, size_tensor, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("interpolate_dynamic_custom", static_cast<at::Tensor(*)(const at::Tensor&, const at::Tensor&)>(&interpolate_dynamic_custom_impl_npu));
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("interpolate_dynamic_custom", &interpolate_dynamic_custom_impl_npu_tuple, "Bilinear interpolate with dynamic size (align_corners=False)");
