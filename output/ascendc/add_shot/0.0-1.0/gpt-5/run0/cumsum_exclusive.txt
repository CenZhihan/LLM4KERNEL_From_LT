#include "cumsum_exclusive_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
inline uint64_t Product(const gert::Shape& s) {
    uint64_t prod = 1;
    for (int i = 0; i < s.GetDimNum(); ++i) {
        prod *= static_cast<uint64_t>(s.GetDim(i));
    }
    return prod;
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    CumsumExclusiveCustomTilingData tiling;

    const auto* inShape = context->GetInputShape(0);
    auto origin = inShape->GetOriginShape();
    int64_t rank = origin.GetDimNum();

    // get axis attribute, support negative
    int64_t axis = 0;
    if (context->GetAttrs() != nullptr) {
        // try fetch attr named "dim"
        axis = context->GetAttrs()->GetInt("dim");
    }
    if (axis < 0) axis += rank;
    if (axis < 0) axis = 0;
    if (axis >= rank) axis = rank - 1;

    uint32_t rowSize = static_cast<uint32_t>(origin.GetDim(axis));
    uint64_t totalSize = Product(origin);
    uint32_t rows = static_cast<uint32_t>(totalSize / (rowSize == 0 ? 1 : rowSize));
    if (rowSize == 0) rows = 0;

    // rows per block, avoid zero
    uint32_t rowsPerBlock = (rows + BLOCK_DIM - 1) / BLOCK_DIM;
    if (rowsPerBlock == 0) rowsPerBlock = 1;

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_rowSize(rowSize);
    tiling.set_rows(rows);
    tiling.set_rowsPerBlock(rowsPerBlock);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "cumsum_exclusive_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 1; // one row per buffer

class KernelCumsumExclusive {
public:
    __aicore__ inline KernelCumsumExclusive() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t rowSize, uint32_t rows, uint32_t rowsPerBlock)
    {
        this->rowSize = rowSize;
        this->rows = rows;
        this->rowsPerBlock = rowsPerBlock;

        uint32_t startRow = AscendC::GetBlockIdx() * rowsPerBlock;
        this->rowStart = startRow;
        uint32_t endRow = startRow + rowsPerBlock;
        if (endRow > rows) endRow = rows;
        this->rowEnd = endRow;

        // set GM base pointers to the start row handled by this block
        uint64_t baseOffset = static_cast<uint64_t>(rowStart) * static_cast<uint64_t>(rowSize);
        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + baseOffset, (rowEnd - rowStart) * rowSize);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + baseOffset, (rowEnd - rowStart) * rowSize);

        // allocate local buffers for one row
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->rowSize * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->rowSize * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        // iterate rows assigned to this block
        for (uint32_t r = 0; r < (rowEnd - rowStart); ++r) {
            CopyIn(r);
            Compute(r);
            CopyOut(r);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t r)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[r * this->rowSize], this->rowSize);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t /*r*/)
    {
        // compute exclusive cumsum for one row using iterative doubling on a shifted buffer
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

        if (this->rowSize == 0) {
            outQueueY.EnQue<DTYPE_Y>(yLocal);
            inQueueX.FreeTensor(xLocal);
            return;
        }

        // y = [0, x[:-1]]
        if (this->rowSize > 0) {
            // first element is zero
            AscendC::Duplicate(yLocal, static_cast<DTYPE_Y>(0), 1);
        }
        if (this->rowSize > 1) {
            AscendC::DataCopy(yLocal[1], xLocal[0], this->rowSize - 1);
        }

        // inclusive scan on y (which is shifted x) by iterative doubling
        for (uint32_t step = 1; step < this->rowSize; step <<= 1) {
            uint32_t len = this->rowSize - step;
            if (len == 0) break;
            AscendC::Add(yLocal[step], yLocal[step], yLocal[0], len);
        }

        outQueueY.EnQue<DTYPE_Y>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t r)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[r * this->rowSize], yLocal, this->rowSize);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t rowSize;
    uint32_t rows;
    uint32_t rowsPerBlock;
    uint32_t rowStart;
    uint32_t rowEnd;
};

extern "C" __global__ __aicore__ void cumsum_exclusive_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumsumExclusive op;
    op.Init(x, y, tiling_data.rowSize, tiling_data.rows, tiling_data.rowsPerBlock);
    op.Process();
}
