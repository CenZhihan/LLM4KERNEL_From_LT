kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "matmul_gelu_softmax_custom_tiling_data.h"
#include <cmath>

extern "C" __global__ __aicore__ void matmul_gelu_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    const uint32_t rows = tiling_data.rows;
    const uint32_t cols = tiling_data.cols;

    // Distribute rows across blocks
    const uint32_t block_num = AscendC::GetBlockNum();
    const uint32_t block_idx = AscendC::GetBlockIdx();

    const uint32_t rows_per_block = (rows + block_num - 1) / block_num;
    const uint32_t row_start = rows_per_block * block_idx;
    const uint32_t row_end = (row_start + rows_per_block) > rows ? rows : (row_start + rows_per_block);

    float* x_ptr = reinterpret_cast<float*>(x);
    float* y_ptr = reinterpret_cast<float*>(y);

    for (uint32_t r = row_start; r < row_end; ++r) {
        // First pass: GELU on input row and track max for softmax
        float maxv = -3.4028235e+38f; // -FLT_MAX
        uint64_t base = static_cast<uint64_t>(r) * cols;
        for (uint32_t c = 0; c < cols; ++c) {
            float v = x_ptr[base + c];
            // Approximate GELU with tanh
            float v3 = v * v * v;
            float gelu = 0.5f * v * (1.0f + tanhf(0.79788456f * (v + 0.044715f * v3)));
            y_ptr[base + c] = gelu;
            if (gelu > maxv) {
                maxv = gelu;
            }
        }
        // Second pass: exponentiate after max-shift and accumulate sum
        float sum_exp = 0.0f;
        for (uint32_t c = 0; c < cols; ++c) {
            float e = expf(y_ptr[base + c] - maxv);
            y_ptr[base + c] = e;
            sum_exp += e;
        }
        // Third pass: normalize to get softmax
        float inv_sum = 1.0f / sum_exp;
        for (uint32_t c = 0; c < cols; ++c) {
            y_ptr[base + c] *= inv_sum;
        }
    }
}
