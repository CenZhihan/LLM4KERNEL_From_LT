namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);
    if (x_shape == nullptr || w_shape == nullptr) {
        return GRAPH_FAILED;
    }
    gert::Shape* y_shape = context->GetOutputShape(0);
    std::vector<int64_t> out;
    if (x_shape->GetDimNum() == 2) {
        out = {x_shape->GetDim(0), w_shape->GetDim(0)};
    } else {
        // Fallback: keep rank, replace last dim
        out.resize(x_shape->GetDimNum());
        for (size_t i = 0; i + 1 < out.size(); ++i) out[i] = x_shape->GetDim(i);
        out.back() = w_shape->GetDim(0);
    }
    y_shape->SetDim(out);
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "gemm_group_norm_hardtanh_custom_tiling_data.h"

using DTYPE = float;
constexpr float EPS = 1e-5f;

// Simple fused kernel: GEMM (X * W^T + b) -> GroupNorm(gamma, beta) -> HardTanh(min, max).
extern "C" __global__ __aicore__ void gemm_group_norm_hardtanh_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR gamma, GM_ADDR beta,
    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {

    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N = tiling_data.batch;
    const uint32_t inF = tiling_data.inFeatures;
    const uint32_t outF = tiling_data.outFeatures;
    const uint32_t groups = tiling_data.numGroups;
    const float minVal = tiling_data.minVal;
    const float maxVal = tiling_data.maxVal;

    const uint32_t blkNum = AscendC::GetBlockNum();
    const uint32_t blkIdx = AscendC::GetBlockIdx();

    const uint32_t rowsPerBlock = (N + blkNum - 1) / blkNum;
    const uint32_t rowStart = blkIdx * rowsPerBlock;
    const uint32_t rowEnd = (rowStart + rowsPerBlock > N) ? N : (rowStart + rowsPerBlock);

    __gm__ DTYPE* x_gm = reinterpret_cast<__gm__ DTYPE*>(x);
    __gm__ DTYPE* w_gm = reinterpret_cast<__gm__ DTYPE*>(weight);
    __gm__ DTYPE* b_gm = reinterpret_cast<__gm__ DTYPE*>(bias);
    __gm__ DTYPE* g_gm = reinterpret_cast<__gm__ DTYPE*>(gamma);
    __gm__ DTYPE* be_gm = reinterpret_cast<__gm__ DTYPE*>(beta);
    __gm__ DTYPE* y_gm = reinterpret_cast<__gm__ DTYPE*>(y);

    // Local buffers for one row (illustrative)
    // NOTE: For demonstration only; real implementation should tile to UB capacity.
    // Max reasonable feature size assumed by the model.
    for (uint32_t r = rowStart; r < rowEnd; ++r) {
        // GEMM: out_row[c] = dot(x_row, W[c, :]) + b[c]
        // Compute per output channel
        for (uint32_t c = 0; c < outF; ++c) {
            float acc = static_cast<float>(b_gm[c]);
            __gm__ DTYPE* w_row = w_gm + static_cast<uint64_t>(c) * inF;
            __gm__ DTYPE* x_row = x_gm + static_cast<uint64_t>(r) * inF;
            // Dot product
            for (uint32_t k = 0; k < inF; ++k) {
                acc += static_cast<float>(x_row[k]) * static_cast<float>(w_row[k]);
            }
            y_gm[static_cast<uint64_t>(r) * outF + c] = static_cast<DTYPE>(acc);
        }

        // GroupNorm + HardTanh
        const uint32_t groupSize = outF / (groups == 0 ? 1 : groups);
        for (uint32_t gidx = 0; gidx < groups; ++gidx) {
            const uint32_t start = gidx * groupSize;
            const uint32_t end = (gidx + 1 == groups) ? outF : (start + groupSize);

            // mean
            float mean = 0.f;
            for (uint32_t c = start; c < end; ++c) {
                mean += static_cast<float>(y_gm[static_cast<uint64_t>(r) * outF + c]);
            }
            uint32_t cnt = end - start;
            if (cnt == 0) continue;
            mean /= static_cast<float>(cnt);

            // variance
            float var = 0.f;
            for (uint32_t c = start; c < end; ++c) {
                float v = static_cast<float>(y_gm[static_cast<uint64_t>(r) * outF + c]) - mean;
                var += v * v;
            }
            var /= static_cast<float>(cnt);
            float invStd = AscendC::Rsqrt(var + EPS);

            // normalize, affine, clamp
            for (uint32_t c = start; c < end; ++c) {
                float v = static_cast<float>(y_gm[static_cast<uint64_t>(r) * outF + c]);
                float n = (v - mean) * invStd;
                float aff = n * static_cast<float>(g_gm[c]) + static_cast<float>(be_gm[c]);
                // HardTanh
                if (aff < minVal) aff = minVal;
                if (aff > maxVal) aff = maxVal;
                y_gm[static_cast<uint64_t>(r) * outF + c] = static_cast<DTYPE>(aff);
            }
        }
    }
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_features, out_features, num_groups, hardtanh_min, hardtanh_max):
        super().__init__()
        self.gemm = torch.nn.Linear(in_features, out_features, bias=True)
        self.group_norm = torch.nn.GroupNorm(num_groups, out_features, affine=True)
        self.hardtanh_min = float(hardtanh_min)
        self.hardtanh_max = float(hardtanh_max)

    def forward(self, x):
        return custom_ops_lib.gemm_group_norm_hardtanh_custom(
            x,
            self.gemm.weight,
            self.gemm.bias,
            self.group_norm.weight,
            self.group_norm.bias,
            self.group_norm.num_groups,
            self.hardtanh_min,
            self.hardtanh_max,
        )
