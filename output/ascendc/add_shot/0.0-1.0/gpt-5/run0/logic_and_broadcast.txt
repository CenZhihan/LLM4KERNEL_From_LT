#include "logic_and_broadcast_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    LogicAndBroadcastCustomTilingData tiling;

    // Assume 3D inputs as in the model: [N, C, W]
    auto x_shape = context->GetInputShape(0)->GetOriginShape();
    auto y_shape = context->GetInputShape(1)->GetOriginShape();

    uint32_t outN = 1, outC = 1, outW = 1;
    if (x_shape.GetDimNum() == 3) {
        outN = static_cast<uint32_t>(x_shape.GetDim(0));
        outC = static_cast<uint32_t>(x_shape.GetDim(1));
        outW = static_cast<uint32_t>(x_shape.GetDim(2));
    } else {
        // Fallback to flat
        uint64_t sz = x_shape.GetShapeSize();
        outN = 1; outC = 1; outW = static_cast<uint32_t>(sz);
    }

    uint32_t bN = 1, bC = 1, bW = 1;
    if (y_shape.GetDimNum() == 3) {
        bN = static_cast<uint32_t>(y_shape.GetDim(0));
        bC = static_cast<uint32_t>(y_shape.GetDim(1));
        bW = static_cast<uint32_t>(y_shape.GetDim(2));
    } else {
        uint64_t sz = y_shape.GetShapeSize();
        bN = 1; bC = 1; bW = static_cast<uint32_t>(sz);
    }

    uint32_t totalLength = static_cast<uint32_t>(outN * outC * outW);

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_totalLength(totalLength);
    tiling.set_tileNum(TILE_NUM);
    tiling.set_outN(outN);
    tiling.set_outC(outC);
    tiling.set_outW(outW);
    tiling.set_bN(bN);
    tiling.set_bC(bC);
    tiling.set_bW(bW);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {

    // Output is the broadcasted shape of x and y; here we assume x already equals the broadcasted shape

    gert::Shape* out_shape = context->GetOutputShape(0);
    *out_shape = *x_shape;
    return GRAPH_SUCCESS;

    return ge::GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "logic_and_broadcast_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

// Helper: safe min
__aicore__ inline uint32_t umin_u32(uint32_t a, uint32_t b) { return a < b ? a : b; }

class KernelLogicAndBroadcast {
public:
    __aicore__ inline KernelLogicAndBroadcast() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR z,
                                uint32_t totalLength, uint32_t tileNum,
                                uint32_t outN, uint32_t outC, uint32_t outW,
                                uint32_t bN, uint32_t bC, uint32_t bW)
    {
        this->totalLength = totalLength;
        this->blockLength = (totalLength + AscendC::GetBlockNum() - 1) / AscendC::GetBlockNum();
        // clamp last block
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t start = blockIdx * this->blockLength;
        uint32_t remain = (start + this->blockLength <= totalLength) ? this->blockLength : (totalLength - start);
        this->blockLength = remain;

        this->tileNum = tileNum;
        this->outN = outN; this->outC = outC; this->outW = outW;
        this->bN = bN; this->bC = bC; this->bW = bW;

        // Choose tileLength so it doesn't cross too many rows; at most outW
        uint32_t baseTile = this->blockLength / (tileNum == 0 ? 1 : tileNum) / BUFFER_NUM;
        baseTile = baseTile == 0 ? umin_u32(this->blockLength, outW) : baseTile;
        this->tileLength = umin_u32(baseTile, outW);

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + start, this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, outN * outC * outW); // y may be broadcast; keep full base
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z + start, this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
        // For broadcast, we optionally prepare a temp Y local buffer per tile
        pipe.InitBuffer(inQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        if (this->blockLength == 0) { return; }
        uint32_t loopCount = (this->blockLength + this->tileLength - 1) / this->tileLength;
        for (uint32_t i = 0; i < loopCount; i++) {
            uint32_t curLen = umin_u32(this->tileLength, this->blockLength - i * this->tileLength);
            CopyIn(i, curLen);
            Compute(i, curLen);
            CopyOut(i, curLen);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t progress, uint32_t curLen)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = inQueueY.AllocTensor<DTYPE_Y>();

        // Copy a slice of x
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], curLen);

        // Prepare broadcasted slice of y
        // Compute global linear start index for this tile
        uint32_t globalStart = globalBlockStart() + progress * this->tileLength;
        // For each element in tile, compute corresponding y index under broadcasting and fetch scalar
        // Note: For performance, real kernels vectorize this. Here we keep it straightforward.
        for (uint32_t t = 0; t < curLen; ++t) {
            uint32_t gid = globalStart + t;
            // map gid -> (n, c, w)
            uint32_t w = gid % outW;
            uint32_t tmp = gid / outW;
            uint32_t c = tmp % outC;
            uint32_t n = tmp / outC;

            uint32_t bn = (bN == 1) ? 0 : n;
            uint32_t bc = (bC == 1) ? 0 : c;
            uint32_t bw = (bW == 1) ? 0 : w;
            uint32_t by = (bn * bC + bc) * bW + bw;

            // Copy one element from GM y to local yLocal[t]
            AscendC::DataCopy(yLocal[t], yGm[by], 1);
        }

        inQueueX.EnQue(xLocal);
        inQueueY.EnQue(yLocal);
    }

    __aicore__ inline void Compute(uint32_t /*progress*/, uint32_t curLen)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = inQueueY.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();

        // Elementwise logical AND
        // AscendC provides basic vector ops; use And for boolean tensors
        AscendC::And(zLocal, xLocal, yLocal, curLen);

        outQueueZ.EnQue<DTYPE_Z>(zLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueY.FreeTensor(yLocal);
    }

    __aicore__ inline void CopyOut(uint32_t progress, uint32_t curLen)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[progress * this->tileLength], zLocal, curLen);
        outQueueZ.FreeTensor(zLocal);
    }

    __aicore__ inline uint32_t globalBlockStart() const {
        return (AscendC::GetBlockIdx()) * ((totalLength + AscendC::GetBlockNum() - 1) / AscendC::GetBlockNum());
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueY;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t totalLength;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;

    uint32_t outN, outC, outW;
    uint32_t bN, bC, bW;
};

extern "C" __global__ __aicore__ void logic_and_broadcast_custom(GM_ADDR x, GM_ADDR y, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogicAndBroadcast op;
    op.Init(x, y, z,
            tiling_data.totalLength, tiling_data.tileNum,
            tiling_data.outN, tiling_data.outC, tiling_data.outW,
            tiling_data.bN, tiling_data.bC, tiling_data.bW);
    op.Process();
}
