kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_transpose3d_max_max_sum_custom_tiling_data.h"

// This kernel performs channel-reduction sum with keepdim=True on an input tensor of shape [N, C, D, H, W].
// It is intended to replace the final torch.sum(x, dim=1, keepdim=True) after the two MaxPool3d ops.

constexpr int32_t BUFFER_NUM = 2;

class KernelReduceCSum {
public:
    __aicore__ inline KernelReduceCSum() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w)
    {
        this->n = n;
        this->c = c;
        this->d = d;
        this->h = h;
        this->w = w;
        this->spatial = d * h * w;

        xGm.SetGlobalBuffer((__gm__ float*)x, (uint64_t)n * c * this->spatial);
        yGm.SetGlobalBuffer((__gm__ float*)y, (uint64_t)n * this->spatial);

        // Allocate double buffers for in/out locals of length spatial
        pipe.InitBuffer(inQueue, BUFFER_NUM, this->spatial * sizeof(float));
        pipe.InitBuffer(outQueue, BUFFER_NUM, this->spatial * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // Iterate over batch; each iteration reduces across C for a full spatial slice
        for (uint32_t ni = 0; ni < n; ++ni) {
            ComputeOne(ni);
        }
    }

private:
    __aicore__ inline void ComputeOne(uint32_t n_idx)
    {
        // Allocate accumulation buffer from outQueue and initialize with channel 0 slice
        AscendC::LocalTensor<float> acc = outQueue.AllocTensor<float>();
        uint64_t base_c0 = ((uint64_t)n_idx * c + 0u) * spatial;
        AscendC::DataCopy(acc, xGm[base_c0], spatial);

        // Accumulate remaining channels
        for (uint32_t ci = 1; ci < c; ++ci) {
            AscendC::LocalTensor<float> tmp = inQueue.AllocTensor<float>();
            uint64_t base_ci = ((uint64_t)n_idx * c + (uint64_t)ci) * spatial;
            AscendC::DataCopy(tmp, xGm[base_ci], spatial);
            AscendC::Add(acc, acc, tmp, spatial);
            inQueue.FreeTensor(tmp);
        }

        // Write out result to yGm at [n_idx, 0, :, :, :]
        uint64_t y_off = (uint64_t)n_idx * spatial;
        AscendC::DataCopy(yGm[y_off], acc, spatial);
        outQueue.FreeTensor(acc);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t n;
    uint32_t c;
    uint32_t d;
    uint32_t h;
    uint32_t w;
    uint32_t spatial;
};

extern "C" __global__ __aicore__ void conv_transpose3d_max_max_sum_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    uint32_t n = tiling_data.n;
    uint32_t c = tiling_data.c;
    uint32_t d = tiling_data.d;
    uint32_t h = tiling_data.h;
    uint32_t w = tiling_data.w;

    KernelReduceCSum op;
    op.Init(x, y, n, c, d, h, w);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: keeps ConvTranspose3d and MaxPool3d in PyTorch, replaces the final channel-reduction sum
    with a custom AscendC kernel conv_transpose3d_max_max_sum_custom for better performance.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding)
        self.max_pool1 = torch.nn.MaxPool3d(kernel_size=2)
        self.max_pool2 = torch.nn.MaxPool3d(kernel_size=3)

    def forward(self, x):
        x = self.conv_transpose(x)
        x = self.max_pool1(x)
        x = self.max_pool2(x)
        # Replace torch.sum(x, dim=1, keepdim=True) with custom op
        x = custom_ops_lib.conv_transpose3d_max_max_sum_custom(x)
        return x
