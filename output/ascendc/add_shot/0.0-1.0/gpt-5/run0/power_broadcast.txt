#include "power_broadcast_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
const uint32_t MAX_BLOCK_DIM = 8;
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    PowerBroadcastCustomTilingData tiling;
    uint32_t totalLength = context->GetInputShape(0)->GetOriginShape().GetShapeSize();
    uint32_t innerSize = context->GetInputShape(1)->GetOriginShape().GetShapeSize();

    // rows across the leading dimensions
    uint32_t rows = innerSize == 0 ? 0 : totalLength / innerSize;
    uint32_t usedBlockDim = rows == 0 ? 1 : (rows < MAX_BLOCK_DIM ? rows : MAX_BLOCK_DIM);
    context->SetBlockDim(usedBlockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_innerSize(innerSize);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "power_broadcast_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 1; // single-buffering per row

class KernelPowerBroadcast {
public:
    __aicore__ inline KernelPowerBroadcast() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR exponent, GM_ADDR y, uint32_t totalLength, uint32_t innerSize)
    {
        this->innerSize = innerSize;
        this->blockLength = totalLength / AscendC::GetBlockNum();
        // rows assigned to this block; we choose blockDim as min(rows, MAX_BLOCK_DIM) in host tiling
        this->rowsPerBlock = (innerSize == 0) ? 0 : (this->blockLength / innerSize);

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        eGm.SetGlobalBuffer((__gm__ DTYPE_Y *)exponent, this->innerSize);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->innerSize * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueE, BUFFER_NUM, this->innerSize * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->innerSize * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < this->rowsPerBlock; ++r) {
            CopyIn(r);
            Compute(r);
            CopyOut(r);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> eLocal = inQueueE.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(xLocal, xGm[rowIdx * this->innerSize], this->innerSize);
        AscendC::DataCopy(eLocal, eGm[0], this->innerSize);
        inQueueX.EnQue(xLocal);
        inQueueE.EnQue(eLocal);
    }

    __aicore__ inline void Compute(uint32_t /*rowIdx*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> eLocal = inQueueE.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();
        AscendC::Pow(yLocal, xLocal, eLocal, this->innerSize);
        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueE.FreeTensor(eLocal);
    }

    __aicore__ inline void CopyOut(uint32_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[rowIdx * this->innerSize], yLocal, this->innerSize);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueE;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> eGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;
    uint32_t innerSize;
    uint32_t blockLength;
    uint32_t rowsPerBlock;
};

extern "C" __global__ __aicore__ void power_broadcast_custom(GM_ADDR x, GM_ADDR exponent, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelPowerBroadcast op;
    op.Init(x, exponent, y, tiling_data.totalLength, tiling_data.innerSize);
    op.Process();
}
"""
