kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_standard_2d_asymmetric_input_asymmetric_kernel_custom_tiling_data.h"

class KernelConv2D {
public:
    __aicore__ inline KernelConv2D() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t c_in, uint32_t h_in, uint32_t w_in,
                                uint32_t c_out, uint32_t k_h, uint32_t k_w,
                                uint32_t stride_h, uint32_t stride_w,
                                uint32_t pad_h, uint32_t pad_w,
                                uint32_t dilation_h, uint32_t dilation_w,
                                uint32_t h_out, uint32_t w_out)
    {
        x_gm = reinterpret_cast<__gm__ float*>(x);
        w_gm = reinterpret_cast<__gm__ float*>(w);
        y_gm = reinterpret_cast<__gm__ float*>(y);

        this->n = n; this->ci = c_in; this->hi = h_in; this->wi = w_in;
        this->co = c_out; this->kh = k_h; this->kw = k_w;
        this->sh = stride_h; this->sw = stride_w;
        this->ph = pad_h; this->pw = pad_w;
        this->dh = dilation_h; this->dw = dilation_w;
        this->ho = h_out; this->wo = w_out;

        totalOut = n * c_out * h_out * w_out;

        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t bid = AscendC::GetBlockIdx();

        base = (totalOut / blocks) * bid + (bid < (totalOut % blocks) ? bid : (totalOut % blocks));
        length = (totalOut / blocks) + (bid < (totalOut % blocks) ? 1 : 0);
    }

    __aicore__ inline void Process()
    {
        for (uint32_t idx = 0; idx < length; ++idx) {
            uint32_t outIndex = base + idx;

            uint32_t ow = outIndex % wo;
            uint32_t tmp = outIndex / wo;
            uint32_t oh = tmp % ho;
            tmp /= ho;
            uint32_t coo = tmp % co;
            uint32_t nn = tmp / co;

            float sum = 0.f;
            for (uint32_t c = 0; c < ci; ++c) {
                for (uint32_t r = 0; r < kh; ++r) {
                    int32_t ih = static_cast<int32_t>(oh) * static_cast<int32_t>(sh) - static_cast<int32_t>(ph) + static_cast<int32_t>(r) * static_cast<int32_t>(dh);
                    if (ih < 0 || ih >= static_cast<int32_t>(hi)) continue;
                    for (uint32_t s = 0; s < kw; ++s) {
                        int32_t iw = static_cast<int32_t>(ow) * static_cast<int32_t>(sw) - static_cast<int32_t>(pw) + static_cast<int32_t>(s) * static_cast<int32_t>(dw);
                        if (iw < 0 || iw >= static_cast<int32_t>(wi)) continue;

                        uint64_t x_index = (((uint64_t)nn * ci + c) * hi + (uint32_t)ih) * wi + (uint32_t)iw;
                        uint64_t w_index = (((uint64_t)coo * ci + c) * kh + r) * kw + s;
                        sum += x_gm[x_index] * w_gm[w_index];
                    }
                }
            }

            uint64_t y_index = (((uint64_t)nn * co + coo) * ho + oh) * wo + ow;
            y_gm[y_index] = sum;
        }
    }

private:
    __gm__ float* x_gm;
    __gm__ float* w_gm;
    __gm__ float* y_gm;

    uint32_t n, ci, hi, wi;
    uint32_t co, kh, kw;
    uint32_t sh, sw, ph, pw, dh, dw;
    uint32_t ho, wo;

    uint32_t totalOut;
    uint32_t base;
    uint32_t length;
};

extern "C" __global__ __aicore__ void conv_standard_2d_asymmetric_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2D op;
    op.Init(x, w, y,
            tiling_data.n, tiling_data.c_in, tiling_data.h_in, tiling_data.w_in,
            tiling_data.c_out, tiling_data.k_h, tiling_data.k_w,
            tiling_data.stride_h, tiling_data.stride_w,
            tiling_data.pad_h, tiling_data.pad_w,
            tiling_data.dilation_h, tiling_data.dilation_w,
            tiling_data.h_out, tiling_data.w_out);
    op.Process();
}

    auto C_out = w.size(0);

    auto W = x.size(3);

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model that replaces nn.Conv2d with a custom AscendC kernel-backed op.
    Supports asymmetric input and kernel sizes. Currently groups==1 and bias=False.
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: tuple,
                 stride: tuple = (1, 1), padding: tuple = (0, 0),
                 dilation: tuple = (1, 1), groups: int = 1, bias: bool = False) -> None:
        super().__init__()
        assert groups == 1, "Custom kernel currently supports groups == 1"
        assert not bias, "Custom kernel currently does not support bias"
        if isinstance(kernel_size, int):
            kh, kw = kernel_size, kernel_size
        else:
            kh, kw = kernel_size
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = (kh, kw)
        self.stride = stride
        self.padding = padding
        self.dilation = dilation
        self.groups = groups

        weight_shape = (out_channels, in_channels, kh, kw)
        self.weight = torch.nn.Parameter(torch.empty(weight_shape))
        torch.nn.init.kaiming_uniform_(self.weight, a=5 ** 0.5)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        sh, sw = self.stride
        ph, pw = self.padding
        dh, dw = self.dilation
        return custom_ops_lib.conv_standard_2d_asymmetric_input_asymmetric_kernel_custom(
            x, self.weight, int(sh), int(sw), int(ph), int(pw), int(dh), int(dw), int(self.groups)
        )
