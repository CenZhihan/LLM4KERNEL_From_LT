kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "matmul_sum_max_avg_pool_log_sum_exp_log_sum_exp_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelDotReduce {
public:
    __aicore__ inline KernelDotReduce() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR wsum, GM_ADDR bsum, GM_ADDR z,
                                uint32_t batch, uint32_t inFeat, uint32_t tileNum)
    {
        this->batch = batch;
        this->inFeat = inFeat;
        this->tileNum = tileNum;
        this->tileLen = (inFeat + tileNum * BUFFER_NUM - 1) / (tileNum * BUFFER_NUM);
        if (this->tileLen == 0) this->tileLen = inFeat;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        this->rowsPerBlock = (batch + blockNum - 1) / blockNum;
        this->rowStart = blockIdx * this->rowsPerBlock;
        uint32_t rowEnd = AscendC::Min(batch, this->rowStart + this->rowsPerBlock);
        this->myRows = rowEnd > this->rowStart ? (rowEnd - this->rowStart) : 0;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->rowStart * inFeat, (uint32_t)this->myRows * inFeat);
        wsumGm.SetGlobalBuffer((__gm__ DTYPE_X *)wsum, inFeat);
        bsumGm.SetGlobalBuffer((__gm__ DTYPE_X *)bsum, 1);
        zGm.SetGlobalBuffer((__gm__ DTYPE_X *)z + this->rowStart, this->myRows);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLen * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueW, BUFFER_NUM, this->tileLen * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, 1 * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        if (this->myRows == 0) {
            return;
        }
        // Load bsum once
        AscendC::LocalTensor<DTYPE_X> bLocal = outQueueZ.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(bLocal, bsumGm[0], 1);
        outQueueZ.EnQue(bLocal);

        for (uint32_t r = 0; r < this->myRows; ++r) {
            ComputeRow(r);
        }
    }

private:
    __aicore__ inline void ComputeRow(uint32_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE_X> accLocal = outQueueZ.AllocTensor<DTYPE_X>();
        AscendC::Dup<DTYPE_X>(accLocal, 0.0f, 1); // initialize accumulator to 0

        uint32_t processed = 0;
        while (processed < inFeat) {
            uint32_t len = AscendC::Min(this->tileLen, inFeat - processed);

            AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_X> wLocal = inQueueW.AllocTensor<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_X> mulLocal = inQueueX.AllocTensor<DTYPE_X>(); // reuse buffer size

            AscendC::DataCopy(xLocal, xGm[rowIdx * inFeat + processed], len);
            AscendC::DataCopy(wLocal, wsumGm[processed], len);

            AscendC::Mul(mulLocal, xLocal, wLocal, len);

            AscendC::LocalTensor<DTYPE_X> partSum = inQueueW.AllocTensor<DTYPE_X>(); // scalar holder
            AscendC::ReduceSum(partSum, mulLocal, len);
            AscendC::Add(accLocal, accLocal, partSum, 1);

            inQueueX.FreeTensor(xLocal);
            inQueueW.FreeTensor(wLocal);
            inQueueX.FreeTensor(mulLocal);
            inQueueW.FreeTensor(partSum);

            processed += len;
        }

        AscendC::LocalTensor<DTYPE_X> bLocal = outQueueZ.DeQue<DTYPE_X>(); // get cached bsum
        AscendC::Add(accLocal, accLocal, bLocal, 1);

        AscendC::DataCopy(zGm[rowIdx], accLocal, 1);

        outQueueZ.EnQue(bLocal); // put bsum back
        outQueueZ.FreeTensor(accLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_X> wsumGm;
    AscendC::GlobalTensor<DTYPE_X> bsumGm;
    AscendC::GlobalTensor<DTYPE_X> zGm;

    uint32_t batch;
    uint32_t inFeat;
    uint32_t tileNum;
    uint32_t tileLen;

    uint32_t rowsPerBlock;
    uint32_t rowStart;
    uint32_t myRows;
};

extern "C" __global__ __aicore__ void matmul_sum_max_avg_pool_log_sum_exp_log_sum_exp_custom(GM_ADDR x, GM_ADDR wsum, GM_ADDR bsum, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelDotReduce op;
    op.Init(x, wsum, bsum, z, tiling_data.batch, tiling_data.inFeatures, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused operator replaces:
      Linear (matmul+bias) -> sum(dim=1, keepdim=True) -> max -> mean -> logsumexp -> logsumexp
    which algebraically reduces to: z = x @ sum(W, dim=0) + sum(bias)
    """
    def __init__(self, in_features, out_features):
        super().__init__()
        self.linear = torch.nn.Linear(in_features, out_features)

        # Precompute collapsed parameters: wsum (in_features,) and bsum (scalar)
        # Register as buffers so they move with .to(device)
        with torch.no_grad():
            wsum = self.linear.weight.sum(dim=0)            # (in_features,)
            bsum = self.linear.bias.sum().reshape(1)        # (1,)
        self.register_buffer("wsum", wsum)
        self.register_buffer("bsum", bsum)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Custom fused AscendC op: returns shape (batch, 1)
        return custom_ops_lib.matmul_sum_max_avg_pool_log_sum_exp_log_sum_exp_custom(x, self.wsum, self.bsum)
