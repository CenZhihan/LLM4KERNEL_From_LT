kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_transposed_2d_square_input_asymmetric_kernel_custom_tiling_data.h"

typedef float DTYPE_X;
typedef float DTYPE_W;
typedef float DTYPE_Y;

class KernelConvTransposed2d {
public:
    __aicore__ inline KernelConvTransposed2d() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t h, uint32_t ww,
                                uint32_t kh, uint32_t kw,
                                uint32_t hout, uint32_t wout,
                                uint32_t stride_h, uint32_t stride_w,
                                uint32_t pad_h, uint32_t pad_w,
                                uint32_t out_pad_h, uint32_t out_pad_w)
    {
        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, static_cast<uint32_t>(n * cin * h * ww));
        wGm.SetGlobalBuffer((__gm__ DTYPE_W*)w, static_cast<uint32_t>(cin * cout * kh * kw));
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y, static_cast<uint32_t>(n * cout * hout * wout));

        this->N = n; this->Cin = cin; this->Cout = cout;
        this->H = h; this->W = ww; this->Kh = kh; this->Kw = kw;
        this->Ho = hout; this->Wo = wout;
        this->Sh = stride_h; this->Sw = stride_w;
        this->Ph = pad_h; this->Pw = pad_w;
        this->OPh = out_pad_h; this->OPw = out_pad_w;

        // UB queues for scalar ops
        pipe.InitBuffer(inQueueX, 1, sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueW, 1, sizeof(DTYPE_W));
        pipe.InitBuffer(inoutQueueY, 1, sizeof(DTYPE_Y));
        pipe.InitBuffer(tmpQueue, 1, sizeof(DTYPE_Y));
    }

    __aicore__ inline void ZeroOutput()
    {
        // Initialize y to zeros
        uint64_t total = (uint64_t)N * (uint64_t)Cout * (uint64_t)Ho * (uint64_t)Wo;
        for (uint64_t idx = 0; idx < total; ++idx) {
            AscendC::LocalTensor<DTYPE_Y> yLocal = inoutQueueY.AllocTensor<DTYPE_Y>();
            AscendC::Duplicate(yLocal, static_cast<DTYPE_Y>(0.0f), 1);
            AscendC::DataCopy(yGm[idx], yLocal, 1);
            inoutQueueY.FreeTensor(yLocal);
        }
    }

    __aicore__ inline void Process()
    {
        // Naive deconvolution (stride/padding supported via tiling params)
        // y[n, oc, oh, ow] += x[n, ic, ih, iw] * w[ic, oc, kh, kw]
        // where oh = ih * Sh - Ph + kh; ow = iw * Sw - Pw + kw
        ZeroOutput();

        for (uint32_t n = 0; n < N; ++n) {
            for (uint32_t ic = 0; ic < Cin; ++ic) {
                for (uint32_t ih = 0; ih < H; ++ih) {
                    for (uint32_t iw = 0; iw < W; ++iw) {
                        uint64_t xoff = (((uint64_t)n * Cin + ic) * H + ih) * W + iw;
                        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                        AscendC::DataCopy(xLocal, xGm[xoff], 1);

                        for (uint32_t oc = 0; oc < Cout; ++oc) {
                            for (uint32_t kh = 0; kh < Kh; ++kh) {
                                int32_t oh = static_cast<int32_t>(ih) * static_cast<int32_t>(Sh) - static_cast<int32_t>(Ph) + static_cast<int32_t>(kh);
                                if (oh < 0 || oh >= static_cast<int32_t>(Ho)) continue;
                                for (uint32_t kw = 0; kw < Kw; ++kw) {
                                    int32_t ow = static_cast<int32_t>(iw) * static_cast<int32_t>(Sw) - static_cast<int32_t>(Pw) + static_cast<int32_t>(kw);
                                    if (ow < 0 || ow >= static_cast<int32_t>(Wo)) continue;

                                    uint64_t woff = (((uint64_t)ic * Cout + oc) * Kh + kh) * Kw + kw;
                                    AscendC::LocalTensor<DTYPE_W> wLocal = inQueueW.AllocTensor<DTYPE_W>();
                                    AscendC::DataCopy(wLocal, wGm[woff], 1);

                                    uint64_t yoff = (((uint64_t)n * Cout + oc) * Ho + (uint32_t)oh) * Wo + (uint32_t)ow;
                                    AscendC::LocalTensor<DTYPE_Y> yLocal = inoutQueueY.AllocTensor<DTYPE_Y>();
                                    AscendC::LocalTensor<DTYPE_Y> tmpLocal = tmpQueue.AllocTensor<DTYPE_Y>();

                                    // y = y + x * w
                                    AscendC::DataCopy(yLocal, yGm[yoff], 1);
                                    AscendC::Mul(tmpLocal, xLocal, wLocal, 1);
                                    AscendC::Add(yLocal, yLocal, tmpLocal, 1);
                                    AscendC::DataCopy(yGm[yoff], yLocal, 1);

                                    inQueueW.FreeTensor(wLocal);
                                    inoutQueueY.FreeTensor(yLocal);
                                    tmpQueue.FreeTensor(tmpLocal);
                                }
                            }
                        }
                        inQueueX.FreeTensor(xLocal);
                    }
                }
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueW;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> inoutQueueY;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> tmpQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t N, Cin, Cout, H, W, Kh, Kw, Ho, Wo;
    uint32_t Sh, Sw, Ph, Pw, OPh, OPw;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_square_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed2d op;
    op.Init(x, w, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.h, tiling_data.w,
            tiling_data.kh, tiling_data.kw,
            tiling_data.hout, tiling_data.wout,
            tiling_data.stride_h, tiling_data.stride_w,
            tiling_data.pad_h, tiling_data.pad_w,
            tiling_data.out_pad_h, tiling_data.out_pad_w);
    op.Process();
}
