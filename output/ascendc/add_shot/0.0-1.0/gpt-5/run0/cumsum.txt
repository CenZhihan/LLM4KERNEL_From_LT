#include "cumsum_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    CumsumCustomTilingData tiling;

    const auto* in_shape_ptr = context->GetInputShape(0);
    const auto& origin = in_shape_ptr->GetOriginShape();
    uint32_t totalLength = origin.GetShapeSize();
    uint32_t rows = origin.GetDimNum() > 0 ? static_cast<uint32_t>(origin.GetDim(0)) : 1;
    if (rows == 0) rows = 1;
    uint32_t innerLength = totalLength / rows;

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_rows(rows);
    tiling.set_innerLength(innerLength);
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {

{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)

    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "cumsum_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // number of tensors per queue

class KernelCumsum {
public:
    __aicore__ inline KernelCumsum() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR z, uint32_t rows, uint32_t innerLength, uint32_t tileNum)
    {
        this->rows = rows;
        this->innerLength = innerLength;
        this->tileNum = tileNum;

        // Partition the inner (contiguous) dimension across blocks
        this->blockInnerLength = this->innerLength / AscendC::GetBlockNum();
        this->baseOffset = this->blockInnerLength * AscendC::GetBlockIdx();

        // tile length per progress
        this->tileLength = this->blockInnerLength / tileNum / BUFFER_NUM;

        // Set whole buffers; we will index using row stride and offsets
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, this->rows * this->innerLength);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z, this->rows * this->innerLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueuePrev, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t p = 0; p < loopCount; ++p) {
            int32_t offset = this->baseOffset + p * this->tileLength;

            // Row 0: y[0, :] = x[0, :]
            {
                AscendC::LocalTensor<DTYPE_X> xLocal0 = inQueueX.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(xLocal0, xGm[0 * this->innerLength + offset], this->tileLength);
                inQueueX.EnQue(xLocal0);

                AscendC::LocalTensor<DTYPE_X> xLocal0c = inQueueX.DeQue<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_Z> zLocal0 = outQueueZ.AllocTensor<DTYPE_Z>();
                AscendC::DataCopy(zLocal0, xLocal0c, this->tileLength);
                outQueueZ.EnQue(zLocal0);
                inQueueX.FreeTensor(xLocal0c);

                AscendC::LocalTensor<DTYPE_Z> zLocal0c = outQueueZ.DeQue<DTYPE_Z>();
                AscendC::DataCopy(zGm[0 * this->innerLength + offset], zLocal0c, this->tileLength);
                outQueueZ.FreeTensor(zLocal0c);
            }

            // Rows 1..rows-1: y[r, :] = x[r, :] + y[r-1, :]
            for (uint32_t r = 1; r < this->rows; ++r) {
                AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(xLocal, xGm[r * this->innerLength + offset], this->tileLength);
                inQueueX.EnQue(xLocal);

                AscendC::LocalTensor<DTYPE_Z> prevLocal = inQueuePrev.AllocTensor<DTYPE_Z>();
                AscendC::DataCopy(prevLocal, zGm[(r - 1) * this->innerLength + offset], this->tileLength);
                inQueuePrev.EnQue(prevLocal);

                AscendC::LocalTensor<DTYPE_X> xComp = inQueueX.DeQue<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_Z> prevComp = inQueuePrev.DeQue<DTYPE_Z>();
                AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();

                AscendC::Add(zLocal, xComp, prevComp, this->tileLength);

                outQueueZ.EnQue(zLocal);
                inQueueX.FreeTensor(xComp);
                inQueuePrev.FreeTensor(prevComp);

                AscendC::LocalTensor<DTYPE_Z> zLocalOut = outQueueZ.DeQue<DTYPE_Z>();
                AscendC::DataCopy(zGm[r * this->innerLength + offset], zLocalOut, this->tileLength);
                outQueueZ.FreeTensor(zLocalOut);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueuePrev;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t rows;
    uint32_t innerLength;
    uint32_t tileNum;
    uint32_t blockInnerLength;
    uint32_t baseOffset;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void cumsum_custom(GM_ADDR x, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumsum op;
    op.Init(x, z, tiling_data.rows, tiling_data.innerLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, dim):
        super().__init__()
        self.dim = dim

    def forward(self, x):
        # Our kernel computes cumsum along dim=0. For arbitrary dim, permute.
        dim = self.dim
        dim = dim if dim >= 0 else dim + x.dim()
        if dim != 0:
            order = [dim] + [i for i in range(x.dim()) if i != dim]
            x_perm = x.permute(order).contiguous()
            y_perm = custom_ops_lib.cumsum_custom(x_perm)
            # invert permutation
            inv_order = [0] * len(order)
            for i, j in enumerate(order):
                inv_order[j] = i
            y = y_perm.permute(inv_order)
        else:
            y = custom_ops_lib.cumsum_custom(x)
        return y
'''
