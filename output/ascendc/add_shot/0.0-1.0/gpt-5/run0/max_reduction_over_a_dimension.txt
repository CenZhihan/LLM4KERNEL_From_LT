namespace {
inline uint32_t GetLastDimSize(const gert::Shape& shape) {
    uint32_t dim_num = shape.GetDimNum();
    if (dim_num == 0) return 1;
    return static_cast<uint32_t>(shape.GetDim(dim_num - 1));
}
inline uint32_t GetTotalSize(const gert::Shape& shape) {
    return static_cast<uint32_t>(shape.GetShapeSize());
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    MaxReductionOverADimensionCustomTilingData tiling;

    const gert::Shape* in_shape = context->GetInputShape(0);
    auto origin = in_shape->GetOriginShape();

    uint32_t total = GetTotalSize(origin);
    uint32_t last_dim = GetLastDimSize(origin);
    uint32_t outer = (last_dim == 0) ? 0 : (total / last_dim);

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_outerNum(outer);
    tiling.set_reduceLen(last_dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);

    auto x_origin = x_shape->GetOriginShape();
    uint32_t dim_num = x_origin.GetDimNum();
    if (dim_num == 0) {
        // scalar -> reduce to scalar
        *y_shape = x_origin;
        return GRAPH_SUCCESS;
    }
    // output dims = input dims minus the last one (since kernel reduces last dim)
    std::vector<int64_t> dims;
    dims.reserve(dim_num > 0 ? dim_num - 1 : 0);
    for (uint32_t i = 0; i + 1 < dim_num; ++i) {
        dims.push_back(x_origin.GetDim(i));
    }
    gert::Shape out;
    out.SetDimNum(static_cast<uint32_t>(dims.size()));
    for (uint32_t i = 0; i < dims.size(); ++i) {
        out.SetDim(i, dims[i]);
    }
    *y_shape = out;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "max_reduction_over_a_dimension_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMaxReduceLastDim {
public:
    __aicore__ inline KernelMaxReduceLastDim() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t outerNum, uint32_t reduceLen)
    {
        this->outerNum = outerNum;
        this->reduceLen = reduceLen;

        // block-wise row partition
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t baseRows = outerNum / blockNum;
        uint32_t extra = outerNum % blockNum;

        this->rowsThisBlock = baseRows + (blockIdx < extra ? 1 : 0);
        this->rowStart = baseRows * blockIdx + (blockIdx < extra ? blockIdx : extra);

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + static_cast<uint64_t>(rowStart) * reduceLen, static_cast<uint64_t>(rowsThisBlock) * reduceLen);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + rowStart, rowsThisBlock);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->reduceLen * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, 1 * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < rowsThisBlock; ++r) {
            CopyIn(r);
            Compute(r);
            CopyOut(r);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[static_cast<uint64_t>(rowIdx) * reduceLen], reduceLen);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t /*rowIdx*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();
        // Reduce over last dimension (entire row)
        AscendC::ReduceMax(yLocal, xLocal, reduceLen);
        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[rowIdx], yLocal, 1);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t outerNum;
    uint32_t reduceLen;
    uint32_t rowsThisBlock;
    uint32_t rowStart;
};

extern "C" __global__ __aicore__ void max_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxReduceLastDim op;
    op.Init(x, y, tiling_data.outerNum, tiling_data.reduceLen);
    op.Process();
}
"""

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

static inline std::vector<int64_t> compute_output_shape_except_last(const at::Tensor& t) {
    auto sizes = t.sizes();
    std::vector<int64_t> out;
    out.reserve(sizes.size() > 0 ? sizes.size() - 1 : 0);
    for (int i = 0; i + 1 < sizes.size(); ++i) out.push_back(sizes[i]);
    return out;
}

at::Tensor max_reduction_over_a_dimension_custom_impl_npu(const at::Tensor& self, int64_t dim) {
    int64_t ndim = self.dim();
    int64_t d = dim < 0 ? dim + ndim : dim;
    TORCH_CHECK(d >= 0 && d < ndim, "Invalid dim");

    at::Tensor x_perm = self;
    if (d != ndim - 1) {
        std::vector<int64_t> perm;
        perm.reserve(ndim);
        for (int64_t i = 0; i < ndim; ++i) if (i != d) perm.push_back(i);
        perm.push_back(d);
        x_perm = self.permute(perm).contiguous();
    } else {
        x_perm = self.contiguous();
    }

    // Flatten to 2D [N, K], reduce over K (last dim)
    int64_t K = x_perm.size(-1);
    at::Tensor x2d = x_perm.reshape({-1, K});
    at::Tensor y1d = at::empty({x2d.size(0)}, x2d.options());

    EXEC_NPU_CMD(aclnnMaxReductionOverADimensionCustom, x2d, y1d);

    // Reshape back to original order without the reduced dim
    auto out_shape = compute_output_shape_except_last(x_perm);
    at::Tensor y = y1d.reshape(out_shape);

    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("max_reduction_over_a_dimension_custom", &max_reduction_over_a_dimension_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("max_reduction_over_a_dimension_custom", &max_reduction_over_a_dimension_custom_impl_npu, "max reduction over given dim");
}
