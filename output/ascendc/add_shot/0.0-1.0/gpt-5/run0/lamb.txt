    return GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "lamb_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelLamb {
public:
    __aicore__ inline KernelLamb() {}
    __aicore__ inline void Init(GM_ADDR m, GM_ADDR v, GM_ADDR e, GM_ADDR r,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        mGm.SetGlobalBuffer((__gm__ float *)m + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        vGm.SetGlobalBuffer((__gm__ float *)v + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        eGm.SetGlobalBuffer((__gm__ float *)e + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        rGm.SetGlobalBuffer((__gm__ float *)r + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        pipe.InitBuffer(inQueueM, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueV, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueE, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(outQueueR, BUFFER_NUM, this->tileLength * sizeof(float));
    }
    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<float> mLocal = inQueueM.AllocTensor<float>();
        AscendC::LocalTensor<float> vLocal = inQueueV.AllocTensor<float>();
        AscendC::LocalTensor<float> eLocal = inQueueE.AllocTensor<float>();
        AscendC::DataCopy(mLocal, mGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(vLocal, vGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(eLocal, eGm[progress * this->tileLength], this->tileLength);
        inQueueM.EnQue(mLocal);
        inQueueV.EnQue(vLocal);
        inQueueE.EnQue(eLocal);
    }
    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<float> mLocal = inQueueM.DeQue<float>();
        AscendC::LocalTensor<float> vLocal = inQueueV.DeQue<float>();
        AscendC::LocalTensor<float> eLocal = inQueueE.DeQue<float>();
        AscendC::LocalTensor<float> rLocal = outQueueR.AllocTensor<float>();

        // tmp buffers
        AscendC::LocalTensor<float> sqrtV = pipe.AllocTmpBuffer<float>(this->tileLength);
        AscendC::LocalTensor<float> denom = pipe.AllocTmpBuffer<float>(this->tileLength);

        AscendC::Sqrt(sqrtV, vLocal, this->tileLength);                 // sqrt(v)
        AscendC::Add(denom, sqrtV, eLocal, this->tileLength);           // sqrt(v) + eps (eps provided as tensor e)
        AscendC::Div(rLocal, mLocal, denom, this->tileLength);          // r = m / (sqrt(v) + eps)

        outQueueR.EnQue<float>(rLocal);
        inQueueM.FreeTensor(mLocal);
        inQueueV.FreeTensor(vLocal);
        inQueueE.FreeTensor(eLocal);
        pipe.FreeTmpBuffer(sqrtV);
        pipe.FreeTmpBuffer(denom);
    }
    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<float> rLocal = outQueueR.DeQue<float>();
        AscendC::DataCopy(rGm[progress * this->tileLength], rLocal, this->tileLength);
        outQueueR.FreeTensor(rLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueM, inQueueV, inQueueE;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueR;
    AscendC::GlobalTensor<float> mGm;
    AscendC::GlobalTensor<float> vGm;
    AscendC::GlobalTensor<float> eGm;
    AscendC::GlobalTensor<float> rGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void lamb_custom(GM_ADDR m, GM_ADDR v, GM_ADDR e, GM_ADDR r, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLamb op;
    op.Init(m, v, e, r, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, lr=1e-3, eps=1e-6):
        super().__init__()
        self.lr = lr
        self.eps = eps

    def forward(self, param, m, v):
        # Use custom kernel to compute r = m / (sqrt(v) + eps)
        eps_tensor = torch.full_like(param, self.eps)
        r = custom_ops_lib.lamb_custom(m, v, eps_tensor)

        # Compute trust ratio with PyTorch reductions
        trust_ratio = param.norm(p=2) / (r.norm(p=2) + self.eps)

        # Update parameter
        param = param - self.lr * trust_ratio * r
        return param
'''
