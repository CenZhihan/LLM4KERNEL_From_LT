kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv2d_avg_pool_sigmoid_sum_custom_tiling_data.h"

// This kernel computes per-batch sum(sigmoid(x)) where x is the input tensor after avg-pooling.
// Each block processes a strided subset of batches to avoid atomic ops.

constexpr int32_t BUFFER_NUM = 2; // tiles per queue

class KernelSigmoidSum {
public:
    __aicore__ inline KernelSigmoidSum() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR z, uint32_t batch, uint32_t innerSize, uint32_t tileNum)
    {
        this->batch = batch;
        this->innerSize = innerSize;
        this->tileNum = tileNum;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint32_t)batch * innerSize);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z, (uint32_t)batch);

        // Tile length per progress (double-buffering)
        this->tileLength = (innerSize / tileNum) / BUFFER_NUM;
        if (this->tileLength == 0) {
            this->tileLength = innerSize; // fallback if inner very small
        }

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(redQueue, BUFFER_NUM, sizeof(DTYPE_X)); // for reduced partial sum
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        for (uint32_t n = blockIdx; n < batch; n += blockNum) {
            // local accumulator for this batch index
            DTYPE_X batchSum = (DTYPE_X)0.0f;

            int32_t loopCount = CalcLoopCount(innerSize);
            for (int32_t i = 0; i < loopCount; i++) {
                CopyIn(n, i);
                batchSum = ComputeAndAccumulate(batchSum);
            }
            // write result for this batch
            zGm.SetValue(n, batchSum);
        }
    }

private:
    __aicore__ inline int32_t CalcLoopCount(uint32_t len) const
    {
        uint32_t perLoop = tileNum * BUFFER_NUM;
        if (perLoop == 0) return 1;
        uint32_t blocks = (len + (tileLength - 1)) / tileLength;
        // align to perLoop rounds
        uint32_t rounds = (blocks + perLoop - 1) / perLoop;
        return (int32_t)(rounds * perLoop);
    }

    __aicore__ inline void CopyIn(uint32_t n, int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        uint32_t offsetBlocks = (uint32_t)progress * tileLength;
        if (offsetBlocks >= innerSize) {
            // zero-pad if out of range
            AscendC::SetVector<DTYPE_X>(xLocal, (DTYPE_X)0.0f, tileLength);
        } else {
            uint32_t valid = AscendC::Min(tileLength, innerSize - offsetBlocks);
            AscendC::DataCopy(xLocal, xGm[n * innerSize + offsetBlocks], valid);
            if (valid < tileLength) {
                // tail zero pad
                AscendC::SetVector<DTYPE_X>(xLocal[valid], (DTYPE_X)0.0f, tileLength - valid);
            }
        }
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline DTYPE_X ComputeAndAccumulate(DTYPE_X acc)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> yLocal = tmpQueue.AllocTensor<DTYPE_X>();
        // y = sigmoid(x) = 1 / (1 + exp(-x))
        AscendC::Neg(yLocal, xLocal, tileLength);
        AscendC::Exp(yLocal, yLocal, tileLength);
        // y = y + 1
        AscendC::Adds(yLocal, yLocal, (DTYPE_X)1.0f, tileLength);
        // y = 1 / y
        AscendC::Reciprocal(yLocal, yLocal, tileLength);

        // Reduce sum over tile
        AscendC::LocalTensor<DTYPE_X> rLocal = redQueue.AllocTensor<DTYPE_X>();
        AscendC::ReduceSum(rLocal, yLocal, tileLength);
        DTYPE_X partial = rLocal.GetValue(0);

        // release queues
        tmpQueue.FreeTensor(yLocal);
        inQueueX.FreeTensor(xLocal);
        redQueue.FreeTensor(rLocal);
        return acc + partial;
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> tmpQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> redQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t batch;
    uint32_t innerSize;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv2d_avg_pool_sigmoid_sum_custom(GM_ADDR x, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSigmoidSum op;
    op.Init(x, z, tiling_data.batch, tiling_data.innerSize, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: conv2d + avg_pool remain as built-ins,
    final sigmoid+sum over CHW fused into a custom AscendC op.
    """
    def __init__(self, in_channels, out_channels, kernel_size, pool_kernel_size):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)
        self.avg_pool = torch.nn.AvgPool2d(pool_kernel_size)

    def forward(self, x):
        x = self.conv(x)
        x = self.avg_pool(x)
        # Apply fused sigmoid + reduction over [C,H,W] per batch on NPU
        x = custom_ops_lib.conv2d_avg_pool_sigmoid_sum_custom(x)
        return x
