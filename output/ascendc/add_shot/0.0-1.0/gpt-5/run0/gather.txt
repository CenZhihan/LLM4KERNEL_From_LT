host_operator_src="""
#include "gather_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
inline uint32_t GetDimSafe(const gert::Shape& s, size_t idx) {
    if (idx < s.GetDimNum()) {
        return static_cast<uint32_t>(s.GetDim(idx));
    }
    return 1;
}
}

namespace optiling {
const uint32_t DEFAULT_BLOCK_DIM = 8;
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    GatherCustomTilingData tiling;

    auto x_origin = context->GetInputShape(0)->GetOriginShape();
    auto idx_origin = context->GetInputShape(1)->GetOriginShape();

    uint32_t nRows = GetDimSafe(x_origin, 0);
    uint32_t inCols = GetDimSafe(x_origin, 1);
    uint32_t outCols = GetDimSafe(idx_origin, 1);

    uint32_t block_dim = DEFAULT_BLOCK_DIM;
    if (nRows < block_dim) {
        block_dim = nRows == 0 ? 1 : nRows;
    }
    context->SetBlockDim(block_dim);

    tiling.set_nRows(nRows);
    tiling.set_inCols(inCols);
    tiling.set_outCols(outCols);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* idx_shape = context->GetInputShape(1);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *idx_shape; // output shape matches index shape
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype); // output dtype same as x
    return ge::GRAPH_SUCCESS;
}
}

namespace ops {
class GatherCustom : public OpDef {
public:
    explicit GatherCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("idx")
            .ParamType(REQUIRED)
            .DataType({ge::DT_INT32})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("z")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore()
            .SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(GatherCustom);
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "gather_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelGather {
public:
    __aicore__ inline KernelGather() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR idx, GM_ADDR z, uint32_t nRows, uint32_t inCols, uint32_t outCols)
    {
        this->nRows = nRows;
        this->inCols = inCols;
        this->outCols = outCols;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint64_t)nRows * (uint64_t)inCols);
        idxGm.SetGlobalBuffer((__gm__ DTYPE_Y *)idx, (uint64_t)nRows * (uint64_t)outCols);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z, (uint64_t)nRows * (uint64_t)outCols);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->inCols * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueIdx, BUFFER_NUM, this->outCols * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->outCols * sizeof(DTYPE_Z));

        // compute row range for this block
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t bid = AscendC::GetBlockIdx();
        uint32_t base = nRows / blocks;
        uint32_t rem = nRows % blocks;
        this->startRow = base * bid + (bid < rem ? bid : rem);
        this->rowCount = base + (bid < rem ? 1 : 0);
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < rowCount; ++r) {
            uint32_t row = startRow + r;

            AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(xLocal, xGm[row * inCols], inCols);
            inQueueX.EnQue(xLocal);

            AscendC::LocalTensor<DTYPE_Y> idxLocal = inQueueIdx.AllocTensor<DTYPE_Y>();
            AscendC::DataCopy(idxLocal, idxGm[row * outCols], outCols);
            inQueueIdx.EnQue(idxLocal);

            AscendC::LocalTensor<DTYPE_X> xLocalDQ = inQueueX.DeQue<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_Y> idxLocalDQ = inQueueIdx.DeQue<DTYPE_Y>();
            AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();

            for (uint32_t i = 0; i < outCols; ++i) {
                int32_t col = (int32_t)idxLocalDQ.GetValue(i);
                // clamp to valid range just in case
                if (col < 0) col = 0;
                if ((uint32_t)col >= inCols) col = (int32_t)(inCols - 1);
                DTYPE_X val = xLocalDQ.GetValue((uint32_t)col);
                zLocal.SetValue(i, (DTYPE_Z)val);
            }

            outQueueZ.EnQue<DTYPE_Z>(zLocal);
            inQueueX.FreeTensor(xLocalDQ);
            inQueueIdx.FreeTensor(idxLocalDQ);

            AscendC::LocalTensor<DTYPE_Z> zLocalDQ = outQueueZ.DeQue<DTYPE_Z>();
            AscendC::DataCopy(zGm[row * outCols], zLocalDQ, outCols);
            outQueueZ.FreeTensor(zLocalDQ);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueIdx;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> idxGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t nRows;
    uint32_t inCols;
    uint32_t outCols;
    uint32_t startRow;
    uint32_t rowCount;
};

extern "C" __global__ __aicore__ void gather_custom(GM_ADDR x, GM_ADDR idx, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGather op;
    op.Init(x, idx, z, tiling_data.nRows, tiling_data.inCols, tiling_data.outCols);
    op.Process();
}

at::Tensor gather_custom_impl_npu(const at::Tensor& self, const at::Tensor& index) {
    at::Tensor index_i32 = index.dtype() == at::kInt ? index : index.to(at::kInt);
    at::Tensor result = at::empty({self.size(0), index.size(1)}, self.options());
    EXEC_NPU_CMD(aclnnGatherCustom, self, index_i32, result);
    return result;
}

    m.impl("gather_custom", &gather_custom_impl_npu);
}
