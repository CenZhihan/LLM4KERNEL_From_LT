kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "square_matrix_multiplication_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2;

__aicore__ inline uint32_t ceil_div_u32(uint32_t a, uint32_t b) {
    return (a + b - 1) / b;
}

class KernelMatMul {
public:
    __aicore__ inline KernelMatMul() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t n, uint32_t tileRows, uint32_t tileCols, uint32_t tileK)
    {
        this->n = n;
        this->tileRows = tileRows;
        this->tileCols = tileCols;
        this->tileK = tileK;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // Distribute rows across blocks as evenly as possible
        uint32_t baseRows = n / blockNum;
        uint32_t rem = n % blockNum;
        this->rowStart = blockIdx * baseRows + (blockIdx < rem ? blockIdx : rem);
        this->myRows = baseRows + (blockIdx < rem ? 1 : 0);

        aGm.SetGlobalBuffer((__gm__ DTYPE_A *)A + this->rowStart * n, this->myRows * n);
        bGm.SetGlobalBuffer((__gm__ DTYPE_B *)B, n * n);
        cGm.SetGlobalBuffer((__gm__ DTYPE_C *)C + this->rowStart * n, this->myRows * n);

        pipe.InitBuffer(inQueueA, BUFFER_NUM, this->tileRows * this->tileK * sizeof(DTYPE_A));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->tileK * this->tileCols * sizeof(DTYPE_B));
        pipe.InitBuffer(outQueueC, BUFFER_NUM, this->tileRows * this->tileCols * sizeof(DTYPE_C));
    }

    __aicore__ inline void Process()
    {
        if (myRows == 0) {
            return;
        }
        uint32_t rowTileCnt = ceil_div_u32(myRows, tileRows);
        uint32_t colTileCnt = ceil_div_u32(n, tileCols);
        uint32_t kTileCnt   = ceil_div_u32(n, tileK);

        for (uint32_t rt = 0; rt < rowTileCnt; ++rt) {
            uint32_t rOff = rt * tileRows;
            uint32_t rSize = AscendC::min(tileRows, myRows - rOff);

            for (uint32_t ct = 0; ct < colTileCnt; ++ct) {
                uint32_t cOff = ct * tileCols;
                uint32_t cSize = AscendC::min(tileCols, n - cOff);

                // Allocate and zero-initialize C local tile
                AscendC::LocalTensor<DTYPE_C> cLocal = outQueueC.AllocTensor<DTYPE_C>();
                uint32_t cLocalSize = tileRows * tileCols;
                for (uint32_t idx = 0; idx < cLocalSize; ++idx) {
                    cLocal[idx] = static_cast<DTYPE_C>(0.0f);
                }

                // Accumulate over K tiles
                for (uint32_t kt = 0; kt < kTileCnt; ++kt) {
                    uint32_t kOff = kt * tileK;
                    uint32_t kSize = AscendC::min(tileK, n - kOff);

                    // Load A tile: [rSize, kSize]
                    AscendC::LocalTensor<DTYPE_A> aLocal = inQueueA.AllocTensor<DTYPE_A>();
                    for (uint32_t r = 0; r < rSize; ++r) {
                        AscendC::DataCopy(aLocal[r * tileK], aGm[(rOff + r) * n + kOff], kSize);
                    }
                    inQueueA.EnQue(aLocal);

                    // Load B tile: [kSize, cSize]
                    AscendC::LocalTensor<DTYPE_B> bLocal = inQueueB.AllocTensor<DTYPE_B>();
                    for (uint32_t k = 0; k < kSize; ++k) {
                        AscendC::DataCopy(bLocal[k * tileCols], bGm[(kOff + k) * n + cOff], cSize);
                    }
                    inQueueB.EnQue(bLocal);

                    // Compute
                    AscendC::LocalTensor<DTYPE_A> aTile = inQueueA.DeQue<DTYPE_A>();
                    AscendC::LocalTensor<DTYPE_B> bTile = inQueueB.DeQue<DTYPE_B>();

                    for (uint32_t r = 0; r < rSize; ++r) {
                        for (uint32_t c = 0; c < cSize; ++c) {
                            DTYPE_C sum = static_cast<DTYPE_C>(0.0f);
                            for (uint32_t k = 0; k < kSize; ++k) {
                                DTYPE_A aval = aTile[r * tileK + k];
                                DTYPE_B bval = bTile[k * tileCols + c];
                                sum += static_cast<DTYPE_C>(aval * bval);
                            }
                            cLocal[r * tileCols + c] += sum;
                        }
                    }

                    inQueueA.FreeTensor(aTile);
                    inQueueB.FreeTensor(bTile);
                }

                // Enqueue result and copy out to GM: C tile [rSize, cSize]
                outQueueC.EnQue<DTYPE_C>(cLocal);

                AscendC::LocalTensor<DTYPE_C> cTile = outQueueC.DeQue<DTYPE_C>();
                for (uint32_t r = 0; r < rSize; ++r) {
                    AscendC::DataCopy(cGm[(rOff + r) * n + cOff], cTile[r * tileCols], cSize);
                }
                outQueueC.FreeTensor(cTile);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA, inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC;

    AscendC::GlobalTensor<DTYPE_A> aGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_C> cGm;

    uint32_t n;
    uint32_t tileRows;
    uint32_t tileCols;
    uint32_t tileK;

    uint32_t rowStart;
    uint32_t myRows;
};

extern "C" __global__ __aicore__ void square_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatMul op;
    op.Init(A, B, C, tiling_data.n, tiling_data.tileRows, tiling_data.tileCols, tiling_data.tileK);
    op.Process();
}
