kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_transposed_2d_asymmetric_input_asymmetric_kernel_padded_custom_tiling_data.h"

// Note: This kernel provides a straightforward reference implementation for
// 2D transposed convolution on AscendC. It focuses on correctness and clarity.
// For production, further tiling, vectorization and fusion should be applied.

class KernelConvTransposed2D {
public:
    __aicore__ inline KernelConvTransposed2D() {}

    __aicore__ inline void Init(
        GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
        uint32_t n, uint32_t c_in, uint32_t h, uint32_t w_in,
        uint32_t c_out, uint32_t k_h, uint32_t k_w,
        uint32_t h_out, uint32_t w_out,
        uint32_t stride_h, uint32_t stride_w,
        uint32_t pad_h, uint32_t pad_w)
    {
        this->N = n;
        this->Cin = c_in;
        this->Hin = h;
        this->Win = w_in;
        this->Cout = c_out;
        this->Kh = k_h;
        this->Kw = k_w;
        this->Hout = h_out;
        this->Wout = w_out;
        this->Sh = stride_h;
        this->Sw = stride_w;
        this->Ph = pad_h;
        this->Pw = pad_w;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint64_t)N * Cin * Hin * Win);
        wGm.SetGlobalBuffer((__gm__ DTYPE_Y *)w, (uint64_t)Cin * Cout * Kh * Kw);
        bGm.SetGlobalBuffer((__gm__ DTYPE_Z *)b, (uint64_t)Cout);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y, (uint64_t)N * Cout * Hout * Wout);

        // Simple block partition over output elements
        totalOut = (uint64_t)N * Cout * Hout * Wout;
        blkNum = AscendC::GetBlockNum();
        blkIdx = AscendC::GetBlockIdx();
        startIdx = (totalOut * blkIdx) / blkNum;
        endIdx = (totalOut * (blkIdx + 1)) / blkNum;
    }

    __aicore__ inline void Process()
    {
        // Initialize output slice to 0
        // We will zero the output region for this block before accumulation
        for (uint64_t idx = startIdx; idx < endIdx; ++idx) {
            // zero y[idx]
            AscendC::LocalTensor<DTYPE_Z> zLocal;
            pipe.InitBuffer(zLocal, 1 * sizeof(DTYPE_Z));
            AscendC::Dup(zLocal, (DTYPE_Z)0, 1);
            AscendC::DataCopy(yGm[idx], zLocal, 1);
        }

        // Accumulate contributions
        for (uint64_t outIndex = startIdx; outIndex < endIdx; ++outIndex) {
            uint32_t wo = outIndex % Wout;
            uint64_t tmp = outIndex / Wout;
            uint32_t ho = tmp % Hout;
            tmp = tmp / Hout;
            uint32_t co = tmp % Cout;
            uint32_t n  = tmp / Cout;

            // Accumulator
            AscendC::LocalTensor<DTYPE_Z> acc;
            pipe.InitBuffer(acc, 1 * sizeof(DTYPE_Z));
            AscendC::Dup(acc, (DTYPE_Z)0, 1);

            // bias if exists (we cannot detect optional in kernel easily, assume zero if empty)
            // We'll try to add bias unconditionally; if bias ptr is invalid, host should pass zeros tensor.
            AscendC::LocalTensor<DTYPE_Z> biasLocal;
            pipe.InitBuffer(biasLocal, 1 * sizeof(DTYPE_Z));
            AscendC::DataCopy(biasLocal, bGm[co], 1);
            AscendC::Add(acc, acc, biasLocal, 1);

            // Iterate over kernel and input channels
            for (uint32_t ci = 0; ci < Cin; ++ci) {
                for (uint32_t r = 0; r < Kh; ++r) {
                    int32_t in_h_numer = (int32_t)ho + (int32_t)Ph - (int32_t)r;
                    if (in_h_numer < 0) continue;
                    if ((uint32_t)in_h_numer > (uint32_t)((Hin - 1) * Sh)) continue;
                    if (in_h_numer % (int32_t)Sh != 0) continue;
                    uint32_t hi = (uint32_t)(in_h_numer / (int32_t)Sh);
                    if (hi >= Hin) continue;

                    for (uint32_t s = 0; s < Kw; ++s) {
                        int32_t in_w_numer = (int32_t)wo + (int32_t)Pw - (int32_t)s;
                        if (in_w_numer < 0) continue;
                        if ((uint32_t)in_w_numer > (uint32_t)((Win - 1) * Sw)) continue;
                        if (in_w_numer % (int32_t)Sw != 0) continue;
                        uint32_t wi = (uint32_t)(in_w_numer / (int32_t)Sw);
                        if (wi >= Win) continue;

                        uint64_t xIdx = (((uint64_t)n * Cin + ci) * Hin + hi) * Win + wi;
                        uint64_t wIdx = (((uint64_t)ci * Cout + co) * Kh + r) * Kw + s;

                        AscendC::LocalTensor<DTYPE_X> xLocal;
                        pipe.InitBuffer(xLocal, 1 * sizeof(DTYPE_X));
                        AscendC::DataCopy(xLocal, xGm[xIdx], 1);

                        AscendC::LocalTensor<DTYPE_Y> wLocal;
                        pipe.InitBuffer(wLocal, 1 * sizeof(DTYPE_Y));
                        AscendC::DataCopy(wLocal, wGm[wIdx], 1);

                        AscendC::LocalTensor<DTYPE_Z> mulLocal;
                        pipe.InitBuffer(mulLocal, 1 * sizeof(DTYPE_Z));
                        AscendC::Mul(mulLocal, xLocal, wLocal, 1);
                        AscendC::Add(acc, acc, mulLocal, 1);
                    }
                }
            }

            // write back
            AscendC::DataCopy(yGm[outIndex], acc, 1);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> wGm;
    AscendC::GlobalTensor<DTYPE_Z> bGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t N, Cin, Hin, Win, Cout, Kh, Kw, Hout, Wout, Sh, Sw, Ph, Pw;
    uint64_t totalOut, startIdx, endIdx;
    uint32_t blkNum, blkIdx;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_padded_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(t, tiling);
    KernelConvTransposed2D op;
    op.Init(
        x, w, bias, y,
        t.n, t.c_in, t.h, t.w,
        t.c_out, t.k_h, t.k_w,
        t.h_out, t.w_out,
        t.stride_h, t.stride_w,
        t.pad_h, t.pad_w
    );
    op.Process();
}

    int64_t Wo = (W - 1) * stride_w - 2 * pad_w + kW;
