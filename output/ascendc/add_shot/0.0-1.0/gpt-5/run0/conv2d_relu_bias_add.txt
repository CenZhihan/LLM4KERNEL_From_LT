kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv2d_relu_bias_add_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConv2dReluBiasAdd {
public:
    __aicore__ inline KernelConv2dReluBiasAdd() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR cb, GM_ADDR ab, GM_ADDR z,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t h, uint32_t w_in, uint32_t kh, uint32_t kw,
                                uint32_t oh, uint32_t ow,
                                uint32_t stride_h, uint32_t stride_w,
                                uint32_t pad_h, uint32_t pad_w,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->n = n; this->cin = cin; this->cout = cout;
        this->h = h; this->w_in = w_in; this->kh = kh; this->kw = kw;
        this->oh = oh; this->ow = ow;
        this->stride_h = stride_h; this->stride_w = stride_w;
        this->pad_h = pad_h; this->pad_w = pad_w;

        this->totalLength = totalLength;
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        // Inputs are read-only and shared by all blocks; no offset for them.
        xPtr = (__gm__ DTYPE_X*)x;
        wPtr = (__gm__ DTYPE_WEIGHT*)w;
        cbPtr = (__gm__ DTYPE_CONV_BIAS*)cb;
        abPtr = (__gm__ DTYPE_ADD_BIAS*)ab;

        // Output is partitioned by blocks.
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void Compute(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();

        uint32_t blockBase = AscendC::GetBlockIdx() * this->blockLength;
        uint32_t outBase = blockBase + progress * this->tileLength;

        for (uint32_t t = 0; t < this->tileLength; ++t) {
            uint32_t outIndex = outBase + t;
            if (outIndex >= this->totalLength) {
                zLocal.SetValue(t, (DTYPE_Z)0);
                continue;
            }

            // unravel index: [n, cout, oh, ow]
            uint32_t n_stride = this->cout * this->oh * this->ow;
            uint32_t oc_stride = this->oh * this->ow;

            uint32_t n_idx  = outIndex / n_stride;
            uint32_t rem1   = outIndex % n_stride;
            uint32_t oc_idx = rem1 / oc_stride;
            uint32_t rem2   = rem1 % oc_stride;
            uint32_t oh_idx = rem2 / this->ow;
            uint32_t ow_idx = rem2 % this->ow;

            float acc = 0.0f;
            // Convolution accumulate
            for (uint32_t ic = 0; ic < this->cin; ++ic) {
                for (uint32_t kh_i = 0; kh_i < this->kh; ++kh_i) {
                    int32_t ih = (int32_t)oh_idx * (int32_t)this->stride_h - (int32_t)this->pad_h + (int32_t)kh_i;
                    if (ih < 0 || ih >= (int32_t)this->h) continue;
                    for (uint32_t kw_i = 0; kw_i < this->kw; ++kw_i) {
                        int32_t iw = (int32_t)ow_idx * (int32_t)this->stride_w - (int32_t)this->pad_w + (int32_t)kw_i;
                        if (iw < 0 || iw >= (int32_t)this->w_in) continue;

                        uint64_t xIndex = (((uint64_t)n_idx * this->cin + ic) * this->h + (uint32_t)ih) * this->w_in + (uint32_t)iw;
                        uint64_t wIndex = ((((uint64_t)oc_idx * this->cin + ic) * this->kh + kh_i) * this->kw + kw_i);

                        float xval = (float)xPtr[xIndex];
                        float wval = (float)wPtr[wIndex];
                        acc += xval * wval;
                    }
                }
            }

            // Add conv bias
            acc += (float)cbPtr[oc_idx];

            // ReLU
            acc = acc > 0.0f ? acc : 0.0f;

            // Add extra bias (broadcast per channel)
            acc += (float)abPtr[oc_idx];

            zLocal.SetValue(t, (DTYPE_Z)acc);
        }

        outQueueZ.EnQue<DTYPE_Z>(zLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[progress * this->tileLength], zLocal, this->tileLength);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    __gm__ DTYPE_X* xPtr;
    __gm__ DTYPE_WEIGHT* wPtr;
    __gm__ DTYPE_CONV_BIAS* cbPtr;
    __gm__ DTYPE_ADD_BIAS* abPtr;

    uint32_t n, cin, cout, h, w_in, kh, kw, oh, ow;
    uint32_t stride_h, stride_w, pad_h, pad_w;

    uint32_t totalLength;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv2d_relu_bias_add_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias, GM_ADDR add_bias, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dReluBiasAdd op;
    op.Init(x, weight, conv_bias, add_bias, z,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.h, tiling_data.w,
            tiling_data.kh, tiling_data.kw,
            tiling_data.oh, tiling_data.ow,
            tiling_data.stride_h, tiling_data.stride_w,
            tiling_data.pad_h, tiling_data.pad_w,
            tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused Conv2d + ReLU + BiasAdd using custom AscendC kernel.
    Assumes stride=1 and padding=0 (same as default in nn.Conv2d when not specified).
    """
    def __init__(self, in_channels, out_channels, kernel_size, bias_shape):
        super(ModelNew, self).__init__()
        # Keep a Conv2d module only to manage weights/bias with PyTorch semantics
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)
        # Extra bias added after ReLU, shape (out_channels, 1, 1)
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))

    def forward(self, x):
        # Call the fused custom op with tensors required:
        # x, conv weights, conv bias, extra bias
        return custom_ops_lib.conv2d_relu_bias_add_custom(x, self.conv.weight, self.conv.bias, self.bias)
