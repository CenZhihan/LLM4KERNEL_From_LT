#include "conv2d_mish_mish_custom_tiling.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static inline int64_t GetDim(const gert::Shape* s, size_t idx, int64_t default_val = 1) {
    if (s == nullptr || s->GetStorageShape().GetDimNum() <= idx) return default_val;
    return s->GetStorageShape().GetDim(idx);
}

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    Conv2dMishMishCustomTilingData tiling;

    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);
    // defaults: stride=1, pad=0, dilation=1, groups=1
    uint32_t stride_h = 1, stride_w = 1;
    uint32_t pad_h = 0, pad_w = 0;
    uint32_t dilation_h = 1, dilation_w = 1;
    uint32_t groups = 1;

    uint32_t n     = static_cast<uint32_t>(GetDim(x_shape, 0, 1));
    uint32_t c_in  = static_cast<uint32_t>(GetDim(x_shape, 1, 1));
    uint32_t h_in  = static_cast<uint32_t>(GetDim(x_shape, 2, 1));
    uint32_t w_in  = static_cast<uint32_t>(GetDim(x_shape, 3, 1));

    uint32_t c_out = static_cast<uint32_t>(GetDim(w_shape, 0, 1));
    uint32_t k_h   = static_cast<uint32_t>(GetDim(w_shape, 2, 1));
    uint32_t k_w   = static_cast<uint32_t>(GetDim(w_shape, 3, 1));

    // Output shape (PyTorch conv2d formula)
    auto out_size = [&](uint32_t in, uint32_t pad, uint32_t dil, uint32_t k, uint32_t str) -> uint32_t {
        int64_t num = static_cast<int64_t>(in) + 2LL * pad - static_cast<int64_t>(dil) * (static_cast<int64_t>(k) - 1) - 1;
        return static_cast<uint32_t>(num / str + 1);
    };
    uint32_t h_out = out_size(h_in, pad_h, dilation_h, k_h, stride_h);
    uint32_t w_out = out_size(w_in, pad_w, dilation_w, k_w, stride_w);

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_n(n);
    tiling.set_c_in(c_in);
    tiling.set_c_out(c_out);
    tiling.set_h_in(h_in);
    tiling.set_w_in(w_in);
    tiling.set_k_h(k_h);
    tiling.set_k_w(k_w);
    tiling.set_stride_h(stride_h);
    tiling.set_stride_w(stride_w);
    tiling.set_pad_h(pad_h);
    tiling.set_pad_w(pad_w);
    tiling.set_dilation_h(dilation_h);
    tiling.set_dilation_w(dilation_w);
    tiling.set_groups(groups);
    tiling.set_h_out(h_out);
    tiling.set_w_out(w_out);

    uint32_t totalLength = n * c_out * h_out * w_out;
    tiling.set_totalLength(totalLength);
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);
    if (x_shape == nullptr || w_shape == nullptr) {
        return GRAPH_FAILED;
    }

    // defaults
    int64_t stride_h = 1, stride_w = 1;
    int64_t pad_h = 0, pad_w = 0;
    int64_t dilation_h = 1, dilation_w = 1;

    int64_t n     = x_shape->GetStorageShape().GetDim(0);
    int64_t c_in  = x_shape->GetStorageShape().GetDim(1);
    int64_t h_in  = x_shape->GetStorageShape().GetDim(2);
    int64_t w_in  = x_shape->GetStorageShape().GetDim(3);

    int64_t c_out = w_shape->GetStorageShape().GetDim(0);
    int64_t k_h   = w_shape->GetStorageShape().GetDim(2);
    int64_t k_w   = w_shape->GetStorageShape().GetDim(3);

    auto out_size = [&](int64_t in, int64_t pad, int64_t dil, int64_t k, int64_t str) -> int64_t {
        return (in + 2 * pad - dil * (k - 1) - 1) / str + 1;
    };
    int64_t h_out = out_size(h_in, pad_h, dilation_h, k_h, stride_h);
    int64_t w_out = out_size(w_in, pad_w, dilation_w, k_w, stride_w);

    gert::Shape* y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(4);
    y_shape->SetDim(0, n);
    y_shape->SetDim(1, c_out);
    y_shape->SetDim(2, h_out);
    y_shape->SetDim(3, w_out);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv2d_mish_mish_custom_tiling_data.h"
#include <cmath>

__aicore__ inline float mish_scalar(float v) {
    // mish(x) = x * tanh(softplus(x)) ; softplus(x) = log(1 + exp(x))
    float sp = log1pf(expf(v));
    float t = tanhf(sp);
    return v * t;
}

extern "C" __global__ __aicore__ void conv2d_mish_mish_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N      = tiling_data.n;
    const uint32_t C_IN   = tiling_data.c_in;
    const uint32_t C_OUT  = tiling_data.c_out;
    const uint32_t H_IN   = tiling_data.h_in;
    const uint32_t W_IN   = tiling_data.w_in;
    const uint32_t K_H    = tiling_data.k_h;
    const uint32_t K_W    = tiling_data.k_w;
    const uint32_t STR_H  = tiling_data.stride_h;
    const uint32_t STR_W  = tiling_data.stride_w;
    const uint32_t PAD_H  = tiling_data.pad_h;
    const uint32_t PAD_W  = tiling_data.pad_w;
    const uint32_t DIL_H  = tiling_data.dilation_h;
    const uint32_t DIL_W  = tiling_data.dilation_w;
    const uint32_t H_OUT  = tiling_data.h_out;
    const uint32_t W_OUT  = tiling_data.w_out;

    const uint32_t total = tiling_data.totalLength;
    const uint32_t blocks = AscendC::GetBlockNum();
    const uint32_t bidx   = AscendC::GetBlockIdx();

    const uint32_t block_len = total / blocks;
    uint32_t start = bidx * block_len;
    uint32_t end   = start + block_len; // assume divisible

    __gm__ DTYPE_X* xGm = reinterpret_cast<__gm__ DTYPE_X*>(x);
    __gm__ DTYPE_W* wGm = reinterpret_cast<__gm__ DTYPE_W*>(w);
    __gm__ DTYPE_B* bGm = reinterpret_cast<__gm__ DTYPE_B*>(b);
    __gm__ DTYPE_Y* yGm = reinterpret_cast<__gm__ DTYPE_Y*>(y);

    for (uint32_t idx = start; idx < end; ++idx) {
        uint32_t tmp = idx;
        uint32_t ow = tmp % W_OUT; tmp /= W_OUT;
        uint32_t oh = tmp % H_OUT; tmp /= H_OUT;
        uint32_t oc = tmp % C_OUT; tmp /= C_OUT;
        uint32_t n  = tmp % N;

        float acc = static_cast<float>(bGm[oc]);

        for (uint32_t ic = 0; ic < C_IN; ++ic) {
            for (uint32_t kh = 0; kh < K_H; ++kh) {
                int32_t ih = static_cast<int32_t>(oh) * static_cast<int32_t>(STR_H) + static_cast<int32_t>(kh) * static_cast<int32_t>(DIL_H) - static_cast<int32_t>(PAD_H);
                for (uint32_t kw = 0; kw < K_W; ++kw) {
                    int32_t iw = static_cast<int32_t>(ow) * static_cast<int32_t>(STR_W) + static_cast<int32_t>(kw) * static_cast<int32_t>(DIL_W) - static_cast<int32_t>(PAD_W);
                    // Bounds check (for generic padding)
                    if (ih >= 0 && ih < static_cast<int32_t>(H_IN) && iw >= 0 && iw < static_cast<int32_t>(W_IN)) {
                        uint64_t xIndex = (((uint64_t)n * C_IN + ic) * H_IN + static_cast<uint32_t>(ih)) * W_IN + static_cast<uint32_t>(iw);
                        uint64_t wIndex = (((uint64_t)oc * C_IN + ic) * K_H + kh) * K_W + kw;
                        acc += static_cast<float>(xGm[xIndex]) * static_cast<float>(wGm[wIndex]);
                    }
                }
            }
        }

        // Apply Mish twice
        float m1 = mish_scalar(acc);
        float m2 = mish_scalar(m1);

        uint64_t yIndex = (((uint64_t)n * C_OUT + oc) * H_OUT + oh) * W_OUT + ow;
        yGm[yIndex] = static_cast<DTYPE_Y>(m2);
    }
}

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor conv2d_mish_mish_custom_impl_npu(const at::Tensor& input,
                                            const at::Tensor& weight,
                                            const c10::optional<at::Tensor>& bias_opt) {
    TORCH_CHECK(input.dim() == 4, \"input must be NCHW\");
    TORCH_CHECK(weight.dim() == 4, \"weight must be OIHW\");
    auto bias = bias_opt.has_value() ? bias_opt.value() : at::zeros({weight.size(0)}, weight.options());

    // defaults: stride=1, padding=0, dilation=1, groups=1
    auto N = input.size(0);
    auto C_in = input.size(1);
    auto H_in = input.size(2);
    auto W_in = input.size(3);

    auto C_out = weight.size(0);
    auto K_h = weight.size(2);
    auto K_w = weight.size(3);

    auto H_out = (H_in - (K_h - 1) - 1) + 1; // stride=1, pad=0, dil=1
    auto W_out = (W_in - (K_w - 1) - 1) + 1;

    at::Tensor result = at::empty({N, C_out, H_out, W_out}, input.options());
    EXEC_NPU_CMD(aclnnConv2dMishMishCustom, input, weight, bias, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"conv2d_mish_mish_custom\", &conv2d_mish_mish_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\"conv2d_mish_mish_custom\", &conv2d_mish_mish_custom_impl_npu, \"fused conv2d + mish + mish\");
}

import torch

import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused Conv2d + Mish + Mish using a custom AscendC kernel.

    """
    def __init__(self, in_channels, out_channels, kernel_size):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)

    def forward(self, x):
        # Use fused custom op: conv2d -> mish -> mish
        return custom_ops_lib.conv2d_mish_mish_custom(x, self.conv.weight, self.conv.bias)
