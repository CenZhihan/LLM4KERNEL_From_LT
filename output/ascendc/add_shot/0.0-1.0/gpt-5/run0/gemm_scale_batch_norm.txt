kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "gemm_scale_batch_norm_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2;     // pipeline buffers for X/Z
constexpr int32_t PARAM_BUFFER_NUM = 1; // buffers for parameter vectors

class KernelGemmScaleBatchNorm {
public:
    __aicore__ inline KernelGemmScaleBatchNorm() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR scale, GM_ADDR bn_weight, GM_ADDR bn_bias,
                                GM_ADDR running_mean, GM_ADDR running_var, GM_ADDR z,
                                uint32_t rows, uint32_t cols, uint32_t tileNum, float eps)
    {
        this->rows = rows;
        this->cols = cols;
        this->rowsPerBlock = tileNum;  // one tile per row
        this->tileNum = this->rowsPerBlock; // loop count per block
        this->tileLength = this->cols; // process one full row each iteration
        this->eps = eps;

        uint32_t blockStartRow = this->rowsPerBlock * AscendC::GetBlockIdx();

        xGm.SetGlobalBuffer((__gm__ float *)x + blockStartRow * this->cols, this->rowsPerBlock * this->cols);
        zGm.SetGlobalBuffer((__gm__ float *)z + blockStartRow * this->cols, this->rowsPerBlock * this->cols);

        scaleGm.SetGlobalBuffer((__gm__ float *)scale, this->cols);
        wGm.SetGlobalBuffer((__gm__ float *)bn_weight, this->cols);
        bGm.SetGlobalBuffer((__gm__ float *)bn_bias, this->cols);
        meanGm.SetGlobalBuffer((__gm__ float *)running_mean, this->cols);
        varGm.SetGlobalBuffer((__gm__ float *)running_var, this->cols);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(float));

        pipe.InitBuffer(qScale, PARAM_BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(qW, PARAM_BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(qB, PARAM_BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(qMean, PARAM_BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(qVar, PARAM_BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(qTmp1, PARAM_BUFFER_NUM, this->tileLength * sizeof(float));   // tmp for var+eps
        pipe.InitBuffer(qInvStd, PARAM_BUFFER_NUM, this->tileLength * sizeof(float)); // 1/sqrt
        pipe.InitBuffer(qGamma, PARAM_BUFFER_NUM, this->tileLength * sizeof(float));  // fused gamma
        pipe.InitBuffer(qBeta, PARAM_BUFFER_NUM, this->tileLength * sizeof(float));   // fused beta
        pipe.InitBuffer(qTmpRow, BUFFER_NUM, this->tileLength * sizeof(float));       // tmp per row
    }

    __aicore__ inline void Process()
    {
        for (uint32_t i = 0; i < this->tileNum; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t progress)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);

        // Load parameter vectors (shared across rows)
        AscendC::LocalTensor<float> sLocal = qScale.AllocTensor<float>();
        AscendC::LocalTensor<float> wLocal = qW.AllocTensor<float>();
        AscendC::LocalTensor<float> bLocal = qB.AllocTensor<float>();
        AscendC::LocalTensor<float> meanLocal = qMean.AllocTensor<float>();
        AscendC::LocalTensor<float> varLocal = qVar.AllocTensor<float>();
        AscendC::DataCopy(sLocal, scaleGm, this->tileLength);
        AscendC::DataCopy(wLocal, wGm, this->tileLength);
        AscendC::DataCopy(bLocal, bGm, this->tileLength);
        AscendC::DataCopy(meanLocal, meanGm, this->tileLength);
        AscendC::DataCopy(varLocal, varGm, this->tileLength);
        qScale.EnQue(sLocal);
        qW.EnQue(wLocal);
        qB.EnQue(bLocal);
        qMean.EnQue(meanLocal);
        qVar.EnQue(varLocal);
    }

    __aicore__ inline void Compute(uint32_t /*progress*/)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();

        AscendC::LocalTensor<float> sLocal = qScale.DeQue<float>();
        AscendC::LocalTensor<float> wLocal = qW.DeQue<float>();
        AscendC::LocalTensor<float> bLocal = qB.DeQue<float>();
        AscendC::LocalTensor<float> meanLocal = qMean.DeQue<float>();
        AscendC::LocalTensor<float> varLocal = qVar.DeQue<float>();

        AscendC::LocalTensor<float> tmp1 = qTmp1.AllocTensor<float>();
        AscendC::LocalTensor<float> invStd = qInvStd.AllocTensor<float>();
        AscendC::LocalTensor<float> gamma = qGamma.AllocTensor<float>();
        AscendC::LocalTensor<float> beta = qBeta.AllocTensor<float>();

        // tmp1 = var + eps
        AscendC::Add(tmp1, varLocal, this->eps, this->tileLength);
        // invStd = 1 / sqrt(tmp1)
        AscendC::Sqrt(invStd, tmp1, this->tileLength);
        AscendC::Reciprocal(invStd, invStd, this->tileLength);

        // gamma = scale * bn_weight * invStd
        AscendC::Mul(gamma, sLocal, wLocal, this->tileLength);
        AscendC::Mul(gamma, gamma, invStd, this->tileLength);

        // beta = bn_bias - gamma * mean
        AscendC::Mul(beta, gamma, meanLocal, this->tileLength);
        AscendC::Sub(beta, bLocal, beta, this->tileLength);

        // z = x * gamma + beta
        AscendC::LocalTensor<float> tmpRow = qTmpRow.AllocTensor<float>();
        AscendC::LocalTensor<float> zLocal = outQueueZ.AllocTensor<float>();
        AscendC::Mul(tmpRow, xLocal, gamma, this->tileLength);
        AscendC::Add(zLocal, tmpRow, beta, this->tileLength);

        outQueueZ.EnQue<float>(zLocal);

        // Free locals
        qTmpRow.FreeTensor(tmpRow);
        qBeta.FreeTensor(beta);
        qGamma.FreeTensor(gamma);
        qInvStd.FreeTensor(invStd);
        qTmp1.FreeTensor(tmp1);

        qVar.FreeTensor(varLocal);
        qMean.FreeTensor(meanLocal);
        qB.FreeTensor(bLocal);
        qW.FreeTensor(wLocal);
        qScale.FreeTensor(sLocal);

        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t progress)
    {
        AscendC::LocalTensor<float> zLocal = outQueueZ.DeQue<float>();
        AscendC::DataCopy(zGm[progress * this->tileLength], zLocal, this->tileLength);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;

    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::TQue<AscendC::TPosition::VECIN, PARAM_BUFFER_NUM> qScale, qW, qB, qMean, qVar;
    AscendC::TQue<AscendC::TPosition::VECIN, PARAM_BUFFER_NUM> qTmp1, qInvStd, qGamma, qBeta;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> qTmpRow;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> zGm;

    AscendC::GlobalTensor<float> scaleGm;
    AscendC::GlobalTensor<float> wGm;
    AscendC::GlobalTensor<float> bGm;
    AscendC::GlobalTensor<float> meanGm;
    AscendC::GlobalTensor<float> varGm;

    uint32_t rows;
    uint32_t cols;
    uint32_t rowsPerBlock;
    uint32_t tileNum;
    uint32_t tileLength;
    float eps;
};

extern "C" __global__ __aicore__ void gemm_scale_batch_norm_custom(GM_ADDR x, GM_ADDR scale, GM_ADDR bn_weight, GM_ADDR bn_bias,
                                                                   GM_ADDR running_mean, GM_ADDR running_var, GM_ADDR z,
                                                                   GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmScaleBatchNorm op;
    op.Init(x, scale, bn_weight, bn_bias, running_mean, running_var, z,
            tiling_data.rows, tiling_data.cols, tiling_data.tileNum, tiling_data.eps);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Same as original but fuses scale + BatchNorm into a custom AscendC operator.
    Linear (GEMM) remains as nn.Linear.
    """
    def __init__(self, in_features, out_features, scale_shape, eps=1e-5, momentum=0.1):
        super(ModelNew, self).__init__()
        self.gemm = torch.nn.Linear(in_features, out_features)
        self.scale = torch.nn.Parameter(torch.randn(scale_shape))
        self.bn = torch.nn.BatchNorm1d(out_features, eps=eps, momentum=momentum)

    def forward(self, x):
        x = self.gemm(x)
        # Fused: (x * scale) -> BatchNorm1d (inference-style using running stats)
        x = custom_ops_lib.gemm_scale_batch_norm_custom(
            x,
            self.scale,
            self.bn.weight,
            self.bn.bias,
            self.bn.running_mean,
            self.bn.running_var,
            float(self.bn.eps),
        )
        return x
