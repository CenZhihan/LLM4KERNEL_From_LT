kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "convtranspose3d_mean_add_softmax_tanh_scaling_custom_tiling_data.h"

// Simplified reference kernel implementing:
// 1) Mean over depth dim (D) -> output D = 1
// 2) Add channel bias (shape [C])
// 3) Softmax over channel dim (C) per (B, H, W)
// 4) Tanh
// 5) Scaling by scalar

extern "C" __global__ __aicore__ void convtranspose3d_mean_add_softmax_tanh_scaling_custom(
    GM_ADDR x, GM_ADDR bias, GM_ADDR scale, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(t, tiling);

    // Expect float tensors
    __gm__ float* xPtr     = reinterpret_cast<__gm__ float*>(x);
    __gm__ float* biasPtr  = reinterpret_cast<__gm__ float*>(bias);
    __gm__ float* scalePtr = reinterpret_cast<__gm__ float*>(scale);
    __gm__ float* zPtr     = reinterpret_cast<__gm__ float*>(z);

    const uint32_t B = t.B();
    const uint32_t C = t.C();
    const uint32_t D = t.D();
    const uint32_t H = t.H();
    const uint32_t W = t.W();

    const float scale_val = scalePtr[0];

    // Iterate over batch and spatial positions, then reduce over D and channel-softmax
    for (uint32_t b = 0; b < B; ++b) {
        for (uint32_t h = 0; h < H; ++h) {
            for (uint32_t w = 0; w < W; ++w) {

                // Temporary per-channel buffer in registers/shared
                // For demonstration, use a conservative channel upper-bound in loops.
                // Compute mean over D and add bias
                // Also find max for softmax stability
                float max_val = -3.402823466e+38F; // -FLT_MAX

                // First pass: compute mean (over D) per channel and add bias
                // Store back to z temporarily at D=0 position (we'll overwrite after softmax)
                for (uint32_t c = 0; c < C; ++c) {
                    float s = 0.f;
                    // sum over depth
                    for (uint32_t d = 0; d < D; ++d) {
                        uint64_t idxIn = ((((uint64_t)b * C + c) * D + d) * H + h) * W + w;
                        s += xPtr[idxIn];
                    }
                    float m = s / static_cast<float>(D);
                    m += biasPtr[c];
                    if (m > max_val) max_val = m;

                    // Temporarily store the pre-softmax value to output buffer
                    uint64_t idxTmp = ((((uint64_t)b * C + c) * 1 + 0) * H + h) * W + w;
                    zPtr[idxTmp] = m;
                }

                // Second pass: softmax denominator
                float denom = 0.f;
                for (uint32_t c = 0; c < C; ++c) {
                    uint64_t idxTmp = ((((uint64_t)b * C + c) * 1 + 0) * H + h) * W + w;
                    float v = zPtr[idxTmp];
                    float e = AscendC::Exp(v - max_val);
                    denom += e;
                    zPtr[idxTmp] = e; // store exp temporarily
                }

                // Third pass: normalize, tanh, and scale
                for (uint32_t c = 0; c < C; ++c) {
                    uint64_t idxOut = ((((uint64_t)b * C + c) * 1 + 0) * H + h) * W + w;
                    float prob = zPtr[idxOut] / denom;  // softmax
                    float th = AscendC::Tanh(prob);
                    zPtr[idxOut] = th * scale_val;
                }
            }
        }
    }
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model:
    - Keep ConvTranspose3d on framework side for parameter handling
    - Fuse the following into a custom kernel:
        Mean over depth -> Bias add -> Softmax over channels -> Tanh -> Scaling
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, scaling_factor):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(
            in_channels, out_channels, kernel_size, stride=stride, padding=padding, bias=True
        )
        # Bias added after mean (broadcast over [B, C, 1, H, W])
        self.bias = torch.nn.Parameter(torch.randn(1, out_channels, 1, 1, 1))
        self.scaling_factor = float(scaling_factor)

    def forward(self, x):
        x = self.conv_transpose(x)  # (B, C, D, H, W)
        scale = torch.tensor([self.scaling_factor], dtype=x.dtype, device=x.device)
        # Custom fused op: mean over depth + bias add + softmax over channels + tanh + scaling
        return custom_ops_lib.convtranspose3d_mean_add_softmax_tanh_scaling_custom(x, self.bias, scale)
