kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv2d_tanh_scaling_bias_add_max_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelFused {
public:
    __aicore__ inline KernelFused() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR scale, GM_ADDR z, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        bGm.SetGlobalBuffer((__gm__ DTYPE_Y *)bias + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        sGm.SetGlobalBuffer((__gm__ DTYPE_Z *)scale + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(inQueueS, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute();
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> bLocal = inQueueB.AllocTensor<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> sLocal = inQueueS.AllocTensor<DTYPE_Z>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(bLocal, bGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(sLocal, sGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
        inQueueB.EnQue(bLocal);
        inQueueS.EnQue(sLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> bLocal = inQueueB.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> sLocal = inQueueS.DeQue<DTYPE_Z>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();

        // z = tanh(x) * scale + bias
        AscendC::Tanh(zLocal, xLocal, this->tileLength);
        AscendC::Mul(zLocal, zLocal, sLocal, this->tileLength);
        AscendC::Add(zLocal, zLocal, bLocal, this->tileLength);

        outQueueZ.EnQue<DTYPE_Z>(zLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueB.FreeTensor(bLocal);
        inQueueS.FreeTensor(sLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[progress * this->tileLength], zLocal, this->tileLength);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueB, inQueueS;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> bGm;
    AscendC::GlobalTensor<DTYPE_Z> sGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv2d_tanh_scaling_bias_add_max_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR scale, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelFused op;
    op.Init(x, bias, scale, z, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: conv -> fused(tanh + scaling + bias add) -> max-pool
    """
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)
        self.scaling_factor = scaling_factor
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))
        self.max_pool = torch.nn.MaxPool2d(pool_kernel_size)

    def forward(self, x):
        # Convolution
        x = self.conv(x)
        # Prepare broadcasted tensors for the custom fused op
        b = self.bias.view(1, -1, 1, 1).expand(x.size(0), -1, x.size(2), x.size(3))
        s = torch.full_like(x, self.scaling_factor)
        # Fused tanh + scaling + bias add via AscendC custom kernel
        x = custom_ops_lib.conv2d_tanh_scaling_bias_add_max_custom(x, b, s)
        # Max-pooling
        x = self.max_pool(x)
        return x
