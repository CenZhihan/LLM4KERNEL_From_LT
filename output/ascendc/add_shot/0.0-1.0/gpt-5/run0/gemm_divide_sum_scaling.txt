    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);

    std::vector<int64_t> out_dims;

        out_dims.push_back(1);

    for (size_t i = 0; i < out_dims.size(); ++i) {
        y_shape->SetDim(i, out_dims[i]);

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "gemm_divide_sum_scaling_custom_tiling_data.h"

using DTYPE = float;

class KernelGemmDivideSumScaling {
public:
    __aicore__ inline KernelGemmDivideSumScaling() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR scale, GM_ADDR y,
                                uint32_t batch, uint32_t K, uint32_t H,
                                uint32_t tileK, uint32_t tileB)
    {
        this->batch = batch;
        this->K = K;
        this->H = H;
        this->tileK = tileK;
        this->tileB = tileB;

        // Block partition along batch dimension
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t bidx = AscendC::GetBlockIdx();
        uint32_t rowsPerBlock = (batch + blocks - 1) / blocks;
        this->rowStart = bidx * rowsPerBlock;
        this->rowEnd = AscendC::Min((uint32_t)(rowStart + rowsPerBlock), batch);

        xGm.SetGlobalBuffer((__gm__ DTYPE*)x, (uint64_t)batch * K);
        wGm.SetGlobalBuffer((__gm__ DTYPE*)w, (uint64_t)H * K);
        sGm.SetGlobalBuffer((__gm__ DTYPE*)scale, 1);
        yGm.SetGlobalBuffer((__gm__ DTYPE*)y, (uint64_t)batch);

        // local buffers
        pipe.InitBuffer(bufSumK, 1, this->tileK * sizeof(DTYPE)); // sBuf
        pipe.InitBuffer(bufXK, 1, this->tileK * sizeof(DTYPE));   // xSeg
        pipe.InitBuffer(bufAccB, 1, this->tileB * sizeof(DTYPE)); // acc buffer for rows
        pipe.InitBuffer(bufRowSeg, 1, this->tileK * sizeof(DTYPE)); // temp W row segment
    }

    __aicore__ inline void Process()
    {
        if (rowStart >= rowEnd) {
            return;
        }

        // Load scaling factor once
        DTYPE scaleVal = 1.0f;
        {
            AscendC::LocalTensor<DTYPE> sLocal = bufXK.AllocTensor<DTYPE>();
            AscendC::DataCopy(sLocal, sGm[0], 1);
            scaleVal = sLocal.GetValue(0);
            bufXK.FreeTensor(sLocal);
        }
        DTYPE coeff = scaleVal * 0.5f;

        for (uint32_t bBase = rowStart; bBase < rowEnd; bBase += tileB) {
            uint32_t curB = AscendC::Min((uint32_t)tileB, (uint32_t)(rowEnd - bBase));

            // acc buffer init to 0
            AscendC::LocalTensor<DTYPE> accLocal = bufAccB.AllocTensor<DTYPE>();
            for (uint32_t i = 0; i < curB; ++i) {
                accLocal.SetValue(i, (DTYPE)0.0f);
            }

            for (uint32_t kBase = 0; kBase < K; kBase += tileK) {
                uint32_t curK = AscendC::Min((uint32_t)tileK, (uint32_t)(K - kBase));

                // Build sBuf = sum over H of W[j, kBase : kBase+curK]
                AscendC::LocalTensor<DTYPE> sBuf = bufSumK.AllocTensor<DTYPE>();
                for (uint32_t kk = 0; kk < curK; ++kk) {
                    sBuf.SetValue(kk, (DTYPE)0.0f);
                }

                AscendC::LocalTensor<DTYPE> wRowSeg = bufRowSeg.AllocTensor<DTYPE>();
                for (uint32_t j = 0; j < H; ++j) {
                    // copy contiguous row segment
                    AscendC::DataCopy(wRowSeg, wGm[j * K + kBase], curK);
                    // accumulate
                    for (uint32_t kk = 0; kk < curK; ++kk) {
                        DTYPE v = sBuf.GetValue(kk) + wRowSeg.GetValue(kk);
                        sBuf.SetValue(kk, v);
                    }
                }

                // For each row in this block tile, accumulate dot(xSeg, sBuf)
                for (uint32_t bi = 0; bi < curB; ++bi) {
                    uint32_t row = bBase + bi;
                    AscendC::LocalTensor<DTYPE> xSeg = bufXK.AllocTensor<DTYPE>();
                    AscendC::DataCopy(xSeg, xGm[row * K + kBase], curK);

                    DTYPE dot = (DTYPE)0.0f;
                    for (uint32_t kk = 0; kk < curK; ++kk) {
                        dot += xSeg.GetValue(kk) * sBuf.GetValue(kk);
                    }
                    bufXK.FreeTensor(xSeg);

                    // accLocal[bi] += dot
                    DTYPE prev = accLocal.GetValue(bi);
                    accLocal.SetValue(bi, prev + dot);
                }

                bufRowSeg.FreeTensor(wRowSeg);
                bufSumK.FreeTensor(sBuf);
            }

            // scale and write out to y
            for (uint32_t bi = 0; bi < curB; ++bi) {
                DTYPE v = accLocal.GetValue(bi) * coeff;
                accLocal.SetValue(bi, v);
            }
            AscendC::DataCopy(yGm[bBase], accLocal, curB);
            bufAccB.FreeTensor(accLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TBuf<AscendC::TPosition::VECIN> bufSumK;
    AscendC::TBuf<AscendC::TPosition::VECIN> bufXK;
    AscendC::TBuf<AscendC::TPosition::VECOUT> bufAccB;
    AscendC::TBuf<AscendC::TPosition::VECIN> bufRowSeg;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> wGm;
    AscendC::GlobalTensor<DTYPE> sGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t batch;
    uint32_t K;
    uint32_t H;
    uint32_t tileK;
    uint32_t tileB;

    uint32_t rowStart;
    uint32_t rowEnd;
};

extern "C" __global__ __aicore__ void gemm_divide_sum_scaling_custom(GM_ADDR x, GM_ADDR w, GM_ADDR scale, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmDivideSumScaling op;
    op.Init(x, w, scale, y,
            tiling_data.batch,
            tiling_data.inputSizeK,
            tiling_data.hiddenSizeH,
            tiling_data.tileK,
            tiling_data.tileB);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused kernel model: matmul -> divide by 2 -> sum over hidden dim -> scale by factor
    """
    def __init__(self, input_size, hidden_size, scaling_factor):
        super(ModelNew, self).__init__()
        self.weight = torch.nn.Parameter(torch.randn(hidden_size, input_size))
        self.scaling_factor = scaling_factor

    def forward(self, x: torch.Tensor):
        scale_tensor = torch.tensor(self.scaling_factor, dtype=x.dtype, device=x.device)
        return custom_ops_lib.gemm_divide_sum_scaling_custom(x, self.weight, scale_tensor)
