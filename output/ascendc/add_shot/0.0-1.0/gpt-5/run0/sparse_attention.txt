kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "sparse_attention_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

// This kernel performs elementwise addition Z = X + Y,
// intended to add a large negative bias from the sparse mask to attention scores.
class KernelSparseAttention {
public:
    __aicore__ inline KernelSparseAttention() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR z, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }
    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = inQueueY.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(yLocal, yGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
        inQueueY.EnQue(yLocal);
    }
    __aicore__ inline void Compute(int32_t)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = inQueueY.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();
        AscendC::Add(zLocal, xLocal, yLocal, this->tileLength);
        outQueueZ.EnQue<DTYPE_Z>(zLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueY.FreeTensor(yLocal);
    }
    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[progress * this->tileLength], zLocal, this->tileLength);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueY;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void sparse_attention_custom(GM_ADDR x, GM_ADDR y, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSparseAttention op;
    op.Init(x, y, z, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import math
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    """
    Sparse attention with custom AscendC operator to fuse masked-bias addition into attention scores.
    The kernel sparse_attention_custom performs elementwise addition between scores and a precomputed mask bias.
    """
    def __init__(self, d_model=256, num_heads=4, window_size=32, max_seq_len=4096):
        super().__init__()
        assert d_model % num_heads == 0, "d_model must be divisible by num_heads"
        self.d_model = d_model
        self.num_heads = num_heads
        self.head_dim = d_model // num_heads
        self.window_size = window_size
        self.max_seq_len = max_seq_len

        # Projection layers (equivalent to basic MHA projections)
        self.q_proj = nn.Linear(d_model, d_model, bias=True)
        self.k_proj = nn.Linear(d_model, d_model, bias=True)
        self.v_proj = nn.Linear(d_model, d_model, bias=True)
        self.out_proj = nn.Linear(d_model, d_model, bias=True)

        # Precompute sparse mask: True means masked (disallowed)
        mask = torch.ones(max_seq_len, max_seq_len)
        for i in range(max_seq_len):
            start = max(0, i - window_size)
            end = min(max_seq_len, i + window_size)
            mask[i, start:end] = 0
        mask = mask.bool()
        self.register_buffer("attn_mask", mask, persistent=False)

    def forward(self, x):
        """
        Args:
            x: (batch, seq_len, d_model)
        Returns:
            output: (batch, seq_len, d_model)
        """
        B, S, _ = x.shape
        H = self.num_heads
        Dh = self.head_dim

        # Projections
        q = self.q_proj(x).view(B, S, H, Dh).transpose(1, 2)  # (B, H, S, Dh)
        k = self.k_proj(x).view(B, S, H, Dh).transpose(1, 2)  # (B, H, S, Dh)
        v = self.v_proj(x).view(B, S, H, Dh).transpose(1, 2)  # (B, H, S, Dh)

        # Scaled dot-product scores
        scores = torch.matmul(q, k.transpose(-2, -1)) / math.sqrt(Dh)  # (B, H, S, S)

        # Prepare mask bias tensor to add to scores: 0 for allowed, large negative for masked positions
        attn_mask = self.attn_mask[:S, :S]
        bias = torch.where(attn_mask, torch.tensor(-1e4, dtype=scores.dtype, device=scores.device),
                           torch.tensor(0.0, dtype=scores.dtype, device=scores.device))
        bias = bias.unsqueeze(0).unsqueeze(0).expand(B, H, S, S).contiguous()

        # Custom kernel: elementwise add scores + bias
        masked_scores = custom_ops_lib.sparse_attention_custom(scores, bias)

        # Softmax over keys dimension
        attn = torch.softmax(masked_scores, dim=-1)

        # Weighted sum
        context = torch.matmul(attn, v)  # (B, H, S, Dh)

        # Merge heads
        context = context.transpose(1, 2).contiguous().view(B, S, H * Dh)
        out = self.out_proj(context)
        return out
'''
