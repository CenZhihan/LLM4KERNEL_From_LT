host_operator_src="""
#include "masked_cumsum_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
inline uint32_t ClampU32(int64_t v) {
    if (v < 0) return 0;
    if (v > static_cast<int64_t>(UINT32_MAX)) return UINT32_MAX;
    return static_cast<uint32_t>(v);
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t DEFAULT_TILE_COLS = 1024;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    MaskedCumsumCustomTilingData tiling;

    auto* inShape = context->GetInputShape(0);
    auto originShape = inShape->GetOriginShape();
    int64_t rank = originShape.GetDimNum();

    int32_t dim = 0;
    if (context->HasAttr("dim")) {
        dim = context->GetAttr<int32_t>("dim");
    }

    // Normalize dim to positive index
    if (rank > 0) {
        if (dim < 0) dim += static_cast<int32_t>(rank);
        if (dim < 0) dim = 0;
        if (dim >= rank) dim = static_cast<int32_t>(rank - 1);
    }

    // Compute rows = prod(d[0..dim-1]), cols = prod(d[dim..rank-1])
    uint64_t rows = 1;
    uint64_t cols = 1;
    for (int64_t i = 0; i < rank; ++i) {
        int64_t d = originShape.GetDim(i);
        if (i < dim) {
            rows *= static_cast<uint64_t>(d);
        } else {
            cols *= static_cast<uint64_t>(d);
        }
    }
    if (rank == 0) { rows = 1; cols = 1; }

    uint32_t urows = ClampU32(rows);
    uint32_t ucols = ClampU32(cols);

    uint32_t rowsPerBlock = (urows + BLOCK_DIM - 1) / BLOCK_DIM;
    if (rowsPerBlock == 0) rowsPerBlock = 1;
    uint32_t tileCols = DEFAULT_TILE_COLS;
    if (tileCols > ucols) tileCols = ucols;

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_rows(urows);
    tiling.set_cols(ucols);
    tiling.set_rowsPerBlock(rowsPerBlock);
    tiling.set_tileCols(tileCols);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* ws = context->GetWorkspaceSizes(1);
    ws[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto xType = context->GetInputDataType(0);
    context->SetOutputDataType(0, xType);
    return GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "masked_cumsum_custom_tiling_data.h"

// We assume:
// - Input0 x: float (DTYPE_X)
// - Input1 mask: bool (DTYPE_Y)
// - Output y: float (DTYPE_Z)

constexpr int32_t BUFFER_NUM = 1;

class KernelMaskedCumsum {
public:
    __aicore__ inline KernelMaskedCumsum() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                                uint32_t rows, uint32_t cols,
                                uint32_t rowsPerBlock, uint32_t tileCols, int32_t dim)
    {
        // We implement cumsum along axis 'dim'. After tiling, we treat data as 2D:
        // rows = prod(d[0..dim-1]), cols = prod(d[dim..]).
        this->rows = rows;
        this->cols = cols;
        this->rowsPerBlock = rowsPerBlock;
        this->tileCols = tileCols;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint64_t)rows * (uint64_t)cols);
        mGm.SetGlobalBuffer((__gm__ DTYPE_Y *)mask, (uint64_t)rows * (uint64_t)cols);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y, (uint64_t)rows * (uint64_t)cols);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileCols * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueM, BUFFER_NUM, this->tileCols * sizeof(DTYPE_Y));
        pipe.InitBuffer(tmpQueueF, BUFFER_NUM, this->tileCols * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileCols * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t startRow = blockIdx * rowsPerBlock;
        uint32_t endRow = startRow + rowsPerBlock;
        if (endRow > rows) endRow = rows;

        for (uint32_t r = startRow; r < endRow; ++r) {
            // Carry for cumsum across tiles within the row
            DTYPE_Z carry = (DTYPE_Z)0;

            for (uint32_t c0 = 0; c0 < cols; c0 += tileCols) {
                uint32_t curCols = AscendC::Min(tileCols, cols - c0);
                // CopyIn
                AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_Y> mLocal = inQueueM.AllocTensor<DTYPE_Y>();
                AscendC::LocalTensor<DTYPE_X> mFloat = tmpQueueF.AllocTensor<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

                AscendC::DataCopy(xLocal, xGm[r * cols + c0], curCols);
                AscendC::DataCopy(mLocal, mGm[r * cols + c0], curCols);

                // Cast mask (bool) -> float, then multiply: yLocal = xLocal * mFloat
                AscendC::Cast(mFloat, mLocal, curCols);
                AscendC::Mul(yLocal, xLocal, mFloat, curCols);

                // Inclusive scan (cumsum) in-place on yLocal with incoming carry
                DTYPE_Z acc = carry;
                for (uint32_t j = 0; j < curCols; ++j) {
                    DTYPE_Z v = (DTYPE_Z)AscendC::GetValue(yLocal, j);
                    acc = acc + v;
                    AscendC::SetValue(yLocal, j, acc);
                }
                carry = acc;

                // CopyOut
                AscendC::DataCopy(yGm[r * cols + c0], yLocal, curCols);

                // Free locals
                outQueueY.FreeTensor(yLocal);
                tmpQueueF.FreeTensor(mFloat);
                inQueueM.FreeTensor(mLocal);
                inQueueX.FreeTensor(xLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueM;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpQueueF;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> mGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t rows;
    uint32_t cols;
    uint32_t rowsPerBlock;
    uint32_t tileCols;
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x, GM_ADDR mask, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaskedCumsum op;
    op.Init(x, mask, y,
            tiling_data.rows, tiling_data.cols,
            tiling_data.rowsPerBlock, tiling_data.tileCols,
            tiling_data.dim);
    op.Process();
}
