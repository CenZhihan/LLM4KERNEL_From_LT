namespace optiling {
const uint32_t BLOCK_DIM = 16;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;

    const auto *aShape = context->GetInputShape(0);
    const auto *bShape = context->GetInputShape(1);
    auto aOrigin = aShape->GetOriginShape();
    auto bOrigin = bShape->GetOriginShape();

    // Assume standard matmul: A [M,K], B [K,N]
    uint64_t M = (aOrigin.GetDimNum() >= 1) ? aOrigin.GetDim(0) : 1;
    uint64_t K = (aOrigin.GetDimNum() >= 2) ? aOrigin.GetDim(1) : 1;
    uint64_t N = (bOrigin.GetDimNum() >= 2) ? bOrigin.GetDim(1) : ((bOrigin.GetDimNum() >= 1) ? bOrigin.GetDim(0) : 1);

    // Heuristic tiling for tall-skinny: prefer larger tileM when M >> N
    uint32_t tM = umin(M, 256);
    uint32_t tN = umin(N, 64);
    uint32_t tK = umin(K, 64);

    context->SetBlockDim(BLOCK_DIM);

    // rows per block
    uint64_t rowsPerBlock = (M + BLOCK_DIM - 1) / BLOCK_DIM;

    tiling.set_M(static_cast<uint32_t>(M));
    tiling.set_N(static_cast<uint32_t>(N));
    tiling.set_K(static_cast<uint32_t>(K));
    tiling.set_tileM(tM);
    tiling.set_tileN(tN);
    tiling.set_tileK(tK);
    tiling.set_blockRows(static_cast<uint32_t>(rowsPerBlock));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "tall_skinny_matrix_multiplication_custom_tiling_data.h"

// A simple GEMM kernel specialized for tall-skinny shapes.
// Compute C = A[M,K] * B[K,N] in tiles.
// This implementation uses local tiles and vector-scalar fused updates.

constexpr int32_t IN_BUFFER_NUM = 1;
constexpr int32_t OUT_BUFFER_NUM = 1;

class KernelTSMM {
public:
    __aicore__ inline KernelTSMM() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK, uint32_t blockRows)
    {
        this->M = M;
        this->N = N;
        this->K = K;
        this->tileM = tileM;
        this->tileN = tileN;
        this->tileK = tileK;

        // Assign rows to block
        uint32_t bid = AscendC::GetBlockIdx();
        uint32_t bnum = AscendC::GetBlockNum();

        uint32_t rowsPerBlock = blockRows;
        this->rowStart = bid * rowsPerBlock;
        this->rowEnd = AscendC::Min(this->rowStart + rowsPerBlock, M);

        aGm.SetGlobalBuffer((__gm__ DTYPE_A *)A, (uint64_t)M * K);
        bGm.SetGlobalBuffer((__gm__ DTYPE_B *)B, (uint64_t)K * N);
        cGm.SetGlobalBuffer((__gm__ DTYPE_C *)C, (uint64_t)M * N);

        // Allocate on-chip buffers
        pipe.InitBuffer(inQueueA, IN_BUFFER_NUM, (uint64_t)tileM * tileK * sizeof(DTYPE_A));
        pipe.InitBuffer(inQueueB, IN_BUFFER_NUM, (uint64_t)tileK * tileN * sizeof(DTYPE_B));
        pipe.InitBuffer(outQueueC, OUT_BUFFER_NUM, (uint64_t)tileM * tileN * sizeof(DTYPE_C));
        pipe.InitBuffer(tmpVecQ, OUT_BUFFER_NUM, AscendC::Max((uint32_t)tileM, (uint32_t)tileN) * sizeof(DTYPE_C));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t m0 = rowStart; m0 < rowEnd; m0 += tileM) {
            uint32_t curTileM = AscendC::Min(tileM, rowEnd - m0);
            for (uint32_t n0 = 0; n0 < N; n0 += tileN) {
                uint32_t curTileN = AscendC::Min(tileN, N - n0);

                // Initialize C tile to zeros
                AscendC::LocalTensor<DTYPE_C> cLocal = outQueueC.AllocTensor<DTYPE_C>();
                AscendC::SetVector<DTYPE_C>(cLocal, 0, curTileM * curTileN);

                // Accumulate across K dimension
                for (uint32_t k0 = 0; k0 < K; k0 += tileK) {
                    uint32_t curTileK = AscendC::Min(tileK, K - k0);

                    // Load A tile [curTileM, curTileK]
                    AscendC::LocalTensor<DTYPE_A> aLocal = inQueueA.AllocTensor<DTYPE_A>();
                    for (uint32_t rm = 0; rm < curTileM; ++rm) {
                        uint64_t gmOff = ((uint64_t)(m0 + rm) * K) + k0;
                        AscendC::DataCopy(aLocal[rm * curTileK], aGm[gmOff], curTileK);
                    }
                    inQueueA.EnQue(aLocal);

                    // Load B tile [curTileK, curTileN]
                    AscendC::LocalTensor<DTYPE_B> bLocal = inQueueB.AllocTensor<DTYPE_B>();
                    for (uint32_t rk = 0; rk < curTileK; ++rk) {
                        uint64_t gmOff = ((uint64_t)(k0 + rk) * N) + n0;
                        AscendC::DataCopy(bLocal[rk * curTileN], bGm[gmOff], curTileN);
                    }
                    inQueueB.EnQue(bLocal);

                    // Compute cLocal += aLocal * bLocal
                    aLocal = inQueueA.DeQue<DTYPE_A>();
                    bLocal = inQueueB.DeQue<DTYPE_B>();

                    // Choose accumulation strategy based on tile shape:
                    // If curTileM >= curTileN, accumulate by columns (vector over M, scalar from B)
                    if (curTileM >= curTileN) {
                        for (uint32_t nk = 0; nk < curTileN; ++nk) {
                            // Column accumulator pointer in cLocal (column-major within tile)
                            // cLocal is stored row-major; index (r,c) -> r*curTileN + c
                            AscendC::LocalTensor<DTYPE_C> tmpVec = tmpVecQ.AllocTensor<DTYPE_C>();
                            // zero column accumulator
                            AscendC::SetVector<DTYPE_C>(tmpVec, 0, curTileM);
                            for (uint32_t kk = 0; kk < curTileK; ++kk) {
                                // scalar = B[kk, nk]
                                DTYPE_B bScalar;
                                AscendC::DataCopy(&bScalar, bLocal[kk * curTileN + nk], 1);
                                // vecA = A[:, kk]
                                AscendC::LocalTensor<DTYPE_A> aCol = aLocal[kk * curTileM]; // treat as base of column
                                // tmp = aCol * bScalar
                                AscendC::LocalTensor<DTYPE_C> tmpMul = tmpVecQ.AllocTensor<DTYPE_C>();
                                AscendC::Cast(tmpMul, aCol, curTileM); // cast A->C
                                AscendC::Muls(tmpMul, tmpMul, (DTYPE_C)bScalar, curTileM);
                                // tmpVec += tmpMul
                                AscendC::Add(tmpVec, tmpVec, tmpMul, curTileM);
                                tmpVecQ.FreeTensor(tmpMul);
                            }
                            // Write accumulated column tmpVec into cLocal at column nk
                            for (uint32_t rm = 0; rm < curTileM; ++rm) {
                                cLocal[rm * curTileN + nk] = tmpVec[rm];
                            }
                            tmpVecQ.FreeTensor(tmpVec);
                        }
                    } else {
                        // Accumulate by rows (vector over N, scalar from A)
                        for (uint32_t rm = 0; rm < curTileM; ++rm) {
                            AscendC::LocalTensor<DTYPE_C> tmpRow = tmpVecQ.AllocTensor<DTYPE_C>();
                            AscendC::SetVector<DTYPE_C>(tmpRow, 0, curTileN);
                            for (uint32_t kk = 0; kk < curTileK; ++kk) {
                                // scalar = A[rm, kk]
                                DTYPE_A aScalar;
                                AscendC::DataCopy(&aScalar, aLocal[kk * curTileM + rm], 1);
                                // vecB = B[kk, :]
                                AscendC::LocalTensor<DTYPE_B> bRow = bLocal[kk * curTileN];
                                AscendC::LocalTensor<DTYPE_C> tmpMul = tmpVecQ.AllocTensor<DTYPE_C>();
                                AscendC::Cast(tmpMul, bRow, curTileN); // cast B->C
                                AscendC::Muls(tmpMul, tmpMul, (DTYPE_C)aScalar, curTileN);
                                AscendC::Add(tmpRow, tmpRow, tmpMul, curTileN);
                                tmpVecQ.FreeTensor(tmpMul);
                            }
                            // Write accumulated row to cLocal
                            AscendC::DataCopy(cLocal[rm * curTileN], tmpRow, curTileN);
                            tmpVecQ.FreeTensor(tmpRow);
                        }
                    }

                    inQueueA.FreeTensor(aLocal);
                    inQueueB.FreeTensor(bLocal);
                }

                // Store C tile back to GM row by row
                for (uint32_t rm = 0; rm < curTileM; ++rm) {
                    uint64_t gmOff = ((uint64_t)(m0 + rm) * N) + n0;
                    AscendC::DataCopy(cGm[gmOff], cLocal[rm * curTileN], curTileN);
                }
                outQueueC.FreeTensor(cLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUFFER_NUM> inQueueA;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUFFER_NUM> inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUFFER_NUM> outQueueC;
    AscendC::TQue<AscendC::TPosition::VECCALC, OUT_BUFFER_NUM> tmpVecQ;

    AscendC::GlobalTensor<DTYPE_A> aGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_C> cGm;

    uint32_t M, N, K;
    uint32_t tileM, tileN, tileK;
    uint32_t rowStart, rowEnd;
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelTSMM op;
    op.Init(A, B, C,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK, tiling_data.blockRows);
    op.Process();
}
