kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "multi_query_attention_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmaxLastDim {
public:
    __aicore__ inline KernelSoftmaxLastDim() {}
    __aicore__ inline void Init(GM_ADDR scores, GM_ADDR probs, uint32_t totalLength, uint32_t rowLength, uint32_t tileNum)
    {
        this->rowLength = rowLength;
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;

        uint32_t baseTile = this->blockLength / tileNum / BUFFER_NUM;
        // ensure tileLength is multiple of rowLength
        uint32_t rowsPerTile = AscendC::Max((uint32_t)1, baseTile / rowLength);
        this->tileLength = rowsPerTile * rowLength;

        sGm.SetGlobalBuffer((__gm__ float *)scores + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        pGm.SetGlobalBuffer((__gm__ float *)probs + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueS, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(outQueueP, BUFFER_NUM, this->tileLength * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<float> sLocal = inQueueS.AllocTensor<float>();
        AscendC::DataCopy(sLocal, sGm[progress * this->tileLength], this->tileLength);
        inQueueS.EnQue(sLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<float> sLocal = inQueueS.DeQue<float>();
        AscendC::LocalTensor<float> pLocal = outQueueP.AllocTensor<float>();

        // number of rows within this tile
        uint32_t rows = this->tileLength / this->rowLength;

        // Perform softmax row-wise along last dimension
        // Assume AscendC provides a fused softmax on contiguous rows
        // If not, this represents a conceptual implementation.
        AscendC::Softmax(pLocal, sLocal, this->rowLength, rows);

        outQueueP.EnQue<float>(pLocal);
        inQueueS.FreeTensor(sLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<float> pLocal = outQueueP.DeQue<float>();
        AscendC::DataCopy(pGm[progress * this->tileLength], pLocal, this->tileLength);
        outQueueP.FreeTensor(pLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueS;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueP;
    AscendC::GlobalTensor<float> sGm;
    AscendC::GlobalTensor<float> pGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
    uint32_t rowLength;
};

extern "C" __global__ __aicore__ void multi_query_attention_custom(GM_ADDR scores, GM_ADDR probs, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxLastDim op;
    op.Init(scores, probs, tiling_data.totalLength, tiling_data.rowLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib
import math
import torch.nn as nn

class ModelNew(torch.nn.Module):
    """
    Multi-Query Attention (MQA) using a custom AscendC softmax kernel
    - Replaces torch.softmax(attn, dim=-1) with custom_ops_lib.multi_query_attention_custom
    """
    def __init__(self, d_model=4096, num_heads=32):
        super().__init__()
        self.q_proj = nn.Linear(d_model, d_model)
        self.k_proj = nn.Linear(d_model, d_model // num_heads)  # shared across heads
        self.v_proj = nn.Linear(d_model, d_model // num_heads)
        self.out_proj = nn.Linear(d_model, d_model)
        self.num_heads = num_heads

    def forward(self, x):
        B, N, D = x.shape
        H = self.num_heads
        d_h = D // H

        Q = self.q_proj(x).view(B, N, H, d_h)
        K = self.k_proj(x).unsqueeze(2)  # shared K/V across heads -> [B, N, 1, d_h]
        V = self.v_proj(x).unsqueeze(2)  # [B, N, 1, d_h]

        # Attention scores: [B, H, N, N]
        attn_scores = torch.einsum("bnhd,bkhd->bhnk", Q, K) / math.sqrt(d_h)

        # Custom softmax along last dimension
        attn = custom_ops_lib.multi_query_attention_custom(attn_scores)

        # Context: [B, N, H, d_h]
        out = torch.einsum("bhnk,bkhd->bnhd", attn, V).reshape(B, N, D)
        return self.out_proj(out)
'''
