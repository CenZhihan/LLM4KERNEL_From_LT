kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "gemm_log_sum_exp_leaky_relu_leaky_relu_gelu_gelu_custom_tiling_data.h"

extern \"C\" __global__ __aicore__ void gemm_log_sum_exp_leaky_relu_leaky_relu_gelu_gelu_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t M = tiling_data.m;
    const uint32_t K = tiling_data.k;
    const uint32_t N = tiling_data.n;
    const uint32_t blockRows = tiling_data.blockRows;
    const float negSlope = tiling_data.negativeSlope;

    uint32_t blockIdx = AscendC::GetBlockIdx();
    uint32_t rowsBegin = blockIdx * blockRows;
    uint32_t rowsEnd = AscendC::Min(rowsBegin + blockRows, M);

    __gm__ float* xG = (__gm__ float*)x;          // [M, K]
    __gm__ float* wG = (__gm__ float*)w;          // [N, K] (PyTorch Linear weight)
    __gm__ float* bG = (__gm__ float*)b;          // [N]
    __gm__ float* yG = (__gm__ float*)y;          // [M, 1]

    const float SQRT_2 = 1.4142135623730951f;
    const float SQRT_2_OVER_PI = 0.7978845608028654f; // sqrt(2/pi)

    auto gelu = [&](float v) -> float {
        // Tanh approximation: 0.5 * v * (1 + tanh(sqrt(2/pi) * (v + 0.044715*v^3)))
        float v3 = v * v * v;
        float t = SQRT_2_OVER_PI * (v + 0.044715f * v3);
        float th = tanhf(t);
        return 0.5f * v * (1.0f + th);
    };

    auto leaky_relu = [&](float v) -> float {
        return v >= 0.f ? v : (negSlope * v);
    };

    for (uint32_t r = rowsBegin; r < rowsEnd; ++r) {
        // First pass: compute row-wise max for numerical stability
        float rowMax = -3.402823466e+38F; // -FLT_MAX
        for (uint32_t n = 0; n < N; ++n) {
            float acc = 0.f;
            // dot(x[r, :], w[n, :])
            const __gm__ float* xRow = xG + (size_t)r * K;
            const __gm__ float* wRow = wG + (size_t)n * K;
            for (uint32_t k = 0; k < K; ++k) {
                acc += xRow[k] * wRow[k];
            }
            acc += bG[n];
            if (acc > rowMax) rowMax = acc;
        }

        // Second pass: sum of exp
        float sumExp = 0.f;
        for (uint32_t n = 0; n < N; ++n) {
            float acc = 0.f;
            const __gm__ float* xRow = xG + (size_t)r * K;
            const __gm__ float* wRow = wG + (size_t)n * K;
            for (uint32_t k = 0; k < K; ++k) {
                acc += xRow[k] * wRow[k];
            }
            acc += bG[n];
            sumExp += expf(acc - rowMax);
        }

        float lse = logf(sumExp) + rowMax;

        // Two LeakyReLU
        float v = leaky_relu(lse);
        v = leaky_relu(v);

        // Two GELU
        v = gelu(v);
        v = gelu(v);

        yG[r] = v; // store to [M,1] flattened
    }
}
