namespace optiling {
const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    MatmulWithLargeKDimensionCustomTilingData tiling;

    const auto* a_shape = context->GetInputShape(0);
    const auto* b_shape = context->GetInputShape(1);

    // Expect A: (M, K), B: (K, N)
    uint32_t m = static_cast<uint32_t>(a_shape->GetOriginShape().GetDim(0));
    uint32_t k = static_cast<uint32_t>(a_shape->GetOriginShape().GetDim(1));
    uint32_t n = static_cast<uint32_t>(b_shape->GetOriginShape().GetDim(1));

    // Choose tiles for large-K friendly streaming
    const uint32_t tileK = 1024;  // chunk K to fit UB
    const uint32_t tileM = 16;
    const uint32_t tileN = 64;

    uint32_t blockDim = BLOCK_DIM;
    if (blockDim > m) blockDim = m == 0 ? 1 : m;
    context->SetBlockDim(blockDim);

    uint32_t rowsPerBlock = (m + blockDim - 1) / blockDim;

    tiling.set_m(m);
    tiling.set_n(n);
    tiling.set_k(k);
    tiling.set_tileK(tileK);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_rowsPerBlock(rowsPerBlock);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    // A: (M, K), B: (K, N) => C: (M, N)
    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);
    gert::Shape* c_shape = context->GetOutputShape(0);

    std::vector<int64_t> out_dims(2);
    out_dims[0] = a_shape->GetDim(0);
    out_dims[1] = b_shape->GetDim(1);
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, out_dims[0]);
    c_shape->SetDim(1, out_dims[1]);

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "matmul_with_large_k_dimension_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMatmulLargeK {
public:
    __aicore__ inline KernelMatmulLargeK() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t m, uint32_t n, uint32_t k,
                                uint32_t tileK, uint32_t tileM, uint32_t tileN,
                                uint32_t rowsPerBlock)
    {
        this->M = m;
        this->N = n;
        this->K = k;
        this->tileK = tileK;
        this->tileM = tileM;
        this->tileN = tileN;

        const uint32_t blockNum = AscendC::GetBlockNum();
        const uint32_t blockIdx = AscendC::GetBlockIdx();

        this->rowStart = rowsPerBlock * blockIdx;
        uint32_t rowEnd = this->rowStart + rowsPerBlock;
        if (rowEnd > M) rowEnd = M;
        this->rowsThisBlock = (rowEnd > this->rowStart) ? (rowEnd - this->rowStart) : 0;

        aGm.SetGlobalBuffer((__gm__ DTYPE_X *)a + (uint64_t)this->rowStart * K, (uint64_t)this->rowsThisBlock * K);
        bGm.SetGlobalBuffer((__gm__ DTYPE_Y *)b, (uint64_t)K * N);
        cGm.SetGlobalBuffer((__gm__ DTYPE_Z *)c + (uint64_t)this->rowStart * N, (uint64_t)this->rowsThisBlock * N);

        // UB buffers: A tile (tileM x tileK), B tile (tileK x tileN), C tile (tileM x tileN)
        pipe.InitBuffer(inQueueA, BUFFER_NUM, (uint32_t)((uint64_t)tileM * tileK * sizeof(DTYPE_X)));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, (uint32_t)((uint64_t)tileK * tileN * sizeof(DTYPE_Y)));
        pipe.InitBuffer(outQueueC, BUFFER_NUM, (uint32_t)((uint64_t)tileM * tileN * sizeof(DTYPE_Z)));
    }

    __aicore__ inline void Process()
    {
        if (rowsThisBlock == 0 || M == 0 || N == 0 || K == 0) {
            return;
        }

        for (uint32_t m0 = 0; m0 < rowsThisBlock; m0 += tileM) {
            uint32_t curM = AscendC::min(tileM, rowsThisBlock - m0);
            for (uint32_t n0 = 0; n0 < N; n0 += tileN) {
                uint32_t curN = AscendC::min(tileN, N - n0);

                AscendC::LocalTensor<DTYPE_Z> cLocal = outQueueC.AllocTensor<DTYPE_Z>();
                // zero init C tile
                AscendC::Dup<DTYPE_Z>(cLocal, static_cast<DTYPE_Z>(0), (uint32_t)((uint64_t)curM * curN));

                for (uint32_t k0 = 0; k0 < K; k0 += tileK) {
                    uint32_t curK = AscendC::min(tileK, K - k0);

                    AscendC::LocalTensor<DTYPE_X> aLocal = inQueueA.AllocTensor<DTYPE_X>();
                    AscendC::LocalTensor<DTYPE_Y> bLocal = inQueueB.AllocTensor<DTYPE_Y>();

                    // Copy A tile: shape (curM, curK), row-major
                    for (uint32_t r = 0; r < curM; ++r) {
                        uint64_t aSrcOffset = ((uint64_t)m0 + r) * K + k0;
                        uint64_t aDstOffset = (uint64_t)r * curK;
                        AscendC::DataCopy(aLocal[aDstOffset], aGm[aSrcOffset], curK);
                    }

                    // Copy B tile: shape (curK, curN), row-major
                    for (uint32_t kk = 0; kk < curK; ++kk) {
                        uint64_t bSrcOffset = ((uint64_t)k0 + kk) * N + n0;
                        uint64_t bDstOffset = (uint64_t)kk * curN;
                        AscendC::DataCopy(bLocal[bDstOffset], bGm[bSrcOffset], curN);
                    }

                    // Enqueue A and B tiles
                    inQueueA.EnQue(aLocal);
                    inQueueB.EnQue(bLocal);

                    // Compute: C[curM x curN] += A[curM x curK] * B[curK x curN]
                    AscendC::LocalTensor<DTYPE_X> aTile = inQueueA.DeQue<DTYPE_X>();
                    AscendC::LocalTensor<DTYPE_Y> bTile = inQueueB.DeQue<DTYPE_Y>();

                    // Use library matmul if available; otherwise this represents block GEMM accumulation
                    AscendC::MatMul(cLocal, aTile, bTile, curM, curN, curK);

                    inQueueA.FreeTensor(aTile);
                    inQueueB.FreeTensor(bTile);
                }

                // Write back C tile: shape (curM, curN)
                for (uint32_t r = 0; r < curM; ++r) {
                    uint64_t cDstOffset = ((uint64_t)m0 + r) * N + n0;
                    uint64_t cSrcOffset = (uint64_t)r * curN;
                    AscendC::DataCopy(cGm[cDstOffset], cLocal[cSrcOffset], curN);
                }
                outQueueC.FreeTensor(cLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA, inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC;

    AscendC::GlobalTensor<DTYPE_X> aGm;
    AscendC::GlobalTensor<DTYPE_Y> bGm;
    AscendC::GlobalTensor<DTYPE_Z> cGm;

    uint32_t M;
    uint32_t N;
    uint32_t K;
    uint32_t tileK;
    uint32_t tileM;
    uint32_t tileN;

    uint32_t rowStart;
    uint32_t rowsThisBlock;
};

extern "C" __global__ __aicore__ void matmul_with_large_k_dimension_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulLargeK op;
    op.Init(a, b, c,
            tiling_data.m, tiling_data.n, tiling_data.k,
            tiling_data.tileK, tiling_data.tileM, tiling_data.tileN,
            tiling_data.rowsPerBlock);
    op.Process();
}
