kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "cosine_similarity_loss_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // double buffer
constexpr float EPS = 1e-12f;

class KernelCosSimLoss {
public:
    __aicore__ inline KernelCosSimLoss() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR z, uint32_t rows, uint32_t cols, uint32_t tileNum)
    {
        this->rows = rows;
        this->cols = cols;
        this->tileNum = tileNum;

        // Split rows across blocks
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        this->rowsPerBlock = rows / blockNum;
        uint32_t tail = rows % blockNum;
        // Distribute remainder rows to first 'tail' blocks
        this->rowStart = blockIdx * this->rowsPerBlock + AscendC::Min(blockIdx, tail);
        this->myRows = this->rowsPerBlock + (blockIdx < tail ? 1 : 0);

        // Choose tile length along cols for local buffer
        this->tileCols = AscendC::Max((uint32_t)1, (uint32_t)(cols / (tileNum * BUFFER_NUM)));
        if (this->tileCols > cols) this->tileCols = cols;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->rowStart * this->cols, this->myRows * this->cols);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->rowStart * this->cols, this->myRows * this->cols);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z + this->rowStart, this->myRows);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileCols * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueY, BUFFER_NUM, this->tileCols * sizeof(DTYPE_Y));
        // we only need single-value buffer for output per row
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, 1 * sizeof(DTYPE_Z));
        pipe.InitBuffer(tmpQueueA, BUFFER_NUM, this->tileCols * sizeof(DTYPE_Z));
        pipe.InitBuffer(tmpQueueB, BUFFER_NUM, this->tileCols * sizeof(DTYPE_Z));
        pipe.InitBuffer(tmpQueueC, BUFFER_NUM, this->tileCols * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < this->myRows; ++r) {
            float accDot = 0.0f;
            float accPX  = 0.0f;
            float accPY  = 0.0f;

            uint32_t processed = 0;
            uint32_t toggle = 0;

            while (processed < this->cols) {
                uint32_t len = AscendC::Min(this->tileCols, this->cols - processed);

                // Copy in tiles for row r
                AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_Y> yLocal = inQueueY.AllocTensor<DTYPE_Y>();
                AscendC::DataCopy(xLocal, xGm[r * this->cols + processed], len);
                AscendC::DataCopy(yLocal, yGm[r * this->cols + processed], len);
                inQueueX.EnQue(xLocal);
                inQueueY.EnQue(yLocal);

                // Compute partials
                AscendC::LocalTensor<DTYPE_X> xTile = inQueueX.DeQue<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_Y> yTile = inQueueY.DeQue<DTYPE_Y>();

                AscendC::LocalTensor<DTYPE_Z> mulXY = tmpQueueA.AllocTensor<DTYPE_Z>();
                AscendC::LocalTensor<DTYPE_Z> sqX   = tmpQueueB.AllocTensor<DTYPE_Z>();
                AscendC::LocalTensor<DTYPE_Z> sqY   = tmpQueueC.AllocTensor<DTYPE_Z>();

                AscendC::Mul(mulXY, xTile, yTile, len);      // x*y
                AscendC::Mul(sqX, xTile, xTile, len);        // x*x
                AscendC::Mul(sqY, yTile, yTile, len);        // y*y

                // ReduceSum over tile
                float sDot = AscendC::ReduceSum(mulXY, len);
                float sPX  = AscendC::ReduceSum(sqX, len);
                float sPY  = AscendC::ReduceSum(sqY, len);

                accDot += sDot;
                accPX  += sPX;
                accPY  += sPY;

                tmpQueueA.FreeTensor(mulXY);
                tmpQueueB.FreeTensor(sqX);
                tmpQueueC.FreeTensor(sqY);

                inQueueX.FreeTensor(xTile);
                inQueueY.FreeTensor(yTile);

                processed += len;
                toggle ^= 1;
            }

            float denom = AscendC::Sqrt(accPX) * AscendC::Sqrt(accPY) + EPS;
            float cosine = accDot / denom;
            float loss = 1.0f - cosine;

            AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();
            zLocal.SetValue(0, (DTYPE_Z)loss);
            outQueueZ.EnQue<DTYPE_Z>(zLocal);

            AscendC::LocalTensor<DTYPE_Z> zOut = outQueueZ.DeQue<DTYPE_Z>();
            AscendC::DataCopy(zGm[r], zOut, 1);
            outQueueZ.FreeTensor(zOut);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueX, inQueueY;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> tmpQueueA, tmpQueueB, tmpQueueC;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t rows;
    uint32_t cols;
    uint32_t tileNum;

    uint32_t rowStart;
    uint32_t rowsPerBlock;
    uint32_t myRows;
    uint32_t tileCols;
};

extern "C" __global__ __aicore__ void cosine_similarity_loss_custom(GM_ADDR x, GM_ADDR y, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCosSimLoss op;
    op.Init(x, y, z, tiling_data.rows, tiling_data.cols, tiling_data.tileNum);
    op.Process();
}
"""
