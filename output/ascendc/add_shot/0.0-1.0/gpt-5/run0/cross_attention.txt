namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_Q = 8;
const uint32_t TILE_KV = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    CrossAttentionCustomTilingData tiling;

    const auto* q_shape = context->GetInputShape(0);
    const auto* kv_shape = context->GetInputShape(1);

    uint32_t batch = static_cast<uint32_t>(q_shape->GetOriginShape().GetDim(0));
    uint32_t lenQ  = static_cast<uint32_t>(q_shape->GetOriginShape().GetDim(1));
    uint32_t dModel = static_cast<uint32_t>(q_shape->GetOriginShape().GetDim(2));
    uint32_t lenKV = static_cast<uint32_t>(kv_shape->GetOriginShape().GetDim(1));

    // Default to 8 heads if not otherwise specified, typical for d_model=512
    uint32_t numHeads = 8;
    if (dModel % numHeads != 0) {
        // Fallback: try to pick a divisor of dModel closest to 8
        if (dModel % 4 == 0) numHeads = 4;
        if (dModel % 16 == 0) numHeads = 16;
    }
    float dk = static_cast<float>(dModel / numHeads);
    float scale = 1.0f / std::sqrt(dk);

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_batch(batch);
    tiling.set_lenQ(lenQ);
    tiling.set_lenKV(lenKV);
    tiling.set_dModel(dModel);
    tiling.set_numHeads(numHeads);
    tiling.set_scale(scale);
    tiling.set_tileNumQ(TILE_Q);
    tiling.set_tileNumKV(TILE_KV);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "cross_attention_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM_Q = 2;
constexpr int32_t BUFFER_NUM_KV = 2;

class KernelCrossAttention {
public:
    __aicore__ inline KernelCrossAttention() {}
    __aicore__ inline void Init(GM_ADDR q, GM_ADDR kv, GM_ADDR out,
                                uint32_t batch, uint32_t lenQ, uint32_t lenKV,
                                uint32_t dModel, uint32_t numHeads,
                                float scale, uint32_t tileNumQ, uint32_t tileNumKV)
    {
        this->B = batch;
        this->Lq = lenQ;
        this->Lkv = lenKV;
        this->D = dModel;
        this->H = numHeads;
        this->Dh = dModel / numHeads;
        this->scale = scale;
        this->tileNumQ = tileNumQ;
        this->tileNumKV = tileNumKV;

        // Split workload across blocks by batch dimension
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t bPerBlock = (B + blocks - 1) / blocks;
        this->bBegin = AscendC::GetBlockIdx() * bPerBlock;
        this->bEnd = AscendC::Min(this->B, this->bBegin + bPerBlock);

        qGm.SetGlobalBuffer((__gm__ DTYPE_Q *)q, (uint32_t)B * Lq * D);
        kvGm.SetGlobalBuffer((__gm__ DTYPE_KV *)kv, (uint32_t)B * Lkv * D);
        outGm.SetGlobalBuffer((__gm__ DTYPE_O *)out, (uint32_t)B * Lq * D);

        pipe.InitBuffer(qQueue, BUFFER_NUM_Q, (uint32_t)(Dh * sizeof(DTYPE_Q)));
        pipe.InitBuffer(kQueue, BUFFER_NUM_KV, (uint32_t)(Dh * sizeof(DTYPE_KV)));
        pipe.InitBuffer(vQueue, BUFFER_NUM_KV, (uint32_t)(Dh * sizeof(DTYPE_KV)));
        pipe.InitBuffer(oQueue, BUFFER_NUM_Q, (uint32_t)(Dh * sizeof(DTYPE_O)));
        // Temporary buffer for attention probabilities per (query-position, head) over Lkv
        // Managed in small chunks implicitly via queues.
    }

    __aicore__ inline void Process()
    {
        // For each batch and query position, compute attention per head
        for (uint32_t b = bBegin; b < bEnd; ++b) {
            for (uint32_t h = 0; h < H; ++h) {
                // Iterate Lq in tiles of 1 (vector per query). A more optimal kernel would tile >1
                for (uint32_t iq = 0; iq < Lq; ++iq) {
                    // Load q[b, iq, h, :] into local buffer
                    __aicore__ LoadQ(b, iq, h);

                    // Compute scores over all keys (kv acts as both K and V)
                    // This is a conceptual flow using vector ops. In a production kernel,
                    // this should be blocked along Lkv and Dh with MMA instructions.
                    // Here we stream K/V in chunks of head dim (Dh) per position.
                    // Softmax accumulation (max + sum) across Lkv
                    float maxScore = -3.4e38f;
                    float sumExp = 0.0f;

                    // First pass: find maxScore for numerical stability
                    for (uint32_t j = 0; j < Lkv; ++j) {
                        float s = DotQK(b, j, h) * scale;
                        if (s > maxScore) maxScore = s;
                    }
                    // Second pass: compute exp and sum
                    for (uint32_t j = 0; j < Lkv; ++j) {
                        float s = DotQK(b, j, h) * scale;
                        float e = AscendC::Exp(s - maxScore);
                        // Accumulate weighted V later; stash e in a register pipeline
                        tmpScores[j % TMP_BUF] = e; // small circular cache
                        if ((j % TMP_BUF) == (TMP_BUF - 1)) {
                            // no-op; serves to avoid large local buffers in concept
                        }
                        sumExp += e;
                    }

                    // Initialize output head vector to zeros
                    AscendC::LocalTensor<DTYPE_O> oLocal = oQueue.AllocTensor<DTYPE_O>();
                    AscendC::Duplicate(oLocal, (DTYPE_O)0, Dh);

                    // Third pass: accumulate probabilities * V
                    for (uint32_t j = 0; j < Lkv; ++j) {
                        float s = DotQK(b, j, h) * scale;
                        float p = AscendC::Exp(s - maxScore) / sumExp;
                        // Load current V[b, j, h, :]
                        AscendC::LocalTensor<DTYPE_KV> vLocal = vQueue.AllocTensor<DTYPE_KV>();
                        LoadHeadVector(b, j, h, vLocal);
                        // oLocal += p * vLocal
                        AscendC::Muls(vLocal, vLocal, (DTYPE_KV)p, Dh);
                        AscendC::Add(oLocal, oLocal, vLocal, Dh);
                        vQueue.FreeTensor(vLocal);
                    }

                    // Store output head slice
                    StoreHeadVector(b, iq, h, oLocal);
                    oQueue.FreeTensor(oLocal);
                }
            }
        }
    }

private:
    static constexpr uint32_t TMP_BUF = 8;

    __aicore__ inline void LoadQ(uint32_t b, uint32_t iq, uint32_t h)
    {
        AscendC::LocalTensor<DTYPE_Q> qLocal = qQueue.AllocTensor<DTYPE_Q>();
        uint32_t base = ((b * Lq + iq) * D) + h * Dh;
        AscendC::DataCopy(qLocal, qGm[base], Dh);
        qQueue.EnQue(qLocal);
    }

    __aicore__ inline float DotQK(uint32_t b, uint32_t j, uint32_t h)
    {
        // Dequeue qLocal prepared by LoadQ
        AscendC::LocalTensor<DTYPE_Q> qLocal = qQueue.DeQue<DTYPE_Q>();
        AscendC::LocalTensor<DTYPE_KV> kLocal = kQueue.AllocTensor<DTYPE_KV>();
        // Load K = kv[b, j, h, :]
        uint32_t baseK = ((b * Lkv + j) * D) + h * Dh;
        AscendC::DataCopy(kLocal, kvGm[baseK], Dh);

        // Vector multiply-accumulate to scalar
        // For brevity, use a simple reduction
        AscendC::LocalTensor<DTYPE_Q> mulLocal;
        AscendC::Mul(mulLocal, qLocal, kLocal, Dh);
        float sum = AscendC::ReduceSum(mulLocal, Dh);

        kQueue.FreeTensor(kLocal);
        qQueue.EnQue(qLocal); // push back qLocal for reuse
        return sum;
    }

    __aicore__ inline void LoadHeadVector(uint32_t b, uint32_t pos, uint32_t h, AscendC::LocalTensor<DTYPE_KV>& vLocal)
    {
        uint32_t baseV = ((b * Lkv + pos) * D) + h * Dh;
        AscendC::DataCopy(vLocal, kvGm[baseV], Dh);
    }

    __aicore__ inline void StoreHeadVector(uint32_t b, uint32_t iq, uint32_t h, AscendC::LocalTensor<DTYPE_O>& oLocal)
    {
        uint32_t baseOut = ((b * Lq + iq) * D) + h * Dh;
        AscendC::DataCopy(outGm[baseOut], oLocal, Dh);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM_Q> qQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM_KV> kQueue, vQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM_Q> oQueue;

    AscendC::GlobalTensor<DTYPE_Q> qGm;
    AscendC::GlobalTensor<DTYPE_KV> kvGm;
    AscendC::GlobalTensor<DTYPE_O> outGm;

    uint32_t B, Lq, Lkv, D, H, Dh;
    uint32_t tileNumQ, tileNumKV;
    float scale;

    uint32_t bBegin, bEnd;
};

extern "C" __global__ __aicore__ void cross_attention_custom(GM_ADDR q, GM_ADDR kv, GM_ADDR out, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCrossAttention op;
    op.Init(q, kv, out,
            tiling_data.batch, tiling_data.lenQ, tiling_data.lenKV,
            tiling_data.dModel, tiling_data.numHeads,
            tiling_data.scale, tiling_data.tileNumQ, tiling_data.tileNumKV);
    op.Process();
}

import custom_ops_lib

class ModelNew(torch.nn.Module):

    Assumes num_heads divides d_model, typical setting d_model=512, num_heads=8.
    """

        super().__init__()
        self.d_model = d_model
        self.num_heads = num_heads

        # q: [B, Lq, D], kv: [B, Lkv, D]
