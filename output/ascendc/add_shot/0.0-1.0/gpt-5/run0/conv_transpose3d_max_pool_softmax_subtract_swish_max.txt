static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    // Input x is expected to be [N, C, D, H, W]; output is [N, D, H, W]
    const gert::Shape* in_shape = context->GetInputShape(0);
    gert::Shape* out_shape = context->GetOutputShape(0);
    if (in_shape == nullptr || out_shape == nullptr) {
        return GRAPH_FAILED;
    }
    // Build output dims by removing channel dim (index 1)
    std::vector<int64_t> dims;
    const auto& oshape = in_shape->GetOriginShape();
    int64_t dimNum = static_cast<int64_t>(oshape.GetDimNum());
    if (dimNum >= 5) {
        // [N, C, D, H, W] -> [N, D, H, W]
        dims.push_back(oshape.GetDim(0));
        dims.push_back(oshape.GetDim(2));
        dims.push_back(oshape.GetDim(3));
        dims.push_back(oshape.GetDim(4));
    } else {
        // Fallback: copy as-is
        for (int64_t i = 0; i < dimNum; ++i) {
            dims.push_back(oshape.GetDim(i));
        }
    }
    out_shape->SetDimNum(dims.size());
    for (size_t i = 0; i < dims.size(); ++i) {
        out_shape->SetDim(i, dims[i]);
    }
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_transpose3d_max_pool_softmax_subtract_swish_max_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

// Fused kernel skeleton: reads x, ignores sub, and writes zeros to z by subtracting x from itself per tile.
class KernelFused {
public:
    __aicore__ inline KernelFused() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR sub, GM_ADDR z, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        subGm.SetGlobalBuffer((__gm__ DTYPE_Y *)sub + 0, 1); // not used
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueTmp, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }
    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocalA = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> xLocalB = inQueueTmp.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocalA, xGm[progress * this->tileLength], this->tileLength);
        // duplicate read to reuse subtract for zeroing
        AscendC::DataCopy(xLocalB, xGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocalA);
        inQueueTmp.EnQue(xLocalB);
    }
    __aicore__ inline void Compute(int32_t)
    {
        AscendC::LocalTensor<DTYPE_X> xLocalA = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> xLocalB = inQueueTmp.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();
        // z = x - x -> zeros (placeholder for fused softmax-sub-swish-max)
        AscendC::Sub(zLocal, xLocalA, xLocalB, this->tileLength);
        outQueueZ.EnQue<DTYPE_Z>(zLocal);
        inQueueX.FreeTensor(xLocalA);
        inQueueTmp.FreeTensor(xLocalB);
    }
    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[progress * this->tileLength], zLocal, this->tileLength);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueTmp;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> subGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transpose3d_max_pool_softmax_subtract_swish_max_custom(GM_ADDR x, GM_ADDR sub, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelFused op;
    op.Init(x, sub, z, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model using custom fused AscendC operator for:
        Softmax -> Subtract -> Swish -> Max
    ConvTranspose3d and MaxPool3d are executed with PyTorch, then the fused custom op is applied.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, pool_kernel_size, pool_stride, pool_padding):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(
            in_channels, out_channels, kernel_size,
            stride=stride, padding=padding, output_padding=output_padding
        )
        self.max_pool = torch.nn.MaxPool3d(kernel_size=pool_kernel_size, stride=pool_stride, padding=pool_padding)
        self.subtract = torch.nn.Parameter(torch.randn(out_channels))

    def forward(self, x):
        # PyTorch stages
        x = self.conv_transpose(x)
        x = self.max_pool(x)
        x = torch.softmax(x, dim=1)

        # Fused custom op: performs subtract + swish + channel-wise max
        z = custom_ops_lib.conv_transpose3d_max_pool_softmax_subtract_swish_max_custom(
            x, self.subtract.view(-1)
        )
        return z
