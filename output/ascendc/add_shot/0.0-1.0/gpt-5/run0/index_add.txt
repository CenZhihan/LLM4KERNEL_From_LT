#include "index_add_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
const uint32_t BLOCK_DIM = 1;
const uint32_t ROWS_PER_TILE = 4;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    IndexAddCustomTilingData tiling;

    const auto* x_shape = context->GetInputShape(0);
    const auto* idx_shape = context->GetInputShape(1);

    // Assume rank-2 for x and source, rank-1 for index
    uint32_t nRows = 1;
    uint32_t nCols = 1;
    if (x_shape->GetOriginShape().GetDimNum() >= 2) {
        nRows = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
        nCols = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    } else {
        // Fallback to flat
        nRows = 1;
        nCols = static_cast<uint32_t>(x_shape->GetOriginShape().GetShapeSize());
    }

    uint32_t kLength = static_cast<uint32_t>(idx_shape->GetOriginShape().GetShapeSize());

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_nRows(nRows);
    tiling.set_nCols(nCols);
    tiling.set_kLength(kLength);
    tiling.set_rowsPerTile(ROWS_PER_TILE);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* z_shape = context->GetOutputShape(0);
    *z_shape = *x_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "index_add_custom_tiling_data.h"

typedef float DTYPE_X;
typedef float DTYPE_SRC;
typedef float DTYPE_Z;
typedef int32_t DTYPE_INDEX;

class KernelIndexAdd {
public:
    __aicore__ inline KernelIndexAdd() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR index, GM_ADDR source, GM_ADDR z,
                                uint32_t nRows, uint32_t nCols, uint32_t kLength, uint32_t rowsPerTile)
    {
        this->nRows = nRows;
        this->nCols = nCols;
        this->kLength = kLength;
        this->rowsPerTile = rowsPerTile;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint64_t)nRows * nCols);
        idxGm.SetGlobalBuffer((__gm__ DTYPE_INDEX *)index, kLength);
        srcGm.SetGlobalBuffer((__gm__ DTYPE_SRC *)source, (uint64_t)kLength * nCols);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z, (uint64_t)nRows * nCols);

        // One buffer for indices tile, one for a single row vector, and one for z row
        pipe.InitBuffer(inQueueIdx, 1, rowsPerTile * sizeof(DTYPE_INDEX));
        pipe.InitBuffer(inQueueVal, 1, nCols * sizeof(DTYPE_SRC));
        pipe.InitBuffer(inoutQueueZ, 1, nCols * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        // Initialize output z with input x
        AscendC::DataCopy(zGm, xGm, (uint64_t)nRows * nCols);

        // Process updates in tiles over the index dimension
        for (uint32_t t = 0; t < kLength; t += rowsPerTile) {
            uint32_t cur = AscendC::min(rowsPerTile, kLength - t);

            // Load a tile of indices
            AscendC::LocalTensor<DTYPE_INDEX> idxLocal = inQueueIdx.AllocTensor<DTYPE_INDEX>();
            AscendC::DataCopy(idxLocal, idxGm[t], cur);
            inQueueIdx.EnQue(idxLocal);
            idxLocal = inQueueIdx.DeQue<DTYPE_INDEX>();

            // For each index in this tile, accumulate source row into z at the target row
            for (uint32_t i = 0; i < cur; ++i) {
                int32_t row = idxLocal.GetValue(i);
                // bounds check defensively
                if (row < 0 || (uint32_t)row >= nRows) {
                    continue;
                }

                // Load source row
                AscendC::LocalTensor<DTYPE_SRC> srcLocal = inQueueVal.AllocTensor<DTYPE_SRC>();
                AscendC::DataCopy(srcLocal, srcGm[(t + i) * nCols], nCols);

                // Load destination row from z
                AscendC::LocalTensor<DTYPE_Z> zLocal = inoutQueueZ.AllocTensor<DTYPE_Z>();
                AscendC::DataCopy(zLocal, zGm[row * nCols], nCols);

                // zLocal += srcLocal
                AscendC::Add(zLocal, zLocal, srcLocal, nCols);

                // Write back to z
                AscendC::DataCopy(zGm[row * nCols], zLocal, nCols);

                inoutQueueZ.FreeTensor(zLocal);
                inQueueVal.FreeTensor(srcLocal);
            }

            inQueueIdx.FreeTensor(idxLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueIdx;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueVal;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> inoutQueueZ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_INDEX> idxGm;
    AscendC::GlobalTensor<DTYPE_SRC> srcGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t nRows;
    uint32_t nCols;
    uint32_t kLength;
    uint32_t rowsPerTile;
};

extern "C" __global__ __aicore__ void index_add_custom(GM_ADDR x, GM_ADDR index, GM_ADDR source, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelIndexAdd op;
    op.Init(x, index, source, z,
            tiling_data.nRows, tiling_data.nCols, tiling_data.kLength, tiling_data.rowsPerTile);
    op.Process();
}
