kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "matmul_avg_pool_gelu_scale_max_custom_tiling_data.h"

// NOTE: This is a simplified reference kernel showing a fused pipeline.
// It computes for each batch element:
// y = max( scale * GELU( AvgPool1d( (x @ W^T) + b ) ), dim=features )

extern "C" __global__ __aicore__ void matmul_avg_pool_gelu_scale_max_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR scale, GM_ADDR pool, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    const uint32_t batch   = tiling_data.batch;
    const uint32_t inFeat  = tiling_data.inFeat;
    const uint32_t outFeat = tiling_data.outFeat;

    __gm__ float* x_gm = reinterpret_cast<__gm__ float*>(x);
    __gm__ float* w_gm = reinterpret_cast<__gm__ float*>(w);
    __gm__ float* b_gm = reinterpret_cast<__gm__ float*>(b);
    __gm__ float* scale_gm = reinterpret_cast<__gm__ float*>(scale);
    __gm__ float* pool_gm = reinterpret_cast<__gm__ float*>(pool);
    __gm__ float* y_gm = reinterpret_cast<__gm__ float*>(y);

    // read scalars from GM
    float scale_val = 1.0f;
    if (scale_gm != nullptr) {
        scale_val = scale_gm[0];
    } else {
        scale_val = tiling_data.scale;
    }
    uint32_t pool_kernel = 1;
    if (pool_gm != nullptr) {
        float pk = pool_gm[0];
        if (pk < 1.0f) pk = 1.0f;
        pool_kernel = static_cast<uint32_t>(pk);
    } else {
        pool_kernel = (tiling_data.poolKernel == 0) ? 1 : tiling_data.poolKernel;
    }
    if (pool_kernel == 0) pool_kernel = 1;

    // distribute batch across blocks
    const uint32_t block_num = AscendC::GetBlockNum();
    const uint32_t block_idx = AscendC::GetBlockIdx();
    uint32_t batch_per_block = (batch + block_num - 1) / block_num;
    uint32_t start_b = block_idx * batch_per_block;
    uint32_t end_b = (start_b + batch_per_block) > batch ? batch : (start_b + batch_per_block);

    for (uint32_t bidx = start_b; bidx < end_b; ++bidx) {
        // streaming computation without large temporaries
        float global_max = -INFINITY;
        uint32_t ofeat_idx = 0;

        float window_acc = 0.0f; // accumulate pooled window sum
        uint32_t window_count = 0;

        while (ofeat_idx < outFeat) {
            // compute one output feature of (x @ W^T) + b
            float acc = 0.0f;
            // bias
            acc += b_gm[ofeat_idx];

            // dot product x[bidx, :] with w[ofeat_idx, :]
            const __gm__ float* x_row = x_gm + static_cast<uint64_t>(bidx) * inFeat;
            const __gm__ float* w_row = w_gm + static_cast<uint64_t>(ofeat_idx) * inFeat;
            for (uint32_t k = 0; k < inFeat; ++k) {
                acc += x_row[k] * w_row[k];
            }

            // accumulate into pooling window
            window_acc += acc;
            window_count += 1;

            if (window_count == pool_kernel) {
                float avg = window_acc / static_cast<float>(pool_kernel);
                // GELU
                float gelu_out = AscendC::Gelu(avg);
                float scaled = gelu_out * scale_val;
                // update max across pooled features
                if (scaled > global_max) {
                    global_max = scaled;
                }
                // reset window
                window_acc = 0.0f;
                window_count = 0;
            }

            ++ofeat_idx;
        }

        // handle tail if outFeat not divisible by pool_kernel
        if (window_count > 0) {
            float avg = window_acc / static_cast<float>(window_count);
            float gelu_out = AscendC::Gelu(avg);
            float scaled = gelu_out * scale_val;
            if (scaled > global_max) {
                global_max = scaled;
            }
        }

        y_gm[bidx] = global_max;
    }
}

                                                          const at::Tensor& w,
                                                          const at::Tensor& b,
                                                          int64_t pool_kernel_size,

    TORCH_CHECK(x.is_cuda(), "x must be on NPU");
    TORCH_CHECK(w.is_cuda(), "w must be on NPU");
    TORCH_CHECK(b.is_cuda(), "b must be on NPU");

    auto out_sizes = std::vector<int64_t>{x.size(0)};

    pool_t.fill_(static_cast<float>(pool_kernel_size));

    return result;
