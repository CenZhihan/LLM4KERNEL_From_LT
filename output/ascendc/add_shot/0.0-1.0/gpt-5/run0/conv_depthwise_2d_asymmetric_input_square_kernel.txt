kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_depthwise_2d_asymmetric_input_square_kernel_custom_tiling_data.h"

// Define default data types
#ifndef DTYPE_X
#define DTYPE_X float
#endif
#ifndef DTYPE_W
#define DTYPE_W float
#endif
#ifndef DTYPE_B
#define DTYPE_B float
#endif
#ifndef DTYPE_Y
#define DTYPE_Y float
#endif

class KernelDWConv2D {
public:
    __aicore__ inline KernelDWConv2D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h_in, uint32_t w_in,
                                uint32_t h_out, uint32_t w_out, uint32_t k,
                                uint32_t stride, uint32_t padding, uint32_t hasBias)
    {
        this->n = n; this->c = c; this->h_in = h_in; this->w_in = w_in;
        this->h_out = h_out; this->w_out = w_out; this->k = k;
        this->stride = stride; this->padding = padding; this->hasBias = hasBias;

        // Split channels across blocks
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        this->c_per_block = c / blockNum;
        this->c_rem = c % blockNum;
        this->c_begin = blockIdx * this->c_per_block + AscendC::Min(blockIdx, this->c_rem);
        this->c_count = this->c_per_block + (blockIdx < this->c_rem ? 1U : 0U);

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint32_t)n * c * h_in * w_in);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W *)w, c * k * k);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, (uint32_t)n * c * h_out * w_out);
        if (hasBias) {
            bGm.SetGlobalBuffer((__gm__ DTYPE_B *)b, c);
        }

        pipe.InitBuffer(bufW, 1, k * k * sizeof(DTYPE_W));
        pipe.InitBuffer(bufXRow, 1, k * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        // Iterate over batch and assigned channels
        for (uint32_t ni = 0; ni < n; ++ni) {
            for (uint32_t ci = 0; ci < c_count; ++ci) {
                uint32_t c_idx = c_begin + ci;

                // Load weight for this channel into local buffer
                AscendC::LocalTensor<DTYPE_W> wLocal = bufW.AllocTensor<DTYPE_W>();
                // weight layout: [C, 1, K, K] => contiguous per channel K*K
                uint64_t w_off = (uint64_t)c_idx * (uint64_t)(k * k);
                AscendC::DataCopy(wLocal, wGm[w_off], k * k);

                DTYPE_Y biasVal = (DTYPE_Y)0;
                if (hasBias) {
                    AscendC::LocalTensor<DTYPE_B> bLocal;
                    // copy one element bias
                    // use row buffer as temp
                    AscendC::LocalTensor<DTYPE_X> tmp = bufXRow.AllocTensor<DTYPE_X>();
                    AscendC::DataCopy(tmp, bGm[c_idx], 1);
                    biasVal = (DTYPE_Y)tmp.GetValue(0);
                    bufXRow.FreeTensor(tmp);
                }

                // Compute per output pixel
                for (uint32_t ho = 0; ho < h_out; ++ho) {
                    int32_t in_h_base = (int32_t)ho * (int32_t)stride - (int32_t)padding;
                    for (uint32_t wo = 0; wo < w_out; ++wo) {
                        int32_t in_w_base = (int32_t)wo * (int32_t)stride - (int32_t)padding;

                        // Accumulate convolution
                        DTYPE_Y acc = biasVal;
                        for (uint32_t kh = 0; kh < k; ++kh) {
                            int32_t in_h = in_h_base + (int32_t)kh;
                            bool h_in_range = (in_h >= 0) && (in_h < (int32_t)h_in);
                            for (uint32_t kw = 0; kw < k; ++kw) {
                                int32_t in_w = in_w_base + (int32_t)kw;
                                DTYPE_X xval = (DTYPE_X)0;
                                if (h_in_range && (in_w >= 0) && (in_w < (int32_t)w_in)) {
                                    // Global offset for x: ((n*C + c) * H + h) * W + w
                                    uint64_t x_off = ((((uint64_t)ni * (uint64_t)c + (uint64_t)c_idx) * (uint64_t)h_in + (uint64_t)in_h) * (uint64_t)w_in + (uint64_t)in_w);
                                    // Copy 1 element to local tmp
                                    AscendC::LocalTensor<DTYPE_X> tmp = bufXRow.AllocTensor<DTYPE_X>();
                                    AscendC::DataCopy(tmp, xGm[x_off], 1);
                                    xval = tmp.GetValue(0);
                                    bufXRow.FreeTensor(tmp);
                                }
                                DTYPE_W wval = wLocal.GetValue(kh * k + kw);
                                acc = (DTYPE_Y)((DTYPE_Y)acc + (DTYPE_Y)((DTYPE_Y)xval * (DTYPE_Y)wval));
                            }
                        }

                        // Store output
                        uint64_t y_off = ((((uint64_t)ni * (uint64_t)c + (uint64_t)c_idx) * (uint64_t)h_out + (uint64_t)ho) * (uint64_t)w_out + (uint64_t)wo);
                        // write acc to GM via a local buffer of size 1
                        AscendC::LocalTensor<DTYPE_Y> tmpOut = bufXRow.AllocTensor<DTYPE_Y>();
                        tmpOut.SetValue(0, acc);
                        AscendC::DataCopy(yGm[y_off], tmpOut, 1);
                        bufXRow.FreeTensor(tmpOut);
                    }
                }
                bufW.FreeTensor(wLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TBuf<AscendC::BufferType::VECCALC> bufW;
    AscendC::TBuf<AscendC::BufferType::VECCALC> bufXRow;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n, c, h_in, w_in, h_out, w_out, k, stride, padding, hasBias;
    uint32_t c_begin, c_count, c_per_block, c_rem;
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_asymmetric_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelDWConv2D op;
    op.Init(x, w, b, y,
            tiling_data.n, tiling_data.c, tiling_data.h_in, tiling_data.w_in,
            tiling_data.h_out, tiling_data.w_out, tiling_data.k,
            tiling_data.stride, tiling_data.padding, tiling_data.hasBias);
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    """
    Depthwise 2D convolution with custom AscendC kernel.
    Only supports groups=in_channels and out_channels==in_channels (multiplier=1).
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super().__init__()
        assert out_channels == in_channels, "Depthwise multiplier=1 supported"
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding

        # weight shape [C, 1, K, K]
        w = torch.empty(out_channels, 1, kernel_size, kernel_size)
        nn.init.kaiming_uniform_(w, a=math.sqrt(5)) if hasattr(nn.init, "kaiming_uniform_") else nn.init.uniform_(w, -0.1, 0.1)
        self.weight = nn.Parameter(w)

        if bias:
            fan_in = in_channels * kernel_size * kernel_size
            bound = 1 / math.sqrt(fan_in)
            b = torch.empty(out_channels).uniform_(-bound, bound)
            self.bias = nn.Parameter(b)
        else:
            self.register_parameter("bias", None)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        b = self.bias if self.bias is not None else torch.empty(0, device=x.device, dtype=x.dtype)
        return custom_ops_lib.conv_depthwise_2d_asymmetric_input_square_kernel_custom(
            x, self.weight, b, self.stride, self.padding
        )
