static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    ThreeDimTensorMatrixMultiplicationCustomTilingData tiling;

    // Input 0: A -> (N, M, K)
    // Input 1: B -> (K, L)
    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);

    uint32_t N = static_cast<uint32_t>(a_shape->GetDim(0));
    uint32_t M = static_cast<uint32_t>(a_shape->GetDim(1));
    uint32_t K = static_cast<uint32_t>(a_shape->GetDim(2));
    uint32_t L = static_cast<uint32_t>(b_shape->GetDim(1));

    uint32_t rows = N * M;

    const uint32_t BLOCK_DIM = 8; // simple default; can be tuned
    context->SetBlockDim(BLOCK_DIM);

    uint32_t rowsPerBlock = (rows + BLOCK_DIM - 1) / BLOCK_DIM;
    // simple column tile size; can be tuned according to L2 size
    uint32_t tileL = L >= 256 ? 256 : L;

    tiling.set_N(N);
    tiling.set_M(M);
    tiling.set_K(K);
    tiling.set_L(L);
    tiling.set_rowsPerBlock(rowsPerBlock);
    tiling.set_tileL(tileL);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "three_dim_tensor_matrix_multiplication_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelBatchedMatMul {
public:
    __aicore__ inline KernelBatchedMatMul() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t N, uint32_t M, uint32_t K, uint32_t L,
                                uint32_t rowsPerBlock, uint32_t tileL)
    {
        this->N = N;
        this->M = M;
        this->K = K;
        this->L = L;
        this->rowsPerBlock = rowsPerBlock;
        this->rowsTotal = N * M;

        this->rowStart = AscendC::GetBlockIdx() * rowsPerBlock;
        this->rowEnd = this->rowStart + rowsPerBlock;
        if (this->rowEnd > this->rowsTotal) {
            this->rowEnd = this->rowsTotal;
        }
        this->assignedRows = (this->rowEnd > this->rowStart) ? (this->rowEnd - this->rowStart) : 0;

        this->tileL = tileL;
        if (this->tileL == 0) this->tileL = this->L;

        aGm.SetGlobalBuffer((__gm__ float *)A + this->rowStart * this->K, (this->assignedRows) * this->K);
        bGm.SetGlobalBuffer((__gm__ float *)B, this->K * this->L);
        cGm.SetGlobalBuffer((__gm__ float *)C + this->rowStart * this->L, (this->assignedRows) * this->L);

        pipe.InitBuffer(bQueue, BUFFER_NUM, this->tileL * sizeof(float));
        pipe.InitBuffer(outQueueC, BUFFER_NUM, this->tileL * sizeof(float));
        pipe.InitBuffer(aScalarQueue, 1, sizeof(float)); // scalar buffer
    }

    __aicore__ inline void Process()
    {
        if (assignedRows == 0) { return; }

        for (uint32_t r = 0; r < assignedRows; ++r) {
            // For each output row (corresponding to (n, m))
            uint32_t lProcessed = 0;
            while (lProcessed < L) {
                uint32_t curLen = AscendC::min(this->tileL, L - lProcessed);

                AscendC::LocalTensor<float> cLocal = outQueueC.AllocTensor<float>();
                // Initialize to zero
                AscendC::Dup(cLocal, 0.0f, curLen);

                // Accumulate over K
                for (uint32_t k = 0; k < K; ++k) {
                    // Load scalar a[r, k]
                    AscendC::LocalTensor<float> aLocal = aScalarQueue.AllocTensor<float>();
                    AscendC::DataCopy(aLocal, aGm[r * K + k], 1);
                    aScalarQueue.EnQue(aLocal);
                    aLocal = aScalarQueue.DeQue<float>();
                    float aScalar = aLocal.GetValue(0);

                    // Load B[k, lProcessed:lProcessed+curLen]
                    AscendC::LocalTensor<float> bLocal = bQueue.AllocTensor<float>();
                    AscendC::DataCopy(bLocal, bGm[k * L + lProcessed], curLen);
                    bQueue.EnQue(bLocal);

                    bLocal = bQueue.DeQue<float>();

                    // cLocal += aScalar * bLocal
                    AscendC::Muls(bLocal, bLocal, aScalar, curLen);
                    AscendC::Add(cLocal, cLocal, bLocal, curLen);

                    bQueue.FreeTensor(bLocal);
                    aScalarQueue.FreeTensor(aLocal);
                }

                outQueueC.EnQue<float>(cLocal);

                // Write back C[r, lProcessed:lProcessed+curLen]
                AscendC::LocalTensor<float> cWrite = outQueueC.DeQue<float>();
                AscendC::DataCopy(cGm[r * L + lProcessed], cWrite, curLen);
                outQueueC.FreeTensor(cWrite);

                lProcessed += curLen;
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> bQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> aScalarQueue;

    AscendC::GlobalTensor<float> aGm;
    AscendC::GlobalTensor<float> bGm;
    AscendC::GlobalTensor<float> cGm;

    uint32_t N, M, K, L;
    uint32_t rowsPerBlock;
    uint32_t rowsTotal;
    uint32_t rowStart, rowEnd, assignedRows;
    uint32_t tileL;
};

extern "C" __global__ __aicore__ void three_dim_tensor_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelBatchedMatMul op;
    op.Init(A, B, C,
            tiling_data.N, tiling_data.M, tiling_data.K, tiling_data.L,
            tiling_data.rowsPerBlock, tiling_data.tileL);
    op.Process();
}
