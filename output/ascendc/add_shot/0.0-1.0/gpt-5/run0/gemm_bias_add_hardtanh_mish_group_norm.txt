kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "gemm_bias_add_hardtanh_mish_group_norm_custom_tiling_data.h"

extern "C" __global__ __aicore__ void gemm_bias_add_hardtanh_mish_group_norm_custom(
    GM_ADDR x,
    GM_ADDR bias,
    GM_ADDR gamma,
    GM_ADDR beta,
    GM_ADDR num_groups,
    GM_ADDR eps,
    GM_ADDR hardtanh_min,
    GM_ADDR hardtanh_max,
    GM_ADDR y,
    GM_ADDR workspace,
    GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    // Cast GM pointers
    __gm__ float* xGm   = (__gm__ float*)x;
    __gm__ float* bGm   = (__gm__ float*)bias;
    __gm__ float* gGm   = (__gm__ float*)gamma;
    __gm__ float* btGm  = (__gm__ float*)beta;
    __gm__ int32_t* groupsGm = (__gm__ int32_t*)num_groups;
    __gm__ float* epsGm = (__gm__ float*)eps;
    __gm__ float* hminGm = (__gm__ float*)hardtanh_min;
    __gm__ float* hmaxGm = (__gm__ float*)hardtanh_max;
    __gm__ float* yGm   = (__gm__ float*)y;

    // Derive sizes
    uint32_t N = tiling_data.nSize;
    uint32_t total = tiling_data.totalLength;
    if (N == 0) { return; }
    uint32_t M = total / N;

    // Read scalars
    int32_t groups = groupsGm[0];
    if (groups <= 0) { groups = 1; }
    float eps_val = epsGm[0];
    float hmin = hminGm[0];
    float hmax = hmaxGm[0];

    // Per-block range (single block used in tiling)
    uint32_t blockIdx = AscendC::GetBlockIdx();
    uint32_t blockNum = AscendC::GetBlockNum();
    uint32_t rowsPerBlock = (M + blockNum - 1) / blockNum;
    uint32_t rowStart = blockIdx * rowsPerBlock;
    uint32_t rowEnd = AscendC::Min(rowStart + rowsPerBlock, M);

    uint32_t groupSize = N / (uint32_t)groups;
    if (groupSize == 0) { groupSize = N; groups = 1; }

    for (uint32_t m = rowStart; m < rowEnd; ++m) {
        // Process each group independently
        for (int32_t g = 0; g < groups; ++g) {
            uint32_t cStart = g * groupSize;
            uint32_t cEnd = (g == groups - 1) ? N : (cStart + groupSize);

            // First pass: apply bias + hardtanh + mish and accumulate stats
            float mean = 0.0f;
            float sqmean = 0.0f;

            for (uint32_t c = cStart; c < cEnd; ++c) {
                uint32_t idx = m * N + c;
                float v = xGm[idx] + bGm[c];
                // Hardtanh
                v = v < hmin ? hmin : (v > hmax ? hmax : v);
                // Mish: v * tanh(softplus(v)) with softplus = log(1 + exp(v))
                float sp = log1pf(expf(v));
                float mish_v = v * tanhf(sp);
                // Temporarily write back mish_v to y as buffer for second pass
                yGm[idx] = mish_v;
                mean += mish_v;
                sqmean += mish_v * mish_v;
            }

            uint32_t cnt = (cEnd - cStart);
            float invCnt = cnt > 0 ? (1.0f / (float)cnt) : 1.0f;
            mean *= invCnt;
            float var = sqmean * invCnt - mean * mean;
            if (var < 0.0f) var = 0.0f;
            float invStd = rsqrtf(var + eps_val);

            // Second pass: GroupNorm affine
            for (uint32_t c = cStart; c < cEnd; ++c) {
                uint32_t idx = m * N + c;
                float normed = (yGm[idx] - mean) * invStd;
                float outv = normed * gGm[c] + btGm[c];
                yGm[idx] = outv;
            }
        }
    }
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model using a custom AscendC kernel to fuse BiasAdd + Hardtanh + Mish + GroupNorm
    after the GEMM (Linear) operation.
    """
    def __init__(self, in_features, out_features, bias_shape, num_groups):
        super(ModelNew, self).__init__()
        self.gemm = torch.nn.Linear(in_features, out_features)
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))
        self.h_min = -1.0
        self.h_max = 1.0
        self.mish = torch.nn.Mish()  # kept only to align attributes; not used directly
        self.num_groups = num_groups
        self.groupnorm = torch.nn.GroupNorm(num_groups=num_groups, num_channels=out_features)

    def forward(self, x):
        # GEMM (with its own bias) is computed by PyTorch module
        x = self.gemm(x)
        # Fused custom kernel applies: +extra bias -> Hardtanh -> Mish -> GroupNorm
        x = custom_ops_lib.gemm_bias_add_hardtanh_mish_group_norm_custom(
            x,
            self.bias,
            self.groupnorm.weight,
            self.groupnorm.bias,
            self.num_groups,
            1e-5,
            self.h_min,
            self.h_max
        )
        return x
'''
