kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_standard_2d_square_input_square_kernel_custom_tiling_data.h"

// This is a simplified illustrative Conv2D kernel specialized to square inputs/kernels and fixed stride/padding.
// It partitions output channels across blocks and iterates directly on GM.
// For demonstration purposes.

constexpr int32_t BUFFER_NUM = 1;

class KernelConv2D {
public:
    __aicore__ inline KernelConv2D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t K, uint32_t kernel, uint32_t stride, uint32_t pad,
                                uint32_t outH, uint32_t outW)
    {
        // Bind global buffers
        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, (uint64_t)N * C * H * W);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W*)w, (uint64_t)K * C * kernel * kernel);
        if (b != nullptr) {
            bGm.SetGlobalBuffer((__gm__ DTYPE_B*)b, (uint64_t)K);
            hasBias = true;
        } else {
            hasBias = false;
        }
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y, (uint64_t)N * K * outH * outW);

        this->N = N; this->C = C; this->H = H; this->W = W;
        this->K = K; this->k = kernel; this->s = stride; this->p = pad;
        this->outH = outH; this->outW = outW;

        // Partition output channels across blocks
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t idx = AscendC::GetBlockIdx();
        ocStart = (K / blocks) * idx + ((idx < (K % blocks)) ? idx : (K % blocks));
        ocLen   = (K / blocks) + ((idx < (K % blocks)) ? 1 : 0);
    }

    __aicore__ inline void Process()
    {
        // Local scalar buffers
        for (uint32_t n = 0; n < N; ++n) {
            for (uint32_t oc = 0; oc < ocLen; ++oc) {
                uint32_t outC = ocStart + oc;
                for (uint32_t oh = 0; oh < outH; ++oh) {
                    for (uint32_t ow = 0; ow < outW; ++ow) {
                        float acc = 0.0f;
                        if (hasBias) {
                            // Load bias[ outC ]
                            // DataCopy a single value to local buffer
                            AscendC::LocalTensor<DTYPE_B> bLocal;
                            bLocal.SetSize(1);
                            AscendC::DataCopy(bLocal, bGm[outC], 1);
                            acc = static_cast<float>(bLocal.GetValue(0));
                        }
                        for (uint32_t ic = 0; ic < C; ++ic) {
                            for (uint32_t kh = 0; kh < k; ++kh) {
                                int32_t ih = (int32_t)oh * (int32_t)s - (int32_t)p + (int32_t)kh;
                                if (ih < 0 || ih >= (int32_t)H) continue;
                                for (uint32_t kw = 0; kw < k; ++kw) {
                                    int32_t iw = (int32_t)ow * (int32_t)s - (int32_t)p + (int32_t)kw;
                                    if (iw < 0 || iw >= (int32_t)W) continue;

                                    uint64_t xIndex = (((uint64_t)n * C + ic) * H + (uint32_t)ih) * W + (uint32_t)iw;
                                    uint64_t wIndex = ((((uint64_t)outC) * C + ic) * k + kh) * k + kw;

                                    AscendC::LocalTensor<DTYPE_X> xLocal;
                                    AscendC::LocalTensor<DTYPE_W> wLocal;
                                    xLocal.SetSize(1);
                                    wLocal.SetSize(1);
                                    AscendC::DataCopy(xLocal, xGm[xIndex], 1);
                                    AscendC::DataCopy(wLocal, wGm[wIndex], 1);

                                    float xv = static_cast<float>(xLocal.GetValue(0));
                                    float wv = static_cast<float>(wLocal.GetValue(0));
                                    acc += xv * wv;
                                }
                            }
                        }
                        uint64_t yIndex = (((uint64_t)n * K + outC) * outH + oh) * outW + ow;
                        AscendC::LocalTensor<DTYPE_Y> yLocal;
                        yLocal.SetSize(1);
                        yLocal.SetValue(0, static_cast<DTYPE_Y>(acc));
                        AscendC::DataCopy(yGm[yIndex], yLocal, 1);
                    }
                }
            }
        }
    }

private:
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t N, C, H, W;
    uint32_t K, k, s, p;
    uint32_t outH, outW;
    uint32_t ocStart, ocLen;
    bool hasBias;
};

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2D op;
    op.Init(x, w, b, y,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.K, tiling_data.kernel, tiling_data.stride, tiling_data.pad,
            tiling_data.outH, tiling_data.outW);
    op.Process();
}
