kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv2d_scaling_min_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelChannelMin {
public:
    __aicore__ inline KernelChannelMin() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t N, uint32_t C, uint32_t H, uint32_t W, uint32_t tileNum)
    {
        this->N = N;
        this->C = C;
        this->H = H;
        this->W = W;
        this->planeSize = H * W;
        this->tileNum = tileNum;
        if (this->tileNum == 0) { this->tileNum = 1; }
        this->tileLen = (this->planeSize + this->tileNum - 1) / this->tileNum;

        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t nPerBlock = (N + blocks - 1) / blocks;
        this->nStart = AscendC::GetBlockIdx() * nPerBlock;
        if (this->nStart > N) this->nStart = N;
        this->nEnd = this->nStart + nPerBlock;
        if (this->nEnd > N) this->nEnd = N;

        // Set GM buffers to the slice this block handles
        xGm.SetGlobalBuffer((__gm__ float *)x + (uint64_t)this->nStart * C * this->planeSize,
                            (uint64_t)(this->nEnd - this->nStart) * C * this->planeSize);
        yGm.SetGlobalBuffer((__gm__ float *)y + (uint64_t)this->nStart * this->planeSize,
                            (uint64_t)(this->nEnd - this->nStart) * this->planeSize);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLen * sizeof(float));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLen * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (this->nStart >= this->nEnd) { return; }
        for (uint32_t n = this->nStart; n < this->nEnd; ++n) {
            for (uint32_t t = 0; t < this->tileNum; ++t) {
                uint32_t tileOffset = t * this->tileLen;
                uint32_t curLen = this->planeSize > tileOffset ? AscendC::Min((uint32_t)this->tileLen, (uint32_t)(this->planeSize - tileOffset)) : 0;
                if (curLen == 0) { continue; }

                // Accumulator buffer for current tile
                AscendC::LocalTensor<float> acc = outQueueY.AllocTensor<float>();

                for (uint32_t c = 0; c < this->C; ++c) {
                    AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
                    // Offset inside the local GM window
                    uint64_t gmOffsetX = ((uint64_t)(n - this->nStart) * this->C + c) * this->planeSize + tileOffset;
                    AscendC::DataCopy(xLocal, xGm[gmOffsetX], curLen);

                    if (c == 0) {
                        AscendC::DataCopy(acc, xLocal, curLen);
                    } else {
                        AscendC::Min(acc, acc, xLocal, curLen);
                    }

                    inQueueX.EnQue(xLocal);
                    AscendC::LocalTensor<float> xUsed = inQueueX.DeQue<float>();
                    inQueueX.FreeTensor(xUsed);
                }

                outQueueY.EnQue(acc);
                AscendC::LocalTensor<float> yLocal = outQueueY.DeQue<float>();
                uint64_t gmOffsetY = ((uint64_t)(n - this->nStart)) * this->planeSize + tileOffset;
                AscendC::DataCopy(yGm[gmOffsetY], yLocal, curLen);
                outQueueY.FreeTensor(yLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t N, C, H, W;
    uint32_t planeSize;
    uint32_t tileNum;
    uint32_t tileLen;
    uint32_t nStart;
    uint32_t nEnd;
};

extern "C" __global__ __aicore__ void conv2d_scaling_min_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelChannelMin op;
    op.Init(x, y,
            tiling_data.N,
            tiling_data.C,
            tiling_data.H,
            tiling_data.W,
            tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model that keeps convolution and scaling in PyTorch,
    and replaces channel-wise min reduction with a custom AscendC op.
    """
    def __init__(self, in_channels, out_channels, kernel_size, scale_factor):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)
        self.scale_factor = scale_factor

    def forward(self, x):
        x = self.conv(x)
        x = x * self.scale_factor
        # Replace torch.min(x, dim=1, keepdim=True)[0]
        x = custom_ops_lib.conv2d_scaling_min_custom(x)
        return x
'''
