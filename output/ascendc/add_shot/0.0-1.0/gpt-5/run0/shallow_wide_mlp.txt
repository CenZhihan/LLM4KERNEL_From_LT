namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    ShallowWideMlpCustomTilingData tiling;

    auto x_shape = context->GetInputShape(0)->GetOriginShape();
    auto w_shape = context->GetInputShape(1)->GetOriginShape();
    auto b_shape = context->GetInputShape(2)->GetOriginShape();

    // x: [M, K], w: [K, N], b: [N]
    uint32_t m = (x_shape.GetDimNum() == 2) ? static_cast<uint32_t>(x_shape.GetDim(0)) : 1;
    uint32_t k_x = (x_shape.GetDimNum() >= 1) ? static_cast<uint32_t>(x_shape.GetDim(x_shape.GetDimNum()-1)) : 1;
    uint32_t k_w = (w_shape.GetDimNum() >= 1) ? static_cast<uint32_t>(w_shape.GetDim(0)) : 1;
    uint32_t n   = (w_shape.GetDimNum() >= 2) ? static_cast<uint32_t>(w_shape.GetDim(1)) : ((b_shape.GetDimNum() >= 1) ? static_cast<uint32_t>(b_shape.GetDim(0)) : 1);
    (void)k_w; // assume consistent

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_m(m);
    tiling.set_k(k_x);
    tiling.set_n(n);
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0; // no workspace
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);
    gert::Shape* z_shape = context->GetOutputShape(0);

    // Infer output as [M, N]
    gert::Shape out;
    if (x_shape->GetDimNum() == 2 && w_shape->GetDimNum() == 2) {
        out.SetDimNum(2);
        out.SetDim(0, x_shape->GetDim(0));
        out.SetDim(1, w_shape->GetDim(1));
    } else {
        // Fallback: treat last dim as K and output last dim as N
        out = *x_shape;
        if (w_shape->GetDimNum() >= 2) {
            if (out.GetDimNum() == 0) out.SetDimNum(2);
            out.SetDim(out.GetDimNum()-1, w_shape->GetDim(1));
        }
    }
    *z_shape = out;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "shallow_wide_mlp_custom_tiling_data.h"

extern "C" __global__ __aicore__ void shallow_wide_mlp_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t M = tiling_data.m;
    const uint32_t K = tiling_data.k;
    const uint32_t N = tiling_data.n;

    __gm__ float* x_gm = reinterpret_cast<__gm__ float*>(x);
    __gm__ float* w_gm = reinterpret_cast<__gm__ float*>(w);
    __gm__ float* b_gm = reinterpret_cast<__gm__ float*>(b);
    __gm__ float* z_gm = reinterpret_cast<__gm__ float*>(z);

    uint32_t blockNum = AscendC::GetBlockNum();
    uint32_t blockIdx = AscendC::GetBlockIdx();

    // simple row partition across blocks
    uint32_t rowsPerBlock = (M + blockNum - 1) / blockNum;
    uint32_t rowStart = blockIdx * rowsPerBlock;
    uint32_t rowEnd = rowStart + rowsPerBlock;
    if (rowEnd > M) rowEnd = M;

    for (uint32_t i = rowStart; i < rowEnd; ++i) {
        __gm__ float* x_row = x_gm + static_cast<uint64_t>(i) * K;
        __gm__ float* z_row = z_gm + static_cast<uint64_t>(i) * N;

        for (uint32_t j = 0; j < N; ++j) {
            float acc = b_gm[j];
            // dot product x_row (1xK) with w (KxN) column j
            for (uint32_t k = 0; k < K; ++k) {
                float xv = x_row[k];
                float wv = w_gm[static_cast<uint64_t>(k) * N + j];
                acc += xv * wv;
            }
            z_row[j] = acc;
        }
    }
}

model_src='''
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    def __init__(self, input_size, hidden_layer_sizes, output_size):
        super().__init__()
        layers = []
        current_input_size = input_size

        for hidden_size in hidden_layer_sizes:
            layers.append(nn.Linear(current_input_size, hidden_size))
            layers.append(nn.ReLU())
            current_input_size = hidden_size

        layers.append(nn.Linear(current_input_size, output_size))
        self.network = nn.Sequential(*layers)

    def forward(self, x):
        # Compute first (L-1) layers using PyTorch
        h = x
        # Expecting structure: [Linear, ReLU, Linear, ReLU, Linear]
        h = self.network[0](h)
        h = self.network[1](h)
        h = self.network[2](h)
        h = self.network[3](h)
        # Final linear via custom AscendC kernel
        last_linear = self.network[4]
        # Transpose weight to [in_features, out_features] as kernel expects (K, N)
        w = last_linear.weight.t().contiguous()
        b = last_linear.bias
        out = custom_ops_lib.shallow_wide_mlp_custom(h, w, b)
        return out
'''
