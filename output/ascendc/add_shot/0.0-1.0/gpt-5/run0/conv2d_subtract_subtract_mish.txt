kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv2d_subtract_subtract_mish_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelConv2dSubtractSubtractMish {
public:
    __aicore__ inline KernelConv2dSubtractSubtractMish() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR sub1, GM_ADDR sub2, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ float *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        s1Gm.SetGlobalBuffer((__gm__ float *)sub1 + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        s2Gm.SetGlobalBuffer((__gm__ float *)sub2 + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ float *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueS1, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueS2, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::LocalTensor<float> s1Local = inQueueS1.AllocTensor<float>();
        AscendC::LocalTensor<float> s2Local = inQueueS2.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(s1Local, s1Gm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(s2Local, s2Gm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
        inQueueS1.EnQue(s1Local);
        inQueueS2.EnQue(s2Local);
    }

    __aicore__ inline void Compute(int32_t progress)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();   // will be reused as temp buffer
        AscendC::LocalTensor<float> s1Local = inQueueS1.DeQue<float>(); // can be reused as constant 1.0 buffer
        AscendC::LocalTensor<float> s2Local = inQueueS2.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();

        // y = ((x - sub1) - sub2)
        AscendC::Sub(yLocal, xLocal, s1Local, this->tileLength);
        AscendC::Sub(yLocal, yLocal, s2Local, this->tileLength);

        // xLocal = exp(y)
        AscendC::Exp(xLocal, yLocal, this->tileLength);

        // s1Local = 1.0
        AscendC::Duplicate(s1Local, static_cast<float>(1.0f), this->tileLength);

        // xLocal = log(1 + exp(y))
        AscendC::Add(xLocal, xLocal, s1Local, this->tileLength);
        AscendC::Log(xLocal, xLocal, this->tileLength);

        // xLocal = tanh(softplus)
        AscendC::Tanh(xLocal, xLocal, this->tileLength);

        // y = y * xLocal (mish)
        AscendC::Mul(yLocal, yLocal, xLocal, this->tileLength);

        outQueueY.EnQue<float>(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueS1.FreeTensor(s1Local);
        inQueueS2.FreeTensor(s2Local);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY.DeQue<float>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueS1, inQueueS2;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> s1Gm;
    AscendC::GlobalTensor<float> s2Gm;
    AscendC::GlobalTensor<float> yGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv2d_subtract_subtract_mish_custom(GM_ADDR x, GM_ADDR sub1, GM_ADDR sub2, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dSubtractSubtractMish op;
    op.Init(x, sub1, sub2, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: keeps Conv2d in PyTorch, fuses two subtractions + Mish into a custom AscendC kernel.
    """
    def __init__(self, in_channels, out_channels, kernel_size, subtract_value_1, subtract_value_2):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)
        self.subtract_value_1 = float(subtract_value_1)
        self.subtract_value_2 = float(subtract_value_2)

    def forward(self, x):
        x = self.conv(x)
        # Create tensors matching x for the custom op (no broadcasting inside kernel)
        sub1 = torch.full_like(x, self.subtract_value_1)
        sub2 = torch.full_like(x, self.subtract_value_2)
        x = custom_ops_lib.conv2d_subtract_subtract_mish_custom(x, sub1, sub2)
        return x
