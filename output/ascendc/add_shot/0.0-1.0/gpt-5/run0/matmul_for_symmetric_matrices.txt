kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "matmul_for_symmetric_matrices_custom_tiling_data.h"

constexpr int32_t IN_BUFFER_NUM = 2;
constexpr int32_t OUT_BUFFER_NUM = 1;

class KernelMatmulSym {
public:
    __aicore__ inline KernelMatmulSym() {}
    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t m, uint32_t n, uint32_t k,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK) {
        this->M = m;
        this->N = n;
        this->K = k;
        this->tileM = tileM;
        this->tileN = tileN;
        this->tileK = tileK;

        // Work partition: split rows by blocks
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        this->rowsPerBlock = (M + blockNum - 1) / blockNum;
        this->rowStart = rowsPerBlock * blockIdx;
        if (this->rowStart > M) this->rowStart = M;
        this->rowEnd = AscendC::min(this->rowStart + this->rowsPerBlock, this->M);
        this->workRows = (this->rowEnd > this->rowStart) ? (this->rowEnd - this->rowStart) : 0;

        // Set GM buffers
        aGm.SetGlobalBuffer((__gm__ DTYPE_A *)a + static_cast<int64_t>(this->rowStart) * this->K,
                            static_cast<int64_t>(this->workRows) * this->K);
        bGm.SetGlobalBuffer((__gm__ DTYPE_B *)b, static_cast<int64_t>(this->K) * this->N);
        cGm.SetGlobalBuffer((__gm__ DTYPE_C *)c + static_cast<int64_t>(this->rowStart) * this->N,
                            static_cast<int64_t>(this->workRows) * this->N);

        // Init local buffers for tiles
        pipe.InitBuffer(aQueue, IN_BUFFER_NUM, static_cast<uint32_t>(this->tileM * this->tileK * sizeof(DTYPE_A)));
        pipe.InitBuffer(bQueue, IN_BUFFER_NUM, static_cast<uint32_t>(this->tileK * this->tileN * sizeof(DTYPE_B)));
        pipe.InitBuffer(cQueue, OUT_BUFFER_NUM, static_cast<uint32_t>(this->tileM * this->tileN * sizeof(DTYPE_C)));
    }

    __aicore__ inline void Process() {
        if (workRows == 0) {
            return;
        }
        // Iterate over output tiles within this block's row range
        for (uint32_t r0 = 0; r0 < workRows; r0 += tileM) {
            uint32_t curM = AscendC::min(tileM, workRows - r0);
            for (uint32_t c0 = 0; c0 < N; c0 += tileN) {
                uint32_t curN = AscendC::min(tileN, N - c0);

                // Allocate and zero local C tile
                AscendC::LocalTensor<DTYPE_C> cLocal = cQueue.AllocTensor<DTYPE_C>();
                // Zero initialize
                for (uint32_t i = 0; i < curM * curN; ++i) {
                    cLocal[i] = static_cast<DTYPE_C>(0);
                }

                // Loop over K dimension tiles
                for (uint32_t k0 = 0; k0 < K; k0 += tileK) {
                    uint32_t curK = AscendC::min(tileK, K - k0);

                    // Copy A tile (curM x curK)
                    AscendC::LocalTensor<DTYPE_A> aLocal = aQueue.AllocTensor<DTYPE_A>();
                    for (uint32_t r = 0; r < curM; ++r) {
                        // Global row index in A relative to aGm base
                        uint32_t gRow = r0 + r;
                        AscendC::DataCopy(aLocal[r * tileK], aGm[gRow * K + k0], curK);
                    }
                    aQueue.EnQue(aLocal);

                    // Copy B tile (curK x curN)
                    AscendC::LocalTensor<DTYPE_B> bLocal = bQueue.AllocTensor<DTYPE_B>();
                    for (uint32_t kk = 0; kk < curK; ++kk) {
                        AscendC::DataCopy(bLocal[kk * tileN], bGm[(k0 + kk) * N + c0], curN);
                    }
                    bQueue.EnQue(bLocal);

                    // DeQue for compute
                    aLocal = aQueue.DeQue<DTYPE_A>();
                    bLocal = bQueue.DeQue<DTYPE_B>();

                    // Naive GEMM on tile: cLocal[curM x curN] += aLocal[curM x curK] * bLocal[curK x curN]
                    for (uint32_t r = 0; r < curM; ++r) {
                        for (uint32_t kk = 0; kk < curK; ++kk) {
                            DTYPE_A a_val = aLocal[r * tileK + kk];
                            for (uint32_t c = 0; c < curN; ++c) {
                                cLocal[r * tileN + c] += static_cast<DTYPE_C>(a_val) * static_cast<DTYPE_C>(bLocal[kk * tileN + c]);
                            }
                        }
                    }

                    // Free consumed tiles
                    aQueue.FreeTensor(aLocal);
                    bQueue.FreeTensor(bLocal);
                }

                // Write back C tile
                for (uint32_t r = 0; r < curM; ++r) {
                    uint32_t gRow = r0 + r;
                    AscendC::DataCopy(cGm[gRow * N + c0], cLocal[r * tileN], curN);
                }
                cQueue.FreeTensor(cLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUFFER_NUM> aQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUFFER_NUM> bQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUFFER_NUM> cQueue;

    AscendC::GlobalTensor<DTYPE_A> aGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_C> cGm;

    uint32_t M;
    uint32_t N;
    uint32_t K;
    uint32_t tileM;
    uint32_t tileN;
    uint32_t tileK;

    uint32_t rowsPerBlock;
    uint32_t rowStart;
    uint32_t rowEnd;
    uint32_t workRows;
};

extern "C" __global__ __aicore__ void matmul_for_symmetric_matrices_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulSym op;
    op.Init(a, b, c,
            tiling_data.m, tiling_data.n, tiling_data.k,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK);
    op.Process();
}
