kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "gemm_relu_divide_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmReluDivide {
public:
    __aicore__ inline KernelGemmReluDivide() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR divisor, GM_ADDR z,
                                uint32_t M, uint32_t N, uint32_t K, uint32_t tileM, uint32_t tileN)
    {
        this->M = M;
        this->N = N;
        this->K = K;
        this->tileM = tileM;
        this->tileN = tileN;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint32_t)(M * K));
        wGm.SetGlobalBuffer((__gm__ DTYPE_W *)w, (uint32_t)(N * K));
        bGm.SetGlobalBuffer((__gm__ DTYPE_B *)b, (uint32_t)N);
        divGm.SetGlobalBuffer((__gm__ DTYPE_DIV *)divisor, (uint32_t)1);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z, (uint32_t)(M * N));

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileM * this->K * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueW, BUFFER_NUM, this->tileN * this->K * sizeof(DTYPE_W));
        pipe.InitBuffer(inQueueB, 1, this->tileN * sizeof(DTYPE_B));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileM * this->tileN * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        uint32_t coreNum = AscendC::GetBlockNum();
        uint32_t coreIdx = AscendC::GetBlockIdx();

        uint32_t rowsPerCore = (M + coreNum - 1) / coreNum;
        uint32_t rowStart = coreIdx * rowsPerCore;
        uint32_t rowEnd = AscendC::Min(rowStart + rowsPerCore, M);

        for (uint32_t m0 = rowStart; m0 < rowEnd; m0 += tileM) {
            uint32_t curTileM = AscendC::Min(tileM, rowEnd - m0);
            for (uint32_t n0 = 0; n0 < N; n0 += tileN) {
                uint32_t curTileN = AscendC::Min(tileN, N - n0);

                AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_W> wLocal = inQueueW.AllocTensor<DTYPE_W>();
                AscendC::LocalTensor<DTYPE_B> bLocal = inQueueB.AllocTensor<DTYPE_B>();
                AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();

                // Copy X[m0:m0+curTileM, :]
                for (uint32_t mi = 0; mi < curTileM; ++mi) {
                    AscendC::DataCopy(xLocal[mi * K], xGm[(m0 + mi) * K], K);
                }
                // Copy W[n0:n0+curTileN, :]
                for (uint32_t ni = 0; ni < curTileN; ++ni) {
                    AscendC::DataCopy(wLocal[ni * K], wGm[(n0 + ni) * K], K);
                }
                // Copy Bias segment
                AscendC::DataCopy(bLocal, bGm[n0], curTileN);

                // Initialize output tile to zero
                AscendC::SetVector(zLocal, (DTYPE_Z)0, curTileM * curTileN);

                // GEMM: Z_tile = X_tile * W_tile^T
                // Accumulate over K
                for (uint32_t kk = 0; kk < K; ++kk) {
                    for (uint32_t mi = 0; mi < curTileM; ++mi) {
                        DTYPE_X xv = xLocal[mi * K + kk];
                        for (uint32_t ni = 0; ni < curTileN; ++ni) {
                            DTYPE_W wv = wLocal[ni * K + kk];
                            zLocal[mi * curTileN + ni] += (DTYPE_Z)(xv * wv);
                        }
                    }
                }

                // Add bias
                for (uint32_t mi = 0; mi < curTileM; ++mi) {
                    for (uint32_t ni = 0; ni < curTileN; ++ni) {
                        zLocal[mi * curTileN + ni] += (DTYPE_Z)bLocal[ni];
                    }
                }

                // Load divisor
                DTYPE_DIV divScalar = divGm[0];

                // ReLU and divide
                for (uint32_t idx = 0; idx < curTileM * curTileN; ++idx) {
                    DTYPE_Z v = zLocal[idx];
                    v = v > (DTYPE_Z)0 ? v : (DTYPE_Z)0;
                    zLocal[idx] = v / (DTYPE_Z)divScalar;
                }

                outQueueZ.EnQue<DTYPE_Z>(zLocal);
                inQueueX.EnQue<DTYPE_X>(xLocal);
                inQueueW.EnQue<DTYPE_W>(wLocal);
                inQueueB.EnQue<DTYPE_B>(bLocal);

                AscendC::LocalTensor<DTYPE_Z> zOut = outQueueZ.DeQue<DTYPE_Z>();
                // Store result to GM
                for (uint32_t mi = 0; mi < curTileM; ++mi) {
                    AscendC::DataCopy(zGm[(m0 + mi) * N + n0], zOut[mi * curTileN], curTileN);
                }
                outQueueZ.FreeTensor(zOut);
                inQueueX.FreeTensor(xLocal);
                inQueueW.FreeTensor(wLocal);
                inQueueB.FreeTensor(bLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueW;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_DIV> divGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t M;
    uint32_t N;
    uint32_t K;
    uint32_t tileM;
    uint32_t tileN;
};

extern "C" __global__ __aicore__ void gemm_relu_divide_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR divisor, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmReluDivide op;
    op.Init(x, w, b, divisor, z,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused custom AscendC op: Y = ReLU(X @ W^T + b) / divisor
    """
    def __init__(self, in_features, out_features, divisor):
        super(ModelNew, self).__init__()
        self.linear = torch.nn.Linear(in_features, out_features)
        self.divisor = float(divisor)

    def forward(self, x):
        div_tensor = torch.tensor(self.divisor, dtype=x.dtype, device=x.device)
        return custom_ops_lib.gemm_relu_divide_custom(x, self.linear.weight, self.linear.bias, div_tensor)
