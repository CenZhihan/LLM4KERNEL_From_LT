kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "gru_bidirectional_hidden_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGruHiddenCopy {
public:
    __aicore__ inline KernelGruHiddenCopy() {}
    __aicore__ inline void Init(GM_ADDR hidden_in, GM_ADDR hidden_out, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        inGm.SetGlobalBuffer((__gm__ DTYPE_X *)hidden_in + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        outGm.SetGlobalBuffer((__gm__ DTYPE_Y *)hidden_out + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> inLocal = inQueue.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(inLocal, inGm[progress * this->tileLength], this->tileLength);
        inQueue.EnQue(inLocal);
    }

    __aicore__ inline void Compute(int32_t)
    {
        AscendC::LocalTensor<DTYPE_X> inLocal = inQueue.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> outLocal = outQueue.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(outLocal, inLocal, this->tileLength);
        outQueue.EnQue(outLocal);
        inQueue.FreeTensor(inLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> outLocal = outQueue.DeQue<DTYPE_Y>();
        AscendC::DataCopy(outGm[progress * this->tileLength], outLocal, this->tileLength);
        outQueue.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue;
    AscendC::GlobalTensor<DTYPE_X> inGm;
    AscendC::GlobalTensor<DTYPE_Y> outGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void gru_bidirectional_hidden_custom(GM_ADDR hidden_in, GM_ADDR hidden_out, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGruHiddenCopy op;
    op.Init(hidden_in, hidden_out, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}
"""
