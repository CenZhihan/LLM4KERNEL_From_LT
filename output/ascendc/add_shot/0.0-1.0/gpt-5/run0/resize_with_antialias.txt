#include "resize_with_antialias_custom_tiling.h"

namespace {
constexpr uint32_t TARGET_OUT_H = 128;
constexpr uint32_t TARGET_OUT_W = 128;
}

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    ResizeWithAntialiasCustomTilingData tiling;
    auto* inShape = context->GetInputShape(0);
    // Expect 4D NCHW
    uint32_t n = inShape->GetDim(0);
    uint32_t c = inShape->GetDim(1);
    uint32_t h = inShape->GetDim(2);
    uint32_t w = inShape->GetDim(3);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_inH(h);
    tiling.set_inW(w);
    tiling.set_outH(TARGET_OUT_H);
    tiling.set_outW(TARGET_OUT_W);

    uint32_t planeNum = n * c;
    uint32_t blockDim = planeNum < 8 ? planeNum : 8;
    blockDim = (blockDim == 0) ? 1 : blockDim;
    context->SetBlockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    // Output is N, C, 128, 128
    std::vector<int64_t> outDims(4);
    outDims[0] = x_shape->GetDim(0);
    outDims[1] = x_shape->GetDim(1);
    outDims[2] = TARGET_OUT_H;
    outDims[3] = TARGET_OUT_W;
    y_shape->SetDimNum(4);
    for (size_t i = 0; i < 4; ++i) {
        y_shape->SetDim(i, outDims[i]);
    }
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto dt = context->GetInputDataType(0);
    context->SetOutputDataType(0, dt);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "resize_with_antialias_custom_tiling_data.h"

// Simplified custom kernel: 2x downsample with 2x2 box filter (antialias) implemented in GM
// Assumes input format NCHW, float32.
// Output size is provided via tiling (typically 128x128 in this example).

extern "C" __global__ __aicore__ void resize_with_antialias_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N = tiling_data.n;
    const uint32_t C = tiling_data.c;
    const uint32_t inH = tiling_data.inH;
    const uint32_t inW = tiling_data.inW;
    const uint32_t outH = tiling_data.outH;
    const uint32_t outW = tiling_data.outW;

    const uint32_t planes = N * C;
    const uint32_t blockNum = AscendC::GetBlockNum();
    const uint32_t blockIdx = AscendC::GetBlockIdx();

    __gm__ const float* inPtr = reinterpret_cast<__gm__ const float*>(x);
    __gm__ float* outPtr = reinterpret_cast<__gm__ float*>(y);

    // Each block handles a subset of (N*C) planes in a strided fashion
    for (uint32_t p = blockIdx; p < planes; p += blockNum) {
        // plane offsets
        uint64_t inPlaneBase  = static_cast<uint64_t>(p) * inH * inW;
        uint64_t outPlaneBase = static_cast<uint64_t>(p) * outH * outW;

        // For simplicity, implement area downsample when scaling factor is <= 1 (here 256->128)
        // 2x2 box filter average: y = 0.25 * (x00 + x01 + x10 + x11)
        // Mapping: ih = floor(oh * inH / outH), iw = floor(ow * inW / outW)
        // For exactly half scale this is ih=oh*2, iw=ow*2
        for (uint32_t oh = 0; oh < outH; ++oh) {
            uint32_t ih0 = (uint64_t)oh * inH / outH;  // generic mapping
            uint32_t ih1 = AscendC::Min(ih0 + 1, inH - 1);
            for (uint32_t ow = 0; ow < outW; ++ow) {
                uint32_t iw0 = (uint64_t)ow * inW / outW;
                uint32_t iw1 = AscendC::Min(iw0 + 1, inW - 1);

                uint64_t i00 = inPlaneBase + (uint64_t)ih0 * inW + iw0;
                uint64_t i01 = inPlaneBase + (uint64_t)ih0 * inW + iw1;
                uint64_t i10 = inPlaneBase + (uint64_t)ih1 * inW + iw0;
                uint64_t i11 = inPlaneBase + (uint64_t)ih1 * inW + iw1;

                float v00 = inPtr[i00];
                float v01 = inPtr[i01];
                float v10 = inPtr[i10];
                float v11 = inPtr[i11];

                float outv = 0.25f * (v00 + v01 + v10 + v11);
                uint64_t oidx = outPlaneBase + (uint64_t)oh * outW + ow;
                outPtr[oidx] = outv;
            }
        }
    }
}

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor resize_with_antialias_custom_impl_npu(const at::Tensor& self) {
    TORCH_CHECK(self.dim() == 4, "Input must be NCHW 4D tensor");
    TORCH_CHECK(self.scalar_type() == at::kFloat, "Only float32 is supported in this custom op");
    auto sizes = self.sizes();
    auto N = sizes[0];
    auto C = sizes[1];
    const int64_t outH = 128;
    const int64_t outW = 128;
    at::Tensor result = at::empty({N, C, outH, outW}, self.options());
    // Custom AclNN entry must be generated by the build system according to project_json
    EXEC_NPU_CMD(aclnnResizeWithAntialiasCustom, self, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("resize_with_antialias_custom", &resize_with_antialias_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("resize_with_antialias_custom", &resize_with_antialias_custom_impl_npu, "Resize bilinear with antialias (custom)");
