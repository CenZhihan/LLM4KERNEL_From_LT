kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "vanilla_rnn_hidden_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelVanillaRnnHidden {
public:
    __aicore__ inline KernelVanillaRnnHidden() {}
    __aicore__ inline void Init(GM_ADDR preact, GM_ADDR hidden, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        preactGm.SetGlobalBuffer((__gm__ DTYPE_X *)preact + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        hiddenGm.SetGlobalBuffer((__gm__ DTYPE_Z *)hidden + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueuePreact, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueHidden, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> preactLocal = inQueuePreact.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(preactLocal, preactGm[progress * this->tileLength], this->tileLength);
        inQueuePreact.EnQue(preactLocal);
    }

    __aicore__ inline void Compute(int32_t)
    {
        AscendC::LocalTensor<DTYPE_X> preactLocal = inQueuePreact.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> hiddenLocal = outQueueHidden.AllocTensor<DTYPE_Z>();
        AscendC::Tanh(hiddenLocal, preactLocal, this->tileLength);
        outQueueHidden.EnQue<DTYPE_Z>(hiddenLocal);
        inQueuePreact.FreeTensor(preactLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> hiddenLocal = outQueueHidden.DeQue<DTYPE_Z>();
        AscendC::DataCopy(hiddenGm[progress * this->tileLength], hiddenLocal, this->tileLength);
        outQueueHidden.FreeTensor(hiddenLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueuePreact;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueHidden;
    AscendC::GlobalTensor<DTYPE_X> preactGm;
    AscendC::GlobalTensor<DTYPE_Z> hiddenGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void vanilla_rnn_hidden_custom(GM_ADDR preact, GM_ADDR hidden, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelVanillaRnnHidden op;
    op.Init(preact, hidden, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, input_size: int, hidden_size: int, output_size: int):
        super(ModelNew, self).__init__()
        self.input_size = input_size
        self.hidden_size = hidden_size
        self.output_size = output_size

        # Same structure as original: fuse only the hidden activation via custom AscendC op
        self.i2h = torch.nn.Linear(input_size + hidden_size, hidden_size)
        self.h2o = torch.nn.Linear(hidden_size, output_size)

    def forward(self, x: torch.Tensor, h0: torch.Tensor) -> torch.Tensor:
        seq_len, batch_size, _ = x.size()
        hidden = h0.to(x.device)
        outputs = []

        for t in range(seq_len):
            combined = torch.cat((x[t], hidden), dim=1)  # (batch, input+hidden)
            preact = self.i2h(combined)                  # (batch, hidden)
            hidden = custom_ops_lib.vanilla_rnn_hidden_custom(preact)
            output = self.h2o(hidden)                    # (batch, output)
            outputs.append(output)

        return torch.stack(outputs, dim=0)
'''
