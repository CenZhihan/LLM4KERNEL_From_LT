    AddBiasFourDimBroadcastCustomTilingData tiling;

    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);

    // Expect NCHW
    uint64_t n = (x_shape->GetOriginShape().GetDimNum() > 0) ? x_shape->GetOriginShape().GetDim(0) : 1;
    uint64_t c = (x_shape->GetOriginShape().GetDimNum() > 1) ? x_shape->GetOriginShape().GetDim(1) : 1;
    uint64_t h = (x_shape->GetOriginShape().GetDimNum() > 2) ? x_shape->GetOriginShape().GetDim(2) : 1;
    uint64_t w = (x_shape->GetOriginShape().GetDimNum() > 3) ? x_shape->GetOriginShape().GetDim(3) : 1;

    uint64_t totalLength = n * c * h * w;
    uint64_t plane = h * w;
    uint64_t channels = (b_shape->GetOriginShape().GetDimNum() >= 1) ? b_shape->GetOriginShape().GetShapeSize() : 1;

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_totalLength(static_cast<uint32_t>(totalLength));
    tiling.set_plane(static_cast<uint32_t>(plane));
    tiling.set_channels(static_cast<uint32_t>(channels));
    // element-wise tiling (tile length = 1)
    tiling.set_tileNum(static_cast<uint32_t>(totalLength));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "add_bias_four_dim_broadcast_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // number of tiles in pipe for each queue

class KernelAddBiasBroadcast {
public:
    __aicore__ inline KernelAddBiasBroadcast() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR z, uint32_t totalLength, uint32_t /*tileNum*/, uint32_t plane, uint32_t channels)
    {
        this->channels = channels;
        this->plane = plane;
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileLength = 1; // element-wise tile
        this->tileNum = this->blockLength / this->tileLength;

        uint32_t blockBase = this->blockLength * AscendC::GetBlockIdx();
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + blockBase, this->blockLength);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z + blockBase, this->blockLength);
        biasGm.SetGlobalBuffer((__gm__ DTYPE_Y *)bias, this->channels);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }
    __aicore__ inline void Process()
    {
        for (uint32_t i = 0; i < this->tileNum; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        // copy one element of x
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);

        // prepare bias scalar for this element
        uint32_t globalIndex = this->blockOffset() + progress * this->tileLength;
        uint32_t cIdx = (globalIndex / this->plane) % this->channels;
        AscendC::LocalTensor<DTYPE_Y> bLocal = inQueueB.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(bLocal, biasGm[cIdx], 1);
        inQueueB.EnQue(bLocal);
    }
    __aicore__ inline void Compute(uint32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> bLocal = inQueueB.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();

        // Vector add on length = 1
        AscendC::Add(zLocal, xLocal, bLocal, this->tileLength);

        outQueueZ.EnQue<DTYPE_Z>(zLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueB.FreeTensor(bLocal);
    }
    __aicore__ inline void CopyOut(uint32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[progress * this->tileLength], zLocal, this->tileLength);
        outQueueZ.FreeTensor(zLocal);
    }
    __aicore__ inline uint32_t blockOffset() const
    {
        return this->blockLength * AscendC::GetBlockIdx();
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> biasGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
    uint32_t plane;
    uint32_t channels;
};

extern "C" __global__ __aicore__ void add_bias_four_dim_broadcast_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelAddBiasBroadcast op;
    op.Init(x, bias, z, tiling_data.totalLength, tiling_data.tileNum, tiling_data.plane, tiling_data.channels);
    op.Process();
}
