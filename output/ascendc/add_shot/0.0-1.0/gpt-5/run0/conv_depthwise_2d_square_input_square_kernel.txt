static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    auto x_shape = context->GetInputShape(0)->GetOriginShape();
    auto w_shape = context->GetInputShape(1)->GetOriginShape();

    int64_t N = x_shape.GetDimNum() > 0 ? x_shape.GetDim(0) : 1;
    int64_t C = x_shape.GetDimNum() > 1 ? x_shape.GetDim(1) : 1;
    int64_t H = x_shape.GetDimNum() > 2 ? x_shape.GetDim(2) : 1;
    int64_t W = x_shape.GetDimNum() > 3 ? x_shape.GetDim(3) : 1;

    int64_t stride = 1;
    int64_t padding = 0;
    // Try to read attributes if provided

#endif
    if (context->HasAttr("stride")) {

    if (context->HasAttr("padding")) {

    }

    int64_t OH = FloorDiv((H + 2 * padding - K), stride) + 1;
    int64_t OW = FloorDiv((W + 2 * padding - K), stride) + 1;

    ConvDepthwise2dSquareInputSquareKernelCustomTilingData tiling;
    context->SetBlockDim(BLOCK_DIM);

    tiling.set_c(static_cast<uint32_t>(C));

    tiling.set_w(static_cast<uint32_t>(W));
    tiling.set_k(static_cast<uint32_t>(K));
    tiling.set_stride(static_cast<uint32_t>(stride));
    tiling.set_pad(static_cast<uint32_t>(padding));
    tiling.set_oh(static_cast<uint32_t>(OH));
    tiling.set_ow(static_cast<uint32_t>(OW));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_depthwise_2d_square_input_square_kernel_custom_tiling_data.h"

#ifndef DTYPE_X
#define DTYPE_X float
#endif
#ifndef DTYPE_W
#define DTYPE_W float
#endif
#ifndef DTYPE_Y
#define DTYPE_Y float
#endif

class KernelDWConv {
public:
    __aicore__ inline KernelDWConv() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w_in,
                                uint32_t k, uint32_t stride, uint32_t pad,
                                uint32_t oh, uint32_t ow)
    {
        this->xPtr = (__gm__ DTYPE_X*)x;
        this->wPtr = (__gm__ DTYPE_W*)w;
        this->yPtr = (__gm__ DTYPE_Y*)y;

        this->N = n; this->C = c; this->H = h; this->W = w_in;
        this->K = k; this->S = stride; this->P = pad; this->OH = oh; this->OW = ow;

        uint64_t totalOut = (uint64_t)N * C * OH * OW;
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t bid = AscendC::GetBlockIdx();

        this->blockStart = (uint64_t)bid * (totalOut / blocks) + AscendC::Min(bid, (uint32_t)(totalOut % blocks));
        this->blockEnd = this->blockStart + (totalOut / blocks) + (bid < (totalOut % blocks) ? 1 : 0);
    }

    __aicore__ inline void Process()
    {
        for (uint64_t idx = blockStart; idx < blockEnd; ++idx) {
            // Decode linear index to (n,c,oh,ow)
            uint64_t t = idx;
            uint32_t ow = (uint32_t)(t % OW); t /= OW;
            uint32_t oh = (uint32_t)(t % OH); t /= OH;
            uint32_t c  = (uint32_t)(t % C);  t /= C;
            uint32_t n  = (uint32_t)(t);

            int32_t h_in0 = (int32_t)oh * (int32_t)S - (int32_t)P;
            int32_t w_in0 = (int32_t)ow * (int32_t)S - (int32_t)P;

            float acc = 0.0f;
            for (uint32_t kh = 0; kh < K; ++kh) {
                int32_t ih = h_in0 + (int32_t)kh;
                if (ih < 0 || ih >= (int32_t)H) continue;
                for (uint32_t kw = 0; kw < K; ++kw) {
                    int32_t iw = w_in0 + (int32_t)kw;
                    if (iw < 0 || iw >= (int32_t)W) continue;

                    uint64_t xoff = ((((uint64_t)n * C + c) * H + (uint32_t)ih) * W + (uint32_t)iw);
                    uint64_t woff = ((uint64_t)c * K + kh) * K + kw;

                    acc += static_cast<float>(xPtr[xoff]) * static_cast<float>(wPtr[woff]);
                }
            }
            yPtr[idx] = static_cast<DTYPE_Y>(acc);
        }
    }

private:
    __gm__ DTYPE_X* xPtr;
    __gm__ DTYPE_W* wPtr;
    __gm__ DTYPE_Y* yPtr;

    uint32_t N, C, H, W;
    uint32_t K, S, P, OH, OW;

    uint64_t blockStart;
    uint64_t blockEnd;
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelDWConv op;
    op.Init(x, w, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.k, tiling_data.stride, tiling_data.pad,
            tiling_data.oh, tiling_data.ow);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Depthwise 2D convolution implemented via custom AscendC kernel.
    Assumes NCHW layout, groups == in_channels, and square kernel.
    """
    def __init__(self, in_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super().__init__()
        assert not bias, "This custom kernel currently does not support bias."
        self.in_channels = in_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        # Weight shape [C, 1, K, K] for depthwise convolution
        self.weight = torch.nn.Parameter(torch.randn(in_channels, 1, kernel_size, kernel_size))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return custom_ops_lib.conv_depthwise_2d_square_input_square_kernel_custom(
            x, self.weight, int(self.stride), int(self.padding)
        )
