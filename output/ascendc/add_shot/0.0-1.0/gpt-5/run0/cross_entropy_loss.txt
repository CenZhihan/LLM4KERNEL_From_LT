kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "cross_entropy_loss_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCrossEntropyLoss {
public:
    __aicore__ inline KernelCrossEntropyLoss() {}
    __aicore__ inline void Init(GM_ADDR predictions, GM_ADDR targets, GM_ADDR loss,
                                uint32_t nBatch, uint32_t nClass, uint32_t tileC)
    {
        this->N = nBatch;
        this->C = nClass;
        this->tileC = tileC;

        predsGm.SetGlobalBuffer((__gm__ DTYPE_X*)predictions, (uint64_t)N * (uint64_t)C);
        targetGm.SetGlobalBuffer((__gm__ DTYPE_Y*)targets, N);
        outGm.SetGlobalBuffer((__gm__ DTYPE_Z*)loss, 1);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileC * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueY, BUFFER_NUM, sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        // Compute mean cross-entropy: mean_i [-log_softmax(x_i)[y_i]]
        DTYPE_Z totalLoss = static_cast<DTYPE_Z>(0);

        for (uint32_t i = 0; i < N; ++i) {
            // sum_k exp(x_ik)
            DTYPE_Z sumExp = static_cast<DTYPE_Z>(0);
            uint64_t rowOffset = (uint64_t)i * (uint64_t)C;

            // Accumulate sum of exp over class dimension by tiles
            for (uint32_t offset = 0; offset < C; offset += tileC) {
                uint32_t len = (offset + tileC <= C) ? tileC : (C - offset);
                AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(xLocal, predsGm[rowOffset + offset], len);
                inQueueX.EnQue(xLocal);

                xLocal = inQueueX.DeQue<DTYPE_X>();
                // in-place exp
                AscendC::Exp(xLocal, xLocal, len);

                // reduce sum for this tile
                DTYPE_Z part = AscendC::ReduceSum(xLocal, len);
                sumExp += part;
                inQueueX.FreeTensor(xLocal);
            }

            // Load target index y_i
            AscendC::LocalTensor<DTYPE_Y> yLocal = inQueueY.AllocTensor<DTYPE_Y>();
            AscendC::DataCopy(yLocal, targetGm[i], 1);
            inQueueY.EnQue(yLocal);

            yLocal = inQueueY.DeQue<DTYPE_Y>();
            int32_t yi = (int32_t)(yLocal.GetValue(0));
            inQueueY.FreeTensor(yLocal);

            // Load x_i_yi
            AscendC::LocalTensor<DTYPE_X> clsLocal = inQueueX.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(clsLocal, predsGm[rowOffset + (uint32_t)yi], 1);
            inQueueX.EnQue(clsLocal);

            clsLocal = inQueueX.DeQue<DTYPE_X>();
            DTYPE_Z x_iy = (DTYPE_Z)(clsLocal.GetValue(0));
            inQueueX.FreeTensor(clsLocal);

            // loss_i = -x_iy + log(sumExp)
            // Use device math logf for scalar
            DTYPE_Z loss_i = (DTYPE_Z)(-x_iy + logf((float)sumExp));
            totalLoss += loss_i;
        }

        DTYPE_Z meanLoss = totalLoss / static_cast<DTYPE_Z>(N);

        // Write out
        AscendC::LocalTensor<DTYPE_Z> outLocal = outQueueZ.AllocTensor<DTYPE_Z>();
        outLocal.SetValue(0, meanLoss);
        outQueueZ.EnQue(outLocal);

        outLocal = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(outGm[0], outLocal, 1);
        outQueueZ.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueY;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<DTYPE_X> predsGm;
    AscendC::GlobalTensor<DTYPE_Y> targetGm;
    AscendC::GlobalTensor<DTYPE_Z> outGm;

    uint32_t N;
    uint32_t C;
    uint32_t tileC;
};

extern "C" __global__ __aicore__ void cross_entropy_loss_custom(GM_ADDR predictions, GM_ADDR targets, GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCrossEntropyLoss op;
    op.Init(predictions, targets, loss, tiling_data.nBatch, tiling_data.nClass, tiling_data.tileC);
    op.Process();
}
