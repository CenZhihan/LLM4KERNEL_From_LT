#include "matmul_mish_mish_custom_tiling.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    MatmulMishMishCustomTilingData tiling;

    auto xShape = context->GetInputShape(0)->GetOriginShape();
    auto wShape = context->GetInputShape(1)->GetOriginShape();

    uint32_t M = xShape.GetDim(0);
    uint32_t K = xShape.GetDim(1);
    uint32_t N = wShape.GetDim(0);

    uint32_t blockNum = BLOCK_DIM;
    context->SetBlockDim(blockNum);

    uint32_t blockRows = (M + blockNum - 1) / blockNum;
    uint32_t tileK = 1024; // simple default K-tile

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileK(tileK);
    tiling.set_blockRows(blockRows);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);
    gert::Shape* y_shape = context->GetOutputShape(0);

    std::vector<int64_t> out_dims = {x_shape->GetDim(0), w_shape->GetDim(0)};
    y_shape->SetDimNum(2);
    y_shape->SetDim(0, out_dims[0]);
    y_shape->SetDim(1, out_dims[1]);

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "matmul_mish_mish_custom_tiling_data.h"
#include <cmath>

inline __aicore__ float mish_scalar(float x) {
    // mish(x) = x * tanh(softplus(x)) = x * tanh(log1p(exp(x)))
    float sp = log1pf(expf(x));
    return x * tanhf(sp);
}

extern "C" __global__ __aicore__ void matmul_mish_mish_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t M = tiling_data.M;
    const uint32_t N = tiling_data.N;
    const uint32_t K = tiling_data.K;
    const uint32_t blockRows = tiling_data.blockRows;

    uint32_t blockIdx = AscendC::GetBlockIdx();
    uint32_t rowStart = blockIdx * blockRows;
    uint32_t rowEnd = rowStart + blockRows;
    if (rowEnd > M) rowEnd = M;

    float* xPtr = reinterpret_cast<float*>(x);
    float* wPtr = reinterpret_cast<float*>(w);
    float* bPtr = reinterpret_cast<float*>(b);
    float* yPtr = reinterpret_cast<float*>(y);

    for (uint32_t m = rowStart; m < rowEnd; ++m) {
        const float* xRow = xPtr + static_cast<size_t>(m) * K;
        for (uint32_t n = 0; n < N; ++n) {
            const float* wRow = wPtr + static_cast<size_t>(n) * K;
            float acc = 0.f;
            for (uint32_t k = 0; k < K; ++k) {
                acc += xRow[k] * wRow[k];
            }
            acc += bPtr[n];
            float v = mish_scalar(mish_scalar(acc));
            yPtr[static_cast<size_t>(m) * N + n] = v;
        }
    }
}

    return result;

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused matmul + Mish + Mish using custom AscendC kernel.
    """
    def __init__(self, in_features, out_features):
        super(ModelNew, self).__init__()
        self.linear = torch.nn.Linear(in_features, out_features)

    def forward(self, x):
        # x: [batch, in_features]
        # weight: [out_features, in_features], bias: [out_features]
        return custom_ops_lib.matmul_mish_mish_custom(x, self.linear.weight, self.linear.bias)
