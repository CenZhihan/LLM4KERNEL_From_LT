kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "squeeze_net_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConcatC {
public:
    __aicore__ inline KernelConcatC() {}
    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR out,
                                uint32_t n, uint32_t c1, uint32_t c2, uint32_t hw, uint32_t tileNum)
    {
        this->N = n;
        this->C1 = c1;
        this->C2 = c2;
        this->HW = hw;
        this->tileNum = tileNum;

        uint32_t outPerBatch = (C1 + C2) * HW;
        // Determine per-tile length; avoid 0
        this->tileLength = outPerBatch / (tileNum * BUFFER_NUM);
        if (this->tileLength == 0) {
            this->tileLength = 1;
        }

        // Split batches across blocks
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blk = AscendC::GetBlockIdx();

        uint32_t base = N / blockNum;
        uint32_t rem = N % blockNum;
        if (blk < rem) {
            this->localN = base + 1;
            this->startN = blk * (base + 1);
        } else {
            this->localN = base;
            this->startN = rem * (base + 1) + (blk - rem) * base;
        }

        aGm.SetGlobalBuffer((__gm__ DTYPE_X *)a + this->startN * (uint64_t)C1 * HW,
                            (uint64_t)(N - this->startN) * C1 * HW);
        bGm.SetGlobalBuffer((__gm__ DTYPE_Y *)b + this->startN * (uint64_t)C2 * HW,
                            (uint64_t)(N - this->startN) * C2 * HW);
        outGm.SetGlobalBuffer((__gm__ DTYPE_Z *)out + this->startN * (uint64_t)(C1 + C2) * HW,
                              (uint64_t)(N - this->startN) * (C1 + C2) * HW);

        pipe.InitBuffer(inQueueA, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        // For each local batch, copy A then B into OUT along channel dimension
        uint64_t lenA = (uint64_t)C1 * HW;
        uint64_t lenB = (uint64_t)C2 * HW;
        uint64_t lenOut = (uint64_t)(C1 + C2) * HW;

        for (uint32_t nb = 0; nb < localN; ++nb) {
            // Copy A segment tiles
            uint64_t tilesA = (lenA + tileLength - 1) / tileLength;
            for (uint64_t t = 0; t < tilesA; ++t) {
                uint32_t cur = (t + 1 == tilesA) ? (uint32_t)(lenA - t * tileLength) : tileLength;

                AscendC::LocalTensor<DTYPE_X> aLocal = inQueueA.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(aLocal, aGm[nb * lenA + t * tileLength], cur);
                inQueueA.EnQue(aLocal);

                AscendC::LocalTensor<DTYPE_X> aLocalUsed = inQueueA.DeQue<DTYPE_X>();
                AscendC::DataCopy(outGm[nb * lenOut + t * tileLength], aLocalUsed, cur);
                inQueueA.FreeTensor(aLocalUsed);
            }

            // Copy B segment tiles
            uint64_t tilesB = (lenB + tileLength - 1) / tileLength;
            for (uint64_t t = 0; t < tilesB; ++t) {
                uint32_t cur = (t + 1 == tilesB) ? (uint32_t)(lenB - t * tileLength) : tileLength;

                AscendC::LocalTensor<DTYPE_Y> bLocal = inQueueB.AllocTensor<DTYPE_Y>();
                AscendC::DataCopy(bLocal, bGm[nb * lenB + t * tileLength], cur);
                inQueueB.EnQue(bLocal);

                AscendC::LocalTensor<DTYPE_Y> bLocalUsed = inQueueB.DeQue<DTYPE_Y>();
                AscendC::DataCopy(outGm[nb * lenOut + lenA + t * tileLength], bLocalUsed, cur);
                inQueueB.FreeTensor(bLocalUsed);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA, inQueueB;
    AscendC::GlobalTensor<DTYPE_X> aGm;
    AscendC::GlobalTensor<DTYPE_Y> bGm;
    AscendC::GlobalTensor<DTYPE_Z> outGm;

    uint32_t N;
    uint32_t C1;
    uint32_t C2;
    uint32_t HW;

    uint32_t tileNum;
    uint32_t tileLength;

    uint32_t startN;
    uint32_t localN;
};

extern "C" __global__ __aicore__ void squeeze_net_custom(GM_ADDR a, GM_ADDR b, GM_ADDR out, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConcatC op;
    op.Init(a, b, out,
            tiling_data.n, tiling_data.c1, tiling_data.c2, tiling_data.hw, tiling_data.tileNum);
    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

static inline void check_cat_shapes(const at::Tensor& a, const at::Tensor& b) {
    TORCH_CHECK(a.dim() == 4 && b.dim() == 4, "squeeze_net_custom expects 4D NCHW tensors");
    TORCH_CHECK(a.size(0) == b.size(0) && a.size(2) == b.size(2) && a.size(3) == b.size(3),
                "Inputs must have same N, H, W for channel concat");
    TORCH_CHECK(a.scalar_type() == b.scalar_type(), "Inputs must have same dtype");
}

at::Tensor squeeze_net_custom_impl_npu(const at::Tensor& a, const at::Tensor& b) {
    check_cat_shapes(a, b);
    auto out_sizes = a.sizes().vec();
    out_sizes[1] = a.size(1) + b.size(1);
    at::Tensor out = at::empty(out_sizes, a.options());
    EXEC_NPU_CMD(aclnnSqueezeNetCustom, a, b, out);
    return out;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("squeeze_net_custom", &squeeze_net_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("squeeze_net_custom", &squeeze_net_custom_impl_npu, "Channel concat for SqueezeNet Fire module");
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

import torch.nn as nn
import torch.nn.functional as F

class FireModule(nn.Module):
    def __init__(self, in_channels, squeeze_channels, expand1x1_channels, expand3x3_channels):
        super(FireModule, self).__init__()
        self.squeeze = nn.Conv2d(in_channels, squeeze_channels, kernel_size=1)
        self.squeeze_activation = nn.ReLU(inplace=True)
        self.expand1x1 = nn.Conv2d(squeeze_channels, expand1x1_channels, kernel_size=1)
        self.expand1x1_activation = nn.ReLU(inplace=True)
        self.expand3x3 = nn.Conv2d(squeeze_channels, expand3x3_channels, kernel_size=3, padding=1)
        self.expand3x3_activation = nn.ReLU(inplace=True)
    
    def forward(self, x):
        x = self.squeeze_activation(self.squeeze(x))
        a = self.expand1x1_activation(self.expand1x1(x))
        b = self.expand3x3_activation(self.expand3x3(x))
        return custom_ops_lib.squeeze_net_custom(a, b)

class ModelNew(nn.Module):
    def __init__(self, num_classes=1000):
        super(ModelNew, self).__init__()
        self.features = nn.Sequential(
            nn.Conv2d(3, 96, kernel_size=7, stride=2),
            nn.ReLU(inplace=True),
            nn.MaxPool2d(kernel_size=3, stride=2, ceil_mode=True),
            FireModule(96, 16, 64, 64),
            FireModule(128, 16, 64, 64),
            FireModule(128, 32, 128, 128),
            nn.MaxPool2d(kernel_size=3, stride=2, ceil_mode=True),
            FireModule(256, 32, 128, 128),
            FireModule(256, 48, 192, 192),
            FireModule(384, 48, 192, 192),
            FireModule(384, 64, 256, 256),
            nn.MaxPool2d(kernel_size=3, stride=2, ceil_mode=True),
            FireModule(512, 64, 256, 256),
        )
        self.classifier = nn.Sequential(
            nn.Dropout(p=0.0),
            nn.Conv2d(512, num_classes, kernel_size=1),
            nn.ReLU(inplace=True),
            nn.AdaptiveAvgPool2d((1, 1))
        )
    
    def forward(self, x):
        x = self.features(x)
        x = self.classifier(x)
        return torch.flatten(x, 1)
