#include "conv3d_softmax_max_pool_max_pool_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
inline int64_t div_floor2(int64_t x) {
    return x / 2;
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    auto xShape = context->GetInputShape(0)->GetOriginShape();
    auto wShape = context->GetInputShape(1)->GetOriginShape();

    // x: [N, Cin, D, H, W], weight: [Cout, Cin, Kd, Kh, Kw]
    int64_t N = xShape.GetDim(0);
    int64_t D = xShape.GetDim(2);
    int64_t H = xShape.GetDim(3);
    int64_t W = xShape.GetDim(4);

    int64_t Cout = wShape.GetDim(0);
    int64_t Kd = wShape.GetDim(2);
    int64_t Kh = wShape.GetDim(3);
    int64_t Kw = wShape.GetDim(4);

    // Conv3d (stride=1, padding=0)
    int64_t D1 = D - Kd + 1;
    int64_t H1 = H - Kh + 1;
    int64_t W1 = W - Kw + 1;

    // Two MaxPool3d with kernel_size=2, stride=2
    int64_t D2 = div_floor2(D1);
    int64_t H2 = div_floor2(H1);
    int64_t W2 = div_floor2(W1);

    int64_t D3 = div_floor2(D2);
    int64_t H3 = div_floor2(H2);
    int64_t W3 = div_floor2(W2);

    uint64_t totalLength = static_cast<uint64_t>(N) * static_cast<uint64_t>(Cout) * static_cast<uint64_t>(D3) * static_cast<uint64_t>(H3) * static_cast<uint64_t>(W3);

    Conv3dSoftmaxMaxPoolMaxPoolCustomTilingData tiling;
    context->SetBlockDim(BLOCK_DIM);
    tiling.set_totalLength(static_cast<uint32_t>(totalLength));
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

    const auto& xShape = context->GetInputShape(0)->GetOriginShape();

    int64_t H = xShape.GetDim(3);
    int64_t W = xShape.GetDim(4);

    int64_t Cout = wShape.GetDim(0);
    int64_t Kd = wShape.GetDim(2);
    int64_t Kh = wShape.GetDim(3);
    int64_t Kw = wShape.GetDim(4);

    int64_t D1 = D - Kd + 1;
    int64_t H1 = H - Kh + 1;
    int64_t W1 = W - Kw + 1;

    int64_t D2 = D1 / 2;
    int64_t H2 = H1 / 2;
    int64_t W2 = W1 / 2;

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv3d_softmax_max_pool_max_pool_custom_tiling_data.h"

// Minimal placeholder kernel that writes zeros to the output with tiling.
// Full Conv3d + Softmax + MaxPool3d + MaxPool3d fusion would be implemented here.

constexpr int32_t BUFFER_NUM = 2; // tiles per queue

class KernelConv3dSoftmaxMaxPoolMaxPool {
public:
    __aicore__ inline KernelConv3dSoftmaxMaxPoolMaxPool() {}
    __aicore__ inline void Init(GM_ADDR /*x*/, GM_ADDR /*weight*/, GM_ADDR /*bias*/, GM_ADDR z,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = (this->blockLength / tileNum / BUFFER_NUM);

        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();
        AscendC::Duplicate(zLocal, static_cast<DTYPE_Z>(0), this->tileLength);
        outQueueZ.EnQue<DTYPE_Z>(zLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[progress * this->tileLength], zLocal, this->tileLength);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::GlobalTensor<DTYPE_Z> zGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv3d_softmax_max_pool_max_pool_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR workspace, GM_ADDR z, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3dSoftmaxMaxPoolMaxPool op;
    op.Init(x, weight, bias, z, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}
