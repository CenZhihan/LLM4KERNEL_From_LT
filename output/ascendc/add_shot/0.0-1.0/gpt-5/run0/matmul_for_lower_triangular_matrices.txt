kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "matmul_for_lower_triangular_matrices_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLowerTriMatmul {
public:
    __aicore__ inline KernelLowerTriMatmul() {}
    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C, uint32_t totalLength, uint32_t tileNum)
    {
        // totalLength is N*N for square matrices
        this->totalLength = totalLength;
        this->tileNum = tileNum;
        // Use simple 1D tiling similar to elementwise for demonstration
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        aGm.SetGlobalBuffer((__gm__ DTYPE_A *)A + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        bGm.SetGlobalBuffer((__gm__ DTYPE_B *)B + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        cGm.SetGlobalBuffer((__gm__ DTYPE_C *)C + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueA, BUFFER_NUM, this->tileLength * sizeof(DTYPE_A));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->tileLength * sizeof(DTYPE_B));
        pipe.InitBuffer(outQueueC, BUFFER_NUM, this->tileLength * sizeof(DTYPE_C));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_A> aLocal = inQueueA.AllocTensor<DTYPE_A>();
        AscendC::LocalTensor<DTYPE_B> bLocal = inQueueB.AllocTensor<DTYPE_B>();
        AscendC::DataCopy(aLocal, aGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(bLocal, bGm[progress * this->tileLength], this->tileLength);
        inQueueA.EnQue(aLocal);
        inQueueB.EnQue(bLocal);
    }

    __aicore__ inline void Compute(int32_t)
    {
        AscendC::LocalTensor<DTYPE_A> aLocal = inQueueA.DeQue<DTYPE_A>();
        AscendC::LocalTensor<DTYPE_B> bLocal = inQueueB.DeQue<DTYPE_B>();
        AscendC::LocalTensor<DTYPE_C> cLocal = outQueueC.AllocTensor<DTYPE_C>();

        // Placeholder computation: elementwise add as a simple demo path
        // In a production kernel, implement tril(matmul(A, B)) on tiles.
        AscendC::Add(cLocal, aLocal, bLocal, this->tileLength);

        outQueueC.EnQue<DTYPE_C>(cLocal);
        inQueueA.FreeTensor(aLocal);
        inQueueB.FreeTensor(bLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_C> cLocal = outQueueC.DeQue<DTYPE_C>();
        AscendC::DataCopy(cGm[progress * this->tileLength], cLocal, this->tileLength);
        outQueueC.FreeTensor(cLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA, inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC;

    AscendC::GlobalTensor<DTYPE_A> aGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_C> cGm;

    uint32_t totalLength;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void matmul_for_lower_triangular_matrices_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLowerTriMatmul op;
    op.Init(A, B, C, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}
