kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_depthwise_separable_2d_custom_tiling_data.h"

using DTYPE_IN = float;
using DTYPE_DW = float;
using DTYPE_PW = float;
using DTYPE_OUT = float;

static __aicore__ inline uint64_t idx4d(uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                        uint32_t C, uint32_t H, uint32_t W) {
    return (((uint64_t)n * C + c) * H + h) * W + w;
}

extern "C" __global__ __aicore__ void conv_depthwise_separable_2d_custom(GM_ADDR x, GM_ADDR dw, GM_ADDR pw, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    uint32_t N = tiling_data.N;
    uint32_t C = tiling_data.C;
    uint32_t H = tiling_data.H;
    uint32_t W = tiling_data.W;
    uint32_t kH = tiling_data.kH;
    uint32_t kW = tiling_data.kW;
    uint32_t outC = tiling_data.outC;
    uint32_t strideH = tiling_data.strideH;
    uint32_t strideW = tiling_data.strideW;
    uint32_t padH = tiling_data.padH;
    uint32_t padW = tiling_data.padW;
    uint32_t dilH = tiling_data.dilH;
    uint32_t dilW = tiling_data.dilW;
    uint32_t Hout = tiling_data.Hout;
    uint32_t Wout = tiling_data.Wout;

    __gm__ DTYPE_IN* xGm = (__gm__ DTYPE_IN*)x;
    __gm__ DTYPE_DW* dwGm = (__gm__ DTYPE_DW*)dw;
    __gm__ DTYPE_PW* pwGm = (__gm__ DTYPE_PW*)pw;
    __gm__ DTYPE_OUT* yGm = (__gm__ DTYPE_OUT*)y;

    uint64_t total = (uint64_t)N * (uint64_t)outC * (uint64_t)Hout * (uint64_t)Wout;

    uint32_t blockNum = AscendC::GetBlockNum();
    uint32_t blockIdx = AscendC::GetBlockIdx();

    uint64_t base = (total / blockNum) * blockIdx;
    uint64_t tail = (blockIdx == blockNum - 1) ? total : (total / blockNum) * (blockIdx + 1);

    for (uint64_t idx = base; idx < tail; ++idx) {
        uint64_t t = idx;
        uint32_t ow = (uint32_t)(t % Wout); t /= Wout;
        uint32_t oh = (uint32_t)(t % Hout); t /= Hout;
        uint32_t oc = (uint32_t)(t % outC); t /= outC;
        uint32_t n  = (uint32_t)t;

        float acc_out = 0.0f;

        for (uint32_t c = 0; c < C; ++c) {
            float acc_dw = 0.0f;
            for (uint32_t kh = 0; kh < kH; ++kh) {
                int32_t ih = (int32_t)oh * (int32_t)strideH - (int32_t)padH + (int32_t)kh * (int32_t)dilH;
                if (ih < 0 || ih >= (int32_t)H) continue;
                for (uint32_t kw = 0; kw < kW; ++kw) {
                    int32_t iw = (int32_t)ow * (int32_t)strideW - (int32_t)padW + (int32_t)kw * (int32_t)dilW;
                    if (iw < 0 || iw >= (int32_t)W) continue;
                    uint64_t xIdx = idx4d(n, c, (uint32_t)ih, (uint32_t)iw, C, H, W);
                    uint64_t dwIdx = (((uint64_t)c * 1 + 0) * kH + kh) * kW + kw; // [C,1,kH,kW]
                    acc_dw += (float)xGm[xIdx] * (float)dwGm[dwIdx];
                }
            }
            // pw [outC, C, 1, 1]
            uint64_t pwIdx = ((uint64_t)oc * C + c);
            acc_out += acc_dw * (float)pwGm[pwIdx];
        }

        uint64_t yIdx = (((uint64_t)n * outC + oc) * Hout + oh) * Wout + ow;
        yGm[yIdx] = (DTYPE_OUT)acc_out;
    }
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized depthwise-separable 2D convolution using custom AscendC kernel.
    Supports NCHW float tensors. Bias is not fused.
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, dilation: int = 1, bias: bool = False):
        super().__init__()
        # Keep nn.Conv2d modules to host and manage weights/buffers (bias ignored/fallback handled by caller)
        self.depthwise = torch.nn.Conv2d(in_channels, in_channels, kernel_size, stride=stride, padding=padding, dilation=dilation, groups=in_channels, bias=False)
        self.pointwise = torch.nn.Conv2d(in_channels, out_channels, kernel_size=1, bias=False)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        s = int(self.depthwise.stride[0])
        p = int(self.depthwise.padding[0])
        d = int(self.depthwise.dilation[0])
        # Fused DW + PW kernel (bias not fused)
        y = custom_ops_lib.conv_depthwise_separable_2d_custom(
            x,
            self.depthwise.weight,
            self.pointwise.weight,
            s, p, d
        )
        return y
