kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "matmul_max_pool_sum_scale_custom_tiling_data.h"
#include <cfloat>

#ifndef DTYPE_X
#define DTYPE_X float
#endif
#ifndef DTYPE_W
#define DTYPE_W float
#endif
#ifndef DTYPE_B
#define DTYPE_B float
#endif
#ifndef DTYPE_Z
#define DTYPE_Z float
#endif

class KernelMatmulMaxPoolSumScale {
public:
    __aicore__ inline KernelMatmulMaxPoolSumScale() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR z,
                                uint32_t batch, uint32_t inFeat, uint32_t outFeat,
                                uint32_t kernelSize, float scaleFactor)
    {
        xGm = (__gm__ DTYPE_X*)x;
        wGm = (__gm__ DTYPE_W*)w;
        bGm = (__gm__ DTYPE_B*)b;
        zGm = (__gm__ DTYPE_Z*)z;

        this->batch = batch;
        this->inFeat = inFeat;
        this->outFeat = outFeat;
        this->kernelSize = kernelSize > 0 ? kernelSize : 1;
        this->scale = scaleFactor;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        // Simple 1D partition over batch
        rowsPerBlock = (batch + blockNum - 1) / blockNum;
        rowStart = rowsPerBlock * blockIdx;
        rowEnd = rowStart + rowsPerBlock;
        if (rowEnd > batch) rowEnd = batch;
    }

    __aicore__ inline void Process()
    {
        for (uint32_t b = rowStart; b < rowEnd; ++b) {
            ComputeRow(b);
        }
    }

private:
    __aicore__ inline void ComputeRow(uint32_t bIdx)
    {
        const DTYPE_X* xRow = xGm + bIdx * inFeat;
        DTYPE_Z accOut = (DTYPE_Z)0;

        // MaxPool1d over last dim with stride=kernelSize
        uint32_t k = kernelSize;
        uint32_t maxStart = 0;
        if (outFeat >= k) {
            uint32_t pooledLen = (outFeat - k) / k + 1;
            // iterate pool windows
            for (uint32_t p = 0; p < pooledLen; ++p) {
                uint32_t oStart = p * k;
                DTYPE_Z maxv = -FLT_MAX;
                // compute k outputs then take max
                for (uint32_t t = 0; t < k; ++t) {
                    uint32_t o = oStart + t;
                    // dot(xRow, w[o, :]) + b[o]
                    DTYPE_Z dotv = (DTYPE_Z)0;
                    const DTYPE_W* wRow = wGm + o * inFeat;
                    for (uint32_t i = 0; i < inFeat; ++i) {
                        dotv += (DTYPE_Z)xRow[i] * (DTYPE_Z)wRow[i];
                    }
                    dotv += (DTYPE_Z)bGm[o];
                    if (dotv > maxv) {
                        maxv = dotv;
                    }
                }
                accOut += maxv;
            }
        }
        zGm[bIdx] = accOut * (DTYPE_Z)scale;
    }

private:
    __gm__ DTYPE_X* xGm;
    __gm__ DTYPE_W* wGm;
    __gm__ DTYPE_B* bGm;
    __gm__ DTYPE_Z* zGm;

    uint32_t batch;
    uint32_t inFeat;
    uint32_t outFeat;
    uint32_t kernelSize;
    float scale;

    uint32_t rowsPerBlock;
    uint32_t rowStart;
    uint32_t rowEnd;
};

extern "C" __global__ __aicore__ void matmul_max_pool_sum_scale_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulMaxPoolSumScale op;
    op.Init(x, weight, bias, z,
            tiling_data.batch, tiling_data.inFeat, tiling_data.outFeat,
            tiling_data.kernelSize, tiling_data.scaleFactor);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_features, out_features, kernel_size, scale_factor):
        super().__init__()
        self.matmul = torch.nn.Linear(in_features, out_features)
        self.kernel_size = int(kernel_size)
        self.scale_factor = float(scale_factor)

    def forward(self, x):
        # Fused custom op: (x @ W^T + b) -> MaxPool1d(kernel_size, stride=kernel_size) -> sum(dim=1) -> scale
        return custom_ops_lib.matmul_max_pool_sum_scale_custom(
            x, self.matmul.weight, self.matmul.bias, self.kernel_size, self.scale_factor
        )
