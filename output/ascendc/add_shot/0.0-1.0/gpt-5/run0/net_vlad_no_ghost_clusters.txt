kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "net_vlad_no_ghost_clusters_custom_tiling_data.h"

// Computes vlad_pre = sum_n assignment[b,n,k]*x[b,n,d] - (sum_n assignment[b,n,k]) * clusters2[0,d,k]
// Shapes:
//  assignment: [B, N, K]
//  x:         [B, N, D]
//  clusters2: [1, D, K]
//  out:       [B, D, K]

extern "C" __global__ __aicore__ void net_vlad_no_ghost_clusters_custom(GM_ADDR assignment,
                                                                        GM_ADDR x,
                                                                        GM_ADDR clusters2,
                                                                        GM_ADDR out,
                                                                        GM_ADDR workspace,
                                                                        GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t B = tiling_data.B;
    const uint32_t N = tiling_data.N;
    const uint32_t D = tiling_data.D;
    const uint32_t K = tiling_data.K;
    const uint32_t total = tiling_data.totalLength;

    const uint32_t block_num = AscendC::GetBlockNum();
    const uint32_t block_idx = AscendC::GetBlockIdx();

    // balanced split
    uint32_t base = (total + block_num - 1) / block_num;
    uint32_t start = base * block_idx;
    uint32_t end = start + base;
    if (end > total) end = total;

    __gm__ float* asn = reinterpret_cast<__gm__ float*>(assignment);
    __gm__ float* xgm = reinterpret_cast<__gm__ float*>(x);
    __gm__ float* c2  = reinterpret_cast<__gm__ float*>(clusters2);
    __gm__ float* y   = reinterpret_cast<__gm__ float*>(out);

    for (uint32_t idx = start; idx < end; ++idx) {
        // map idx -> (b, d, k) with layout [B, D, K]
        uint32_t tmp = idx;
        uint32_t k = tmp % K; tmp /= K;
        uint32_t d = tmp % D; tmp /= D;
        uint32_t b = tmp;

        float dot = 0.f;
        float asum = 0.f;

        // offsets base
        uint64_t asn_bn_base = (uint64_t)b * (uint64_t)N * (uint64_t)K;
        uint64_t x_bn_base   = (uint64_t)b * (uint64_t)N * (uint64_t)D;

        for (uint32_t n = 0; n < N; ++n) {
            uint64_t a_off = asn_bn_base + (uint64_t)n * (uint64_t)K + (uint64_t)k;
            uint64_t x_off = x_bn_base   + (uint64_t)n * (uint64_t)D + (uint64_t)d;
            float aval = asn[a_off];
            float xval = xgm[x_off];
            dot  += aval * xval;
            asum += aval;
        }

        uint64_t c2_off = (uint64_t)d * (uint64_t)K + (uint64_t)k;
        float c2v = c2[c2_off];

        float outv = dot - asum * c2v;

        uint64_t y_off = ((uint64_t)b * (uint64_t)D + (uint64_t)d) * (uint64_t)K + (uint64_t)k;
        y[y_off] = outv;
    }
}

model_src='''
import torch
import torch.nn as nn
import torch.nn.functional as F
import torch as th
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    def __init__(self, cluster_size, feature_size, ghost_clusters):
        super(ModelNew, self).__init__()
        self.feature_size = feature_size
        self.cluster_size = cluster_size
        self.ghost_clusters = ghost_clusters

        import math
        init_sc = (1 / math.sqrt(feature_size))
        clusters = cluster_size + ghost_clusters

        self.clusters = nn.Parameter(init_sc * th.randn(feature_size, clusters))
        self.batch_norm = nn.BatchNorm1d(clusters)
        self.clusters2 = nn.Parameter(init_sc * th.randn(1, feature_size, cluster_size))
        self.out_dim = self.cluster_size * feature_size

    def forward(self, x, mask=None):
        # x: [B, N, D]
        B, N, D = x.shape[0], x.shape[1], x.shape[2]
        x_flat = x.view(-1, self.feature_size)  # BN x D

        if x_flat.device != self.clusters.device:
            raise ValueError(f"x.device {x_flat.device} != cluster.device {self.clusters.device}")

        assignment = th.matmul(x_flat, self.clusters)  # BN x (K+G)
        assignment = self.batch_norm(assignment)
        assignment = F.softmax(assignment, dim=1)
        assignment = assignment[:, :self.cluster_size]  # BN x K
        assignment = assignment.view(B, N, self.cluster_size)  # B x N x K

        # Custom fused kernel to compute:
        # vlad_pre = sum_n a[b,n,k]*x[b,n,d] - (sum_n a[b,n,k]) * c2[0,d,k]
        vlad_pre = custom_ops_lib.net_vlad_no_ghost_clusters_custom(assignment, x, self.clusters2)  # B x D x K

        # L2 intra norm along D
        vlad = F.normalize(vlad_pre, dim=1)

        # flattening + L2 norm
        vlad = vlad.reshape(B, self.cluster_size * self.feature_size)
        vlad = F.normalize(vlad, dim=1)
        return vlad
'''
