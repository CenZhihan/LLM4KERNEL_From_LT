    ConvolutionalVisionTransformerCustomTilingData tiling;

    uint32_t batch = static_cast<uint32_t>(shape0.GetDim(0));
    uint32_t hidden = static_cast<uint32_t>(shape0.GetDim(2)); // (B, 1, D)

    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* in0 = context->GetInputShape(0);
    const gert::Shape* in1 = context->GetInputShape(1);
    if (in0 == nullptr || in1 == nullptr) {
        return ge::GRAPH_FAILED;
    }
    gert::Shape* out = context->GetOutputShape(0);
    *out = *in0;
    // out dim1 = in0 dim1 + in1 dim1
    int64_t dim1_0 = in0->GetDim(1);
    int64_t dim1_1 = in1->GetDim(1);
    out->SetDim(1, dim1_0 + dim1_1);
    return GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "convolutional_vision_transformer_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 1; // buffers per queue

class KernelCViTConcat {
public:
    __aicore__ inline KernelCViTConcat() {}
    __aicore__ inline void Init(GM_ADDR cls, GM_ADDR emb, GM_ADDR out,
                                uint32_t totalLength, uint32_t tileNum,
                                uint32_t batch, uint32_t hidden)
    {
        this->batch = batch;
        this->hidden = hidden;
        (void)totalLength;
        (void)tileNum;

        clsGm.SetGlobalBuffer((__gm__ DTYPE_X *)cls, batch * hidden);
        embGm.SetGlobalBuffer((__gm__ DTYPE_Y *)emb, batch * hidden);
        outGm.SetGlobalBuffer((__gm__ DTYPE_Z *)out, batch * 2 * hidden);

        pipe.InitBuffer(inQueueCls, BUFFER_NUM, this->hidden * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueEmb, BUFFER_NUM, this->hidden * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t b = 0; b < this->batch; ++b) {
            // Copy CLS token: out[b, 0, :]
            AscendC::LocalTensor<DTYPE_X> clsLocal = inQueueCls.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(clsLocal, clsGm[b * this->hidden], this->hidden);
            inQueueCls.EnQue(clsLocal);
            AscendC::LocalTensor<DTYPE_X> clsLocalD = inQueueCls.DeQue<DTYPE_X>();
            AscendC::DataCopy(outGm[b * (this->hidden * 2) + 0], clsLocalD, this->hidden);
            inQueueCls.FreeTensor(clsLocalD);

            // Copy EMB token: out[b, 1, :]
            AscendC::LocalTensor<DTYPE_Y> embLocal = inQueueEmb.AllocTensor<DTYPE_Y>();
            AscendC::DataCopy(embLocal, embGm[b * this->hidden], this->hidden);
            inQueueEmb.EnQue(embLocal);
            AscendC::LocalTensor<DTYPE_Y> embLocalD = inQueueEmb.DeQue<DTYPE_Y>();
            AscendC::DataCopy(outGm[b * (this->hidden * 2) + this->hidden], embLocalD, this->hidden);
            inQueueEmb.FreeTensor(embLocalD);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueCls, inQueueEmb;
    AscendC::GlobalTensor<DTYPE_X> clsGm;
    AscendC::GlobalTensor<DTYPE_Y> embGm;
    AscendC::GlobalTensor<DTYPE_Z> outGm;
    uint32_t batch;
    uint32_t hidden;
};

extern "C" __global__ __aicore__ void convolutional_vision_transformer_custom(GM_ADDR cls, GM_ADDR emb, GM_ADDR out, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCViTConcat op;
    op.Init(cls, emb, out,
            tiling_data.totalLength,
            tiling_data.tileNum,
            tiling_data.batch,
            tiling_data.hiddenSize);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, num_classes, embed_dim=512, num_heads=8, num_layers=6, 
                 mlp_ratio=4.0, patch_size=4, in_channels=3, image_size=32):
        super(ModelNew, self).__init__()

        self.patch_size = patch_size
        self.image_size = image_size
        self.embed_dim = embed_dim

        self.conv1 = torch.nn.Conv2d(in_channels, embed_dim, kernel_size=patch_size, stride=patch_size)
        num_patches = (image_size // patch_size) ** 2
        self.linear_proj = torch.nn.Linear(embed_dim * num_patches, embed_dim)

        self.transformer_layers = torch.nn.ModuleList([
            torch.nn.TransformerEncoderLayer(
                d_model=embed_dim,
                nhead=num_heads,
                dim_feedforward=int(embed_dim * mlp_ratio),
                dropout=0.0,
                batch_first=True
            ) for _ in range(num_layers)
        ])

        self.cls_token = torch.nn.Parameter(torch.zeros(1, 1, embed_dim))
        self.fc_out = torch.nn.Linear(embed_dim, num_classes)

    def forward(self, x):
        B = x.size(0)
        x = self.conv1(x)                  # (B, embed_dim, H/patch_size, W/patch_size)
        x = x.flatten(start_dim=1)         # (B, embed_dim * num_patches)
        x = self.linear_proj(x)            # (B, embed_dim)

        cls_tokens = self.cls_token.expand(B, -1, -1)  # (B, 1, embed_dim)
        x = custom_ops_lib.convolutional_vision_transformer_custom(cls_tokens, x.unsqueeze(1))  # (B, 2, embed_dim)

        for layer in self.transformer_layers:
            x = layer(x)

        return self.fc_out(x[:, 0])
'''
