inline int64_t PoolOutDim(int64_t in, int64_t kernel, int64_t stride, int64_t pad) {
    // PyTorch AvgPool3d default: ceil_mode = false
    return (in + 2 * pad - kernel) / stride + 1;
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM  = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    AveragePooling3dCustomTilingData tiling;

    // Input shape: N, C, D, H, W
    auto inShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t n  = static_cast<uint32_t>(inShape.GetDim(0));
    uint32_t c  = static_cast<uint32_t>(inShape.GetDim(1));
    uint32_t inD = static_cast<uint32_t>(inShape.GetDim(2));
    uint32_t inH = static_cast<uint32_t>(inShape.GetDim(3));
    uint32_t inW = static_cast<uint32_t>(inShape.GetDim(4));

    // Get attributes
    int64_t k  = 1;
    int64_t s  = 1;
    int64_t p  = 0;
    (void)context->GetAttrs()->GetInt("kernel_size", k);
    (void)context->GetAttrs()->GetInt("stride", s);
    (void)context->GetAttrs()->GetInt("padding", p);

    uint32_t outD = static_cast<uint32_t>(PoolOutDim(inD, k, s, p));
    uint32_t outH = static_cast<uint32_t>(PoolOutDim(inH, k, s, p));
    uint32_t outW = static_cast<uint32_t>(PoolOutDim(inW, k, s, p));

    uint64_t totalOut = static_cast<uint64_t>(n) * c * outD * outH * outW;

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_inD(inD);
    tiling.set_inH(inH);
    tiling.set_inW(inW);
    tiling.set_outD(outD);
    tiling.set_outH(outH);
    tiling.set_outW(outW);
    tiling.set_kernelSize(static_cast<uint32_t>(k));
    tiling.set_stride(static_cast<uint32_t>(s));
    tiling.set_padding(static_cast<uint32_t>(p));
    tiling.set_totalOutLength(static_cast<uint32_t>(totalOut));
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    // Input shape: N, C, D, H, W
    const auto* x = context->GetInputShape(0);
    auto n  = x->GetOriginShape().GetDim(0);
    auto c  = x->GetOriginShape().GetDim(1);
    auto inD = x->GetOriginShape().GetDim(2);
    auto inH = x->GetOriginShape().GetDim(3);
    auto inW = x->GetOriginShape().GetDim(4);

    int64_t k = 1, s = 1, p = 0;
    (void)context->GetAttrs()->GetInt("kernel_size", k);
    (void)context->GetAttrs()->GetInt("stride", s);
    (void)context->GetAttrs()->GetInt("padding", p);

    auto PoolOutDim = [](int64_t in, int64_t kernel, int64_t stride, int64_t pad) -> int64_t {
        return (in + 2 * pad - kernel) / stride + 1;
    };

    int64_t outD = PoolOutDim(inD, k, s, p);
    int64_t outH = PoolOutDim(inH, k, s, p);
    int64_t outW = PoolOutDim(inW, k, s, p);

}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "average_pooling_3d_custom_tiling_data.h"

class KernelAvgPool3D {
public:
    __aicore__ inline KernelAvgPool3D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t n, uint32_t c,
                                uint32_t inD, uint32_t inH, uint32_t inW,
                                uint32_t outD, uint32_t outH, uint32_t outW,
                                uint32_t k, uint32_t s, uint32_t p,
                                uint32_t totalOutLength)
    {
        this->n = n;
        this->c = c;
        this->inD = inD; this->inH = inH; this->inW = inW;
        this->outD = outD; this->outH = outH; this->outW = outW;
        this->k = k; this->s = s; this->p = p;

        this->totalOut = static_cast<uint64_t>(n) * c * outD * outH * outW;
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t blkIdx = AscendC::GetBlockIdx();
        uint64_t perBlock = (this->totalOut + blocks - 1) / blocks;
        this->start = perBlock * blkIdx;
        this->end = AscendC::Min(this->start + perBlock, this->totalOut);

        xPtr = (__gm__ DTYPE_X*)x;
        yPtr = (__gm__ DTYPE_Z*)y;
    }

    __aicore__ inline void Process()
    {
        for (uint64_t idx = this->start; idx < this->end; ++idx) {
            // Linear index to N,C,OD,OH,OW
            uint64_t t = idx;
            uint32_t ow = static_cast<uint32_t>(t % outW); t /= outW;
            uint32_t oh = static_cast<uint32_t>(t % outH); t /= outH;
            uint32_t od = static_cast<uint32_t>(t % outD); t /= outD;
            uint32_t cc = static_cast<uint32_t>(t % c);   t /= c;
            uint32_t nn = static_cast<uint32_t>(t);

            int64_t id_start = static_cast<int64_t>(od) * s - p;
            int64_t ih_start = static_cast<int64_t>(oh) * s - p;
            int64_t iw_start = static_cast<int64_t>(ow) * s - p;

            float sum = 0.0f;
            uint32_t count = 0;

            for (uint32_t kd = 0; kd < k; ++kd) {
                int64_t id = id_start + static_cast<int64_t>(kd);
                if (id < 0 || id >= static_cast<int64_t>(inD)) continue;
                for (uint32_t kh = 0; kh < k; ++kh) {
                    int64_t ih = ih_start + static_cast<int64_t>(kh);
                    if (ih < 0 || ih >= static_cast<int64_t>(inH)) continue;
                    for (uint32_t kw = 0; kw < k; ++kw) {
                        int64_t iw = iw_start + static_cast<int64_t>(kw);
                        if (iw < 0 || iw >= static_cast<int64_t>(inW)) continue;

                        uint64_t inIndex =
                            (((static_cast<uint64_t>(nn) * c + cc) * inD + static_cast<uint32_t>(id)) * inH
                              + static_cast<uint32_t>(ih)) * inW + static_cast<uint32_t>(iw);
                        sum += static_cast<float>(xPtr[inIndex]);
                        ++count;
                    }
                }
            }
            float avg = (count > 0) ? (sum / static_cast<float>(count)) : 0.0f;

            uint64_t outIndex = idx;
            yPtr[outIndex] = static_cast<DTYPE_Z>(avg);
        }
    }

private:
    __gm__ DTYPE_X* xPtr;
    __gm__ DTYPE_Z* yPtr;

    uint32_t n, c, inD, inH, inW;
    uint32_t outD, outH, outW;
    uint32_t k, s, p;

    uint64_t totalOut;
    uint64_t start, end;
};

extern "C" __global__ __aicore__ void average_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelAvgPool3D op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c,
            tiling_data.inD, tiling_data.inH, tiling_data.inW,
            tiling_data.outD, tiling_data.outH, tiling_data.outW,
            tiling_data.kernelSize, tiling_data.stride, tiling_data.padding,
            tiling_data.totalOutLength);
    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

static inline int64_t pool_out(int64_t in, int64_t k, int64_t s, int64_t p) {
    return (in + 2 * p - k) / s + 1;
}

at::Tensor average_pooling_3d_custom_impl_npu(const at::Tensor& x, int64_t kernel_size, int64_t stride, int64_t padding) {
    TORCH_CHECK(x.dim() == 5, "Input must be 5D (N,C,D,H,W)");
    auto sizes = x.sizes();
    int64_t N = sizes[0], C = sizes[1], D = sizes[2], H = sizes[3], W = sizes[4];

    if (stride <= 0) {
        stride = kernel_size;
    }

    int64_t outD = pool_out(D, kernel_size, stride, padding);
    int64_t outH = pool_out(H, kernel_size, stride, padding);
    int64_t outW = pool_out(W, kernel_size, stride, padding);

    at::Tensor result = at::empty({N, C, outD, outH, outW}, x.options());
    EXEC_NPU_CMD(aclnnAveragePooling3dCustom, x, (int)kernel_size, (int)stride, (int)padding, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("average_pooling_3d_custom", &average_pooling_3d_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("average_pooling_3d_custom", &average_pooling_3d_custom_impl_npu, "3D Average Pooling");
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Model using custom AscendC Average Pooling 3D kernel.
    """
    def __init__(self, kernel_size: int, stride: int = None, padding: int = 0):
        super().__init__()
        self.kernel_size = int(kernel_size)
        self.stride = int(stride) if stride is not None else None
        self.padding = int(padding)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        k = self.kernel_size
        s = self.stride if self.stride is not None else k
        p = self.padding
        return custom_ops_lib.average_pooling_3d_custom(x, k, s, p)
