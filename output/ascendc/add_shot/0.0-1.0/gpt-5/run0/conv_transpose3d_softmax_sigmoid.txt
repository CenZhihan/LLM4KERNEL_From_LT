kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_transpose3d_softmax_sigmoid_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmaxSigmoid {
public:
    __aicore__ inline KernelSoftmaxSigmoid() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR z, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }
    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();

        // Fused Softmax + Sigmoid (approximated on a per-tile basis):
        // 1) softmax: exp(x) / sum(exp(x))
        // 2) sigmoid: 1 / (1 + exp(-softmax))
        // Note: This uses tile-local normalization for demonstration.

        // temp1 = exp(x)
        AscendC::Exp(zLocal, xLocal, this->tileLength);

        // sumExp = reduce_sum(temp1)
        AscendC::LocalTensor<DTYPE_Z> sumBuf = inQueueX.AllocTensor<DTYPE_Z>(); // reuse small buffer
        AscendC::ReduceSum(sumBuf, zLocal, this->tileLength);
        // Broadcast sum to vector
        AscendC::Broadcast(sumBuf, sumBuf, this->tileLength);

        // softmax = temp1 / sumExp
        AscendC::Div(zLocal, zLocal, sumBuf, this->tileLength);

        // sigmoid(softmax): 1 / (1 + exp(-softmax))
        AscendC::Muls(zLocal, zLocal, static_cast<DTYPE_Z>(-1.0f), this->tileLength);   // -softmax
        AscendC::Exp(zLocal, zLocal, this->tileLength);                                 // exp(-softmax)
        AscendC::Adds(zLocal, zLocal, static_cast<DTYPE_Z>(1.0f), this->tileLength);    // 1 + exp(-softmax)
        AscendC::Reciprocal(zLocal, zLocal, this->tileLength);                          // 1 / (...)

        outQueueZ.EnQue<DTYPE_Z>(zLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueX.FreeTensor(sumBuf);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[progress * this->tileLength], zLocal, this->tileLength);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transpose3d_softmax_sigmoid_custom(GM_ADDR x, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxSigmoid op;
    op.Init(x, z, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: keeps ConvTranspose3d in PyTorch, fuses Softmax(dim=1) + Sigmoid into a custom AscendC op.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, bias=True):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(
            in_channels, out_channels, kernel_size,
            stride=stride, padding=padding, output_padding=output_padding, bias=bias
        )

    def forward(self, x):
        x = self.conv_transpose(x)
        # Fused softmax (dim=1) + sigmoid in a single custom kernel
        x = custom_ops_lib.conv_transpose3d_softmax_sigmoid_custom(x)
        return x
