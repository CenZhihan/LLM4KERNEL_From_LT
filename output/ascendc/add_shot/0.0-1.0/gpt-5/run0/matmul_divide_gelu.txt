kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "matmul_divide_gelu_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMatmulDivGelu {
public:
    __aicore__ inline KernelMatmulDivGelu() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR d, GM_ADDR y,
                                uint32_t B, uint32_t K, uint32_t N,
                                uint32_t rowsPerBlock, uint32_t tileN)
    {
        this->B = B;
        this->K = K;
        this->N = N;
        this->rowsPerBlock = rowsPerBlock;
        this->tileN = tileN;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        startRow = blockIdx * rowsPerBlock;
        endRow = AscendC::Min(startRow + rowsPerBlock, B);

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + startRow * K, (endRow - startRow) * K);
        // w: [K, N] laid out row-major
        wGm.SetGlobalBuffer((__gm__ DTYPE_W*)w, K * N);
        bGm.SetGlobalBuffer((__gm__ DTYPE_B*)b, N);
        dGm.SetGlobalBuffer((__gm__ DTYPE_D*)d, 1);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + startRow * N, (endRow - startRow) * N);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, K * sizeof(DTYPE_X));            // one row of x
        pipe.InitBuffer(inQueueW, BUFFER_NUM, tileN * sizeof(DTYPE_W));        // tile of w along N
        pipe.InitBuffer(inQueueB, BUFFER_NUM, tileN * sizeof(DTYPE_B));        // tile of bias
        pipe.InitBuffer(outQueueY, BUFFER_NUM, tileN * sizeof(DTYPE_Y));       // tile of y
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, tileN * sizeof(DTYPE_W));        // temporary for mul/add

        // Load divisor once
        AscendC::LocalTensor<DTYPE_D> dLocal = tmpScalar.AllocTensor<DTYPE_D>();
        AscendC::DataCopy(dLocal, dGm[0], 1);
        tmpScalar.EnQue(dLocal);
        dLocal = tmpScalar.DeQue<DTYPE_D>();
        // read scalar value
        divisor = static_cast<float>(dLocal.GetValue(0));
        invDiv = 1.0f / divisor;
        tmpScalar.FreeTensor(dLocal);
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < (endRow - startRow); ++r) {
            // Load x row
            AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(xLocal, xGm[r * K], K);
            inQueueX.EnQue(xLocal);
            xLocal = inQueueX.DeQue<DTYPE_X>();

            for (uint32_t nStart = 0; nStart < N; nStart += tileN) {
                uint32_t curTile = AscendC::Min(tileN, N - nStart);

                // accumulator init to 0
                AscendC::LocalTensor<DTYPE_Y> acc = outQueueY.AllocTensor<DTYPE_Y>();
                AscendC::Dup(acc, 0.0f, curTile);

                // GEMM accumulate: acc += x[r, k] * w[k, nStart:nStart+curTile]
                for (uint32_t k = 0; k < K; ++k) {
                    // load w tile (row k, columns nStart..)
                    AscendC::LocalTensor<DTYPE_W> wTile = inQueueW.AllocTensor<DTYPE_W>();
                    AscendC::DataCopy(wTile, wGm[k * N + nStart], curTile);

                    // multiply by scalar x[r, k] in place
                    float xk = static_cast<float>(xLocal.GetValue(k));
                    AscendC::Muls(wTile, wTile, xk, curTile);

                    // acc += wTile
                    AscendC::Add(acc, acc, wTile, curTile);

                    inQueueW.FreeTensor(wTile);
                }

                // add bias
                AscendC::LocalTensor<DTYPE_B> bTile = inQueueB.AllocTensor<DTYPE_B>();
                AscendC::DataCopy(bTile, bGm[nStart], curTile);
                AscendC::Add(acc, acc, bTile, curTile);
                inQueueB.FreeTensor(bTile);

                // divide by scalar
                AscendC::Muls(acc, acc, invDiv, curTile);

                // GELU activation (vector)
                AscendC::Gelu(acc, acc, curTile);

                // write back
                AscendC::DataCopy(yGm[r * N + nStart], acc, curTile);
                outQueueY.FreeTensor(acc);
            }

            inQueueX.FreeTensor(xLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpScalar;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_D> dGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t B;
    uint32_t K;
    uint32_t N;
    uint32_t rowsPerBlock;
    uint32_t tileN;
    uint32_t startRow;
    uint32_t endRow;
    float divisor;
    float invDiv;
};

extern "C" __global__ __aicore__ void matmul_divide_gelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR d, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulDivGelu op;
    op.Init(x, w, b, d, y,
            tiling_data.batch, tiling_data.inK, tiling_data.outN,
            tiling_data.rowsPerBlock, tiling_data.tileN);
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    """
    Fused NPU op: y = GELU( (x @ W^T + b) / divisor )
    We pass W^T (i.e., weight transposed) as [K, N] to the custom kernel.
    """
    def __init__(self, input_size, output_size, divisor):
        super(ModelNew, self).__init__()
        self.linear = nn.Linear(input_size, output_size)
        self.divisor = float(divisor)

    def forward(self, x):
        # Prepare operands for fused custom kernel:
        # x: [B, K], w_t: [K, N], b: [N]
        w_t = self.linear.weight.t().contiguous()
        b = self.linear.bias.contiguous()
        return custom_ops_lib.matmul_divide_gelu_custom(x, w_t, b, self.divisor)
