kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv2d_subtract_tanh_subtract_avg_pool_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelFusedEltwise {
public:
    __aicore__ inline KernelFusedEltwise() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR sub1, GM_ADDR sub2, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        s1Gm.SetGlobalBuffer((__gm__ DTYPE_X *)sub1 + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        s2Gm.SetGlobalBuffer((__gm__ DTYPE_X *)sub2 + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_X *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueS1, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueS2, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> s1Local = inQueueS1.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> s2Local = inQueueS2.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal,  xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(s1Local, s1Gm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(s2Local, s2Gm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
        inQueueS1.EnQue(s1Local);
        inQueueS2.EnQue(s2Local);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal  = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> s1Local = inQueueS1.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> s2Local = inQueueS2.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> yLocal  = outQueueY.AllocTensor<DTYPE_X>();

        // y = tanh(x - sub1) - sub2
        AscendC::Sub(yLocal, xLocal, s1Local, this->tileLength);
        AscendC::Tanh(yLocal, yLocal, this->tileLength);
        AscendC::Sub(yLocal, yLocal, s2Local, this->tileLength);

        outQueueY.EnQue<DTYPE_X>(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueS1.FreeTensor(s1Local);
        inQueueS2.FreeTensor(s2Local);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> yLocal = outQueueY.DeQue<DTYPE_X>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueX, inQueueS1, inQueueS2;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_X> s1Gm;
    AscendC::GlobalTensor<DTYPE_X> s2Gm;
    AscendC::GlobalTensor<DTYPE_X> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv2d_subtract_tanh_subtract_avg_pool_custom(GM_ADDR x, GM_ADDR sub1, GM_ADDR sub2, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelFusedEltwise op;
    op.Init(x, sub1, sub2, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model that keeps PyTorch Conv2d and AvgPool2d, and fuses the sequence:
    subtract -> tanh -> subtract into a custom AscendC operator.
    """
    def __init__(self, in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)
        self.subtract1_value = float(subtract1_value)
        self.subtract2_value = float(subtract2_value)
        self.avgpool = torch.nn.AvgPool2d(kernel_size_pool)

    def forward(self, x):
        x = self.conv(x)
        sub1_tensor = torch.full_like(x, self.subtract1_value)
        sub2_tensor = torch.full_like(x, self.subtract2_value)
        x = custom_ops_lib.conv2d_subtract_tanh_subtract_avg_pool_custom(x, sub1_tensor, sub2_tensor)
        x = self.avgpool(x)
        return x
