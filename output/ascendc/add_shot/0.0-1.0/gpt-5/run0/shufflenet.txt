const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)

    ShufflenetCustomTilingData tiling;

    uint32_t dims = inShape.GetDimNum();

    uint32_t c = static_cast<uint32_t>(inShape.GetDim(1));
    uint32_t h = static_cast<uint32_t>(inShape.GetDim(2));
    uint32_t w = static_cast<uint32_t>(inShape.GetDim(3));
    uint32_t hw = h * w;
    uint32_t totalLength = static_cast<uint32_t>(inShape.GetShapeSize());

    int64_t groups_attr = 1;
    // Try get attr 'groups' if provided
    if (context->HasAttr("groups")) {
        groups_attr = context->GetAttr<int64_t>("groups");

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_totalLength(totalLength);
    tiling.set_tileNum(TILE_NUM);
    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_hw(hw);
    tiling.set_groups(static_cast<uint32_t>(groups_attr));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "shufflenet_custom_tiling_data.h"

typedef float DTYPE_X;
typedef float DTYPE_Y;

constexpr int32_t BUFFER_NUM = 1;

class KernelShuffle {
public:
    __aicore__ inline KernelShuffle() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileNum,
                                uint32_t n, uint32_t c, uint32_t hw, uint32_t groups)
    {
        this->totalLength = totalLength;
        this->tileNum = tileNum;
        this->n = n;
        this->c = c;
        this->hw = hw;
        this->g = groups;
        this->cpg = c / groups;

        // Determine plane distribution across blocks
        uint32_t totalPlanes = n * c;
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t basePlanes = totalPlanes / blockNum;
        uint32_t rem = totalPlanes % blockNum;
        this->planesStart = blockIdx * basePlanes + (blockIdx < rem ? blockIdx : rem);
        uint32_t myPlanes = basePlanes + (blockIdx < rem ? 1 : 0);
        this->planesEnd = this->planesStart + myPlanes;

        // Set global buffers to full tensors to allow arbitrary indexing
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, totalLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, totalLength);

        // Tile length per transfer within one channel plane
        uint32_t suggested = hw / (tileNum == 0 ? 1 : tileNum);
        if (suggested == 0) suggested = hw;
        // Cap to at least 1 and at most hw
        this->tileLength = suggested > hw ? hw : suggested;
        if (this->tileLength == 0) this->tileLength = 1;

        pipe.InitBuffer(inQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t plane = this->planesStart; plane < this->planesEnd; ++plane) {
            // plane maps to (n_idx, c_out)
            uint32_t n_idx = plane / this->c;
            uint32_t c_out = plane % this->c;

            // Channel shuffle mapping
            uint32_t i = c_out / this->g;         // index in cpg
            uint32_t j = c_out % this->g;         // group index
            uint32_t c_in = j * this->cpg + i;

            uint64_t src_base = (static_cast<uint64_t>(n_idx) * this->c + c_in) * this->hw;
            uint64_t dst_base = (static_cast<uint64_t>(plane)) * this->hw;

            uint32_t offset = 0;
            while (offset < this->hw) {
                uint32_t len = this->tileLength;
                if (offset + len > this->hw) {
                    len = this->hw - offset;
                }
                CopyIn(src_base + offset, len);
                Compute(len);
                CopyOut(dst_base + offset, len);
                offset += len;
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint64_t gm_index, uint32_t length)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueue.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[gm_index], length);
        inQueue.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t length)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueue.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueue.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(yLocal, xLocal, length);
        outQueue.EnQue<DTYPE_Y>(yLocal);
        inQueue.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint64_t gm_index, uint32_t length)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueue.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[gm_index], yLocal, length);
        outQueue.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t totalLength;
    uint32_t tileNum;
    uint32_t n;
    uint32_t c;
    uint32_t hw;
    uint32_t g;
    uint32_t cpg;
    uint32_t planesStart;
    uint32_t planesEnd;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void shufflenet_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelShuffle op;
    op.Init(x, y,
            tiling_data.totalLength, tiling_data.tileNum,
            tiling_data.n, tiling_data.c, tiling_data.hw, tiling_data.groups);
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch.nn.functional as F
import torch_npu
import custom_ops_lib

class ChannelShuffleCustom(nn.Module):
    def __init__(self, groups: int):
        super().__init__()
        self.groups = groups

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return custom_ops_lib.shufflenet_custom(x, self.groups)

class ShuffleNetUnitNew(nn.Module):
    def __init__(self, in_channels, out_channels, groups=3):
        super(ShuffleNetUnitNew, self).__init__()
        assert out_channels % 4 == 0
        mid_channels = out_channels // 4

        self.conv1 = nn.Conv2d(in_channels, mid_channels, kernel_size=1, stride=1, padding=0, groups=groups, bias=False)
        self.bn1 = nn.BatchNorm2d(mid_channels)

        self.conv2 = nn.Conv2d(mid_channels, mid_channels, kernel_size=3, stride=1, padding=1, groups=mid_channels, bias=False)
        self.bn2 = nn.BatchNorm2d(mid_channels)

        self.conv3 = nn.Conv2d(mid_channels, out_channels, kernel_size=1, stride=1, padding=0, groups=groups, bias=False)
        self.bn3 = nn.BatchNorm2d(out_channels)

        self.shuffle = ChannelShuffleCustom(groups)

        if in_channels == out_channels:
            self.shortcut = nn.Sequential()
        else:
            self.shortcut = nn.Sequential(
                nn.Conv2d(in_channels, out_channels, kernel_size=1, stride=1, padding=0, bias=False),
                nn.BatchNorm2d(out_channels)
            )

    def forward(self, x):
        out = F.relu(self.bn1(self.conv1(x)))
        out = self.bn2(self.conv2(out))
        out = self.shuffle(out)
        out = F.relu(self.bn3(self.conv3(out)))
        out = out + self.shortcut(x)
        return out

class ModelNew(nn.Module):
    def __init__(self, num_classes=1000, groups=3, stages_repeats=[3, 7, 3], stages_out_channels=[24, 240, 480, 960]):
        super(ModelNew, self).__init__()

        self.conv1 = nn.Conv2d(3, stages_out_channels[0], kernel_size=3, stride=2, padding=1, bias=False)
        self.bn1 = nn.BatchNorm2d(stages_out_channels[0])
        self.maxpool = nn.MaxPool2d(kernel_size=3, stride=2, padding=1)

        self.stage2 = self._make_stage(stages_out_channels[0], stages_out_channels[1], stages_repeats[0], groups)
        self.stage3 = self._make_stage(stages_out_channels[1], stages_out_channels[2], stages_repeats[1], groups)
        self.stage4 = self._make_stage(stages_out_channels[2], stages_out_channels[3], stages_repeats[2], groups)

        self.conv5 = nn.Conv2d(stages_out_channels[3], 1024, kernel_size=1, stride=1, padding=0, bias=False)
        self.bn5 = nn.BatchNorm2d(1024)

        self.fc = nn.Linear(1024, num_classes)

    def _make_stage(self, in_channels, out_channels, repeats, groups):
        layers = []
        layers.append(ShuffleNetUnitNew(in_channels, out_channels, groups))
        for _ in range(1, repeats):
            layers.append(ShuffleNetUnitNew(out_channels, out_channels, groups))
        return nn.Sequential(*layers)

    def forward(self, x):
        x = F.relu(self.bn1(self.conv1(x)))
        x = self.maxpool(x)

        x = self.stage2(x)
        x = self.stage3(x)
        x = self.stage4(x)

        x = F.relu(self.bn5(self.conv5(x)))
        x = F.adaptive_avg_pool2d(x, (1, 1))
        x = x.view(x.size(0), -1)
        x = self.fc(x)
        return x
'''
