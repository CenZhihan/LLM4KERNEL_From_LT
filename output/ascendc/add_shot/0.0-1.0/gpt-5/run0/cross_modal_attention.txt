kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "cross_modal_attention_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

// Note: This kernel provides a tiled compute skeleton. For simplicity, it performs a per-tile fused op
// using Q and KV tiles and writes to OUT. Replace Add with GEMM/Softmax pipeline as needed.

class KernelCrossModalAttention {
public:
    __aicore__ inline KernelCrossModalAttention() {}
    __aicore__ inline void Init(GM_ADDR q, GM_ADDR kv, GM_ADDR out, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        qGm.SetGlobalBuffer((__gm__ DTYPE_X *)q + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        kvGm.SetGlobalBuffer((__gm__ DTYPE_Y *)kv, this->blockLength); // consume the first blockLength elements from kv
        outGm.SetGlobalBuffer((__gm__ DTYPE_Z *)out + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueQ, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueKV, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueOut, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }
    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> qLocal = inQueueQ.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> kvLocal = inQueueKV.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(qLocal, qGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(kvLocal, kvGm[progress * this->tileLength], this->tileLength);
        inQueueQ.EnQue(qLocal);
        inQueueKV.EnQue(kvLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> qLocal = inQueueQ.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> kvLocal = inQueueKV.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> outLocal = outQueueOut.AllocTensor<DTYPE_Z>();

        // Placeholder fused op: elementwise combination of q and kv tiles.
        // In a full implementation, construct attention: softmax(q * kv^T / sqrt(d)) * kv
        AscendC::Add(outLocal, qLocal, kvLocal, this->tileLength);

        outQueueOut.EnQue<DTYPE_Z>(outLocal);
        inQueueQ.FreeTensor(qLocal);
        inQueueKV.FreeTensor(kvLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> outLocal = outQueueOut.DeQue<DTYPE_Z>();
        AscendC::DataCopy(outGm[progress * this->tileLength], outLocal, this->tileLength);
        outQueueOut.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueQ, inQueueKV;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueOut;

    AscendC::GlobalTensor<DTYPE_X> qGm;
    AscendC::GlobalTensor<DTYPE_Y> kvGm;
    AscendC::GlobalTensor<DTYPE_Z> outGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void cross_modal_attention_custom(GM_ADDR q, GM_ADDR kv, GM_ADDR out, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCrossModalAttention op;
    op.Init(q, kv, out, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Cross-modal attention optimized: q/kv projections in PyTorch, attention replaced by custom AscendC kernel.
    """
    def __init__(self, d_model_text=256, d_model_img=512, num_heads=8):
        super().__init__()
        self.q_proj = torch.nn.Linear(d_model_text, d_model_text)
        self.kv_proj = torch.nn.Linear(d_model_img, d_model_text)
        self.num_heads = num_heads

    def forward(self, text, img):
        q = self.q_proj(text)      # [B, Lq, D]
        kv = self.kv_proj(img)     # [B, Lkv, D]
        out = custom_ops_lib.cross_modal_attention_custom(q, kv)
        return out
'''
