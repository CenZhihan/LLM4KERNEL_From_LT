kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "mse_loss_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

// This kernel computes elementwise squared difference: (predictions - targets)^2
// The mean reduction is handled in the Python binding to keep kernel simple and efficient.
class KernelMseLoss {
public:
    __aicore__ inline KernelMseLoss() {}
    __aicore__ inline void Init(GM_ADDR predictions, GM_ADDR targets, GM_ADDR loss,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->totalLength = totalLength;
        this->blockNum = AscendC::GetBlockNum();
        this->blockIdx = AscendC::GetBlockIdx();

        // Simple partition; assume totalLength divisible for demo purposes
        this->blockLength = totalLength / this->blockNum;
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        predGm.SetGlobalBuffer((__gm__ DTYPE_X *)predictions + this->blockLength * this->blockIdx, this->blockLength);
        tgtGm.SetGlobalBuffer((__gm__ DTYPE_Y *)targets + this->blockLength * this->blockIdx, this->blockLength);
        outGm.SetGlobalBuffer((__gm__ DTYPE_Z *)loss + this->blockLength * this->blockIdx, this->blockLength);

        pipe.InitBuffer(inQueuePred, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueTgt, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> pLocal = inQueuePred.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> tLocal = inQueueTgt.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(pLocal, predGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(tLocal, tgtGm[progress * this->tileLength], this->tileLength);
        inQueuePred.EnQue(pLocal);
        inQueueTgt.EnQue(tLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> pLocal = inQueuePred.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> tLocal = inQueueTgt.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> oLocal = outQueue.AllocTensor<DTYPE_Z>();

        // oLocal = (pLocal - tLocal) ^ 2
        AscendC::Sub(oLocal, pLocal, tLocal, this->tileLength);
        AscendC::Mul(oLocal, oLocal, oLocal, this->tileLength);

        outQueue.EnQue<DTYPE_Z>(oLocal);
        inQueuePred.FreeTensor(pLocal);
        inQueueTgt.FreeTensor(tLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> oLocal = outQueue.DeQue<DTYPE_Z>();
        AscendC::DataCopy(outGm[progress * this->tileLength], oLocal, this->tileLength);
        outQueue.FreeTensor(oLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueuePred, inQueueTgt;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue;

    AscendC::GlobalTensor<DTYPE_X> predGm;
    AscendC::GlobalTensor<DTYPE_Y> tgtGm;
    AscendC::GlobalTensor<DTYPE_Z> outGm;

    uint32_t totalLength;
    uint32_t blockNum;
    uint32_t blockIdx;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void mse_loss_custom(GM_ADDR predictions, GM_ADDR targets, GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMseLoss op;
    op.Init(predictions, targets, loss, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor mse_loss_custom_impl_npu(const at::Tensor& predictions, const at::Tensor& targets) {
    // Kernel produces elementwise squared difference; reduce to mean here
    at::Tensor squared = at::empty_like(predictions);
    EXEC_NPU_CMD(aclnnMseLossCustom, predictions, targets, squared);
    return squared.mean();
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("mse_loss_custom", &mse_loss_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("mse_loss_custom", &mse_loss_custom_impl_npu, "Mean Squared Error loss");
}
