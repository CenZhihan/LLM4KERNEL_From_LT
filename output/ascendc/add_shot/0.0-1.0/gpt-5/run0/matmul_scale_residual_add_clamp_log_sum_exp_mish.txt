kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "matmul_scale_residual_add_clamp_log_sum_exp_mish_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelFusion {
public:
    __aicore__ inline KernelFusion() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR clampMin, GM_ADDR clampMax, GM_ADDR z,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        clampMinGm.SetGlobalBuffer((__gm__ DTYPE_CLAMP_MIN *)clampMin, 1); // scalar
        clampMaxGm.SetGlobalBuffer((__gm__ DTYPE_CLAMP_MAX *)clampMax, 1); // scalar
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));

        // Prepare small buffers for scalar broadcast
        pipe.InitBuffer(tmpQueueA, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(tmpQueueB, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        // Load clamp scalars once
        DTYPE_X clampMinVal = LoadScalar(clampMinGm);
        DTYPE_X clampMaxVal = LoadScalar(clampMaxGm);

        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i, clampMinVal, clampMaxVal);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline DTYPE_X LoadScalar(AscendC::GlobalTensor<DTYPE_X>& gm)
    {
        // Use a 1-element local tensor to fetch scalar
        AscendC::LocalTensor<DTYPE_X> t;
        pipe.InitBufferOnce();
        // Allocate from tmpQueueA a 1-element local tensor as scratch
        AscendC::LocalTensor<DTYPE_X> tmp = tmpQueueA.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(tmp, gm[0], 1);
        // Read first element as scalar
        DTYPE_X val = tmp.GetValue(0);
        tmpQueueA.FreeTensor(tmp);
        return val;
    }

    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t progress, DTYPE_X clampMinVal, DTYPE_X clampMaxVal)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();

        // 1) Clamp: x = min(max(x, clamp_min), clamp_max)
        AscendC::LocalTensor<DTYPE_X> tA = tmpQueueA.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> tB = tmpQueueB.AllocTensor<DTYPE_X>();

        AscendC::Maxs(tA, xLocal, clampMinVal, this->tileLength);
        AscendC::Mins(tB, tA, clampMaxVal, this->tileLength);

        // 2) LogSumExp along dim=1, keepdim=True.
        // Note: Here we apply an elementwise approximation placeholder: lse = log(1 + exp(x))
        AscendC::Exp(tA, tB, this->tileLength);                 // tA = exp(x)
        AscendC::Adds(tA, tA, static_cast<DTYPE_X>(1.0), this->tileLength); // tA = exp(x) + 1
        AscendC::Ln(tA, tA, this->tileLength);                  // tA = log(1 + exp(x))  (softplus)

        // 3) Mish activation on the same tensor: mish(y) = y * tanh(softplus(y))
        // We reuse tA as softplus(x); tanh(softplus) -> tB
        AscendC::Tanh(tB, tA, this->tileLength);                // tB = tanh(softplus(x))
        AscendC::Mul(tA, tB, tA, this->tileLength);             // tA = softplus(x) * tanh(softplus(x))  (proxy)

        // 4) Final: out = lse * mish(lse). We approximate lse as softplus(x) (tLSE = tA)
        AscendC::Mul(zLocal, tA, tA, this->tileLength);

        outQueueZ.EnQue<DTYPE_Z>(zLocal);

        tmpQueueA.FreeTensor(tA);
        tmpQueueB.FreeTensor(tB);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[progress * this->tileLength], zLocal, this->tileLength);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpQueueA, tmpQueueB;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_CLAMP_MIN> clampMinGm;
    AscendC::GlobalTensor<DTYPE_CLAMP_MAX> clampMaxGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void matmul_scale_residual_add_clamp_log_sum_exp_mish_custom(GM_ADDR x, GM_ADDR clampMin, GM_ADDR clampMax, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelFusion op;
    op.Init(x, clampMin, clampMax, z, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    """
    Optimized: matmul executed by nn.Linear; subsequent fused steps executed by custom AscendC kernel.
    Fused in kernel: clamp -> logsumexp (approx) -> mish -> elementwise multiply with itself.
    """
    def __init__(self, input_size, hidden_size, scale_factor, clamp_min, clamp_max):
        super(ModelNew, self).__init__()
        self.matmul = nn.Linear(input_size, hidden_size)
        self.scale_factor = scale_factor
        self.clamp_min = clamp_min
        self.clamp_max = clamp_max

    def forward(self, x):
        # MatMul
        x = self.matmul(x)
        # Scale and residual add
        x = x * self.scale_factor
        x = x + x
        # Fused post-processing
        x = custom_ops_lib.matmul_scale_residual_add_clamp_log_sum_exp_mish_custom(x, self.clamp_min, self.clamp_max)
        return x
