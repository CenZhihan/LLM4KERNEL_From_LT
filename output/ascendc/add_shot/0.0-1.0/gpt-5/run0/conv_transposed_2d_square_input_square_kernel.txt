    uint32_t cout = static_cast<uint32_t>(wShapeOrigin.GetDim(1));
    uint32_t k = static_cast<uint32_t>(wShapeOrigin.GetDim(2)); // square kernel, so k == wShapeOrigin.GetDim(3)

    // constraints: stride=1, padding=0, output_padding=0, groups=1
    uint32_t ho = h + k - 1;
    uint32_t wo = w + k - 1;

    ConvTransposed2dSquareInputSquareKernelCustomTilingData tiling;
    context->SetBlockDim(BLOCK_DIM);
    tiling.set_n(n);
    tiling.set_cin(cin);
    tiling.set_cout(cout);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_k(k);
    tiling.set_ho(ho);
    tiling.set_wo(wo);
    uint64_t total = static_cast<uint64_t>(n) * cout * ho * wo;
    tiling.set_totalElems(static_cast<uint32_t>(total)); // assuming fits uint32_t for demo

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);

    auto n = x_shape->GetDim(0);
    auto cin = x_shape->GetDim(1);
    auto h = x_shape->GetDim(2);
    auto w = x_shape->GetDim(3);

    auto cin_w = w_shape->GetDim(0);
    auto cout = w_shape->GetDim(1);
    auto k = w_shape->GetDim(2);

    if (cin != cin_w) {
        return ge::GRAPH_FAILED;

    auto wo = w + k - 1;

    y_shape->SetDim(0, n);
    y_shape->SetDim(1, cout);
    y_shape->SetDim(2, ho);
    y_shape->SetDim(3, wo);

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_transposed_2d_square_input_square_kernel_custom_tiling_data.h"

typedef float DTYPE_X;
typedef float DTYPE_W;
typedef float DTYPE_Y;

extern "C" __global__ __aicore__ void conv_transposed_2d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N = tiling_data.n;
    const uint32_t Cin = tiling_data.cin;
    const uint32_t Cout = tiling_data.cout;
    const uint32_t H = tiling_data.h;
    const uint32_t W = tiling_data.w;
    const uint32_t K = tiling_data.k;
    const uint32_t Ho = tiling_data.ho;
    const uint32_t Wo = tiling_data.wo;
    const uint64_t total = (uint64_t)N * Cout * Ho * Wo;

    __gm__ DTYPE_X* xGm = reinterpret_cast<__gm__ DTYPE_X*>(x);
    __gm__ DTYPE_W* wGm = reinterpret_cast<__gm__ DTYPE_W*>(w);
    __gm__ DTYPE_Y* yGm = reinterpret_cast<__gm__ DTYPE_Y*>(y);

    // simple 1D block partition over output elements
    uint32_t blockNum = AscendC::GetBlockNum();
    uint32_t blockIdx = AscendC::GetBlockIdx();

    uint64_t blkSize = (total + blockNum - 1) / blockNum;
    uint64_t start = (uint64_t)blockIdx * blkSize;
    uint64_t end = start + blkSize;
    if (end > total) end = total;

    for (uint64_t idx = start; idx < end; ++idx) {
        uint64_t t = idx;
        uint32_t wo = (uint32_t)(t % Wo); t /= Wo;
        uint32_t ho = (uint32_t)(t % Ho); t /= Ho;
        uint32_t co = (uint32_t)(t % Cout); t /= Cout;
        uint32_t n = (uint32_t)t;

        float acc = 0.0f;
        // accumulate over input channels and kernel window
        for (uint32_t ci = 0; ci < Cin; ++ci) {
            for (uint32_t kh = 0; kh < K; ++kh) {
                int32_t hi = (int32_t)ho - (int32_t)kh;
                if (hi < 0 || hi >= (int32_t)H) continue;
                for (uint32_t kw = 0; kw < K; ++kw) {
                    int32_t wi = (int32_t)wo - (int32_t)kw;
                    if (wi < 0 || wi >= (int32_t)W) continue;

                    uint64_t xOff = (((uint64_t)n * Cin + ci) * H + (uint32_t)hi) * W + (uint32_t)wi;
                    uint64_t wOff = (((uint64_t)ci * Cout + co) * K + kh) * K + kw;

                    acc += (float)xGm[xOff] * (float)wGm[wOff];
                }
            }
        }

        uint64_t yOff = (((uint64_t)n * Cout + co) * Ho + ho) * Wo + wo;
        yGm[yOff] = (DTYPE_Y)acc;
    }
}
"""

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor conv_transposed_2d_square_input_square_kernel_custom_impl_npu(const at::Tensor& x, const at::Tensor& w) {
    TORCH_CHECK(x.dim() == 4, \"x must be NCHW\");
    TORCH_CHECK(w.dim() == 4, \"w must be Cin x Cout x K x K\");
    TORCH_CHECK(x.scalar_type() == at::kFloat, \"only float32 supported\");
    TORCH_CHECK(w.scalar_type() == at::kFloat, \"only float32 supported\");

    auto N = x.size(0);
    auto Cin = x.size(1);
    auto H = x.size(2);
    auto W = x.size(3);

    TORCH_CHECK(w.size(0) == Cin, \"weight Cin mismatch\");
    auto Cout = w.size(1);
    auto K = w.size(2);
    TORCH_CHECK(w.size(3) == K, \"weight must be square kernel\");

    auto Ho = H + K - 1;
    auto Wo = W + K - 1;

    auto options = x.options();
    at::Tensor y = at::empty({N, Cout, Ho, Wo}, options);

    EXEC_NPU_CMD(aclnnConvTransposed2dSquareInputSquareKernelCustom, x, w, y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"conv_transposed_2d_square_input_square_kernel_custom\", &conv_transposed_2d_square_input_square_kernel_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int) -> None:
        super().__init__()
        # weight layout: Cin x Cout x K x K (matches conv-transpose2d)
        w = torch.randn(in_channels, out_channels, kernel_size, kernel_size, dtype=torch.float32)
        self.weight = torch.nn.Parameter(w)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # constraints assumed: stride=1, padding=0, output_padding=0, groups=1, no bias
        return custom_ops_lib.conv_transposed_2d_square_input_square_kernel_custom(x, self.weight)
