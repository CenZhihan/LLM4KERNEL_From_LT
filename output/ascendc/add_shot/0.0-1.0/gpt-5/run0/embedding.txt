host_operator_src="""
#include "embedding_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    EmbeddingCustomTilingData tiling;

    // weight: [num_embeddings, embedDim]
    const auto* w_shape = context->GetInputShape(0);
    // indices: arbitrary shape
    const auto* idx_shape = context->GetInputShape(1);

    uint32_t embedDim = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(1));

    uint32_t totalRows = 1;
    auto& idxOrigin = idx_shape->GetOriginShape();
    for (int i = 0; i < idxOrigin.GetDimNum(); ++i) {
        totalRows *= static_cast<uint32_t>(idxOrigin.GetDim(i));
    }

    uint32_t tableSize = static_cast<uint32_t>(w_shape->GetOriginShape().GetShapeSize());

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_totalRows(totalRows);
    tiling.set_embedDim(embedDim);
    tiling.set_tableSize(tableSize);
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* w_shape = context->GetInputShape(0);
    const gert::Shape* idx_shape = context->GetInputShape(1);
    gert::Shape* out_shape = context->GetOutputShape(0);

    // output shape = indices.shape + [embedDim]
    auto& wOrigin = w_shape->GetOriginShape();
    auto& idxOrigin = idx_shape->GetOriginShape();

    std::vector<int64_t> outDims;
    outDims.reserve(idxOrigin.GetDimNum() + 1);
    for (int i = 0; i < idxOrigin.GetDimNum(); ++i) {
        outDims.push_back(idxOrigin.GetDim(i));
    }
    outDims.push_back(wOrigin.GetDim(1));

    out_shape->SetOriginShape(ge::Shape(outDims));
    out_shape->SetStorageShape(ge::Shape(outDims));
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // output dtype == weight dtype
    const auto weightDType = context->GetInputDataType(0);
    context->SetOutputDataType(0, weightDType);
    return ge::GRAPH_SUCCESS;
}
}

namespace ops {
class EmbeddingCustom : public OpDef {
public:
    explicit EmbeddingCustom(const char* name) : OpDef(name)
    {
        this->Input("weight")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Input("indices")
            .ParamType(REQUIRED)
            .DataType({ge::DT_INT32})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Output("output")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore()
            .SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(EmbeddingCustom);
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "embedding_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // tensor buffers for each queue

class KernelEmbedding {
public:
    __aicore__ inline KernelEmbedding() {}
    __aicore__ inline void Init(GM_ADDR weight, GM_ADDR indices, GM_ADDR output, uint32_t totalRows, uint32_t embedDim, uint32_t tableSize, uint32_t tileNum)
    {
        this->embedDim = embedDim;
        this->tileNum = tileNum;
        this->blockRows = totalRows / AscendC::GetBlockNum();
        // protect against too small tiles
        this->tileRows = AscendC::Max((uint32_t)1, this->blockRows / (tileNum * BUFFER_NUM));

        weightGm.SetGlobalBuffer((__gm__ DTYPE_WEIGHT *)weight, tableSize);
        indicesGm.SetGlobalBuffer((__gm__ DTYPE_INDICES *)indices + this->blockRows * AscendC::GetBlockIdx(), this->blockRows);
        outputGm.SetGlobalBuffer((__gm__ DTYPE_OUTPUT *)output + (uint64_t)this->blockRows * this->embedDim * AscendC::GetBlockIdx(), (uint64_t)this->blockRows * this->embedDim);

        pipe.InitBuffer(inQueueIdx, BUFFER_NUM, this->tileRows * sizeof(DTYPE_INDICES));
        pipe.InitBuffer(outQueueBuf, BUFFER_NUM, this->embedDim * sizeof(DTYPE_OUTPUT));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        uint32_t offset = (uint32_t)progress * this->tileRows;
        if (offset >= this->blockRows) {
            // nothing to copy (out of range)
            return;
        }
        uint32_t rowsThisTile = AscendC::Min(this->tileRows, this->blockRows - offset);

        AscendC::LocalTensor<DTYPE_INDICES> idxLocal = inQueueIdx.AllocTensor<DTYPE_INDICES>();
        AscendC::DataCopy(idxLocal, indicesGm[offset], rowsThisTile);
        inQueueIdx.EnQue(idxLocal);

        // record valid rows for this tile
        this->validRows = rowsThisTile;
    }

    __aicore__ inline void Compute(int32_t progress)
    {
        uint32_t offset = (uint32_t)progress * this->tileRows;
        if (offset >= this->blockRows) {
            return;
        }
        AscendC::LocalTensor<DTYPE_INDICES> idxLocal = inQueueIdx.DeQue<DTYPE_INDICES>();

        for (uint32_t j = 0; j < this->validRows; ++j) {
            int32_t idx = static_cast<int32_t>(idxLocal.GetValue(j));
            // guard index if necessary (optional)
            // copy one embedding row from weight to output via UB buffer
            AscendC::LocalTensor<DTYPE_OUTPUT> buf = outQueueBuf.AllocTensor<DTYPE_OUTPUT>();
            AscendC::DataCopy(buf, weightGm[(uint64_t)idx * this->embedDim], this->embedDim);
            outQueueBuf.EnQue(buf);

            AscendC::LocalTensor<DTYPE_OUTPUT> buf2 = outQueueBuf.DeQue<DTYPE_OUTPUT>();
            AscendC::DataCopy(outputGm[((uint64_t)offset + j) * this->embedDim], buf2, this->embedDim);
            outQueueBuf.FreeTensor(buf2);
        }

        inQueueIdx.FreeTensor(idxLocal);
    }

    __aicore__ inline void CopyOut(int32_t)
    {
        // no-op, copy out is done in Compute for each row
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueIdx;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueBuf;

    AscendC::GlobalTensor<DTYPE_WEIGHT> weightGm;
    AscendC::GlobalTensor<DTYPE_INDICES> indicesGm;
    AscendC::GlobalTensor<DTYPE_OUTPUT> outputGm;

    uint32_t blockRows;
    uint32_t tileNum;
    uint32_t tileRows;
    uint32_t embedDim;
    uint32_t validRows;
};

extern "C" __global__ __aicore__ void embedding_custom(GM_ADDR weight, GM_ADDR indices, GM_ADDR output, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelEmbedding op;
    op.Init(weight, indices, output, tiling_data.totalRows, tiling_data.embedDim, tiling_data.tableSize, tiling_data.tileNum);
    op.Process();
}
"""

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor embedding_custom_impl_npu(const at::Tensor& weight, const at::Tensor& indices_in) {
    at::Tensor indices = indices_in;
    if (indices.scalar_type() != at::kInt) {
        indices = indices_in.to(at::kInt);
    }
    std::vector<int64_t> out_sizes = indices.sizes().vec();
    out_sizes.push_back(weight.size(-1));
    at::Tensor result = at::empty(out_sizes, weight.options());
    EXEC_NPU_CMD(aclnnEmbeddingCustom, weight, indices, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("embedding_custom", &embedding_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("embedding_custom", &embedding_custom_impl_npu, "Embedding lookup with custom AscendC kernel");
}
