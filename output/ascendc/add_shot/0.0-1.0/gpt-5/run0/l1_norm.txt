host_operator_src="""
#include "l1_norm_custom_tiling.h"
#include "register/op_def_registry.h"
#include <algorithm>

namespace {
inline uint32_t ClampTile(uint32_t d) {
    const uint32_t kMaxTile = 4096; // number of elements per tile
    return std::min(d, kMaxTile);
}
}

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    L1NormCustomTilingData tiling;

    const gert::Shape* inShape = context->GetInputShape(0);
    const auto& origin = inShape->GetOriginShape();
    const uint32_t dimNum = origin.GetDimNum();
    if (dimNum < 1) {
        return ge::GRAPH_PARAM_INVALID;
    }

    uint64_t n64 = origin.GetDim(0);
    uint64_t d64 = 1;
    if (dimNum >= 2) {
        d64 = origin.GetDim(1);
    }

    uint32_t n = static_cast<uint32_t>(n64);
    uint32_t d = static_cast<uint32_t>(d64);

    uint32_t blockDim = std::min<uint32_t>(n, 64); // choose up to 64 aicores or limited by N
    if (blockDim == 0) {
        blockDim = 1;
    }
    context->SetBlockDim(blockDim);

    uint32_t rowsPerBlock = (n + blockDim - 1) / blockDim;
    uint32_t tileCols = ClampTile(d);

    tiling.set_n(n);
    tiling.set_d(d);
    tiling.set_rowsPerBlock(rowsPerBlock);
    tiling.set_tileCols(tileCols);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

namespace ops {
class L1NormCustom : public OpDef {
public:
    explicit L1NormCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore()
            .SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(L1NormCustom);
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "l1_norm_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelL1Norm {
public:
    __aicore__ inline KernelL1Norm() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t d,
                                uint32_t rowsPerBlock, uint32_t tileCols)
    {
        this->n = n;
        this->d = d;
        this->rowsPerBlock = rowsPerBlock;
        this->tileCols = tileCols;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        this->rowStart = blockIdx * rowsPerBlock;
        this->rowsThisBlock = (rowStart < n) ? AscendC::min(rowsPerBlock, n - rowStart) : 0;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + (uint64_t)rowStart * d, (uint64_t)rowsThisBlock * d);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + (uint64_t)rowStart * d, (uint64_t)rowsThisBlock * d);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileCols * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileCols * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        if (rowsThisBlock == 0) {
            return;
        }
        for (uint32_t r = 0; r < rowsThisBlock; ++r) {
            RowNormalize(r);
        }
    }

private:
    __aicore__ inline void RowNormalize(uint32_t rowIdxInBlock)
    {
        uint64_t base = (uint64_t)rowIdxInBlock * d;

        // Pass 1: compute mean(|x|)
        float sumAbs = 0.f;
        for (uint32_t c = 0; c < d; c += tileCols) {
            uint32_t curr = AscendC::min<uint32_t>(tileCols, d - c);
            AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(xLocal, xGm[base + c], curr);
            inQueueX.EnQue(xLocal);

            xLocal = inQueueX.DeQue<DTYPE_X>();
            AscendC::Abs(xLocal, xLocal, curr);
            float part = AscendC::ReduceSum(xLocal, curr);
            sumAbs += part;
            inQueueX.FreeTensor(xLocal);
        }
        float meanAbs = sumAbs / static_cast<float>(d);
        float invMean = 1.0f / meanAbs;

        // Pass 2: scale by invMean
        for (uint32_t c = 0; c < d; c += tileCols) {
            uint32_t curr = AscendC::min<uint32_t>(tileCols, d - c);

            AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(xLocal, xGm[base + c], curr);
            inQueueX.EnQue(xLocal);

            xLocal = inQueueX.DeQue<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
            AscendC::Muls(yLocal, xLocal, invMean, curr);
            outQueueY.EnQue(yLocal);

            AscendC::LocalTensor<DTYPE_Y> yOut = outQueueY.DeQue<DTYPE_Y>();
            AscendC::DataCopy(yGm[base + c], yOut, curr);
            outQueueY.FreeTensor(yOut);
            inQueueX.FreeTensor(xLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n;
    uint32_t d;
    uint32_t rowsPerBlock;
    uint32_t tileCols;

    uint32_t rowStart;
    uint32_t rowsThisBlock;
};

extern "C" __global__ __aicore__ void l1_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelL1Norm op;
    op.Init(x, y, tiling_data.n, tiling_data.d, tiling_data.rowsPerBlock, tiling_data.tileCols);
    op.Process();
}
