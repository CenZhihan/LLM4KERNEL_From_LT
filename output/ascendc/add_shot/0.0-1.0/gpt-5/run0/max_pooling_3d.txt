inline int64_t floor_div(int64_t a, int64_t b) { return a / b; }
inline int64_t ceil_div(int64_t a, int64_t b) { return (a + b - 1) / b; }
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    MaxPooling3dCustomTilingData tiling;

    const gert::Shape* in_shape = context->GetInputShape(0);
    if (in_shape->GetDimNum() != 5) {
        return ge::GRAPH_FAILED;
    }

    uint32_t n = static_cast<uint32_t>(in_shape->GetDim(0));
    uint32_t c = static_cast<uint32_t>(in_shape->GetDim(1));
    uint32_t d = static_cast<uint32_t>(in_shape->GetDim(2));
    uint32_t h = static_cast<uint32_t>(in_shape->GetDim(3));
    uint32_t w = static_cast<uint32_t>(in_shape->GetDim(4));

    int64_t k_attr = 1, s_attr = 1, p_attr = 0;
    bool ceil_mode = false;

    (void)context->GetIntAttr("kernel_size", k_attr);
    (void)context->GetIntAttr("stride", s_attr);
    (void)context->GetIntAttr("padding", p_attr);
    (void)context->GetBoolAttr("ceil_mode", ceil_mode);

    int64_t k = k_attr;
    int64_t s = s_attr > 0 ? s_attr : k; // default stride = kernel_size
    int64_t p = p_attr;

    auto out_dim = [&](int64_t in) -> int64_t {
        int64_t numer = in + 2 * p - k;
        if (ceil_mode) {
            return static_cast<int64_t>(ceil_div(numer + s, s)) + 0; // same as ceil((in+2p-k)/s)+1 but fused
        } else {
            return static_cast<int64_t>(floor_div(numer, s)) + 1;
        }
    };

    int64_t od = out_dim(static_cast<int64_t>(d));
    int64_t oh = out_dim(static_cast<int64_t>(h));
    int64_t ow = out_dim(static_cast<int64_t>(w));
    if (od <= 0 || oh <= 0 || ow <= 0) {
        return ge::GRAPH_FAILED;
    }

    uint64_t totalOut = static_cast<uint64_t>(n) * c * static_cast<uint64_t>(od) * static_cast<uint64_t>(oh) * static_cast<uint64_t>(ow);

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_d(d);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_k(static_cast<uint32_t>(k));
    tiling.set_stride(static_cast<uint32_t>(s));
    tiling.set_padding(static_cast<uint32_t>(p));
    tiling.set_od(static_cast<uint32_t>(od));
    tiling.set_oh(static_cast<uint32_t>(oh));
    tiling.set_ow(static_cast<uint32_t>(ow));
    tiling.set_totalOut(static_cast<uint32_t>(totalOut));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    if (x_shape->GetDimNum() != 5) {
        return GRAPH_FAILED;
    }
    int64_t n = x_shape->GetDim(0);
    int64_t c = x_shape->GetDim(1);
    int64_t d = x_shape->GetDim(2);
    int64_t h = x_shape->GetDim(3);
    int64_t w = x_shape->GetDim(4);

    int64_t k = 1, s = 1, p = 0;
    bool ceil_mode = false;

    (void)context->GetIntAttr("kernel_size", k);
    (void)context->GetIntAttr("stride", s);
    (void)context->GetIntAttr("padding", p);
    (void)context->GetBoolAttr("ceil_mode", ceil_mode);

    if (s <= 0) s = k;

    auto out_dim = [&](int64_t in) -> int64_t {
        int64_t numer = in + 2 * p - k;
        if (ceil_mode) {
            // ceil((in + 2p - k)/s) + 1  implemented as (numer + s - 1)/s + 1
            return (numer + s - 1) / s + 1;
        } else {
            return numer / s + 1;
        }
    };

    int64_t od = out_dim(d);
    int64_t oh = out_dim(h);
    int64_t ow = out_dim(w);

    if (od <= 0 || oh <= 0 || ow <= 0) {
        return GRAPH_FAILED;
    }

    gert::Shape* y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(5);
    y_shape->SetDim(0, n);
    y_shape->SetDim(1, c);
    y_shape->SetDim(2, od);
    y_shape->SetDim(3, oh);
    y_shape->SetDim(4, ow);

    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "max_pooling_3d_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMaxPool3D {
public:
    __aicore__ inline KernelMaxPool3D() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w,
                                uint32_t k, uint32_t stride, uint32_t padding,
                                uint32_t od, uint32_t oh, uint32_t ow,
                                uint32_t totalOut)
    {
        this->n = n; this->c = c; this->d = d; this->h = h; this->w = w;
        this->k = k; this->s = stride; this->p = padding;
        this->od = od; this->oh = oh; this->ow = ow;
        this->totalOut = totalOut;

        // Partition output among blocks
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t base = (totalOut / blocks) * blockIdx;
        uint32_t rem = totalOut % blocks;
        if (blockIdx < rem) {
            this->startOut = base + blockIdx;
            this->lenOut = totalOut / blocks + 1;
        } else {
            this->startOut = base + rem;
            this->lenOut = totalOut / blocks;
        }

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint64_t)n * c * d * h * w);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, (uint64_t)n * c * od * oh * ow);

        pipe.InitBuffer(tmpQueue, BUFFER_NUM, k * k * k * sizeof(DTYPE_X)); // window buffer
        pipe.InitBuffer(outQueue, BUFFER_NUM, 1 * sizeof(DTYPE_Y)); // single element out
    }

    __aicore__ inline void Process()
    {
        for (uint32_t idx = 0; idx < this->lenOut; ++idx) {
            uint32_t outIndex = this->startOut + idx;

            // Map flat outIndex -> (n, c, od, oh, ow)
            uint32_t oc_stride_d = this->oh * this->ow;
            uint32_t oc_stride_c = this->od * oc_stride_d;
            uint32_t n_idx = outIndex / (c * oc_stride_c);
            uint32_t rem1 = outIndex % (c * oc_stride_c);
            uint32_t c_idx = rem1 / oc_stride_c;
            uint32_t rem2 = rem1 % oc_stride_c;
            uint32_t od_idx = rem2 / oc_stride_d;
            uint32_t rem3 = rem2 % oc_stride_d;
            uint32_t oh_idx = rem3 / this->ow;
            uint32_t ow_idx = rem3 % this->ow;

            // compute input window origin (front/top/left)
            int32_t d0 = (int32_t)od_idx * (int32_t)s - (int32_t)p;
            int32_t h0 = (int32_t)oh_idx * (int32_t)s - (int32_t)p;
            int32_t w0 = (int32_t)ow_idx * (int32_t)s - (int32_t)p;

            // Prepare local window buffer
            AscendC::LocalTensor<DTYPE_X> winLocal = tmpQueue.AllocTensor<DTYPE_X>();
            uint32_t filled = 0;

            // Iterate pooling window and copy valid elements into local buffer
            for (uint32_t kd = 0; kd < k; ++kd) {
                int32_t id = d0 + (int32_t)kd;
                if (id < 0 || id >= (int32_t)d) continue;
                for (uint32_t kh = 0; kh < k; ++kh) {
                    int32_t ih = h0 + (int32_t)kh;
                    if (ih < 0 || ih >= (int32_t)h) continue;
                    for (uint32_t kw = 0; kw < k; ++kw) {
                        int32_t iw = w0 + (int32_t)kw;
                        if (iw < 0 || iw >= (int32_t)w) continue;

                        uint64_t inOffset = (((uint64_t)n_idx * c + c_idx) * d + (uint32_t)id) * h;
                        inOffset = (inOffset + (uint32_t)ih) * w + (uint32_t)iw;

                        AscendC::DataCopy(winLocal[filled], xGm[inOffset], 1);
                        ++filled;
                    }
                }
            }

            // Reduce max over the collected elements
            AscendC::LocalTensor<DTYPE_Y> outLocal = outQueue.AllocTensor<DTYPE_Y>();
            if (filled == 0) {
                // Should not happen, but set to minimal
                // Write zero
                AscendC::SetValue(outLocal, (DTYPE_Y)0, 1);
            } else if (filled == 1) {
                AscendC::DataCopy(outLocal, winLocal, 1);
            } else {
                // iterative pairwise max
                // Use outLocal as accumulator: copy first value
                AscendC::DataCopy(outLocal, winLocal, 1);
                for (uint32_t i = 1; i < filled; ++i) {
                    // temp buffer of 1
                    AscendC::LocalTensor<DTYPE_X> tmpVal = tmpQueue.AllocTensor<DTYPE_X>();
                    AscendC::DataCopy(tmpVal, winLocal[i], 1);
                    // outLocal = max(outLocal, tmpVal)
                    AscendC::Max(outLocal, outLocal, tmpVal, 1);
                    tmpQueue.FreeTensor(tmpVal);
                }
            }

            // queue management
            tmpQueue.FreeTensor(winLocal);
            outQueue.EnQue(outLocal);

            // write back
            AscendC::LocalTensor<DTYPE_Y> yVal = outQueue.DeQue<DTYPE_Y>();
            uint64_t outOffset = ((((uint64_t)n_idx * c + c_idx) * od + od_idx) * oh + oh_idx) * ow + ow_idx;
            AscendC::DataCopy(yGm[outOffset], yVal, 1);
            outQueue.FreeTensor(yVal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n, c, d, h, w;
    uint32_t k, s, p;
    uint32_t od, oh, ow;
    uint32_t totalOut;
    uint32_t startOut;
    uint32_t lenOut;
};

extern "C" __global__ __aicore__ void max_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPool3D op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c, tiling_data.d, tiling_data.h, tiling_data.w,
            tiling_data.k, tiling_data.stride, tiling_data.padding,
            tiling_data.od, tiling_data.oh, tiling_data.ow,
            tiling_data.totalOut);
    op.Process();
}
