kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv2d_multiply_leaky_relu_gelu_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue
constexpr float kNegSlope = 0.01f; // LeakyReLU negative slope
constexpr float kInvSqrt2 = 0.707106781186547524f; // 1/sqrt(2)

class KernelFused {
public:
    __aicore__ inline KernelFused() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR m, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        mGm.SetGlobalBuffer((__gm__ DTYPE_Y *)m + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueM, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> mLocal = inQueueM.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(mLocal, mGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
        inQueueM.EnQue(mLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> mLocal = inQueueM.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();
        AscendC::LocalTensor<DTYPE_Z> tmpLocal = tmpQueue.AllocTensor<DTYPE_Z>();

        // 1) Multiply: tmp = x * m
        AscendC::Mul(tmpLocal, xLocal, mLocal, this->tileLength);

        // 2) LeakyReLU: y = max(tmp, k * tmp)
        AscendC::Muls(yLocal, tmpLocal, (DTYPE_Z)kNegSlope, this->tileLength);
        AscendC::Maximum(yLocal, yLocal, tmpLocal, this->tileLength);

        // 3) GELU: y = 0.5 * y * (1 + erf(y / sqrt(2)))
        AscendC::Muls(tmpLocal, yLocal, (DTYPE_Z)kInvSqrt2, this->tileLength);      // tmp = y / sqrt(2)
        AscendC::Erf(tmpLocal, tmpLocal, this->tileLength);                         // tmp = erf(tmp)
        AscendC::Adds(tmpLocal, tmpLocal, (DTYPE_Z)1.0f, this->tileLength);         // tmp = 1 + tmp
        AscendC::Muls(tmpLocal, tmpLocal, (DTYPE_Z)0.5f, this->tileLength);         // tmp = 0.5 * tmp
        AscendC::Mul(yLocal, yLocal, tmpLocal, this->tileLength);                   // y = y * tmp

        outQueueY.EnQue<DTYPE_Z>(yLocal);

        tmpQueue.FreeTensor(tmpLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueM.FreeTensor(mLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueM;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> mGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv2d_multiply_leaky_relu_gelu_custom(GM_ADDR x, GM_ADDR m, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelFused op;
    op.Init(x, m, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Conv2d followed by fused multiply + LeakyReLU + GELU implemented via custom AscendC kernel.
    The multiplier is broadcast over spatial dimensions and batch.
    """
    def __init__(self, in_channels, out_channels, kernel_size, multiplier_shape):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)
        self.multiplier = torch.nn.Parameter(torch.randn(multiplier_shape))
        self.leaky_relu = torch.nn.LeakyReLU()  # kept for parity; kernel uses default slope 0.01

    def forward(self, x):
        x = self.conv(x)
        # Expand multiplier to match x shape for elementwise fused kernel
        scale = self.multiplier.view(1, -1, 1, 1).expand_as(x)
        x = custom_ops_lib.conv2d_multiply_leaky_relu_gelu_custom(x, scale)
        return x
