kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_standard_1d_dilated_strided_custom_tiling_data.h"

// Naive reference 1D convolution kernel (NCHW-like [N, C_in, L] with weight [C_out, C_in, K])
// This kernel is written for clarity; production kernels should use optimized vectorization/tiling.

constexpr int32_t BUFFER_NUM = 2;

class KernelConv1D {
public:
    __aicore__ inline KernelConv1D() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t c_in, uint32_t c_out,
                                uint32_t l_in, uint32_t l_out, uint32_t k,
                                uint32_t stride, uint32_t dilation, uint32_t tileNum)
    {
        this->n = n;
        this->c_in = c_in;
        this->c_out = c_out;
        this->l_in = l_in;
        this->l_out = l_out;
        this->k = k;
        this->stride = stride;
        this->dilation = dilation;
        this->tileNum = tileNum;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint64_t)n * c_in * l_in);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W *)w, (uint64_t)c_out * c_in * k);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, (uint64_t)n * c_out * l_out);

        // Temporary local buffers (scalar-sized ops)
        pipe.InitBuffer(tmpX, BUFFER_NUM, sizeof(DTYPE_X));
        pipe.InitBuffer(tmpW, BUFFER_NUM, sizeof(DTYPE_W));
        pipe.InitBuffer(tmpP, BUFFER_NUM, sizeof(DTYPE_Y));
        pipe.InitBuffer(tmpAcc, BUFFER_NUM, sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        if (l_out == 0) { return; }

        // Split output length across blocks
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t start = (uint64_t)l_out * blockIdx / blockNum;
        uint32_t end   = (uint64_t)l_out * (blockIdx + 1) / blockNum;

        for (uint32_t ni = 0; ni < n; ++ni) {
            for (uint32_t co = 0; co < c_out; ++co) {
                for (uint32_t to = start; to < end; ++to) {
                    ComputeOne(ni, co, to);
                }
            }
        }
    }

private:
    __aicore__ inline void ComputeOne(uint32_t ni, uint32_t co, uint32_t to)
    {
        AscendC::LocalTensor<DTYPE_Y> acc = tmpAcc.AllocTensor<DTYPE_Y>();
        // Initialize accumulator to 0
        AscendC::SetValue(acc, (DTYPE_Y)0, 1);

        for (uint32_t ci = 0; ci < c_in; ++ci) {
            for (uint32_t kk = 0; kk < k; ++kk) {
                int64_t xi = (int64_t)to * (int64_t)stride + (int64_t)kk * (int64_t)dilation;
                if (xi < 0 || xi >= (int64_t)l_in) {
                    continue;
                }

                uint64_t xOff = ((uint64_t)ni * c_in + ci) * l_in + (uint64_t)xi;
                uint64_t wOff = ((uint64_t)co * c_in + ci) * k + kk;

                AscendC::LocalTensor<DTYPE_X> lx = tmpX.AllocTensor<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_W> lw = tmpW.AllocTensor<DTYPE_W>();
                AscendC::LocalTensor<DTYPE_Y> prod = tmpP.AllocTensor<DTYPE_Y>();

                AscendC::DataCopy(lx, xGm[xOff], 1);
                AscendC::DataCopy(lw, wGm[wOff], 1);

                // prod = lx * lw
                AscendC::Mul(prod, lx, lw, 1);
                // acc += prod
                AscendC::Add(acc, acc, prod, 1);

                tmpX.FreeTensor(lx);
                tmpW.FreeTensor(lw);
                tmpP.FreeTensor(prod);
            }
        }

        uint64_t yOff = ((uint64_t)ni * c_out + co) * l_out + to;
        AscendC::DataCopy(yGm[yOff], acc, 1);
        tmpAcc.FreeTensor(acc);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> tmpX, tmpW;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> tmpP, tmpAcc; // use VECIN for local temp buffers as well
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n, c_in, c_out;
    uint32_t l_in, l_out, k;
    uint32_t stride, dilation, tileNum;
};

extern "C" __global__ __aicore__ void conv_standard_1d_dilated_strided_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv1D op;
    op.Init(x, w, y,
            tiling_data.n, tiling_data.c_in, tiling_data.c_out,
            tiling_data.l_in, tiling_data.l_out, tiling_data.k,
            tiling_data.stride, tiling_data.dilation, tiling_data.tileNum);
    op.Process();
}

#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor conv_standard_1d_dilated_strided_custom_impl_npu(const at::Tensor& x,
                                                            const at::Tensor& w,
                                                            int64_t stride,
                                                            int64_t dilation) {
    // x: [N, Cin, L], w: [Cout, Cin, K]
    // y: [N, Cout, L_out]
    auto N = x.size(0);
    auto Cout = w.size(0);
    // Infer output length on host to allocate result
    auto Lin = x.size(2);
    auto K = w.size(2);
    int64_t effective = Lin - dilation * (K - 1) - 1;
    int64_t Lout = effective >= 0 ? (effective / stride + 1) : 0;

    at::Tensor y = at::empty({N, Cout, Lout}, x.options());

    // Call custom NPU op with attributes
    EXEC_NPU_CMD(aclnnConvStandard1dDilatedStridedCustom, x, w, stride, dilation, y);
    return y;
}
