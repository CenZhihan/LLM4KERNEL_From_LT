kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_transpose3d_layer_norm_gelu_scaling_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

// Fused kernel: channel-wise LayerNorm (across C), then GELU, then scaling
class KernelLnGeluScale {
public:
    __aicore__ inline KernelLnGeluScale() {}

    __aicore__ inline void Init(GM_ADDR x,
                                GM_ADDR gamma,
                                GM_ADDR beta,
                                GM_ADDR eps,
                                GM_ADDR scale,
                                GM_ADDR y,
                                uint32_t totalLength,
                                uint32_t tileNum,
                                uint32_t C,
                                uint32_t groups)
    {
        this->C = C;
        this->groups = groups; // number of (n,d,h,w) positions
        this->tileNum = tileNum;

        // Split by groups so each group has C contiguous elements (NDHWC layout friendly)
        uint32_t groupsPerBlock = groups / AscendC::GetBlockNum();
        this->groupsPerTile = AscendC::max((uint32_t)1, (groupsPerBlock / tileNum / BUFFER_NUM));
        this->tileLength = this->groupsPerTile * C;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + (uint64_t)AscendC::GetBlockIdx() * (uint64_t)groupsPerBlock * (uint64_t)C,
                            (uint64_t)groupsPerBlock * (uint64_t)C);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + (uint64_t)AscendC::GetBlockIdx() * (uint64_t)groupsPerBlock * (uint64_t)C,
                            (uint64_t)groupsPerBlock * (uint64_t)C);

        gammaGm.SetGlobalBuffer((__gm__ DTYPE_X *)gamma, C);
        betaGm.SetGlobalBuffer((__gm__ DTYPE_X *)beta, C);
        epsGm.SetGlobalBuffer((__gm__ DTYPE_X *)eps, 1);
        scaleGm.SetGlobalBuffer((__gm__ DTYPE_X *)scale, 1);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));

        pipe.InitBuffer(bufGamma, 1, this->C * sizeof(DTYPE_X));
        pipe.InitBuffer(bufBeta, 1, this->C * sizeof(DTYPE_X));
        pipe.InitBuffer(bufScalar, 1, 2 * sizeof(DTYPE_X)); // eps and scale
        // Load gamma, beta, eps, scale once into local buffers
        gammaLocal = bufGamma.AllocTensor<DTYPE_X>();
        betaLocal = bufBeta.AllocTensor<DTYPE_X>();
        scalarsLocal = bufScalar.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(gammaLocal, gammaGm, this->C);
        AscendC::DataCopy(betaLocal, betaGm, this->C);
        AscendC::DataCopy(scalarsLocal, epsGm, 1);
        AscendC::DataCopy(scalarsLocal[1], scaleGm, 1); // place scale at index 1
        bufGamma.EnQue(gammaLocal);
        bufBeta.EnQue(betaLocal);
        bufScalar.EnQue(scalarsLocal);
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[(uint64_t)progress * (uint64_t)this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        AscendC::LocalTensor<DTYPE_X> gammaL = bufGamma.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> betaL = bufBeta.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> scalarsL = bufScalar.DeQue<DTYPE_X>();

        // Read scalars
        DTYPE_X eps = scalarsL.GetValue(0);
        DTYPE_X scl = scalarsL.GetValue(1);

        // Process per group
        uint32_t groupsThisTile = this->groupsPerTile;
        for (uint32_t g = 0; g < groupsThisTile; ++g) {
            uint32_t base = g * this->C;

            // Compute mean
            DTYPE_X mean = (DTYPE_X)0;
            for (uint32_t c = 0; c < this->C; ++c) {
                mean += xLocal.GetValue(base + c);
            }
            mean = mean / (DTYPE_X)this->C;

            // Compute variance
            DTYPE_X var = (DTYPE_X)0;
            for (uint32_t c = 0; c < this->C; ++c) {
                DTYPE_X v = xLocal.GetValue(base + c) - mean;
                var += v * v;
            }
            var = var / (DTYPE_X)this->C;
            DTYPE_X inv_std = AscendC::rsqrt(var + eps);

            // Normalize, affine, GELU, scale
            for (uint32_t c = 0; c < this->C; ++c) {
                DTYPE_X xn = (xLocal.GetValue(base + c) - mean) * inv_std;
                DTYPE_X aff = xn * gammaL.GetValue(c) + betaL.GetValue(c);
                // Apply GELU using built-in if available, otherwise use approximation or identity
                // Here we use a simple tanh-based approximation
                DTYPE_X x3 = aff * aff * aff;
                DTYPE_X inner = (DTYPE_X)0.7978845608f * (aff + (DTYPE_X)0.044715f * x3); // sqrt(2/pi)
                DTYPE_X gelu = (DTYPE_X)0.5f * aff * ((DTYPE_X)1.0f + AscendC::tanh(inner));
                yLocal.SetValue(base + c, gelu * scl);
            }
        }

        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueX.FreeTensor(xLocal);

        bufGamma.EnQue(gammaL);
        bufBeta.EnQue(betaL);
        bufScalar.EnQue(scalarsL);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[(uint64_t)progress * (uint64_t)this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::TBuf<AscendC::TPosition::VECCALC> bufGamma;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bufBeta;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bufScalar;

    AscendC::LocalTensor<DTYPE_X> gammaLocal;
    AscendC::LocalTensor<DTYPE_X> betaLocal;
    AscendC::LocalTensor<DTYPE_X> scalarsLocal;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_X> gammaGm;
    AscendC::GlobalTensor<DTYPE_X> betaGm;
    AscendC::GlobalTensor<DTYPE_X> epsGm;
    AscendC::GlobalTensor<DTYPE_X> scaleGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t C;
    uint32_t groups;        // N*D*H*W
    uint32_t groupsPerTile; // per tile groups
    uint32_t tileNum;
    uint32_t tileLength;    // groupsPerTile * C
};

extern "C" __global__ __aicore__ void conv_transpose3d_layer_norm_gelu_scaling_custom(GM_ADDR x,
                                                                                       GM_ADDR gamma,
                                                                                       GM_ADDR beta,
                                                                                       GM_ADDR eps,
                                                                                       GM_ADDR scale,
                                                                                       GM_ADDR y,
                                                                                       GM_ADDR workspace,
                                                                                       GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLnGeluScale op;
    op.Init(x, gamma, beta, eps, scale, y,
            tiling_data.totalLength,
            tiling_data.tileNum,
            tiling_data.C,
            tiling_data.groups);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Model that performs a 3D transposed convolution, then a fused channel-wise LayerNorm + GELU + scaling
    using a custom AscendC operator.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride=1, padding=0, bias=True, eps=1e-5, scaling_factor=1.0):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, bias=bias)
        # Keep LayerNorm module to own the affine parameters (gamma/beta)
        self.layer_norm = torch.nn.LayerNorm(out_channels, eps=eps)
        self.scaling_factor = scaling_factor

    def forward(self, x):
        x = self.conv_transpose(x)
        # Fused: channel-wise LayerNorm (over C), then GELU, then scaling
        # Expect input to be in channels-last-3d memory format for contiguous channels per spatial location
        x = x.contiguous(memory_format=torch.channels_last_3d)
        y = custom_ops_lib.conv_transpose3d_layer_norm_gelu_scaling_custom(
            x,
            self.layer_norm.weight,
            self.layer_norm.bias,
            float(self.layer_norm.eps),
            float(self.scaling_factor),
        )
        return y
