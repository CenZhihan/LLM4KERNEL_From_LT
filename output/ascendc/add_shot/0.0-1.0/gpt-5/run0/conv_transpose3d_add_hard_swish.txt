kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_transpose3d_add_hard_swish_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelConvTranspose3dAddHardSwish {
public:
    __aicore__ inline KernelConvTranspose3dAddHardSwish() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR add, GM_ADDR out, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        addGm.SetGlobalBuffer((__gm__ DTYPE_Y *)add + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        outGm.SetGlobalBuffer((__gm__ DTYPE_Z *)out + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueAdd, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueOut, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> addLocal = inQueueAdd.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(addLocal, addGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
        inQueueAdd.EnQue(addLocal);
    }

    __aicore__ inline void Compute(int32_t)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> addLocal = inQueueAdd.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> outLocal = outQueueOut.AllocTensor<DTYPE_Z>();

        // t = x + add
        AscendC::Add(outLocal, xLocal, addLocal, this->tileLength);

        // Use addLocal as scratch: s = clip(outLocal + 3, 0, 6)
        AscendC::Adds(addLocal, outLocal, static_cast<DTYPE_Y>(3.0f), this->tileLength); // s = t + 3
        AscendC::Maxs(addLocal, addLocal, static_cast<DTYPE_Y>(0.0f), this->tileLength); // s = max(s, 0)
        AscendC::Mins(addLocal, addLocal, static_cast<DTYPE_Y>(6.0f), this->tileLength); // s = min(s, 6)

        // outLocal = t^2
        AscendC::Mul(outLocal, outLocal, outLocal, this->tileLength);

        // outLocal = outLocal * s
        AscendC::Mul(outLocal, outLocal, addLocal, this->tileLength);

        // outLocal = outLocal * (1/6)
        AscendC::Muls(outLocal, outLocal, static_cast<DTYPE_Z>(1.0f / 6.0f), this->tileLength);

        outQueueOut.EnQue<DTYPE_Z>(outLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueAdd.FreeTensor(addLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> outLocal = outQueueOut.DeQue<DTYPE_Z>();
        AscendC::DataCopy(outGm[progress * this->tileLength], outLocal, this->tileLength);
        outQueueOut.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueAdd;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueOut;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> addGm;
    AscendC::GlobalTensor<DTYPE_Z> outGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transpose3d_add_hard_swish_custom(GM_ADDR x, GM_ADDR add, GM_ADDR out, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTranspose3dAddHardSwish op;
    op.Init(x, add, out, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused model: ConvTranspose3d -> Add -> x * HardSwish(x)
    The custom kernel computes: out = t * hardswish(t) where t = conv_out + add_input.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, bias_shape):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(
            in_channels,
            out_channels,
            kernel_size,
            stride=stride,
            padding=padding,
            output_padding=output_padding
        )
        # keep the same attribute for compatibility, though unused in forward
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))

    def forward(self, x, add_input):
        x = self.conv_transpose(x)
        # Fused custom op for (x + add_input) * hardswish(x + add_input)
        x = custom_ops_lib.conv_transpose3d_add_hard_swish_custom(x, add_input)
        return x
