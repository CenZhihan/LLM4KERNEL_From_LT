host_operator_src="""
#include "cumprod_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
inline uint32_t GetLastDimSize(const gert::Shape& s) {
    auto rank = s.GetDimNum();
    if (rank == 0) return 1;
    return static_cast<uint32_t>(s.GetDim(rank - 1));
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    CumprodCustomTilingData tiling;

    const auto* in_shape = context->GetInputShape(0);
    auto origin = in_shape->GetOriginShape();
    uint64_t total = origin.GetShapeSize();
    uint32_t cols = GetLastDimSize(origin);
    if (cols == 0) {
        return ge::GRAPH_FAILED;
    }
    uint32_t rows = static_cast<uint32_t>(total / cols);

    // Choose a reasonable tile length for local processing along last dimension.
    // Keep it power-of-two-ish and under typical UB limits.
    const uint32_t DEFAULT_TILE = 1024;
    uint32_t tileLen = cols < DEFAULT_TILE ? cols : DEFAULT_TILE;

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_rows(rows);
    tiling.set_cols(cols);
    tiling.set_tileLen(tileLen);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

namespace ops {
class CumprodCustom : public OpDef {
public:
    explicit CumprodCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);
        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(CumprodCustom);
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "cumprod_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 1; // one buffer per queue for simplicity

// NOTE:
// This kernel computes cumulative product along the last dimension only.
// For a tensor of shape [N0, N1, ..., Nk], it performs cumprod over axis k.
// Parallelism is across the outer rows.

class KernelCumprod {
public:
    __aicore__ inline KernelCumprod() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t rows, uint32_t cols, uint32_t tileLen)
    {
        this->rowsPerBlock = rows / AscendC::GetBlockNum();
        this->cols = cols;
        this->tileLen = tileLen;

        uint64_t offset = static_cast<uint64_t>(this->rowsPerBlock) * static_cast<uint64_t>(this->cols) * AscendC::GetBlockIdx();
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + offset, this->rowsPerBlock * this->cols);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + offset, this->rowsPerBlock * this->cols);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLen * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLen * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < this->rowsPerBlock; ++r) {
            uint32_t processed = 0;
            DTYPE_Y carry = static_cast<DTYPE_Y>(1.0f);
            while (processed < this->cols) {
                uint32_t len = AscendC::min(this->tileLen, this->cols - processed);

                // Copy in a tile
                AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(xLocal, xGm[r * this->cols + processed], len);
                inQueueX.EnQue(xLocal);

                // Compute prefix products within tile with carry from previous tiles
                AscendC::LocalTensor<DTYPE_X> xT = inQueueX.DeQue<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

                // Elementwise scalar loop over the tile
                DTYPE_Y run = carry;
                for (uint32_t i = 0; i < len; ++i) {
                    // Assuming LocalTensor supports operator[] for element access
                    DTYPE_Y v = static_cast<DTYPE_Y>(xT[i]);
                    run = run * v;
                    yLocal[i] = run;
                }
                carry = run;

                outQueueY.EnQue(yLocal);
                inQueueX.FreeTensor(xT);

                // Copy out
                AscendC::LocalTensor<DTYPE_Y> yT = outQueueY.DeQue<DTYPE_Y>();
                AscendC::DataCopy(yGm[r * this->cols + processed], yT, len);
                outQueueY.FreeTensor(yT);

                processed += len;
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t rowsPerBlock;
    uint32_t cols;
    uint32_t tileLen;
};

extern "C" __global__ __aicore__ void cumprod_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumprod op;
    op.Init(x, y, tiling_data.rows, tiling_data.cols, tiling_data.tileLen);
    op.Process();
}
"""

at::Tensor cumprod_custom_impl_npu(const at::Tensor& self, int64_t dim) {
    // Kernel currently supports cumprod along the last dimension.
    // 'dim' is accepted for API consistency but not used in the device call.
    at::Tensor result = at::empty_like(self);
    EXEC_NPU_CMD(aclnnCumprodCustom, self, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("cumprod_custom", &cumprod_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("cumprod_custom", &cumprod_custom_impl_npu, "cumprod along a given dim (last-dim supported)");
}
