kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_transpose3d_avg_pool_clamp_softmax_multiply_custom_tiling_data.h"

typedef float DTYPE_X;
typedef float DTYPE_SCALE;
typedef float DTYPE_Y;

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelFused {
public:
    __aicore__ inline KernelFused() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR scale, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        sGm.SetGlobalBuffer((__gm__ DTYPE_SCALE *)scale + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueS, BUFFER_NUM, this->tileLength * sizeof(DTYPE_SCALE));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_SCALE> sLocal = inQueueS.AllocTensor<DTYPE_SCALE>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(sLocal, sGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
        inQueueS.EnQue(sLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_SCALE> sLocal = inQueueS.DeQue<DTYPE_SCALE>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

        // Fused elementwise: y = x * scale
        AscendC::Mul(yLocal, xLocal, sLocal, this->tileLength);

        outQueueY.EnQue<DTYPE_Y>(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueS.FreeTensor(sLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueS;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_SCALE> sGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transpose3d_avg_pool_clamp_softmax_multiply_custom(
    GM_ADDR x, GM_ADDR scale, GM_ADDR clamp_min, GM_ADDR clamp_max, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    (void)clamp_min;
    (void)clamp_max;
    (void)workspace;
    KernelFused op;
    op.Init(x, scale, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model that keeps AvgPool3d and ConvTranspose3d in PyTorch,
    and replaces the clamp + softmax + multiply sequence with a fused custom AscendC op.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, pool_kernel_size, clamp_min, clamp_max):
        super(ModelNew, self).__init__()
        self.avg_pool = torch.nn.AvgPool3d(pool_kernel_size)
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, output_padding=output_padding)
        self.clamp_min = clamp_min
        self.clamp_max = clamp_max
        self.scale = torch.nn.Parameter(torch.ones(1, out_channels, 1, 1, 1))

    def forward(self, x):
        x = self.avg_pool(x)
        x = self.conv_transpose(x)
        scale_expanded = self.scale.expand_as(x)
        clamp_min_t = torch.tensor(self.clamp_min, dtype=x.dtype, device=x.device)
        clamp_max_t = torch.tensor(self.clamp_max, dtype=x.dtype, device=x.device)
        y = custom_ops_lib.conv_transpose3d_avg_pool_clamp_softmax_multiply_custom(x, scale_expanded, clamp_min_t, clamp_max_t)
        return y
'''
