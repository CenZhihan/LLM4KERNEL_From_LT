#include "instance_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1) / b; }
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    InstanceNormCustomTilingData tiling;

    const auto* inShape = context->GetInputShape(0);
    const auto& shape = inShape->GetOriginShape();
    // Expect NCHW
    if (shape.GetDimNum() < 4) {
        return ge::GRAPH_PARAM_INVALID;
    }
    uint32_t N = static_cast<uint32_t>(shape.GetDim(0));
    uint32_t C = static_cast<uint32_t>(shape.GetDim(1));
    uint32_t H = static_cast<uint32_t>(shape.GetDim(2));
    uint32_t W = static_cast<uint32_t>(shape.GetDim(3));
    uint32_t groupSize = H * W;
    uint32_t groupNum  = N * C;

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_groupSize(groupSize);
    tiling.set_groupNum(groupNum);
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "instance_norm_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue
constexpr float EPSILON = 1e-5f;

template<typename T>
__aicore__ inline uint32_t ceil_div_u32(T a, T b) {
    return static_cast<uint32_t>((a + b - 1) / b);
}

class KernelInstanceNorm {
public:
    __aicore__ inline KernelInstanceNorm() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t groupSize, uint32_t groupNum, uint32_t tileNum)
    {
        this->groupSize = groupSize;
        this->groupNum = groupNum;
        this->tileNum = tileNum;

        // Partition groups across blocks to avoid crossing group boundaries
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t base = groupNum / blockNum;
        uint32_t rem  = groupNum % blockNum;
        this->myGroupStart = base * blockIdx + (blockIdx < rem ? blockIdx : rem);
        this->myGroupCount = base + (blockIdx < rem ? 1 : 0);

        // Determine tile length inside each group
        uint32_t pieces = this->tileNum * BUFFER_NUM;
        this->tileLength = ceil_div_u32<uint32_t>(this->groupSize, pieces);
        if (this->tileLength == 0) this->tileLength = this->groupSize;

        // Set GM windows
        xGm.SetGlobalBuffer((__gm__ float *)x + static_cast<uint64_t>(this->myGroupStart) * this->groupSize,
                            static_cast<uint64_t>(this->myGroupCount) * this->groupSize);
        yGm.SetGlobalBuffer((__gm__ float *)y + static_cast<uint64_t>(this->myGroupStart) * this->groupSize,
                            static_cast<uint64_t>(this->myGroupCount) * this->groupSize);

        // Init local buffers
        pipe.InitBuffer(inQueue, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(outQueue, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileLength * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t g = 0; g < this->myGroupCount; ++g) {
            compute_one_group(g);
        }
    }

private:
    __aicore__ inline void compute_one_group(uint32_t gidxInBlock)
    {
        uint64_t baseOffset = static_cast<uint64_t>(gidxInBlock) * this->groupSize;
        uint32_t tiles = ceil_div_u32<uint32_t>(this->groupSize, this->tileLength);

        // Pass 1: compute mean and variance
        float sum = 0.0f;
        float sumsq = 0.0f;
        for (uint32_t t = 0; t < tiles; ++t) {
            uint32_t offset = t * this->tileLength;
            uint32_t currLen = (offset + this->tileLength <= this->groupSize) ? this->tileLength
                                                                              : (this->groupSize - offset);

            AscendC::LocalTensor<float> xLocal = inQueue.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm[baseOffset + offset], currLen);
            inQueue.EnQue(xLocal);

            xLocal = inQueue.DeQue<float>();
            AscendC::LocalTensor<float> tmpLocal = tmpQueue.AllocTensor<float>();
            AscendC::Mul(tmpLocal, xLocal, xLocal, currLen); // x^2

            float partSum   = AscendC::ReduceSum(xLocal, currLen);
            float partSumSq = AscendC::ReduceSum(tmpLocal, currLen);
            sum   += partSum;
            sumsq += partSumSq;

            tmpQueue.FreeTensor(tmpLocal);
            inQueue.FreeTensor(xLocal);
        }

        float mean = sum / static_cast<float>(this->groupSize);
        float var  = sumsq / static_cast<float>(this->groupSize) - mean * mean;
        float invStd = AscendC::Rsqrt(var + EPSILON);

        // Pass 2: normalize and write out
        for (uint32_t t = 0; t < tiles; ++t) {
            uint32_t offset = t * this->tileLength;
            uint32_t currLen = (offset + this->tileLength <= this->groupSize) ? this->tileLength
                                                                              : (this->groupSize - offset);

            AscendC::LocalTensor<float> xLocal = inQueue.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm[baseOffset + offset], currLen);
            inQueue.EnQue(xLocal);

            xLocal = inQueue.DeQue<float>();
            AscendC::LocalTensor<float> yLocal = outQueue.AllocTensor<float>();

            AscendC::Adds(yLocal, xLocal, -mean, currLen);
            AscendC::Muls(yLocal, yLocal, invStd, currLen);

            outQueue.EnQue(yLocal);
            inQueue.FreeTensor(xLocal);

            yLocal = outQueue.DeQue<float>();
            AscendC::DataCopy(yGm[baseOffset + offset], yLocal, currLen);
            outQueue.FreeTensor(yLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> tmpQueue; // scratch for x^2
    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t groupSize;
    uint32_t groupNum;
    uint32_t tileNum;

    uint32_t myGroupStart;
    uint32_t myGroupCount;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void instance_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelInstanceNorm op;
    op.Init(x, y, tiling_data.groupSize, tiling_data.groupNum, tiling_data.tileNum);
    op.Process();
}
