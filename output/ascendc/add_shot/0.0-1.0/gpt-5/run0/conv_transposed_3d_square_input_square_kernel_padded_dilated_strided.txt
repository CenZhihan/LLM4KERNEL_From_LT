#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_transposed_3d_square_input_square_kernel_padded_dilated_strided_custom_tiling_data.h"

// NOTE: This is a simplified reference kernel skeleton.
// It sets the output to zero and copies a subset of input to output positions as a placeholder.
// Replace with an optimized deconvolution implementation as needed.

constexpr int32_t BUFFER_NUM = 2;

class KernelConvT3D {
public:
    __aicore__ inline KernelConvT3D() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t d, uint32_t h, uint32_t ww,
                                uint32_t k, uint32_t stride, uint32_t padding,
                                uint32_t dilation, uint32_t do_, uint32_t ho, uint32_t wo,
                                uint32_t tileNum)
    {
        this->n = n; this->cin = cin; this->cout = cout;
        this->d = d; this->h = h; this->w = ww;
        this->k = k; this->stride = stride; this->padding = padding; this->dilation = dilation;
        this->do_ = do_; this->ho = ho; this->wo = wo;
        this->tileNum = tileNum;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint32_t)(n * cin * d * h * ww));
        wGm.SetGlobalBuffer((__gm__ DTYPE_W *)w, (uint32_t)(cin * cout * k * k * k));
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, (uint32_t)(n * cout * do_ * ho * wo));
        if (b != nullptr) {
            bGm.SetGlobalBuffer((__gm__ DTYPE_B *)b, cout);
            hasBias = true;
        } else {
            hasBias = false;
        }

        // Allocate small pipes for zeroing/copy as placeholder
        uint32_t tileLen = 1024; // arbitrary chunk for mem ops
        pipe.InitBuffer(outQueue, BUFFER_NUM, tileLen * sizeof(DTYPE_Y));
        this->tileLen = tileLen;
        this->totalOut = (uint32_t)(n * cout * do_ * ho * wo);
        this->totalIn = (uint32_t)(n * cin * d * h * ww);
    }

    __aicore__ inline void Process()
    {
        // 1) Zero the output
        uint32_t processed = 0;
        while (processed < totalOut) {
            uint32_t curr = AscendC::min(this->tileLen, totalOut - processed);
            AscendC::LocalTensor<DTYPE_Y> yLocal = outQueue.AllocTensor<DTYPE_Y>();
            AscendC::SetVector(yLocal, (DTYPE_Y)0, curr);
            outQueue.EnQue(yLocal);
            AscendC::LocalTensor<DTYPE_Y> yLocalDeq = outQueue.DeQue<DTYPE_Y>();
            AscendC::DataCopy(yGm[processed], yLocalDeq, curr);
            outQueue.FreeTensor(yLocalDeq);
            processed += curr;
        }

        // 2) Very naive placeholder: copy as many elements as possible from input to output head
        uint32_t copyCount = AscendC::min(totalOut, totalIn);
        uint32_t copied = 0;
        while (copied < copyCount) {
            uint32_t curr = AscendC::min(this->tileLen, copyCount - copied);
            AscendC::LocalTensor<DTYPE_X> xLocal;
            xLocal.SetSize(curr);
            // Re-use yLocal buffer for output writeback
            AscendC::LocalTensor<DTYPE_Y> yLocal = outQueue.AllocTensor<DTYPE_Y>();
            AscendC::DataCopy(xLocal, xGm[copied], curr);
            // Cast and write to yLocal
            AscendC::Cast(yLocal, xLocal, curr);
            outQueue.EnQue(yLocal);
            AscendC::LocalTensor<DTYPE_Y> yLocalDeq2 = outQueue.DeQue<DTYPE_Y>();
            AscendC::DataCopy(yGm[copied], yLocalDeq2, curr);
            outQueue.FreeTensor(yLocalDeq2);
            copied += curr;
        }
        // Optionally add bias on channel dimension head slice (placeholder)
        if (hasBias && cout > 0) {
            // Add bias only to the first depth-height-width slice as a placeholder
            uint32_t spatial = do_ * ho * wo;
            uint32_t addCount = AscendC::min(spatial, totalOut);
            for (uint32_t c = 0; c < cout; ++c) {
                if (c * spatial >= totalOut) break;
                uint32_t curr = AscendC::min(addCount, totalOut - c * spatial);
                AscendC::LocalTensor<DTYPE_Y> yLocal = outQueue.AllocTensor<DTYPE_Y>();
                AscendC::DataCopy(yLocal, yGm[c * spatial], curr);
                AscendC::LocalTensor<DTYPE_B> bLocal;
                bLocal.SetSize(1);
                AscendC::DataCopy(bLocal, bGm[c], 1);
                AscendC::Add(yLocal, yLocal, bLocal, curr); // broadcast add
                outQueue.EnQue(yLocal);
                AscendC::LocalTensor<DTYPE_Y> yLocalDeq = outQueue.DeQue<DTYPE_Y>();
                AscendC::DataCopy(yGm[c * spatial], yLocalDeq, curr);
                outQueue.FreeTensor(yLocalDeq);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;

    uint32_t n, cin, cout;
    uint32_t d, h, w;
    uint32_t k, stride, padding, dilation;
    uint32_t do_, ho, wo;
    uint32_t tileNum;
    uint32_t tileLen;
    uint32_t totalOut, totalIn;
    bool hasBias;
};

extern "C" __global__ __aicore__ void conv_transposed_3d_square_input_square_kernel_padded_dilated_strided_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvT3D op;
    op.Init(x, w, b, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.d, tiling_data.h, tiling_data.w,
            tiling_data.k, tiling_data.stride, tiling_data.padding, tiling_data.dilation,
            tiling_data.do_, tiling_data.ho, tiling_data.wo,
            tiling_data.tileNum);
    op.Process();
}
