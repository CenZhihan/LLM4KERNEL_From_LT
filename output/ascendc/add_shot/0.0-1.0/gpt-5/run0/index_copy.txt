namespace optiling {
const uint32_t BLOCK_DIM = 1;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    IndexCopyCustomTilingData tiling;

    const auto* x_shape = context->GetInputShape(0);
    const auto* idx_shape = context->GetInputShape(1);
    // x is 2D [rows, cols] for our usage, but allow ND as long as first dim is rows
    uint32_t rows = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    uint64_t total = x_shape->GetOriginShape().GetShapeSize();
    uint32_t cols = static_cast<uint32_t>(total / rows);
    uint32_t srcRows = static_cast<uint32_t>(idx_shape->GetOriginShape().GetShapeSize());

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_rows(rows);
    tiling.set_cols(cols);
    tiling.set_srcRows(srcRows);
    tiling.set_tileRows(1);  // process one row per tile
    tiling.set_tileNum(rows); // one tile per row

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "index_copy_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 1; // one row per tile

class KernelIndexCopy {
public:
    __aicore__ inline KernelIndexCopy() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR indices, GM_ADDR src, GM_ADDR z,
                                uint32_t rows, uint32_t cols, uint32_t srcRows)
    {
        this->rows = rows;
        this->cols = cols;
        this->srcRows = srcRows;
        this->tileLen = cols; // process one full row at a time

        // Single block processing for simplicity and to avoid write conflicts
        uint32_t blockOffset = 0;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + blockOffset, rows * cols);
        indicesGm.SetGlobalBuffer((__gm__ DTYPE_INDICES *)indices, srcRows);
        srcGm.SetGlobalBuffer((__gm__ DTYPE_SRC *)src, srcRows * cols);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z + blockOffset, rows * cols);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLen * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueSrc, BUFFER_NUM, this->tileLen * sizeof(DTYPE_SRC));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLen * sizeof(DTYPE_Z));
        pipe.InitBuffer(inQueueIdx, BUFFER_NUM, 1 * sizeof(DTYPE_INDICES));
    }

    __aicore__ inline void Process()
    {
        // Phase 1: copy x -> z
        for (uint32_t r = 0; r < rows; ++r) {
            CopyInRowFromX(r);
            PassThrough();
            CopyOutRowToZ(r);
        }
        // Phase 2: apply index_copy updates: z[indices[m], :] = src[m, :]
        for (uint32_t m = 0; m < srcRows; ++m) {
            ApplyUpdate(m);
        }
    }

private:
    __aicore__ inline void CopyInRowFromX(uint32_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[rowIdx * cols], this->tileLen);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void PassThrough()
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();
        AscendC::DataCopy(zLocal, xLocal, this->tileLen);
        outQueueZ.EnQue<DTYPE_Z>(zLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOutRowToZ(uint32_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[rowIdx * cols], zLocal, this->tileLen);
        outQueueZ.FreeTensor(zLocal);
    }

    __aicore__ inline void ApplyUpdate(uint32_t m)
    {
        // load index value
        AscendC::LocalTensor<DTYPE_INDICES> idxLocal = inQueueIdx.AllocTensor<DTYPE_INDICES>();
        AscendC::DataCopy(idxLocal, indicesGm[m], 1);
        inQueueIdx.EnQue(idxLocal);
        AscendC::LocalTensor<DTYPE_INDICES> idxLocal2 = inQueueIdx.DeQue<DTYPE_INDICES>();
        // Get index scalar from local tensor
        uint64_t rowIndex = static_cast<uint64_t>(idxLocal2.GetValue(0));
        inQueueIdx.FreeTensor(idxLocal2);

        // load src row
        AscendC::LocalTensor<DTYPE_SRC> srcLocal = inQueueSrc.AllocTensor<DTYPE_SRC>();
        AscendC::DataCopy(srcLocal, srcGm[m * cols], this->tileLen);
        inQueueSrc.EnQue(srcLocal);

        // write to z at rowIndex
        AscendC::LocalTensor<DTYPE_SRC> srcLocal2 = inQueueSrc.DeQue<DTYPE_SRC>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();
        AscendC::DataCopy(zLocal, srcLocal2, this->tileLen);
        outQueueZ.EnQue<DTYPE_Z>(zLocal);
        inQueueSrc.FreeTensor(srcLocal2);

        AscendC::LocalTensor<DTYPE_Z> zLocal2 = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[rowIndex * cols], zLocal2, this->tileLen);
        outQueueZ.FreeTensor(zLocal2);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueSrc;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueIdx;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_INDICES> indicesGm;
    AscendC::GlobalTensor<DTYPE_SRC> srcGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t rows;
    uint32_t cols;
    uint32_t srcRows;
    uint32_t tileLen;
};

extern "C" __global__ __aicore__ void index_copy_custom(GM_ADDR x, GM_ADDR indices, GM_ADDR src, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelIndexCopy op;
    op.Init(x, indices, src, z, tiling_data.rows, tiling_data.cols, tiling_data.srcRows);
    op.Process();
}
