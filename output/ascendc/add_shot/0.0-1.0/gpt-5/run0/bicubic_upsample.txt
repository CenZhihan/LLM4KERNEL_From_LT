#include "bicubic_upsample_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    BicubicUpsampleCustomTilingData tiling;

    const gert::Shape* in_shape = context->GetInputShape(0);
    // Expect NCHW
    uint32_t dims = in_shape->GetActualShape().GetDimNum();
    if (dims != 4) {
        return ge::GRAPH_FAILED;
    }

    uint32_t n = static_cast<uint32_t>(in_shape->GetActualShape().GetDim(0));
    uint32_t c = static_cast<uint32_t>(in_shape->GetActualShape().GetDim(1));
    uint32_t inH = static_cast<uint32_t>(in_shape->GetActualShape().GetDim(2));
    uint32_t inW = static_cast<uint32_t>(in_shape->GetActualShape().GetDim(3));
    // Fixed target size per the model: (256, 256)
    uint32_t outH = 256;
    uint32_t outW = 256;

    uint64_t totalOut = static_cast<uint64_t>(n) * c * outH * outW;

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_inH(inH);
    tiling.set_inW(inW);
    tiling.set_outH(outH);
    tiling.set_outW(outW);
    tiling.set_totalOut(totalOut);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* ws = context->GetWorkspaceSizes(1);
    ws[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);

    // N, C preserved; H, W -> 256, 256
    gert::Shape out;
    out.SetDimNum(4);
    out.SetDim(0, x_shape->GetDim(0));
    out.SetDim(1, x_shape->GetDim(1));
    out.SetDim(2, 256);
    out.SetDim(3, 256);

    *y_shape = out;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "bicubic_upsample_custom_tiling_data.h"

class KernelBicubicUpsample {
public:
    __aicore__ inline KernelBicubicUpsample() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c,
                                uint32_t inH, uint32_t inW,
                                uint32_t outH, uint32_t outW,
                                uint64_t totalOut)
    {
        xPtr = reinterpret_cast<__gm__ float*>(x);
        yPtr = reinterpret_cast<__gm__ float*>(y);

        this->n = n;
        this->c = c;
        this->inH = inH;
        this->inW = inW;
        this->outH = outH;
        this->outW = outW;
        this->totalOut = totalOut;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint64_t baseLen = totalOut / blockNum;
        uint64_t rem = totalOut % blockNum;
        this->start = baseLen * blockIdx + (blockIdx < rem ? blockIdx : rem);
        this->end = this->start + baseLen + (blockIdx < rem ? 1 : 0);
        if (this->end > totalOut) this->end = totalOut;
    }

    __aicore__ inline void Process()
    {
        const float a = -0.75f; // cubic coefficient (Keys)
        // Precompute strides
        uint64_t inStrideN = static_cast<uint64_t>(c) * inH * inW;
        uint64_t inStrideC = static_cast<uint64_t>(inH) * inW;
        uint64_t inStrideH = static_cast<uint64_t>(inW);

        uint64_t outStrideN = static_cast<uint64_t>(c) * outH * outW;
        uint64_t outStrideC = static_cast<uint64_t>(outH) * outW;
        uint64_t outStrideH = static_cast<uint64_t>(outW);

        float scaleY = (outH > 1 && inH > 1) ? (static_cast<float>(inH - 1) / static_cast<float>(outH - 1)) : 0.0f;
        float scaleX = (outW > 1 && inW > 1) ? (static_cast<float>(inW - 1) / static_cast<float>(outW - 1)) : 0.0f;

        for (uint64_t idx = this->start; idx < this->end; ++idx) {
            // Map linear out index -> (n, c, oh, ow)
            uint64_t t = idx;
            uint32_t nIdx = static_cast<uint32_t>(t / outStrideN); t %= outStrideN;
            uint32_t cIdx = static_cast<uint32_t>(t / outStrideC); t %= outStrideC;
            uint32_t oh   = static_cast<uint32_t>(t / outStrideH);
            uint32_t ow   = static_cast<uint32_t>(t % outStrideH);

            float inYf = (outH == 1 ? 0.0f : scaleY * static_cast<float>(oh));
            float inXf = (outW == 1 ? 0.0f : scaleX * static_cast<float>(ow));

            int32_t yInt = static_cast<int32_t>(AscendC::Floor(inYf));
            int32_t xInt = static_cast<int32_t>(AscendC::Floor(inXf));
            float ty = inYf - static_cast<float>(yInt);
            float tx = inXf - static_cast<float>(xInt);

            float wy[4];
            float wx[4];
            cubicWeights(ty, a, wy);
            cubicWeights(tx, a, wx);

            // Accumulate 4x4 neighborhood
            float acc = 0.0f;
            for (int ky = 0; ky < 4; ++ky) {
                int y = clamp(yInt + ky - 1, 0, static_cast<int>(inH - 1));
                float wyv = wy[ky];
                for (int kx = 0; kx < 4; ++kx) {
                    int x = clamp(xInt + kx - 1, 0, static_cast<int>(inW - 1));
                    float wxv = wx[kx];

                    uint64_t inIndex = static_cast<uint64_t>(nIdx) * inStrideN
                                     + static_cast<uint64_t>(cIdx) * inStrideC
                                     + static_cast<uint64_t>(y) * inStrideH
                                     + static_cast<uint64_t>(x);
                    float v = xPtr[inIndex];
                    acc += wyv * wxv * v;
                }
            }

            uint64_t outIndex = static_cast<uint64_t>(nIdx) * outStrideN
                              + static_cast<uint64_t>(cIdx) * outStrideC
                              + static_cast<uint64_t>(oh) * outStrideH
                              + static_cast<uint64_t>(ow);
            yPtr[outIndex] = acc;
        }
    }

private:
    __aicore__ inline int clamp(int v, int lo, int hi) {
        return v < lo ? lo : (v > hi ? hi : v);
    }

    __aicore__ inline void cubicWeights(float t, float a, float w[4]) {
        float t2 = t * t;
        float t3 = t2 * t;
        // Keys cubic with parameter a
        w[0] = ((-a * t + 2.0f * a) * t - a) * t;                   // (t+1)
        w[1] = ((a + 2.0f) * t - (a + 3.0f)) * t2 + 1.0f;           // t
        w[2] = ((-a - 2.0f) * t + (2.0f * a + 3.0f)) * t2;          // (1-t)
        w[3] = (a * t - a) * t2;                                    // (2-t)
    }

private:
    __gm__ float* xPtr;
    __gm__ float* yPtr;
    uint32_t n, c, inH, inW, outH, outW;
    uint64_t totalOut;
    uint64_t start, end;
};

extern "C" __global__ __aicore__ void bicubic_upsample_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelBicubicUpsample op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c,
            tiling_data.inH, tiling_data.inW,
            tiling_data.outH, tiling_data.outW,
            tiling_data.totalOut);
    op.Process();
}
