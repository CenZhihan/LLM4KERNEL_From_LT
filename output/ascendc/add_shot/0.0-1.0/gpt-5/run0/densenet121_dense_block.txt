kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "densenet121_dense_block_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // number of local tensors per queue

class KernelDenseBlockConcat {
public:
    __aicore__ inline KernelDenseBlockConcat() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR z,
                                uint32_t totalLengthX, uint32_t totalLengthY, uint32_t tileNum)
    {
        this->tileNum = tileNum;
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        this->blockLengthX = totalLengthX / blockNum;
        this->blockLengthY = totalLengthY / blockNum;

        // Tile lengths
        this->tileLengthX = this->blockLengthX / tileNum / BUFFER_NUM;
        this->tileLengthY = this->blockLengthY / tileNum / BUFFER_NUM;

        // Global memory pointers for inputs
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLengthX * blockIdx, this->blockLengthX);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->blockLengthY * blockIdx, this->blockLengthY);

        // Output: first copy X segment, then Y segment after totalLengthX
        zGmX.SetGlobalBuffer((__gm__ DTYPE_Z *)z + this->blockLengthX * blockIdx, this->blockLengthX);
        zGmY.SetGlobalBuffer((__gm__ DTYPE_Z *)z + totalLengthX + this->blockLengthY * blockIdx, this->blockLengthY);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLengthX * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueZX, BUFFER_NUM, this->tileLengthX * sizeof(DTYPE_Z));

        pipe.InitBuffer(inQueueY, BUFFER_NUM, this->tileLengthY * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueZY, BUFFER_NUM, this->tileLengthY * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        // Copy X part
        int32_t loopCountX = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCountX; ++i) {
            CopyInX(i);
            ComputeX(i);
            CopyOutX(i);
        }
        // Copy Y part
        int32_t loopCountY = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCountY; ++i) {
            CopyInY(i);
            ComputeY(i);
            CopyOutY(i);
        }
    }

private:
    __aicore__ inline void CopyInX(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLengthX], this->tileLengthX);
        inQueueX.EnQue(xLocal);
    }
    __aicore__ inline void ComputeX(int32_t)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZX.AllocTensor<DTYPE_Z>();
        AscendC::DataCopy(zLocal, xLocal, this->tileLengthX);
        outQueueZX.EnQue<DTYPE_Z>(zLocal);
        inQueueX.FreeTensor(xLocal);
    }
    __aicore__ inline void CopyOutX(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZX.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGmX[progress * this->tileLengthX], zLocal, this->tileLengthX);
        outQueueZX.FreeTensor(zLocal);
    }

    __aicore__ inline void CopyInY(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = inQueueY.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(yLocal, yGm[progress * this->tileLengthY], this->tileLengthY);
        inQueueY.EnQue(yLocal);
    }
    __aicore__ inline void ComputeY(int32_t)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = inQueueY.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZY.AllocTensor<DTYPE_Z>();
        AscendC::DataCopy(zLocal, yLocal, this->tileLengthY);
        outQueueZY.EnQue<DTYPE_Z>(zLocal);
        inQueueY.FreeTensor(yLocal);
    }
    __aicore__ inline void CopyOutY(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGmY[progress * this->tileLengthY], zLocal, this->tileLengthY);
        outQueueZY.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;

    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZX;

    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueY;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;
    AscendC::GlobalTensor<DTYPE_Z> zGmX;
    AscendC::GlobalTensor<DTYPE_Z> zGmY;

    uint32_t blockLengthX;
    uint32_t blockLengthY;
    uint32_t tileNum;
    uint32_t tileLengthX;
    uint32_t tileLengthY;
};

extern "C" __global__ __aicore__ void densenet121_dense_block_custom(GM_ADDR x, GM_ADDR y, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelDenseBlockConcat op;
    op.Init(x, y, z, tiling_data.totalLengthX, tiling_data.totalLengthY, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, num_layers: int, num_input_features: int, growth_rate: int):
        super(ModelNew, self).__init__()
        layers = []
        for i in range(num_layers):
            layers.append(self._make_layer(num_input_features + i * growth_rate, growth_rate))
        self.layers = torch.nn.ModuleList(layers)

    def _make_layer(self, in_features: int, growth_rate: int):
        return torch.nn.Sequential(
            torch.nn.BatchNorm2d(in_features),
            torch.nn.ReLU(inplace=True),
            torch.nn.Conv2d(in_features, growth_rate, kernel_size=3, padding=1, bias=False),
            torch.nn.Dropout(0.0)
        )

    def forward(self, x):
        features = [x]
        for layer in self.layers:
            new_feature = layer(x)
            features.append(new_feature)
            # Replace torch.cat with custom AscendC kernel that concatenates along channel axis (dim=1)
            x = custom_ops_lib.densenet121_dense_block_custom(x, new_feature)
        return x
