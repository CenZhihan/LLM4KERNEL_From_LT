kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "grid_sample_affine_custom_tiling_data.h"
#include <cmath>

class KernelGridSampleAffine {
public:
    __aicore__ inline KernelGridSampleAffine() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR theta, GM_ADDR y,
                                uint32_t totalLength, uint32_t N, uint32_t C, uint32_t H, uint32_t W) {
        this->xGm = (__gm__ DTYPE_X*)x;
        this->thetaGm = (__gm__ DTYPE_Y*)theta;
        this->yGm = (__gm__ DTYPE_Z*)y;
        this->totalLength = totalLength;
        this->N = N; this->C = C; this->H = H; this->W = W;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t base = totalLength / blockNum;
        uint32_t rem  = totalLength % blockNum;
        this->blockStart = blockIdx * base + (blockIdx < rem ? blockIdx : rem);
        this->blockLen   = base + (blockIdx < rem ? 1 : 0);
        this->blockEnd   = this->blockStart + this->blockLen;
    }

    __aicore__ inline void Process() {
        uint32_t CHW = C * H * W;
        uint32_t HW  = H * W;

        float Wf = static_cast<float>(W);
        float Hf = static_cast<float>(H);

        int32_t lastN = -1;
        float t00 = 0.f, t01 = 0.f, t02 = 0.f;
        float t10 = 0.f, t11 = 0.f, t12 = 0.f;

        for (uint32_t idx = this->blockStart; idx < this->blockEnd; ++idx) {
            uint32_t n  = idx / CHW;
            uint32_t rc = idx % CHW;
            uint32_t c  = rc / HW;
            uint32_t rhw= rc % HW;
            uint32_t h  = rhw / W;
            uint32_t w  = rhw % W;

            if ((int32_t)n != lastN) {
                uint32_t baseTheta = n * 6;
                t00 = static_cast<float>(thetaGm[baseTheta + 0]);
                t01 = static_cast<float>(thetaGm[baseTheta + 1]);
                t02 = static_cast<float>(thetaGm[baseTheta + 2]);
                t10 = static_cast<float>(thetaGm[baseTheta + 3]);
                t11 = static_cast<float>(thetaGm[baseTheta + 4]);
                t12 = static_cast<float>(thetaGm[baseTheta + 5]);
                lastN = (int32_t)n;
            }

            float xs = ( (static_cast<float>(w) + 0.5f) / Wf ) * 2.f - 1.f;
            float ys = ( (static_cast<float>(h) + 0.5f) / Hf ) * 2.f - 1.f;

            float gx = t00 * xs + t01 * ys + t02;
            float gy = t10 * xs + t11 * ys + t12;

            float ix = (gx + 1.f) * Wf * 0.5f - 0.5f;
            float iy = (gy + 1.f) * Hf * 0.5f - 0.5f;

            int x0 = static_cast<int>(floorf(ix));
            int y0 = static_cast<int>(floorf(iy));
            int x1 = x0 + 1;
            int y1 = y0 + 1;

            float wx1 = ix - static_cast<float>(x0);
            float wy1 = iy - static_cast<float>(y0);
            float wx0 = 1.f - wx1;
            float wy0 = 1.f - wy1;

            float v = 0.f;

            uint32_t baseNC = (n * C + c) * H * W;

            if (x0 >= 0 && x0 < (int)W && y0 >= 0 && y0 < (int)H) {
                uint32_t off = baseNC + (uint32_t)y0 * W + (uint32_t)x0;
                v += static_cast<float>(xGm[off]) * (wx0 * wy0);
            }
            if (x1 >= 0 && x1 < (int)W && y0 >= 0 && y0 < (int)H) {
                uint32_t off = baseNC + (uint32_t)y0 * W + (uint32_t)x1;
                v += static_cast<float>(xGm[off]) * (wx1 * wy0);
            }
            if (x0 >= 0 && x0 < (int)W && y1 >= 0 && y1 < (int)H) {
                uint32_t off = baseNC + (uint32_t)y1 * W + (uint32_t)x0;
                v += static_cast<float>(xGm[off]) * (wx0 * wy1);
            }
            if (x1 >= 0 && x1 < (int)W && y1 >= 0 && y1 < (int)H) {
                uint32_t off = baseNC + (uint32_t)y1 * W + (uint32_t)x1;
                v += static_cast<float>(xGm[off]) * (wx1 * wy1);
            }

            yGm[idx] = static_cast<DTYPE_Z>(v);
        }
    }

private:
    __gm__ DTYPE_X* xGm;
    __gm__ DTYPE_Y* thetaGm;
    __gm__ DTYPE_Z* yGm;

    uint32_t totalLength;
    uint32_t N, C, H, W;
    uint32_t blockStart, blockEnd, blockLen;
};

extern "C" __global__ __aicore__ void grid_sample_affine_custom(GM_ADDR x, GM_ADDR theta, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGridSampleAffine op;
    op.Init(x, theta, y,
            tiling_data.totalLength,
            tiling_data.N, tiling_data.C,
            tiling_data.H, tiling_data.W);
    op.Process();
}
