kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_transpose2d_min_sum_gelu_add_custom_tiling_data.h"

// Fused kernel: given x[N,C,H,W] and bias (broadcastable), compute
// y[N,1,1,W] where y[n,0,0,w] = GELU( sum_h( min_c( x[n,c,h,w] ) ) ) + bias

// GELU approximation with tanh
__aicore__ inline float gelu_tanh(float x) {
    const float kAlpha = 0.7978845608f; // sqrt(2/pi)
    const float kBeta = 0.044715f;
    float x3 = x * x * x;
    float inner = kAlpha * (x + kBeta * x3);
    float t = AscendC::Tanh(inner);
    return 0.5f * x * (1.0f + t);
}

class KernelFused {
public:
    __aicore__ inline KernelFused() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                uint32_t outLength, uint32_t biasLen, uint32_t tileNum)
    {
        this->n = n;
        this->c = c;
        this->h = h;
        this->w = w;
        this->outLength = outLength;
        this->biasLen = biasLen;
        this->tileNum = tileNum;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint32_t)(n * c * h * w));
        biasGm.SetGlobalBuffer((__gm__ DTYPE_Y *)bias, biasLen);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y, outLength);
    }

    __aicore__ inline void Process()
    {
        // Distribute [0, outLength) over blockDim
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t perBlock = (outLength + blockNum - 1) / blockNum;
        uint32_t start = perBlock * blockIdx;
        uint32_t end = start + perBlock;
        if (end > outLength) end = outLength;

        for (uint32_t idx = start; idx < end; ++idx) {
            uint32_t nIdx = idx / w;
            uint32_t wIdx = idx % w;

            // reduce over h and c
            float sumMin = 0.0f;
            for (uint32_t hi = 0; hi < h; ++hi) {
                float minVal = FLT_MAX;
                // address base for (n, :, h, wIdx)
                for (uint32_t ci = 0; ci < c; ++ci) {
                    uint32_t offset = (((nIdx * c + ci) * h + hi) * w + wIdx);
                    float v = (float)xGm[offset];
                    minVal = (v < minVal) ? v : minVal;
                }
                sumMin += minVal;
            }

            float g = gelu_tanh(sumMin);
            float b = (biasLen == 1) ? (float)biasGm[0] : (float)biasGm[wIdx % biasLen];
            yGm[idx] = (DTYPE_Z)(g + b);
        }
    }

private:
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> biasGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t n, c, h, w;
    uint32_t outLength;
    uint32_t biasLen;
    uint32_t tileNum;
};

extern "C" __global__ __aicore__ void conv_transpose2d_min_sum_gelu_add_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelFused op;
    op.Init(x,
            bias,
            y,
            tiling_data.n,
            tiling_data.c,
            tiling_data.h,
            tiling_data.w,
            tiling_data.outLength,
            tiling_data.biasLen,
            tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    A model that performs a convolution transpose, then a custom fused kernel:
    min over channels -> sum over height -> GELU -> add bias.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, bias_shape):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose2d(in_channels, out_channels, kernel_size, stride, padding, output_padding)
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))

    def forward(self, x):
        x = self.conv_transpose(x)
        # Fused custom op computes: min(dim=1, keepdim=True) -> sum(dim=2, keepdim=True) -> gelu -> add bias
        x = custom_ops_lib.conv_transpose2d_min_sum_gelu_add_custom(x, self.bias)
        return x
