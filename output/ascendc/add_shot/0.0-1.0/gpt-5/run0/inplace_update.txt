kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "inplace_update_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelInplaceUpdate {
public:
    __aicore__ inline KernelInplaceUpdate() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR idx, GM_ADDR value, GM_ADDR z,
                                uint32_t xRows, uint32_t cols, uint32_t mRows, uint32_t tileNum)
    {
        this->xRows = xRows;
        this->cols = cols;
        this->mRows = mRows;
        this->tileNum = tileNum;

        this->totalElems = xRows * cols;

        this->blockNum = AscendC::GetBlockNum();
        this->blockIdx = AscendC::GetBlockIdx();

        // Ranges for copy phase
        this->copyStart = (uint64_t)this->totalElems * this->blockIdx / this->blockNum;
        uint64_t copyEnd = (uint64_t)this->totalElems * (this->blockIdx + 1) / this->blockNum;
        this->copyLen = (uint32_t)(copyEnd - this->copyStart);

        // Ranges for update phase (over M rows of updates)
        this->updStart = (uint64_t)this->mRows * this->blockIdx / this->blockNum;
        uint64_t updEnd = (uint64_t)this->mRows * (this->blockIdx + 1) / this->blockNum;
        this->updLen = (uint32_t)(updEnd - this->updStart);

        // Set global buffers to full range; we'll index with absolute offsets
        xGm.SetGlobalBuffer((__gm__ float*)x, this->totalElems);
        zGm.SetGlobalBuffer((__gm__ float*)z, this->totalElems);
        valGm.SetGlobalBuffer((__gm__ float*)value, this->mRows * this->cols);
        idxGm.SetGlobalBuffer((__gm__ int32_t*)idx, this->mRows);

        // Tile sizes
        this->tileLenCopy = this->copyLen / (this->tileNum * BUFFER_NUM);
        if (this->tileLenCopy == 0) this->tileLenCopy = this->copyLen > 0 ? this->copyLen : 1;

        this->tileCols = this->cols / (this->tileNum * BUFFER_NUM);
        if (this->tileCols == 0) this->tileCols = this->cols > 0 ? this->cols : 1;

        // Initialize buffers
        pipe.InitBuffer(inQueueCopy, BUFFER_NUM, this->tileLenCopy * sizeof(float));
        pipe.InitBuffer(inQueueIdx, BUFFER_NUM, sizeof(int32_t));
        pipe.InitBuffer(inQueueVal, BUFFER_NUM, this->tileCols * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // Phase 1: copy x to z for the assigned range
        if (copyLen > 0) {
            uint32_t loops = (copyLen + tileLenCopy - 1) / tileLenCopy;
            for (uint32_t i = 0; i < loops; ++i) {
                uint32_t curLen = (i + 1 == loops) ? (copyLen - i * tileLenCopy) : tileLenCopy;
                AscendC::LocalTensor<float> xLocal = inQueueCopy.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm[copyStart + (uint64_t)i * tileLenCopy], curLen);
                inQueueCopy.EnQue(xLocal);

                AscendC::LocalTensor<float> xLocal2 = inQueueCopy.DeQue<float>();
                AscendC::DataCopy(zGm[copyStart + (uint64_t)i * tileLenCopy], xLocal2, curLen);
                inQueueCopy.FreeTensor(xLocal2);
            }
        }

        // Phase 2: apply updates for assigned subset of M rows
        for (uint32_t r = 0; r < updLen; ++r) {
            uint32_t i = updStart + r;

            // Load index idx[i]
            AscendC::LocalTensor<int32_t> idxLocal = inQueueIdx.AllocTensor<int32_t>();
            AscendC::DataCopy(idxLocal, idxGm[i], 1);
            inQueueIdx.EnQue(idxLocal);
            AscendC::LocalTensor<int32_t> idxLocal2 = inQueueIdx.DeQue<int32_t>();
            // Read scalar from local tensor
            int32_t rowDst = idxLocal2.GetValue(0);
            inQueueIdx.FreeTensor(idxLocal2);

            // Copy one row value[i, :] -> z[rowDst, :]
            uint64_t dstBase = (uint64_t)rowDst * cols;
            uint64_t srcBase = (uint64_t)i * cols;

            uint32_t colLoops = (cols + tileCols - 1) / tileCols;
            for (uint32_t c = 0; c < colLoops; ++c) {
                uint32_t curCols = (c + 1 == colLoops) ? (cols - c * tileCols) : tileCols;

                AscendC::LocalTensor<float> vLocal = inQueueVal.AllocTensor<float>();
                AscendC::DataCopy(vLocal, valGm[srcBase + (uint64_t)c * tileCols], curCols);
                inQueueVal.EnQue(vLocal);

                AscendC::LocalTensor<float> vLocal2 = inQueueVal.DeQue<float>();
                AscendC::DataCopy(zGm[dstBase + (uint64_t)c * tileCols], vLocal2, curCols);
                inQueueVal.FreeTensor(vLocal2);
            }
        }
    }

private:
    AscendC::TPipe pipe;

    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  inQueueCopy;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  inQueueIdx;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  inQueueVal;

    AscendC::GlobalTensor<float>   xGm;
    AscendC::GlobalTensor<float>   zGm;
    AscendC::GlobalTensor<float>   valGm;
    AscendC::GlobalTensor<int32_t> idxGm;

    uint32_t xRows;
    uint32_t cols;
    uint32_t mRows;
    uint32_t tileNum;
    uint32_t totalElems;

    uint32_t blockNum;
    uint32_t blockIdx;

    uint64_t copyStart;
    uint32_t copyLen;
    uint64_t updStart;
    uint32_t updLen;

    uint32_t tileLenCopy;
    uint32_t tileCols;
};

extern "C" __global__ __aicore__ void inplace_update_custom(GM_ADDR x, GM_ADDR idx, GM_ADDR value, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelInplaceUpdate op;
    op.Init(x, idx, value, z,
            tiling_data.xRows, tiling_data.cols, tiling_data.mRows, tiling_data.tileNum);
    op.Process();
}
"""
