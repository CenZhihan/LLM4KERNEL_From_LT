kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "bmm_instance_norm_sum_residual_add_multiply_custom_tiling_data.h"

// Simple fused kernel implementing:
// z = ((x @ weight^T + bias) IN instance-norm along features) + y; then z *= y
// Shapes:
// x: (N, In), weight: (Out, In), bias: (Out), y: (N, Out), z: (N, Out)

extern \"C\" __global__ __aicore__ void bmm_instance_norm_sum_residual_add_multiply_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    const uint32_t N = tiling_data.batchSize;
    const uint32_t In = tiling_data.inFeatures;
    const uint32_t Out = tiling_data.outFeatures;

    const uint32_t blockNum = AscendC::GetBlockNum();
    const uint32_t blockIdx = AscendC::GetBlockIdx();

    const uint32_t rowsPerBlock = (N + blockNum - 1) / blockNum;
    const uint32_t rowStart = blockIdx * rowsPerBlock;
    const uint32_t rowEnd = (rowStart + rowsPerBlock) < N ? (rowStart + rowsPerBlock) : N;

    const float eps = 1e-5f;

    __gm__ float* xGm = (__gm__ float*)x;               // N * In
    __gm__ float* wGm = (__gm__ float*)weight;          // Out * In (row-major by Out)
    __gm__ float* bGm = (__gm__ float*)bias;            // Out
    __gm__ float* yGm = (__gm__ float*)y;               // N * Out
    __gm__ float* zGm = (__gm__ float*)z;               // N * Out

    for (uint32_t r = rowStart; r < rowEnd; ++r) {
        // First pass: accumulate mean and variance statistics across Out features
        float sum = 0.0f;
        float sumsq = 0.0f;
        for (uint32_t j = 0; j < Out; ++j) {
            // compute dot(x[r, :], w[j, :]) + b[j]
            float acc = bGm[j];
            __gm__ float* wRow = wGm + j * In;
            __gm__ float* xRow = xGm + r * In;
            for (uint32_t k = 0; k < In; ++k) {
                acc += xRow[k] * wRow[k];
            }
            sum += acc;
            sumsq += acc * acc;
        }
        float mean = sum / (float)Out;
        float var = sumsq / (float)Out - mean * mean;
        if (var < 0.0f) var = 0.0f;
        float invStd = AscendC::hsqrt_recip(var + eps);

        // Second pass: compute normalized, add y and multiply by y
        for (uint32_t j = 0; j < Out; ++j) {
            float acc = bGm[j];
            __gm__ float* wRow = wGm + j * In;
            __gm__ float* xRow = xGm + r * In;
            for (uint32_t k = 0; k < In; ++k) {
                acc += xRow[k] * wRow[k];
            }
            float norm = (acc - mean) * invStd;
            float yv = yGm[r * Out + j];
            float tmp = norm + yv;
            zGm[r * Out + j] = tmp * yv;
        }
    }
}
