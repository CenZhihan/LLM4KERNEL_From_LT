kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_transposed_3d_square_input_asymmetric_kernel_custom_tiling_data.h"

#define DTYPE_X float
#define DTYPE_W float
#define DTYPE_Y float

extern "C" __global__ __aicore__ void conv_transposed_3d_square_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N  = tiling_data.N;
    const uint32_t Ci = tiling_data.Ci;
    const uint32_t Di = tiling_data.Di;
    const uint32_t Hi = tiling_data.Hi;
    const uint32_t Wi = tiling_data.Wi;

    const uint32_t Co = tiling_data.Co;
    const uint32_t Kd = tiling_data.Kd;
    const uint32_t Kh = tiling_data.Kh;
    const uint32_t Kw = tiling_data.Kw;

    const uint32_t Do = tiling_data.Do;
    const uint32_t Ho = tiling_data.Ho;
    const uint32_t Wo = tiling_data.Wo;

    const uint64_t totalOut = (uint64_t)N * Co * Do * Ho * Wo;

    __gm__ DTYPE_X* x_ptr = reinterpret_cast<__gm__ DTYPE_X*>(x);
    __gm__ DTYPE_W* w_ptr = reinterpret_cast<__gm__ DTYPE_W*>(w);
    __gm__ DTYPE_Y* y_ptr = reinterpret_cast<__gm__ DTYPE_Y*>(y);

    // Compute per-block range
    uint32_t blockNum = AscendC::GetBlockNum();
    uint32_t blockIdx = AscendC::GetBlockIdx();
    uint64_t blockLen = (totalOut + blockNum - 1) / blockNum;
    uint64_t start = (uint64_t)blockIdx * blockLen;
    uint64_t end = start + blockLen;
    if (end > totalOut) end = totalOut;

    // Precompute strides
    const uint64_t SxN = (uint64_t)Ci * Di * Hi * Wi;
    const uint64_t SxC = (uint64_t)Di * Hi * Wi;
    const uint64_t SxD = (uint64_t)Hi * Wi;
    const uint64_t SxH = (uint64_t)Wi;
    const uint64_t SxW = 1ULL;

    const uint64_t SwCi = (uint64_t)Co * Kd * Kh * Kw;
    const uint64_t SwCo = (uint64_t)Kd * Kh * Kw;
    const uint64_t SwKd = (uint64_t)Kh * Kw;
    const uint64_t SwKh = (uint64_t)Kw;
    const uint64_t SwKw = 1ULL;

    const uint64_t SyN = (uint64_t)Co * Do * Ho * Wo;
    const uint64_t SyC = (uint64_t)Do * Ho * Wo;
    const uint64_t SyD = (uint64_t)Ho * Wo;
    const uint64_t SyH = (uint64_t)Wo;
    const uint64_t SyW = 1ULL;

    // For each output element in this block, compute accumulation
    for (uint64_t linear = start; linear < end; ++linear) {
        // Decode linear index to (n, oc, od, oh, ow)
        uint64_t t = linear;
        uint32_t n  = t / SyN;  t -= (uint64_t)n * SyN;
        uint32_t oc = t / SyC;  t -= (uint64_t)oc * SyC;
        uint32_t od = t / SyD;  t -= (uint64_t)od * SyD;
        uint32_t oh = t / SyH;  t -= (uint64_t)oh * SyH;
        uint32_t ow = t; // remaining

        DTYPE_Y acc = static_cast<DTYPE_Y>(0);

        // Sum over input channels and kernel
        for (uint32_t ci = 0; ci < Ci; ++ci) {
            for (uint32_t kd = 0; kd < Kd; ++kd) {
                int64_t id = (int64_t)od - (int64_t)kd; // stride=1, pad=0
                if (id < 0 || id >= (int64_t)Di) continue;
                for (uint32_t kh = 0; kh < Kh; ++kh) {
                    int64_t ih = (int64_t)oh - (int64_t)kh;
                    if (ih < 0 || ih >= (int64_t)Hi) continue;
                    for (uint32_t kw = 0; kw < Kw; ++kw) {
                        int64_t iw = (int64_t)ow - (int64_t)kw;
                        if (iw < 0 || iw >= (int64_t)Wi) continue;

                        uint64_t x_index = (uint64_t)n * SxN + (uint64_t)ci * SxC +
                                           (uint64_t)id * SxD + (uint64_t)ih * SxH + (uint64_t)iw * SxW;

                        uint64_t w_index = (uint64_t)ci * SwCi + (uint64_t)oc * SwCo +
                                           (uint64_t)kd * SwKd + (uint64_t)kh * SwKh + (uint64_t)kw * SwKw;

                        acc += (DTYPE_Y)(x_ptr[x_index] * w_ptr[w_index]);
                    }
                }
            }
        }

        uint64_t y_index = (uint64_t)n * SyN + (uint64_t)oc * SyC +
                           (uint64_t)od * SyD + (uint64_t)oh * SyH + (uint64_t)ow * SyW;

        y_ptr[y_index] = acc;
    }
}
