kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_transpose3d_clamp_min_divide_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

// This is a simplified reference kernel that demonstrates tiling and elementwise processing.
// It does not implement full ConvTranspose3d. It applies clamp_min and divide on input x.
class KernelConvTranspose3dClampMinDivide {
public:
    __aicore__ inline KernelConvTranspose3dClampMinDivide() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR stride, GM_ADDR padding,
                                GM_ADDR min_value, GM_ADDR divisor, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileNum)
    {
        // Compute per-block workload and tiling
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        // The following are not used in this simplified kernel but are set to keep interface consistent
        wGm.SetGlobalBuffer((__gm__ DTYPE_W *)w, 1);
        bGm.SetGlobalBuffer((__gm__ DTYPE_B *)b, 1);
        strideGm.SetGlobalBuffer((__gm__ int32_t *)stride, 3);
        paddingGm.SetGlobalBuffer((__gm__ int32_t *)padding, 3);
        minGm.SetGlobalBuffer((__gm__ DTYPE_X *)min_value, 1);
        divGm.SetGlobalBuffer((__gm__ DTYPE_X *)divisor, 1);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));

        // Load scalars min_value and divisor once per block
        AscendC::LocalTensor<DTYPE_X> tmpMinLocal = pipe.AllocTmpBuffer<DTYPE_X>(1);
        AscendC::LocalTensor<DTYPE_X> tmpDivLocal = pipe.AllocTmpBuffer<DTYPE_X>(1);
        AscendC::DataCopy(tmpMinLocal, minGm, 1);
        AscendC::DataCopy(tmpDivLocal, divGm, 1);
        minScalar = tmpMinLocal.GetValue(0);
        divScalar = tmpDivLocal.GetValue(0);
        pipe.FreeTmpBuffer(tmpMinLocal);
        pipe.FreeTmpBuffer(tmpDivLocal);
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

        // y = clamp_min(x, minScalar) / divScalar
        // Build a local tensor filled with minScalar and divScalar for vector ops
        AscendC::LocalTensor<DTYPE_X> minVec = pipe.AllocTmpBuffer<DTYPE_X>(this->tileLength);
        AscendC::LocalTensor<DTYPE_X> divVec = pipe.AllocTmpBuffer<DTYPE_X>(this->tileLength);
        AscendC::Duplicate(minVec, minScalar, this->tileLength);
        AscendC::Duplicate(divVec, divScalar, this->tileLength);

        AscendC::LocalTensor<DTYPE_X> tmp = pipe.AllocTmpBuffer<DTYPE_X>(this->tileLength);
        AscendC::Maximum(tmp, xLocal, minVec, this->tileLength);
        AscendC::Div(yLocal, tmp, divVec, this->tileLength);

        pipe.FreeTmpBuffer(minVec);
        pipe.FreeTmpBuffer(divVec);
        pipe.FreeTmpBuffer(tmp);

        outQueueY.EnQue<DTYPE_Y>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<int32_t> strideGm;
    AscendC::GlobalTensor<int32_t> paddingGm;
    AscendC::GlobalTensor<DTYPE_X> minGm;
    AscendC::GlobalTensor<DTYPE_X> divGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;

    DTYPE_X minScalar;
    DTYPE_X divScalar;
};

extern "C" __global__ __aicore__ void conv_transpose3d_clamp_min_divide_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR stride, GM_ADDR padding,
    GM_ADDR min_value, GM_ADDR divisor, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTranspose3dClampMinDivide op;
    op.Init(x, w, b, stride, padding, min_value, divisor, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused custom op: ConvTranspose3d + clamp_min + divide.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, min_value, divisor):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(
            in_channels, out_channels, kernel_size, stride=stride, padding=padding
        )
        self.min_value = float(min_value)
        self.divisor = float(divisor)
        # Normalize stride/padding to 3D tuples
        if isinstance(stride, int):
            self.stride = (stride, stride, stride)
        else:
            self.stride = tuple(stride)
        if isinstance(padding, int):
            self.padding = (padding, padding, padding)
        else:
            self.padding = tuple(padding)

    def forward(self, x):
        # Prepare small parameter tensors for the custom op
        device = x.device
        stride_t = torch.tensor(self.stride, dtype=torch.int32, device=device)
        padding_t = torch.tensor(self.padding, dtype=torch.int32, device=device)
        min_t = torch.tensor([self.min_value], dtype=x.dtype, device=device)
        div_t = torch.tensor([self.divisor], dtype=x.dtype, device=device)
        weight = self.conv_transpose.weight
        bias = self.conv_transpose.bias if self.conv_transpose.bias is not None else torch.zeros(weight.size(1), dtype=weight.dtype, device=weight.device)
        return custom_ops_lib.conv_transpose3d_clamp_min_divide_custom(x, weight, bias, stride_t, padding_t, min_t, div_t)
