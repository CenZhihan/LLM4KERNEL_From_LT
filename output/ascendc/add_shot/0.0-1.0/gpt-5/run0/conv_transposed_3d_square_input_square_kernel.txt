kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_transposed_3d_square_input_square_kernel_custom_tiling_data.h"

extern \"C\" __global__ __aicore__ void conv_transposed_3d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(t, tiling);

    const uint32_t N  = t.N;
    const uint32_t Ci = t.Ci;
    const uint32_t Co = t.Co;
    const uint32_t D  = t.D;
    const uint32_t H  = t.H;
    const uint32_t W  = t.W;
    const uint32_t K  = t.K;
    const uint32_t Do = t.Do;
    const uint32_t Ho = t.Ho;
    const uint32_t Wo = t.Wo;

    __gm__ DTYPE_X* xGm = (__gm__ DTYPE_X*)x;   // [N, Ci, D, H, W]
    __gm__ DTYPE_Y* wGm = (__gm__ DTYPE_Y*)w;   // [Ci, Co, K, K, K]
    __gm__ DTYPE_Z* yGm = (__gm__ DTYPE_Z*)y;   // [N, Co, Do, Ho, Wo]

    // Partition work across blocks along Co dimension
    uint32_t blocks = AscendC::GetBlockNum();
    uint32_t blockId = AscendC::GetBlockIdx();

    uint32_t coPerBlock = (Co + blocks - 1) / blocks;
    uint32_t coStart = blockId * coPerBlock;
    uint32_t coEnd = coStart + coPerBlock;
    if (coEnd > Co) coEnd = Co;

    if (coStart >= Co) {
        return;
    }

    // Precompute strides
    uint64_t x_stride_w = 1;
    uint64_t x_stride_h = (uint64_t)W;
    uint64_t x_stride_d = (uint64_t)H * (uint64_t)W;
    uint64_t x_stride_c = (uint64_t)D * (uint64_t)H * (uint64_t)W;
    uint64_t x_stride_n = (uint64_t)Ci * x_stride_c;

    uint64_t w_stride_kw = 1;
    uint64_t w_stride_kh = (uint64_t)K;
    uint64_t w_stride_kd = (uint64_t)K * (uint64_t)K;
    uint64_t w_stride_co = (uint64_t)K * (uint64_t)K * (uint64_t)K;
    uint64_t w_stride_ci = (uint64_t)Co * w_stride_co;

    uint64_t y_stride_w = 1;
    uint64_t y_stride_h = (uint64_t)Wo;
    uint64_t y_stride_d = (uint64_t)Ho * (uint64_t)Wo;
    uint64_t y_stride_c = (uint64_t)Do * (uint64_t)Ho * (uint64_t)Wo;
    uint64_t y_stride_n = (uint64_t)Co * y_stride_c;

    // Loop over batches and assigned output channels
    for (uint32_t n = 0; n < N; ++n) {
        // Zero initialize output for assigned Co range
        for (uint32_t oc = coStart; oc < coEnd; ++oc) {
            uint64_t y_base = (uint64_t)n * y_stride_n + (uint64_t)oc * y_stride_c;
            for (uint32_t od = 0; od < Do; ++od) {
                for (uint32_t oh = 0; oh < Ho; ++oh) {
                    // vectorize along width if desired; here scalar clear
                    for (uint32_t ow = 0; ow < Wo; ++ow) {
                        yGm[y_base + (uint64_t)od * y_stride_d + (uint64_t)oh * y_stride_h + (uint64_t)ow] = (DTYPE_Z)0;
                    }
                }
            }
        }

        // Accumulate transposed convolution
        for (uint32_t ic = 0; ic < Ci; ++ic) {
            for (uint32_t id = 0; id < D; ++id) {
                for (uint32_t ih = 0; ih < H; ++ih) {
                    for (uint32_t iw = 0; iw < W; ++iw) {
                        uint64_t x_idx = (uint64_t)n * x_stride_n + (uint64_t)ic * x_stride_c + (uint64_t)id * x_stride_d + (uint64_t)ih * x_stride_h + (uint64_t)iw * x_stride_w;
                        DTYPE_X xval = xGm[x_idx];

                        for (uint32_t kd = 0; kd < K; ++kd) {
                            uint32_t od = id + kd; // stride=1, padding=0
                            for (uint32_t kh = 0; kh < K; ++kh) {
                                uint32_t oh = ih + kh;
                                for (uint32_t kw = 0; kw < K; ++kw) {
                                    uint32_t ow = iw + kw;
                                    // for assigned oc range
                                    for (uint32_t oc = coStart; oc < coEnd; ++oc) {
                                        uint64_t w_idx = (uint64_t)ic * w_stride_ci + (uint64_t)oc * w_stride_co + (uint64_t)kd * w_stride_kd + (uint64_t)kh * w_stride_kh + (uint64_t)kw * w_stride_kw;
                                        DTYPE_Y wval = wGm[w_idx];

                                        uint64_t y_idx = (uint64_t)n * y_stride_n + (uint64_t)oc * y_stride_c + (uint64_t)od * y_stride_d + (uint64_t)oh * y_stride_h + (uint64_t)ow * y_stride_w;
                                        yGm[y_idx] = (DTYPE_Z)(yGm[y_idx] + (DTYPE_Z)(xval * wval));
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }
    }
}
"""

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

static inline at::Tensor conv_transposed_3d_square_input_square_kernel_custom_out_shape(const at::Tensor& x, const at::Tensor& w) {
    // x: [N, Ci, D, H, W], w: [Ci, Co, K, K, K]
    auto N = x.size(0);
    auto Ci = x.size(1);
    auto D = x.size(2);
    auto H = x.size(3);
    auto W = x.size(4);
    TORCH_CHECK(w.dim() == 5, \"weight must be 5D\");
    TORCH_CHECK(w.size(0) == Ci, \"weight Ci must match input Ci\");
    auto Co = w.size(1);
    TORCH_CHECK(w.size(2) == w.size(3) && w.size(3) == w.size(4), \"kernel must be square\");
    auto K = w.size(2);
    auto Do = D + K - 1;
    auto Ho = H + K - 1;
    auto Wo = W + K - 1;
    return at::empty({N, Co, Do, Ho, Wo}, x.options().dtype(x.dtype()));
}

at::Tensor conv_transposed_3d_square_input_square_kernel_custom_impl_npu(const at::Tensor& x, const at::Tensor& w) {
    at::Tensor y = conv_transposed_3d_square_input_square_kernel_custom_out_shape(x, w);
    EXEC_NPU_CMD(aclnnConvTransposed3dSquareInputSquareKernelCustom, x, w, y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"conv_transposed_3d_square_input_square_kernel_custom\", &conv_transposed_3d_square_input_square_kernel_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
