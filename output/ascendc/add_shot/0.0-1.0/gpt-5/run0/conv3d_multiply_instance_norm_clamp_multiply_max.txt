kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv3d_multiply_instance_norm_clamp_multiply_max_custom_tiling_data.h"
#include <float.h>
#include <cmath>

using DTYPE = float;

class KernelFused {
public:
    __aicore__ inline KernelFused() {}

    __aicore__ inline void Init(GM_ADDR x,
                                GM_ADDR multiplier,
                                GM_ADDR clamp_min,
                                GM_ADDR clamp_max,
                                GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t D, uint32_t H, uint32_t W) {
        this->N = N;
        this->C = C;
        this->D = D;
        this->H = H;
        this->W = W;
        this->S = D * H * W;
        xGm.SetGlobalBuffer((__gm__ DTYPE*)x, (uint32_t)(N * C * S));
        mGm.SetGlobalBuffer((__gm__ DTYPE*)multiplier, C);
        minGm.SetGlobalBuffer((__gm__ DTYPE*)clamp_min, 1);
        maxGm.SetGlobalBuffer((__gm__ DTYPE*)clamp_max, 1);
        yGm.SetGlobalBuffer((__gm__ DTYPE*)y, (uint32_t)(N * S));
    }

    __aicore__ inline void Process() {
        // Read clamp scalars
        DTYPE clamp_min_v = ((__gm__ DTYPE*)minGm.GetPhyAddr())[0];
        DTYPE clamp_max_v = ((__gm__ DTYPE*)maxGm.GetPhyAddr())[0];

        uint32_t blkNum = AscendC::GetBlockNum();
        uint32_t blkIdx = AscendC::GetBlockIdx();

        for (uint32_t n = blkIdx; n < N; n += blkNum) {
            // Initialize output y[n, :, :, :] to -inf
            uint32_t yBase = n * S;
            for (uint32_t s = 0; s < S; ++s) {
                ((__gm__ DTYPE*)yGm.GetPhyAddr())[yBase + s] = -FLT_MAX;
            }

            // Process each channel independently
            for (uint32_t c = 0; c < C; ++c) {
                DTYPE mval = ((__gm__ DTYPE*)mGm.GetPhyAddr())[c];

                // Compute mean and variance over spatial dims for t = x * mval
                uint32_t xBase = (n * C + c) * S;
                DTYPE sum = static_cast<DTYPE>(0.0f);
                DTYPE sumsq = static_cast<DTYPE>(0.0f);
                for (uint32_t s = 0; s < S; ++s) {
                    DTYPE t = ((__gm__ DTYPE*)xGm.GetPhyAddr())[xBase + s] * mval;
                    sum += t;
                    sumsq += t * t;
                }
                DTYPE mean = sum / static_cast<DTYPE>(S);
                DTYPE var = sumsq / static_cast<DTYPE>(S) - mean * mean;
                if (var < static_cast<DTYPE>(0.0f)) var = static_cast<DTYPE>(0.0f);
                const DTYPE eps = static_cast<DTYPE>(1e-5f);
                DTYPE inv_std = static_cast<DTYPE>(1.0f) / AscendC::Sqrt(var + eps);

                // Second pass: normalize, clamp, multiply again, and reduce max across channels
                for (uint32_t s = 0; s < S; ++s) {
                    DTYPE t = ((__gm__ DTYPE*)xGm.GetPhyAddr())[xBase + s] * mval;
                    DTYPE normed = (t - mean) * inv_std;
                    // clamp
                    if (normed < clamp_min_v) normed = clamp_min_v;
                    if (normed > clamp_max_v) normed = clamp_max_v;
                    // second multiply
                    DTYPE outv = normed * mval;
                    // reduce max with current y
                    DTYPE* yPtr = ((__gm__ DTYPE*)yGm.GetPhyAddr());
                    DTYPE cur = yPtr[yBase + s];
                    yPtr[yBase + s] = (outv > cur) ? outv : cur;
                }
            }
        }
    }

private:
    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> mGm;
    AscendC::GlobalTensor<DTYPE> minGm;
    AscendC::GlobalTensor<DTYPE> maxGm;
    AscendC::GlobalTensor<DTYPE> yGm;
    uint32_t N, C, D, H, W, S;
};

extern "C" __global__ __aicore__ void conv3d_multiply_instance_norm_clamp_multiply_max_custom(GM_ADDR x,
                                                                                              GM_ADDR multiplier,
                                                                                              GM_ADDR clamp_min,
                                                                                              GM_ADDR clamp_max,
                                                                                              GM_ADDR y,
                                                                                              GM_ADDR workspace,
                                                                                              GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelFused op;
    op.Init(x, multiplier, clamp_min, clamp_max, y,
            tiling_data.N, tiling_data.C, tiling_data.D, tiling_data.H, tiling_data.W);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fuses: x = conv3d(x); x = x * multiplier; x = instance_norm(x); x = clamp(x); x = x * multiplier; x = max(x, dim=1)
    into a custom NPU operator after the Conv3d layer.
    """
    def __init__(self, in_channels, out_channels, kernel_size, multiplier_shape, clamp_min, clamp_max):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv3d(in_channels, out_channels, kernel_size)
        self.multiplier = torch.nn.Parameter(torch.randn(multiplier_shape))
        self.clamp_min = clamp_min
        self.clamp_max = clamp_max

    def forward(self, x):
        x = self.conv(x)
        min_t = torch.tensor(self.clamp_min, dtype=x.dtype, device=x.device)
        max_t = torch.tensor(self.clamp_max, dtype=x.dtype, device=x.device)
        return custom_ops_lib.conv3d_multiply_instance_norm_clamp_multiply_max_custom(x, self.multiplier, min_t, max_t)
