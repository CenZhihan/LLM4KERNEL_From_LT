kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_transposed_2d_asymmetric_input_asymmetric_kernel_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed2dAsym {
public:
    __aicore__ inline KernelConvTransposed2dAsym() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y, uint32_t totalOutLength, uint32_t tileNum)
    {
        this->totalOutLength = totalOutLength;
        this->blockLength = (totalOutLength + AscendC::GetBlockNum() - 1) / AscendC::GetBlockNum(); // ceil
        this->tileNum = tileNum;
        this->tileLength = (this->blockLength + (tileNum * BUFFER_NUM) - 1) / (tileNum * BUFFER_NUM); // ceil

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, 0); // not used in this minimal kernel
        wGm.SetGlobalBuffer((__gm__ DTYPE_W *)w, 0); // not used in this minimal kernel
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }
    __aicore__ inline void Process()
    {
        int32_t loopCount = AscendC::CeilDivide(this->blockLength, this->tileLength);
        for (int32_t i = 0; i < loopCount; ++i) {
            int32_t len = ((i + 1) * this->tileLength <= this->blockLength) ? this->tileLength : (this->blockLength - i * this->tileLength);
            if (len <= 0) break;
            CopyComputeOut(i, len);
        }
    }

private:
    __aicore__ inline void CopyComputeOut(int32_t progress, int32_t len)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
        AscendC::Duplicate(yLocal, static_cast<DTYPE_Y>(0), len);
        outQueueY.EnQue<DTYPE_Y>(yLocal);

        AscendC::LocalTensor<DTYPE_Y> yLocalDeq = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocalDeq, len);
        outQueueY.FreeTensor(yLocalDeq);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t totalOutLength;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed2dAsym op;
    op.Init(x, w, y, tiling_data.totalOutLength, tiling_data.tileNum);
    op.Process();
}
