kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "matmul_add_swish_tanh_gelu_hardtanh_custom_tiling_data.h"

// Note: This is a reference AscendC kernel illustrating a fused GEMM + activations pipeline.
// It uses straightforward tiling and elementwise math. Further optimization (cube MMA, better tiling)
// is recommended for production-performance workloads.

using DTYPE = float;

__aicore__ inline DTYPE sigmoid_scalar(DTYPE x) {
    // 1 / (1 + exp(-x))
    return static_cast<DTYPE>(1.0f) / (static_cast<DTYPE>(1.0f) + AscendC::Exp(-x));
}

__aicore__ inline DTYPE tanh_scalar(DTYPE x) {
    return AscendC::Tanh(x);
}

__aicore__ inline DTYPE gelu_approx_scalar(DTYPE x) {
    // Approximate GELU: 0.5*x*(1 + tanh(sqrt(2/pi)*(x + 0.044715*x^3)))
    const DTYPE k0 = static_cast<DTYPE>(0.7978845608f); // sqrt(2/pi)
    const DTYPE k1 = static_cast<DTYPE>(0.044715f);
    DTYPE x3 = x * x * x;
    DTYPE t = k0 * (x + k1 * x3);
    DTYPE u = AscendC::Tanh(t);
    return static_cast<DTYPE>(0.5f) * x * (static_cast<DTYPE>(1.0f) + u);
}

__aicore__ inline DTYPE hardtanh_scalar(DTYPE x, DTYPE min_v, DTYPE max_v) {
    if (x < min_v) return min_v;
    if (x > max_v) return max_v;
    return x;
}

class KernelMatmulAddAct {
public:
    __aicore__ inline KernelMatmulAddAct() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR a, GM_ADDR y,
                                uint32_t m, uint32_t n, uint32_t k,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK)
    {
        this->M = m; this->N = n; this->K = k;
        this->tM = tileM; this->tN = tileN; this->tK = tileK;

        xGm.SetGlobalBuffer((__gm__ DTYPE*)x, (uint64_t)m * k);
        wGm.SetGlobalBuffer((__gm__ DTYPE*)w, (uint64_t)n * k);
        bGm.SetGlobalBuffer((__gm__ DTYPE*)b, (uint64_t)n);
        aGm.SetGlobalBuffer((__gm__ DTYPE*)a, (uint64_t)n);
        yGm.SetGlobalBuffer((__gm__ DTYPE*)y, (uint64_t)m * n);

        // Buffers: tile-level. Keep simple: single-buffered
        pipe.InitBuffer(xTileQue, 1, tM * tK * sizeof(DTYPE));
        pipe.InitBuffer(wTileQue, 1, tN * tK * sizeof(DTYPE));
        pipe.InitBuffer(yTileQue, 1, tM * tN * sizeof(DTYPE));
        pipe.InitBuffer(biasQue,   1, tN * sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        // Split rows across blocks
        uint32_t blk = AscendC::GetBlockNum();
        uint32_t bid = AscendC::GetBlockIdx();
        uint32_t rowStart = (uint64_t)M * bid / blk;
        uint32_t rowEnd   = (uint64_t)M * (bid + 1) / blk;
        uint32_t rowsThisBlock = rowEnd - rowStart;

        for (uint32_t rm = 0; rm < rowsThisBlock; rm += tM) {
            uint32_t curM = AscendC::Min(tM, rowsThisBlock - rm);

            for (uint32_t cn = 0; cn < N; cn += tN) {
                uint32_t curN = AscendC::Min(tN, N - cn);

                // Allocate output tile and set to 0
                AscendC::LocalTensor<DTYPE> yTile = yTileQue.AllocTensor<DTYPE>();
                for (uint32_t i = 0; i < curM * curN; ++i) {
                    yTile.SetValue(i, static_cast<DTYPE>(0));
                }

                // GEMM accumulation over K
                for (uint32_t ck = 0; ck < K; ck += tK) {
                    uint32_t curK = AscendC::Min(tK, K - ck);

                    // Load X tile [curM, curK]
                    AscendC::LocalTensor<DTYPE> xTile = xTileQue.AllocTensor<DTYPE>();
                    for (uint32_t i = 0; i < curM; ++i) {
                        // gm offset for row (rowStart+rm+i), col ck
                        uint64_t off = ((uint64_t)(rowStart + rm + i) * K + ck);
                        AscendC::DataCopy(xTile[i * curK], xGm[off], curK);
                    }
                    xTileQue.EnQue(xTile);

                    // Load W tile [curN, curK] (rows of W are output channels)
                    AscendC::LocalTensor<DTYPE> wTile = wTileQue.AllocTensor<DTYPE>();
                    for (uint32_t j = 0; j < curN; ++j) {
                        // gm offset for row (cn+j), col ck
                        uint64_t off = ((uint64_t)(cn + j) * K + ck);
                        AscendC::DataCopy(wTile[j * curK], wGm[off], curK);
                    }
                    wTileQue.EnQue(wTile);

                    // DeQue to compute
                    AscendC::LocalTensor<DTYPE> xLoc = xTileQue.DeQue<DTYPE>();
                    AscendC::LocalTensor<DTYPE> wLoc = wTileQue.DeQue<DTYPE>();

                    // Naive matmul: yTile[i, j] += dot(xLoc[i, :curK], wLoc[j, :curK])
                    for (uint32_t i = 0; i < curM; ++i) {
                        for (uint32_t j = 0; j < curN; ++j) {
                            DTYPE acc = yTile.GetValue(i * curN + j);
                            for (uint32_t p = 0; p < curK; ++p) {
                                DTYPE xv = xLoc.GetValue(i * curK + p);
                                DTYPE wv = wLoc.GetValue(j * curK + p);
                                acc += xv * wv;
                            }
                            yTile.SetValue(i * curN + j, acc);
                        }
                    }

                    xTileQue.FreeTensor(xLoc);
                    wTileQue.FreeTensor(wLoc);
                }

                // Load bias and add_value slices [curN]
                AscendC::LocalTensor<DTYPE> biasLoc = biasQue.AllocTensor<DTYPE>();
                AscendC::DataCopy(biasLoc, bGm[cn], curN);
                // Reuse biasLoc as temp to store bias + addv
                for (uint32_t j = 0; j < curN; ++j) {
                    DTYPE bj = biasLoc.GetValue(j);
                    DTYPE aj = aGm.GetValue(cn + j);
                    biasLoc.SetValue(j, bj + aj);
                }

                // Add bias+addv and apply activations: Swish -> Tanh -> GELU -> Hardtanh
                for (uint32_t i = 0; i < curM; ++i) {
                    for (uint32_t j = 0; j < curN; ++j) {
                        DTYPE v = yTile.GetValue(i * curN + j) + biasLoc.GetValue(j);
                        DTYPE s = sigmoid_scalar(v) * v;            // Swish
                        DTYPE t = tanh_scalar(s);                    // Tanh
                        DTYPE g = gelu_approx_scalar(t);             // GELU approx
                        DTYPE h = hardtanh_scalar(g, -1.0f, 1.0f);   // Hardtanh
                        yTile.SetValue(i * curN + j, h);
                    }
                }

                // Store output tile to GM
                for (uint32_t i = 0; i < curM; ++i) {
                    uint64_t offY = ((uint64_t)(rowStart + rm + i) * N + cn);
                    AscendC::DataCopy(yGm[offY], yTile[i * curN], curN);
                }

                biasQue.FreeTensor(biasLoc);
                yTileQue.FreeTensor(yTile);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  xTileQue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  wTileQue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  biasQue;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> yTileQue;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> wGm;
    AscendC::GlobalTensor<DTYPE> bGm;
    AscendC::GlobalTensor<DTYPE> aGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t M, N, K;
    uint32_t tM, tN, tK;
};

extern "C" __global__ __aicore__ void matmul_add_swish_tanh_gelu_hardtanh_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR add_value, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulAddAct op;
    op.Init(x, weight, bias, add_value, y,
            tiling_data.m, tiling_data.n, tiling_data.k,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK);
    op.Process();
}
