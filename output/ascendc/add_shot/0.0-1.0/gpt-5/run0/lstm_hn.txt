kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "lstm_hn_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelLstmHn {
public:
    __aicore__ inline KernelLstmHn() {}
    __aicore__ inline void Init(GM_ADDR h, GM_ADDR hn, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        hGm.SetGlobalBuffer((__gm__ DTYPE_X *)h + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        hnGm.SetGlobalBuffer((__gm__ DTYPE_Z *)hn + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueH, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueHN, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> hLocal = inQueueH.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(hLocal, hGm[progress * this->tileLength], this->tileLength);
        inQueueH.EnQue(hLocal);
    }

    __aicore__ inline void Compute(int32_t)
    {
        AscendC::LocalTensor<DTYPE_X> hLocal = inQueueH.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> hnLocal = outQueueHN.AllocTensor<DTYPE_Z>();
        AscendC::DataCopy(hnLocal, hLocal, this->tileLength);
        outQueueHN.EnQue<DTYPE_Z>(hnLocal);
        inQueueH.FreeTensor(hLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> hnLocal = outQueueHN.DeQue<DTYPE_Z>();
        AscendC::DataCopy(hnGm[progress * this->tileLength], hnLocal, this->tileLength);
        outQueueHN.FreeTensor(hnLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueH;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueHN;

    AscendC::GlobalTensor<DTYPE_X> hGm;
    AscendC::GlobalTensor<DTYPE_Z> hnGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void lstm_hn_custom(GM_ADDR h, GM_ADDR hn, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLstmHn op;
    op.Init(h, hn, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}
"""

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, input_size, hidden_size, num_layers, output_size, dropout=0.0):
        super(ModelNew, self).__init__()
        self.lstm = torch.nn.LSTM(input_size, hidden_size, num_layers, batch_first=True, dropout=dropout, bidirectional=False)
        self.fc = torch.nn.Linear(hidden_size, output_size)

    def forward(self, x, h0, c0):
        out, state = self.lstm(x, (h0, c0))
        out = self.fc(out[:, -1, :])
        return custom_ops_lib.lstm_hn_custom(state[0])
