kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_standard_2d_square_input_asymmetric_kernel_dilated_padded_custom_tiling_data.h"

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_asymmetric_kernel_dilated_padded_custom(
    GM_ADDR x,
    GM_ADDR w,
    GM_ADDR b,
    GM_ADDR y,
    GM_ADDR workspace,
    GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    // Cast GM addresses
    __gm__ float* x_gm = reinterpret_cast<__gm__ float*>(x);
    __gm__ float* w_gm = reinterpret_cast<__gm__ float*>(w);
    __gm__ float* b_gm = reinterpret_cast<__gm__ float*>(b);
    __gm__ float* y_gm = reinterpret_cast<__gm__ float*>(y);

    // Read tiling parameters
    uint32_t N  = tiling_data.n;
    uint32_t Cin = tiling_data.ci;
    uint32_t Hin = tiling_data.hi;
    uint32_t Win = tiling_data.wi;
    uint32_t Cout = tiling_data.co;
    uint32_t Kh = tiling_data.kh;
    uint32_t Kw = tiling_data.kw;
    uint32_t Hout = tiling_data.ho;
    uint32_t Wout = tiling_data.wo;

    uint32_t stride_h = tiling_data.stride_h;
    uint32_t stride_w = tiling_data.stride_w;
    uint32_t pad_h = tiling_data.pad_h;
    uint32_t pad_w = tiling_data.pad_w;
    uint32_t dilation_h = tiling_data.dilation_h;
    uint32_t dilation_w = tiling_data.dilation_w;
    uint32_t hasBias = tiling_data.hasBias;

    uint32_t total = tiling_data.totalOutLength;

    uint32_t numBlocks = AscendC::GetBlockNum();
    uint32_t blockIdx = AscendC::GetBlockIdx();
    uint32_t perBlock = (total + numBlocks - 1) / numBlocks;
    uint32_t start = blockIdx * perBlock;
    uint32_t end = start + perBlock;
    if (end > total) end = total;

    // Strides for indexing
    uint64_t x_n_stride = static_cast<uint64_t>(Cin) * Hin * Win;
    uint64_t x_c_stride = static_cast<uint64_t>(Hin) * Win;
    uint64_t x_h_stride = static_cast<uint64_t>(Win);

    uint64_t w_co_stride = static_cast<uint64_t>(Cin) * Kh * Kw;
    uint64_t w_ci_stride = static_cast<uint64_t>(Kh) * Kw;
    uint64_t w_kh_stride = static_cast<uint64_t>(Kw);

    uint64_t y_n_stride = static_cast<uint64_t>(Cout) * Hout * Wout;
    uint64_t y_co_stride = static_cast<uint64_t>(Hout) * Wout;
    uint64_t y_ho_stride = static_cast<uint64_t>(Wout);

    for (uint32_t idx = start; idx < end; ++idx) {
        uint32_t tmp = idx;

        uint32_t n = tmp / y_n_stride;
        tmp = tmp % y_n_stride;

        uint32_t co = tmp / y_co_stride;
        tmp = tmp % y_co_stride;

        uint32_t ho = tmp / y_ho_stride;
        uint32_t wo = tmp % y_ho_stride;

        int64_t h_in_base = static_cast<int64_t>(ho) * stride_h - static_cast<int64_t>(pad_h);
        int64_t w_in_base = static_cast<int64_t>(wo) * stride_w - static_cast<int64_t>(pad_w);

        float acc = 0.0f;
        if (hasBias) {
            acc = b_gm[co];
        }

        uint64_t y_off = static_cast<uint64_t>(n) * y_n_stride + static_cast<uint64_t>(co) * y_co_stride + static_cast<uint64_t>(ho) * y_ho_stride + wo;

        for (uint32_t ci = 0; ci < Cin; ++ci) {
            for (uint32_t kh = 0; kh < Kh; ++kh) {
                int64_t hi = h_in_base + static_cast<int64_t>(kh) * dilation_h;
                if (hi < 0 || hi >= static_cast<int64_t>(Hin)) {
                    continue;
                }
                for (uint32_t kw = 0; kw < Kw; ++kw) {
                    int64_t wi = w_in_base + static_cast<int64_t>(kw) * dilation_w;
                    if (wi < 0 || wi >= static_cast<int64_t>(Win)) {
                        continue;
                    }

                    uint64_t x_off = static_cast<uint64_t>(n) * x_n_stride
                                   + static_cast<uint64_t>(ci) * x_c_stride
                                   + static_cast<uint64_t>(hi) * x_h_stride
                                   + static_cast<uint64_t>(wi);

                    uint64_t w_off = static_cast<uint64_t>(co) * w_co_stride
                                   + static_cast<uint64_t>(ci) * w_ci_stride
                                   + static_cast<uint64_t>(kh) * w_kh_stride
                                   + static_cast<uint64_t>(kw);

                    acc += x_gm[x_off] * w_gm[w_off];
                }
            }
        }

        y_gm[y_off] = acc;
    }
}

at::Tensor conv_standard_2d_square_input_asymmetric_kernel_dilated_padded_custom_impl_npu(
    const at::Tensor& x,            // N, Cin, H, W
    const at::Tensor& w,            // Cout, Cin, Kh, Kw
    c10::optional<at::Tensor> b,    // [Cout] or None
    int64_t stride_h,
    int64_t stride_w,
    int64_t pad_h,
    int64_t pad_w,
    int64_t dilation_h,
    int64_t dilation_w)
{
    TORCH_CHECK(x.dim() == 4, "x must be 4D NCHW");
    TORCH_CHECK(w.dim() == 4, "w must be 4D OIHW");
    auto N = x.size(0);
    auto Cin = x.size(1);
    auto Hin = x.size(2);
    auto Win = x.size(3);

    auto Cout = w.size(0);
    auto Kh = w.size(2);
    auto Kw = w.size(3);

    TORCH_CHECK(w.size(1) == Cin, "weight Cin mismatch with input");

    int64_t Hout = (Hin + 2 * pad_h - (Kh - 1) * dilation_h - 1) / stride_h + 1;
    int64_t Wout = (Win + 2 * pad_w - (Kw - 1) * dilation_w - 1) / stride_w + 1;
    TORCH_CHECK(Hout > 0 && Wout > 0, "Computed output size is non-positive");

    at::Tensor y = at::empty({N, Cout, Hout, Wout}, x.options());

    at::Tensor biasTensor;
    if (b.has_value() && b.value().defined()) {
        biasTensor = b.value();
        TORCH_CHECK(biasTensor.dim() == 1 && biasTensor.size(0) == Cout, "bias must be 1D of size Cout");

        biasTensor = at::Tensor(); // undefined optional

                 x, w, biasTensor,
                 stride_h, stride_w,
                 pad_h, pad_w,

model_src='''
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    """
    Custom AscendC-optimized 2D convolution with asymmetric kernel, dilation, and symmetric axis padding.
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: tuple, stride: int = 1, padding: tuple = (0, 0), dilation: tuple = (1, 1), bias: bool = False):
        super(ModelNew, self).__init__()
        # Use nn.Conv2d only as parameter container; forward uses custom op
        self.conv2d = nn.Conv2d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, dilation=dilation, bias=bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        stride_h, stride_w = (self.conv2d.stride if isinstance(self.conv2d.stride, tuple) else (self.conv2d.stride, self.conv2d.stride))
        pad_h, pad_w = (self.conv2d.padding if isinstance(self.conv2d.padding, tuple) else (self.conv2d.padding, self.conv2d.padding))
        dil_h, dil_w = (self.conv2d.dilation if isinstance(self.conv2d.dilation, tuple) else (self.conv2d.dilation, self.conv2d.dilation))
        bias = self.conv2d.bias if self.conv2d.bias is not None else None
        return custom_ops_lib.conv_standard_2d_square_input_asymmetric_kernel_dilated_padded_custom(
            x, self.conv2d.weight, bias, int(stride_h), int(stride_w), int(pad_h), int(pad_w), int(dil_h), int(dil_w)
        )
