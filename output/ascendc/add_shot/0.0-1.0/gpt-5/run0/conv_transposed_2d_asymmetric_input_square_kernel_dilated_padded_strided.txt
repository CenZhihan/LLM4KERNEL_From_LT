#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_transposed_2d_asymmetric_input_square_kernel_dilated_padded_strided_custom_tiling_data.h"

using i64 = long long;
using u32 = unsigned int;
using f32 = float;

class KernelConvTransposed2d {
public:
    __aicore__ inline KernelConvTransposed2d() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR z,
                                u32 n, u32 cin, u32 hin, u32 win,
                                u32 cout, u32 k, u32 stride, u32 padding, u32 dilation,
                                u32 hout, u32 wout, u32 hasBias) {
        x_ptr = (__gm__ f32*)x;
        w_ptr = (__gm__ f32*)w;
        b_ptr = (__gm__ f32*)b;
        z_ptr = (__gm__ f32*)z;

        this->n = n; this->cin = cin; this->hin = hin; this->win = win;
        this->cout = cout; this->k = k; this->stride = stride; this->padding = padding; this->dilation = dilation;
        this->hout = hout; this->wout = wout; this->hasBias = hasBias;

        // Zero initialize output in block-chunk
        i64 total = (i64)n * (i64)cout * (i64)hout * (i64)wout;
        i64 blockNum = (i64)AscendC::GetBlockNum();
        i64 blockId  = (i64)AscendC::GetBlockIdx();
        i64 chunk = (total + blockNum - 1) / blockNum;
        i64 start = blockId * chunk;
        i64 end   = start + chunk;
        if (end > total) end = total;

        for (i64 idx = start; idx < end; ++idx) {
            z_ptr[idx] = 0.0f;
        }
        AscendC::PipeBarrier();
    }

    __aicore__ inline void Process() {
        // Deconvolution compute per output element
        i64 total = (i64)n * (i64)cout * (i64)hout * (i64)wout;
        i64 blockNum = (i64)AscendC::GetBlockNum();
        i64 blockId  = (i64)AscendC::GetBlockIdx();
        i64 chunk = (total + blockNum - 1) / blockNum;
        i64 start = blockId * chunk;
        i64 end   = start + chunk;
        if (end > total) end = total;

        for (i64 idx = start; idx < end; ++idx) {
            i64 t = idx;
            u32 ow = (u32)(t % wout); t /= wout;
            u32 oh = (u32)(t % hout); t /= hout;
            u32 oc = (u32)(t % cout); t /= cout;
            u32 nn = (u32)(t);

            f32 acc = 0.0f;
            if (hasBias) {
                acc = b_ptr[oc];
            }

            for (u32 ic = 0; ic < cin; ++ic) {
                for (u32 kh = 0; kh < k; ++kh) {
                    for (u32 kw = 0; kw < k; ++kw) {
                        // map output (oh, ow) to input (ih, iw)
                        i64 h_nom = (i64)oh + (i64)padding - (i64)kh * (i64)dilation;
                        i64 w_nom = (i64)ow + (i64)padding - (i64)kw * (i64)dilation;

                        if (h_nom >= 0 && w_nom >= 0) {
                            if ((h_nom % (i64)stride) == 0 && (w_nom % (i64)stride) == 0) {
                                i64 ih = h_nom / (i64)stride;
                                i64 iw = w_nom / (i64)stride;
                                if (ih >= 0 && ih < (i64)hin && iw >= 0 && iw < (i64)win) {
                                    i64 x_index = (((i64)nn * (i64)cin + (i64)ic) * (i64)hin + ih) * (i64)win + iw;
                                    i64 w_index = ((((i64)ic * (i64)cout + (i64)oc) * (i64)k + (i64)kh) * (i64)k + (i64)kw);
                                    acc += x_ptr[x_index] * w_ptr[w_index];
                                }
                            }
                        }
                    }
                }
            }

            z_ptr[idx] = acc;
        }
    }

private:
    __gm__ f32* x_ptr;
    __gm__ f32* w_ptr;
    __gm__ f32* b_ptr;
    __gm__ f32* z_ptr;

    u32 n, cin, hin, win;
    u32 cout, k, stride, padding, dilation;
    u32 hout, wout, hasBias;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_square_kernel_dilated_padded_strided_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {

    GET_TILING_DATA(tiling_data, tiling);

    KernelConvTransposed2d op;
    op.Init(x, w, b, z,
            tiling_data.n, tiling_data.cin, tiling_data.hin, tiling_data.win,
            tiling_data.cout, tiling_data.k, tiling_data.stride, tiling_data.padding, tiling_data.dilation,
            tiling_data.hout, tiling_data.wout, tiling_data.hasBias);
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    """
    Custom NPU implementation of ConvTranspose2d using AscendC kernel:
    conv_transposed_2d_asymmetric_input_square_kernel_dilated_padded_strided_custom
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, dilation: int = 1, bias: bool = False):
        super().__init__()
        assert isinstance(kernel_size, int), "Only square kernel is supported"
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.dilation = dilation
        # Weight layout for ConvTranspose2d in PyTorch: [in_c, out_c, k, k]
        weight = torch.empty(in_channels, out_channels, kernel_size, kernel_size)
        nn.init.kaiming_uniform_(weight, a=5 ** 0.5)
        self.weight = nn.Parameter(weight)
        if bias:
            self.bias = nn.Parameter(torch.zeros(out_channels))
        else:
            self.bias = None

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return custom_ops_lib.conv_transposed_2d_asymmetric_input_square_kernel_dilated_padded_strided_custom(
            x, self.weight, self.bias, self.stride, self.padding, self.dilation
        )
