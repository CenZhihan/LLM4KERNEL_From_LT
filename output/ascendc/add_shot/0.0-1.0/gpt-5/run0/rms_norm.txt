kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "rms_norm_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // tensor buffers per queue

class KernelRmsNorm {
public:
    __aicore__ inline KernelRmsNorm() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength,
                                uint32_t reduceSize,
                                uint32_t outerSize,
                                uint32_t tileOuter,
                                float eps)
    {
        this->reduceSize = reduceSize;
        this->outerSize = outerSize;
        this->tileOuter = tileOuter;
        this->eps = eps;

        // Work partition by outer groups
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t base = outerSize / blockNum;
        uint32_t rem = outerSize % blockNum;
        this->groupStart = blockIdx * base + (blockIdx < rem ? blockIdx : rem);
        this->groupsThisBlock = base + (blockIdx < rem ? 1 : 0);

        // Tile by groups
        uint32_t denom = this->tileOuter * BUFFER_NUM;
        this->tileGroups = (groupsThisBlock + denom - 1) / denom;
        if (this->tileGroups == 0) this->tileGroups = 1;
        this->tileLength = this->tileGroups * this->reduceSize; // elements per tile

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + (uint64_t)this->groupStart * this->reduceSize, this->groupsThisBlock * this->reduceSize);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + (uint64_t)this->groupStart * this->reduceSize, this->groupsThisBlock * this->reduceSize);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        uint32_t totalTiles = AscendC::CeilDiv(this->groupsThisBlock, this->tileGroups);
        for (uint32_t t = 0; t < totalTiles; ++t) {
            CopyIn(t);
            Compute(t);
            CopyOut(t);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t tileIdx)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        uint32_t groupsDone = tileIdx * this->tileGroups;
        uint32_t groupsLeft = (groupsDone < this->groupsThisBlock) ? (this->groupsThisBlock - groupsDone) : 0;
        this->curGroups = AscendC::Min(this->tileGroups, groupsLeft);
        this->curLen = this->curGroups * this->reduceSize;
        AscendC::DataCopy(xLocal, xGm[groupsDone * this->reduceSize], this->curLen);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t tileIdx)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

        // Compute RMS per group (along last dim) and normalize
        uint32_t K = this->reduceSize;
        uint32_t G = this->curGroups;

        for (uint32_t g = 0; g < G; ++g) {
            uint32_t base = g * K;
            float sumsq = 0.f;
            for (uint32_t k = 0; k < K; ++k) {
                float v = static_cast<float>(xLocal.GetValue(base + k));
                sumsq += v * v;
            }
            float mean = sumsq / static_cast<float>(K);
            float denom = AscendC::Sqrt(mean + this->eps);
            float rcp = 1.0f / denom;
            for (uint32_t k = 0; k < K; ++k) {
                float v = static_cast<float>(xLocal.GetValue(base + k));
                float outv = v * rcp;
                yLocal.SetValue(base + k, static_cast<DTYPE_Y>(outv));
            }
        }

        outQueueY.EnQue<DTYPE_Y>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t tileIdx)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        uint32_t groupsDone = tileIdx * this->tileGroups;
        AscendC::DataCopy(yGm[groupsDone * this->reduceSize], yLocal, this->curLen);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t reduceSize;
    uint32_t outerSize;
    uint32_t tileOuter;

    uint32_t groupStart;
    uint32_t groupsThisBlock;

    uint32_t tileGroups;
    uint32_t tileLength;

    uint32_t curGroups;
    uint32_t curLen;

    float eps;
};

extern "C" __global__ __aicore__ void rms_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelRmsNorm op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.reduceSize,
            tiling_data.outerSize,
            tiling_data.tileOuter,
            tiling_data.eps);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model using custom AscendC RMSNorm kernel.
    Note: The custom kernel reduces over the last dimension, so we permute to NHWC,
    apply the kernel, then permute back to NCHW to match original semantics (reduce over dim=1).
    """
    def __init__(self, num_features: int, eps: float = 1e-5):
        super().__init__()
        self.num_features = num_features
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # x is assumed NCHW...; move channel dim to last, compute RMSNorm over last dim, then move back
        # This preserves original behavior of reducing over dim=1.
        x_last = x.permute(0, *range(2, x.dim()), 1).contiguous()
        y_last = custom_ops_lib.rms_norm_custom(x_last, self.eps)
        y = y_last.permute(0, x.dim()-1, *range(1, x.dim()-1)).contiguous()
        return y
