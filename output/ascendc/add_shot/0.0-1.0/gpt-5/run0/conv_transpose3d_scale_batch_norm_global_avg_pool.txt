kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_transpose3d_scale_batch_norm_global_avg_pool_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // double buffer
// Fused per-channel affine transform:
// y = x * alpha[c] + beta2[c], where
// alpha[c] = scale * gamma[c] * rsqrt(var[c] + eps)
// beta2[c] = beta[c] - mean[c] * gamma[c] * rsqrt(var[c] + eps)

class KernelFusedBNScale {
public:
    __aicore__ inline KernelFusedBNScale() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR alpha, GM_ADDR beta2, GM_ADDR y,
                                uint32_t totalLength, uint32_t channelNum, uint32_t stridePerChannel)
    {
        this->totalLength = totalLength;
        this->channelNum = channelNum;
        this->stridePerChannel = stridePerChannel;

        uint32_t ncTotal = totalLength / stridePerChannel; // N * C
        uint32_t blocks = AscendC::GetBlockNum();
        this->ncPerBlock = ncTotal / blocks;
        this->ncStart = this->ncPerBlock * AscendC::GetBlockIdx();

        // Set GM buffers
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + (uint64_t)this->ncStart * stridePerChannel,
                            (uint64_t)this->ncPerBlock * stridePerChannel);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + (uint64_t)this->ncStart * stridePerChannel,
                            (uint64_t)this->ncPerBlock * stridePerChannel);

        alphaGm.SetGlobalBuffer((__gm__ DTYPE_X *)alpha, channelNum);
        beta2Gm.SetGlobalBuffer((__gm__ DTYPE_X *)beta2, channelNum);

        // For simplicity, process one full channel slice per tile
        this->tileLength = stridePerChannel;

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
        // Small scalar buffers
        pipe.InitBuffer(paramQueue, BUFFER_NUM, 2 * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t i = 0; i < this->ncPerBlock; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t ncIdx)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[(uint64_t)ncIdx * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t ncIdx)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        // determine channel index within the global NC index
        uint32_t globalNc = this->ncStart + ncIdx;
        uint32_t cIdx = (globalNc) % this->channelNum;

        // Load alpha[cIdx] and beta2[cIdx] as scalars from GM
        AscendC::LocalTensor<DTYPE_X> paramLocal = paramQueue.AllocTensor<DTYPE_X>();
        // alpha
        AscendC::DataCopy(paramLocal, alphaGm[cIdx], 1);
        DTYPE_X alphaVal = paramLocal.GetValue(0);
        // beta2
        AscendC::DataCopy(paramLocal, beta2Gm[cIdx], 1);
        DTYPE_X beta2Val = paramLocal.GetValue(0);
        paramQueue.FreeTensor(paramLocal);

        // Perform y = x * alpha + beta2 for this whole channel slice
        // Assume scalar broadcasting interfaces exist
        AscendC::Mul(yLocal, xLocal, alphaVal, this->tileLength);
        AscendC::Adds(yLocal, yLocal, beta2Val, this->tileLength);

        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t ncIdx)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[(uint64_t)ncIdx * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> paramQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_X> alphaGm;
    AscendC::GlobalTensor<DTYPE_X> beta2Gm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t totalLength;
    uint32_t channelNum;
    uint32_t stridePerChannel;
    uint32_t ncPerBlock;
    uint32_t ncStart;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transpose3d_scale_batch_norm_global_avg_pool_custom(
    GM_ADDR x, GM_ADDR alpha, GM_ADDR beta2, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelFusedBNScale op;
    op.Init(x, alpha, beta2, y,
            tiling_data.totalLength,
            tiling_data.channelNum,
            tiling_data.stridePerChannel);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model:
    - ConvTranspose3d executed by PyTorch
    - Fused scale + BatchNorm (inference form) executed by custom AscendC op
    - Global Average Pooling executed by PyTorch
    """
    def __init__(self, in_channels, out_channels, kernel_size, scale_factor, eps=1e-5, momentum=0.1):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size)
        self.scale_factor = scale_factor
        self.batch_norm = torch.nn.BatchNorm3d(out_channels, eps=eps, momentum=momentum)
        self.global_avg_pool = torch.nn.AdaptiveAvgPool3d((1, 1, 1))

    def forward(self, x):
        x = self.conv_transpose(x)

        # Precompute fused affine coefficients per channel for inference-style BN
        # alpha[c] = scale * gamma[c] * rsqrt(var[c] + eps)
        # beta2[c] = beta[c] - mean[c] * gamma[c] * rsqrt(var[c] + eps)
        bn = self.batch_norm
        # Ensure parameters are on the same device/dtype
        device = x.device
        dtype = x.dtype
        eps_tensor = torch.tensor([bn.eps], device=device, dtype=dtype)
        inv_std = torch.rsqrt(bn.running_var.to(device=device, dtype=dtype) + eps_tensor)
        alpha = (torch.tensor([self.scale_factor], device=device, dtype=dtype) *
                 bn.weight.to(device=device, dtype=dtype) * inv_std)
        beta2 = bn.bias.to(device=device, dtype=dtype) - bn.running_mean.to(device=device, dtype=dtype) * bn.weight.to(device=device, dtype=dtype) * inv_std

        x = custom_ops_lib.conv_transpose3d_scale_batch_norm_global_avg_pool_custom(x, alpha, beta2)
        x = self.global_avg_pool(x)
        return x
'''
