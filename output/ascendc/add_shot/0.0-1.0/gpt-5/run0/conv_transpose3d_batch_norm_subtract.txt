kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_transpose3d_batch_norm_subtract_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSubtractSpatialMean {
public:
    __aicore__ inline KernelSubtractSpatialMean() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w,
                                uint32_t tileNum)
    {
        this->n = n;
        this->c = c;
        this->d = d;
        this->h = h;
        this->w = w;
        this->channels = n * c;
        this->spatial = d * h * w;
        this->tileNum = tileNum;
        this->tileLength = AscendC::max((uint32_t)1, (uint32_t)(this->spatial / (tileNum * BUFFER_NUM)));
        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, this->channels * this->spatial);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y, this->channels * this->spatial);
        pipe.InitBuffer(inQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        // a tiny buffer to hold scalar reductions
        pipe.InitBuffer(tmpQueue, 1, sizeof(DTYPE_X)); // one scalar buffer
    }

    __aicore__ inline void Process()
    {
        uint32_t grid = AscendC::GetBlockNum();
        uint32_t bid = AscendC::GetBlockIdx();

        for (uint32_t ch = bid; ch < this->channels; ch += grid) {
            // First pass: compute sum over spatial dims for channel ch
            AscendC::LocalTensor<DTYPE_X> sumLocal = tmpQueue.AllocTensor<DTYPE_X>();
            // initialize sumLocal to 0
            AscendC::SetValue(sumLocal, (DTYPE_X)0, 1);

            uint32_t processed = 0;
            uint32_t tilesTotal = (this->spatial + this->tileLength - 1) / this->tileLength;
            for (uint32_t t = 0; t < tilesTotal; ++t) {
                uint32_t len = AscendC::min(this->tileLength, this->spatial - processed);
                AscendC::LocalTensor<DTYPE_X> xLocal = inQueue.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(xLocal, xGm[ch * this->spatial + processed], len);
                inQueue.EnQue(xLocal);

                AscendC::LocalTensor<DTYPE_X> xTile = inQueue.DeQue<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_X> tileSum = tmpQueue.AllocTensor<DTYPE_X>();
                // reduce tile to scalar
                AscendC::ReduceSum(tileSum, xTile, len);
                // accumulate into sumLocal
                AscendC::Add(sumLocal, sumLocal, tileSum, 1);

                tmpQueue.FreeTensor(tileSum);
                inQueue.FreeTensor(xTile);
                processed += len;
            }

            // compute mean = sum / spatial
            DTYPE_X invSpatial = (DTYPE_X)(1.0f / (float)this->spatial);
            AscendC::Muls(sumLocal, sumLocal, invSpatial, 1); // now sumLocal holds mean

            // Second pass: subtract mean from each element and write out
            processed = 0;
            for (uint32_t t = 0; t < tilesTotal; ++t) {
                uint32_t len = AscendC::min(this->tileLength, this->spatial - processed);

                AscendC::LocalTensor<DTYPE_X> xLocal = inQueue.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(xLocal, xGm[ch * this->spatial + processed], len);
                inQueue.EnQue(xLocal);

                AscendC::LocalTensor<DTYPE_X> xTile = inQueue.DeQue<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_Y> yTile = outQueue.AllocTensor<DTYPE_Y>();

                // broadcast mean to a temporary buffer
                AscendC::LocalTensor<DTYPE_X> meanVec = outQueue.AllocTensor<DTYPE_X>();
                AscendC::Duplicate(meanVec, sumLocal.GetValue(0), len);

                AscendC::Sub(yTile, xTile, meanVec, len);

                outQueue.EnQue<DTYPE_Y>(yTile);

                // free temps
                outQueue.FreeTensor(meanVec);
                inQueue.FreeTensor(xTile);

                AscendC::LocalTensor<DTYPE_Y> yOut = outQueue.DeQue<DTYPE_Y>();
                AscendC::DataCopy(yGm[ch * this->spatial + processed], yOut, len);
                outQueue.FreeTensor(yOut);

                processed += len;
            }

            tmpQueue.FreeTensor(sumLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> tmpQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n, c, d, h, w;
    uint32_t channels;
    uint32_t spatial;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transpose3d_batch_norm_subtract_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSubtractSpatialMean op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c, tiling_data.d, tiling_data.h, tiling_data.w,
            tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    A 3D convolutional transpose layer followed by Batch Normalization,
    then a custom fused AscendC operator that subtracts the per-(N,C) spatial mean.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, bias=True):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, bias=bias)
        self.batch_norm = torch.nn.BatchNorm3d(out_channels)

    def forward(self, x):
        x = self.conv_transpose(x)
        x = self.batch_norm(x)
        # Replace: x = x - torch.mean(x, dim=(2,3,4), keepdim=True)
        x = custom_ops_lib.conv_transpose3d_batch_norm_subtract_custom(x)
        return x
