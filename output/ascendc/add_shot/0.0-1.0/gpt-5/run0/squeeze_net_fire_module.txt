    // For simplicity, set output shape same as first input.

    const gert::Shape* a_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *a_shape;
    return GRAPH_SUCCESS;

{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "squeeze_net_fire_module_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConcatC {
public:
    __aicore__ inline KernelConcatC() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR y,
                                uint32_t totalLenA, uint32_t totalLenB, uint32_t tileNum)
    {
        this->tileNum = tileNum;

        // Split work across blocks
        this->blockLenA = totalLenA / AscendC::GetBlockNum();
        this->blockLenB = totalLenB / AscendC::GetBlockNum();
        this->blockOffsetA = this->blockLenA * AscendC::GetBlockIdx();
        this->blockOffsetB = this->blockLenB * AscendC::GetBlockIdx();

        // tile sizes for A and B
        this->tileLenA = AscendC::Max((uint32_t)1, this->blockLenA / (this->tileNum * BUFFER_NUM));
        this->tileLenB = AscendC::Max((uint32_t)1, this->blockLenB / (this->tileNum * BUFFER_NUM));

        aGm.SetGlobalBuffer((__gm__ DTYPE_X *)a + this->blockOffsetA, this->blockLenA);
        bGm.SetGlobalBuffer((__gm__ DTYPE_Y *)b + this->blockOffsetB, this->blockLenB);

        // y block covers A part followed by B part
        uint32_t yBlockLen = this->blockLenA + this->blockLenB;
        uint32_t yBlockOffset = yBlockLen * AscendC::GetBlockIdx();
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + yBlockOffset, yBlockLen);

        pipe.InitBuffer(inQueueA, BUFFER_NUM, this->tileLenA * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->tileLenB * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        ProcessA();
        ProcessB();
    }

private:
    __aicore__ inline void ProcessA()
    {
        if (this->blockLenA == 0) { return; }
        uint32_t loopCount = (this->blockLenA + this->tileLenA - 1) / this->tileLenA;
        for (uint32_t i = 0; i < loopCount; i++) {
            uint32_t srcOffset = i * this->tileLenA;
            uint32_t valid = AscendC::Min(this->tileLenA, this->blockLenA - srcOffset);

            AscendC::LocalTensor<DTYPE_X> aLocal = inQueueA.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(aLocal, aGm[srcOffset], valid);
            inQueueA.EnQue(aLocal);

            AscendC::LocalTensor<DTYPE_X> aLocal2 = inQueueA.DeQue<DTYPE_X>();
            AscendC::DataCopy(yGm[srcOffset], aLocal2, valid);
            inQueueA.FreeTensor(aLocal2);
        }
    }

    __aicore__ inline void ProcessB()
    {
        if (this->blockLenB == 0) { return; }
        uint32_t loopCount = (this->blockLenB + this->tileLenB - 1) / this->tileLenB;
        for (uint32_t i = 0; i < loopCount; i++) {
            uint32_t srcOffset = i * this->tileLenB;
            uint32_t dstOffset = this->blockLenA + srcOffset; // place after A
            uint32_t valid = AscendC::Min(this->tileLenB, this->blockLenB - srcOffset);

            AscendC::LocalTensor<DTYPE_Y> bLocal = inQueueB.AllocTensor<DTYPE_Y>();
            AscendC::DataCopy(bLocal, bGm[srcOffset], valid);
            inQueueB.EnQue(bLocal);

            AscendC::LocalTensor<DTYPE_Y> bLocal2 = inQueueB.DeQue<DTYPE_Y>();
            AscendC::DataCopy(yGm[dstOffset], bLocal2, valid);
            inQueueB.FreeTensor(bLocal2);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA, inQueueB;

    AscendC::GlobalTensor<DTYPE_X> aGm;
    AscendC::GlobalTensor<DTYPE_Y> bGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t tileNum;
    uint32_t blockLenA;
    uint32_t blockLenB;
    uint32_t blockOffsetA;
    uint32_t blockOffsetB;
    uint32_t tileLenA;
    uint32_t tileLenB;
};

extern "C" __global__ __aicore__ void squeeze_net_fire_module_custom(GM_ADDR a, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConcatC op;
    op.Init(a, b, y, tiling_data.totalLenA, tiling_data.totalLenB, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, squeeze_channels, expand1x1_channels, expand3x3_channels):
        super(ModelNew, self).__init__()
        self.squeeze = torch.nn.Conv2d(in_channels, squeeze_channels, kernel_size=1)
        self.squeeze_activation = torch.nn.ReLU(inplace=True)

        self.expand1x1 = torch.nn.Conv2d(squeeze_channels, expand1x1_channels, kernel_size=1)
        self.expand1x1_activation = torch.nn.ReLU(inplace=True)

        self.expand3x3 = torch.nn.Conv2d(squeeze_channels, expand3x3_channels, kernel_size=3, padding=1)
        self.expand3x3_activation = torch.nn.ReLU(inplace=True)

    def forward(self, x):
        x = self.squeeze_activation(self.squeeze(x))
        out1 = self.expand1x1_activation(self.expand1x1(x))
        out3 = self.expand3x3_activation(self.expand3x3(x))
        return custom_ops_lib.squeeze_net_fire_module_custom(out1, out3)
