host_operator_src="""
#include "shufflenet_unit_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    ShufflenetUnitCustomTilingData tiling;

    auto* shape = context->GetInputShape(0);
    uint32_t totalLength = shape->GetOriginShape().GetShapeSize();
    const auto& dims = shape->GetOriginShape().GetDims();

    uint32_t n = 1, c = 1, h = 1, w = 1;
    if (dims.size() == 4) {
        n = static_cast<uint32_t>(dims[0]);
        c = static_cast<uint32_t>(dims[1]);
        h = static_cast<uint32_t>(dims[2]);
        w = static_cast<uint32_t>(dims[3]);
    }

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_totalLength(totalLength);
    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

namespace ops {
class ShufflenetUnitCustom : public OpDef {
public:
    explicit ShufflenetUnitCustom(const char* name) : OpDef(name)

        this->Input("x")

            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "shufflenet_unit_custom_tiling_data.h"

constexpr int32_t GROUPS = 3; // Fixed groups consistent with model default
template<typename T>
__aicore__ inline T my_min(T a, T b) { return a < b ? a : b; }

extern "C" __global__ __aicore__ void shufflenet_unit_custom(GM_ADDR x, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N = tiling_data.n;
    const uint32_t C = tiling_data.c;
    const uint32_t H = tiling_data.h;
    const uint32_t W = tiling_data.w;

    if (C == 0 || H == 0 || W == 0 || N == 0) {
        return;
    }

    const uint32_t blocks = AscendC::GetBlockNum();
    const uint32_t bid = AscendC::GetBlockIdx();

    // Split work by batch dimension
    const uint32_t nPerBlock = (N + blocks - 1) / blocks;
    const uint32_t nStart = bid * nPerBlock;
    const uint32_t nEnd = my_min<uint32_t>(N, nStart + nPerBlock);

    const uint64_t HW = static_cast<uint64_t>(H) * static_cast<uint64_t>(W);
    const uint64_t CHW = static_cast<uint64_t>(C) * HW;

    const uint32_t channelsPerGroup = C / GROUPS;

    __gm__ DTYPE_X* x_ptr = (__gm__ DTYPE_X*)x;
    __gm__ DTYPE_Z* z_ptr = (__gm__ DTYPE_Z*)z;

    for (uint32_t n = nStart; n < nEnd; ++n) {
        const uint64_t nOffset = static_cast<uint64_t>(n) * CHW;

        for (uint32_t c = 0; c < C; ++c) {
            const uint32_t cOut = c;
            const uint32_t cIn = (cOut % GROUPS) * channelsPerGroup + (cOut / GROUPS);

            const uint64_t outBase = nOffset + static_cast<uint64_t>(cOut) * HW;
            const uint64_t inBase  = nOffset + static_cast<uint64_t>(cIn) * HW;

            // Copy spatial plane H*W
            for (uint64_t s = 0; s < HW; ++s) {
                z_ptr[outBase + s] = x_ptr[inBase + s];
            }
        }
    }
}
"""

model_src='''
import torch
import torch_npu
import custom_ops_lib

import torch.nn as nn
import torch.nn.functional as F

class ModelNew(nn.Module):
    def __init__(self, in_channels, out_channels, groups=3):
        super(ModelNew, self).__init__()
        assert out_channels % 4 == 0
        mid_channels = out_channels // 4

        self.conv1 = nn.Conv2d(in_channels, mid_channels, kernel_size=1, stride=1, padding=0, groups=groups, bias=False)
        self.bn1 = nn.BatchNorm2d(mid_channels)

        self.conv2 = nn.Conv2d(mid_channels, mid_channels, kernel_size=3, stride=1, padding=1, groups=mid_channels, bias=False)
        self.bn2 = nn.BatchNorm2d(mid_channels)

        self.conv3 = nn.Conv2d(mid_channels, out_channels, kernel_size=1, stride=1, padding=0, groups=groups, bias=False)
        self.bn3 = nn.BatchNorm2d(out_channels)

        # Shortcut
        if in_channels == out_channels:
            self.shortcut = nn.Sequential()
        else:
            self.shortcut = nn.Sequential(
                nn.Conv2d(in_channels, out_channels, kernel_size=1, stride=1, padding=0, bias=False),
                nn.BatchNorm2d(out_channels)
            )

    def forward(self, x):
        out = F.relu(self.bn1(self.conv1(x)))
        out = self.bn2(self.conv2(out))
        # Replace ChannelShuffle with custom AscendC kernel
        out = custom_ops_lib.shufflenet_unit_custom(out)
        out = F.relu(self.bn3(self.conv3(out)))
        out = out + self.shortcut(x)
        return out
'''
