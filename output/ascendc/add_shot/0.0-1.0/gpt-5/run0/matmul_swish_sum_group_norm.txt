kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "matmul_swish_sum_group_norm_custom_tiling_data.h"

constexpr float EPS = 1e-5f;

class KernelMatmulSwishSumGroupNorm {
public:
    __aicore__ inline KernelMatmulSwishSumGroupNorm() {}

    __aicore__ inline void Init(GM_ADDR x,
                                GM_ADDR bias,
                                GM_ADDR gn_weight,
                                GM_ADDR gn_bias,
                                GM_ADDR y,
                                uint32_t batch,
                                uint32_t channels,
                                uint32_t groups)
    {
        this->batch = batch;
        this->channels = channels;
        this->groups = groups;
        uint32_t blkNum = AscendC::GetBlockNum();

        // Split batch across blocks
        this->rowsPerBlock = (batch + blkNum - 1) / blkNum;
        this->rowStart = AscendC::GetBlockIdx() * this->rowsPerBlock;
        this->rowEnd = AscendC::Min(this->rowStart + this->rowsPerBlock, this->batch);

        xGm.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(batch * channels));
        biasGm.SetGlobalBuffer((__gm__ float*)bias, channels);
        gammaGm.SetGlobalBuffer((__gm__ float*)gn_weight, channels);
        betaGm.SetGlobalBuffer((__gm__ float*)gn_bias, channels);
        yGm.SetGlobalBuffer((__gm__ float*)y, static_cast<uint32_t>(batch * channels));
    }

    __aicore__ inline void Process()
    {
        if (rowStart >= rowEnd) {
            return;
        }

        // Allocate UB buffers for a single row (C)
        AscendC::LocalTensor<float> xLocal = AscendC::LocalTensor<float>(channels);
        AscendC::LocalTensor<float> yLocal = AscendC::LocalTensor<float>(channels);
        AscendC::LocalTensor<float> biasLocal = AscendC::LocalTensor<float>(channels);
        AscendC::LocalTensor<float> gammaLocal = AscendC::LocalTensor<float>(channels);
        AscendC::LocalTensor<float> betaLocal = AscendC::LocalTensor<float>(channels);

        AscendC::DataCopy(biasLocal, biasGm, channels);
        AscendC::DataCopy(gammaLocal, gammaGm, channels);
        AscendC::DataCopy(betaLocal, betaGm, channels);

        uint32_t groupSize = channels / groups;

        for (uint32_t b = rowStart; b < rowEnd; ++b) {
            // Load row
            AscendC::DataCopy(xLocal, xGm[b * channels], channels);

            // Swish: y = x * sigmoid(x)
            AscendC::Sigmoid(yLocal, xLocal, channels);
            AscendC::Mul(yLocal, yLocal, xLocal, channels);

            // Add bias (broadcast over batch)
            AscendC::Add(yLocal, yLocal, biasLocal, channels);

            // GroupNorm per sample
            for (uint32_t g = 0; g < groups; ++g) {
                uint32_t offset = g * groupSize;

                // Compute mean
                AscendC::LocalTensor<float> tmp = AscendC::LocalTensor<float>(groupSize);
                AscendC::DataCopy(tmp, yLocal[offset], groupSize);
                float mean = AscendC::ReduceSum(tmp, groupSize) / static_cast<float>(groupSize);

                // Compute variance
                AscendC::Sub(tmp, tmp, mean, groupSize);
                AscendC::Mul(tmp, tmp, tmp, groupSize);
                float var = AscendC::ReduceSum(tmp, groupSize) / static_cast<float>(groupSize);
                float invStd = AscendC::InvSqrt(var + EPS);

                // Normalize and apply affine
                AscendC::LocalTensor<float> yg = AscendC::LocalTensor<float>(groupSize);
                AscendC::DataCopy(yg, yLocal[offset], groupSize);

                // (yg - mean) * invStd
                AscendC::Sub(yg, yg, mean, groupSize);
                AscendC::Muls(yg, yg, invStd, groupSize);

                // Scale and shift with per-channel gamma/beta
                AscendC::LocalTensor<float> gammaG = AscendC::LocalTensor<float>(groupSize);
                AscendC::LocalTensor<float> betaG = AscendC::LocalTensor<float>(groupSize);
                AscendC::DataCopy(gammaG, gammaLocal[offset], groupSize);
                AscendC::DataCopy(betaG, betaLocal[offset], groupSize);

                AscendC::Mul(yg, yg, gammaG, groupSize);
                AscendC::Add(yg, yg, betaG, groupSize);

                // Store back
                AscendC::DataCopy(yLocal[offset], yg, groupSize);
            }

            // Write out
            AscendC::DataCopy(yGm[b * channels], yLocal, channels);
        }
    }

private:
    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> biasGm;
    AscendC::GlobalTensor<float> gammaGm;
    AscendC::GlobalTensor<float> betaGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t batch;
    uint32_t channels;
    uint32_t groups;

    uint32_t rowsPerBlock;
    uint32_t rowStart;
    uint32_t rowEnd;
};

extern "C" __global__ __aicore__ void matmul_swish_sum_group_norm_custom(GM_ADDR x,
                                                                         GM_ADDR bias,
                                                                         GM_ADDR gn_weight,
                                                                         GM_ADDR gn_bias,
                                                                         GM_ADDR y,
                                                                         GM_ADDR workspace,
                                                                         GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulSwishSumGroupNorm op;
    op.Init(x, bias, gn_weight, gn_bias, y, tiling_data.batch, tiling_data.channels, tiling_data.groups);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model that uses a fused custom kernel for Swish + bias add + GroupNorm
    after the matmul (Linear) layer.
    """
    def __init__(self, in_features, out_features, num_groups, bias_shape):
        super(ModelNew, self).__init__()
        self.matmul = torch.nn.Linear(in_features, out_features)
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))
        self.group_norm = torch.nn.GroupNorm(num_groups, out_features)

    def forward(self, x):
        # Keep matmul with highly-optimized library, then fuse the rest in a custom kernel
        x = self.matmul(x)  # shape: (B, C)
        x = custom_ops_lib.matmul_swish_sum_group_norm_custom(
            x,
            self.bias,
            self.group_norm.weight,
            self.group_norm.bias,
            self.group_norm.num_groups
        )
        return x
