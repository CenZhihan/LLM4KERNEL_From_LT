#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_standard_3d_square_input_square_kernel_custom_tiling_data.h"

extern "C" __global__ __aicore__ void conv_standard_3d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N    = tiling_data.n;
    const uint32_t Cin  = tiling_data.cin;
    const uint32_t Cout = tiling_data.cout;
    const uint32_t D    = tiling_data.d;
    const uint32_t H    = tiling_data.h;
    const uint32_t W    = tiling_data.w;
    const uint32_t K    = tiling_data.k;
    const uint32_t Dout = tiling_data.dout;
    const uint32_t Hout = tiling_data.hout;
    const uint32_t Wout = tiling_data.wout;

    const uint64_t total = static_cast<uint64_t>(N) * Cout * Dout * Hout * Wout;

    const float* __restrict__ x_ptr = reinterpret_cast<const float*>(x);
    const float* __restrict__ w_ptr = reinterpret_cast<const float*>(w);
    float* __restrict__ z_ptr = reinterpret_cast<float*>(z);

    uint32_t block_num = AscendC::GetBlockNum();
    uint32_t block_idx = AscendC::GetBlockIdx();

    uint64_t elems_per_block = (total + block_num - 1) / block_num;
    uint64_t start = elems_per_block * block_idx;
    uint64_t end = start + elems_per_block;
    if (end > total) end = total;

    // Strides for input and output (N, C, D, H, W) layout
    const uint64_t x_stride_c = static_cast<uint64_t>(D) * H * W;
    const uint64_t x_stride_n = static_cast<uint64_t>(Cin) * x_stride_c;
    const uint64_t x_stride_d = static_cast<uint64_t>(H) * W;
    const uint64_t x_stride_h = static_cast<uint64_t>(W);

    const uint64_t z_stride_c = static_cast<uint64_t>(Dout) * Hout * Wout;
    const uint64_t z_stride_n = static_cast<uint64_t>(Cout) * z_stride_c;
    const uint64_t z_stride_d = static_cast<uint64_t>(Hout) * Wout;
    const uint64_t z_stride_h = static_cast<uint64_t>(Wout);

    const uint64_t w_stride_cin = static_cast<uint64_t>(K) * K * K;
    const uint64_t w_stride_cout = static_cast<uint64_t>(Cin) * w_stride_cin;
    const uint64_t w_stride_kd = static_cast<uint64_t>(K) * K;
    const uint64_t w_stride_kh = static_cast<uint64_t>(K);

    for (uint64_t p = start; p < end; ++p) {
        uint64_t t = p;
        uint32_t ow = static_cast<uint32_t>(t % Wout); t /= Wout;
        uint32_t oh = static_cast<uint32_t>(t % Hout); t /= Hout;
        uint32_t od = static_cast<uint32_t>(t % Dout); t /= Dout;
        uint32_t oc = static_cast<uint32_t>(t % Cout); t /= Cout;
        uint32_t on = static_cast<uint32_t>(t);

        float acc = 0.0f;

        for (uint32_t ic = 0; ic < Cin; ++ic) {
            for (uint32_t kd = 0; kd < K; ++kd) {
                for (uint32_t kh = 0; kh < K; ++kh) {
                    for (uint32_t kw = 0; kw < K; ++kw) {
                        uint64_t id = static_cast<uint64_t>(od + kd);
                        uint64_t ih = static_cast<uint64_t>(oh + kh);
                        uint64_t iw = static_cast<uint64_t>(ow + kw);

                        uint64_t x_index =
                            static_cast<uint64_t>(on) * x_stride_n +
                            static_cast<uint64_t>(ic) * x_stride_c +
                            id * x_stride_d +
                            ih * x_stride_h +
                            iw;

                        uint64_t w_index =
                            static_cast<uint64_t>(oc) * w_stride_cout +
                            static_cast<uint64_t>(ic) * w_stride_cin +
                            static_cast<uint64_t>(kd) * w_stride_kd +
                            static_cast<uint64_t>(kh) * w_stride_kh +
                            static_cast<uint64_t>(kw);

                        acc += x_ptr[x_index] * w_ptr[w_index];
                    }
                }
            }
        }

        uint64_t z_index =
            static_cast<uint64_t>(on) * z_stride_n +
            static_cast<uint64_t>(oc) * z_stride_c +
            static_cast<uint64_t>(od) * z_stride_d +
            static_cast<uint64_t>(oh) * z_stride_h +
            static_cast<uint64_t>(ow);

        z_ptr[z_index] = acc;
    }
}
