kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_transpose3d_log_sum_exp_hard_swish_subtract_clamp_max_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

// Elementwise kernel: z = clamp(hardswish(x) - bias, -1, 1)
// Assumes 'bias' tensor has same flattened length as 'x' (broadcast should be handled on host/python side).
class KernelFused {
public:
    __aicore__ inline KernelFused() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR z, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        bGm.SetGlobalBuffer((__gm__ DTYPE_Y *)bias + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> bLocal = inQueueB.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(bLocal, bGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
        inQueueB.EnQue(bLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> bLocal = inQueueB.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();

        // hardswish: x * sigmoid(x + 3) / 6
        // Use built-ins for convenience; implementations may map to hardware vector ops.
        // t = x + 3
        AscendC::LocalTensor<DTYPE_Z> tLocal = outQueueZ.AllocTensor<DTYPE_Z>();
        AscendC::Add(tLocal, xLocal, (DTYPE_Z)3.0f, this->tileLength);
        // sigmoid(t) = 1 / (1 + exp(-t))
        AscendC::Neg(tLocal, tLocal, this->tileLength);
        AscendC::Exp(tLocal, tLocal, this->tileLength);
        AscendC::Add(tLocal, tLocal, (DTYPE_Z)1.0f, this->tileLength);
        AscendC::Reciprocal(tLocal, tLocal, this->tileLength);
        // z = x * sigmoid(t) / 6
        AscendC::Mul(zLocal, xLocal, tLocal, this->tileLength);
        AscendC::Div(zLocal, zLocal, (DTYPE_Z)6.0f, this->tileLength);
        // z = z - bias
        AscendC::Sub(zLocal, zLocal, bLocal, this->tileLength);
        // clamp to [-1, 1]
        AscendC::Maxs(zLocal, zLocal, (DTYPE_Z)(-1.0f), this->tileLength);
        AscendC::Mins(zLocal, zLocal, (DTYPE_Z)(1.0f), this->tileLength);

        outQueueZ.EnQue<DTYPE_Z>(zLocal);
        outQueueZ.FreeTensor(tLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueB.FreeTensor(bLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[progress * this->tileLength], zLocal, this->tileLength);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> bGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transpose3d_log_sum_exp_hard_swish_subtract_clamp_max_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelFused op;
    op.Init(x, bias, z, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model using custom AscendC kernel to fuse HardSwish, subtract and clamp after LogSumExp.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, bias_shape):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding)
        self.bias = torch.nn.Parameter(torch.randn(*bias_shape))

    def forward(self, x):
        x = self.conv_transpose(x)
        x = torch.logsumexp(x, dim=1, keepdim=True)
        # Expand bias to match x and ensure contiguous memory for the custom kernel
        b = self.bias.expand_as(x).contiguous()
        x = custom_ops_lib.conv_transpose3d_log_sum_exp_hard_swish_subtract_clamp_max_custom(x, b)
        return x
