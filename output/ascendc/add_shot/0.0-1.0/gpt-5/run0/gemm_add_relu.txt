kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "gemm_add_relu_custom_tiling_data.h"

class KernelGemmAddRelu {
public:
    __aicore__ inline KernelGemmAddRelu() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t m, uint32_t n, uint32_t k,
                                uint32_t rowsPerBlock, uint32_t tileK, uint32_t tileN)
    {
        this->m = m;
        this->n = n;
        this->k = k;
        this->rowsPerBlock = rowsPerBlock;
        this->tileK = tileK;
        this->tileN = tileN;

        uint32_t startRow = rowsPerBlock * AscendC::GetBlockIdx();
        startRow = startRow > m ? m : startRow;
        uint32_t endRow = startRow + rowsPerBlock;
        endRow = endRow > m ? m : endRow;

        this->rowStart = startRow;
        this->rowEnd = endRow;

        xGm = reinterpret_cast<__gm__ DTYPE_X *>(x);
        wGm = reinterpret_cast<__gm__ DTYPE_W *>(w);
        bGm = reinterpret_cast<__gm__ DTYPE_B *>(b);
        yGm = reinterpret_cast<__gm__ DTYPE_Y *>(y);
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = rowStart; r < rowEnd; ++r) {
            uint32_t xRowBase = r * k;
            uint32_t yRowBase = r * n;

            for (uint32_t n0 = 0; n0 < n; n0 += tileN) {
                uint32_t curN = (n0 + tileN <= n) ? tileN : (n - n0);

                // local accumulators
                // initialize to 0
                for (uint32_t j = 0; j < curN; ++j) {
                    acc[j] = static_cast<DTYPE_Y>(0);
                }

                // GEMM accumulation: y[r, n0:n0+curN] = sum_k x[r,k] * w[n0:n0+curN, k]
                for (uint32_t k0 = 0; k0 < k; k0 += tileK) {
                    uint32_t curK = (k0 + tileK <= k) ? tileK : (k - k0);

                    // multiply-accumulate
                    for (uint32_t t = 0; t < curK; ++t) {
                        DTYPE_X xval = xGm[xRowBase + (k0 + t)];
                        // for each output channel in this tile
                        for (uint32_t j = 0; j < curN; ++j) {
                            // w is [n,k] row-major: row (n0+j), col (k0+t)
                            DTYPE_W wval = wGm[(n0 + j) * k + (k0 + t)];
                            acc[j] += static_cast<DTYPE_Y>(xval * wval);
                        }
                    }
                }

                // add bias and apply ReLU
                for (uint32_t j = 0; j < curN; ++j) {
                    DTYPE_Y v = acc[j] + static_cast<DTYPE_Y>(bGm[n0 + j]);
                    // ReLU
                    v = v > static_cast<DTYPE_Y>(0) ? v : static_cast<DTYPE_Y>(0);
                    yGm[yRowBase + n0 + j] = v;
                }
            }
        }
    }

private:
    // GM pointers
    __gm__ DTYPE_X* xGm;
    __gm__ DTYPE_W* wGm;
    __gm__ DTYPE_B* bGm;
    __gm__ DTYPE_Y* yGm;

    // dims
    uint32_t m;
    uint32_t n;
    uint32_t k;
    uint32_t rowsPerBlock;
    uint32_t tileK;
    uint32_t tileN;

    uint32_t rowStart;
    uint32_t rowEnd;

    // local accumulator buffer sized by maximum tileN
    // Assuming tileN won't exceed 1024 here; can be tuned by tiling config
    DTYPE_Y acc[1024];
};

extern "C" __global__ __aicore__ void gemm_add_relu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmAddRelu op;
    op.Init(x, w, b, y,
            tiling_data.m, tiling_data.n, tiling_data.k,
            tiling_data.rowsPerBlock, tiling_data.tileK, tiling_data.tileN);
    op.Process();
}

at::Tensor gemm_add_relu_custom_impl_npu(const at::Tensor& x,
                                         const at::Tensor& w,

    EXEC_NPU_CMD(aclnnGemmAddReluCustom, x, w, b, y);
    return y;

    m.impl("gemm_add_relu_custom", &gemm_add_relu_custom_impl_npu);

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused Gemm + Bias + ReLU using custom AscendC kernel.
    """
    def __init__(self, in_features, out_features, bias_shape):
        super(ModelNew, self).__init__()
        self.gemm = torch.nn.Linear(in_features, out_features, bias=False)
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))

    def forward(self, x):
        # Fused call: y = relu(x @ W^T + b)
        return custom_ops_lib.gemm_add_relu_custom(x, self.gemm.weight, self.bias)
