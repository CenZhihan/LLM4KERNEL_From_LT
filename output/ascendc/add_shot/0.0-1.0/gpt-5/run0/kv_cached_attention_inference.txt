kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "kv_cached_attention_inference_custom_tiling_data.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 1;

class KernelKvCachedAttentionInference {
public:
    __aicore__ inline KernelKvCachedAttentionInference() {}
    __aicore__ inline void Init(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR o,
                                uint32_t batch, uint32_t qLen, uint32_t kvLen, uint32_t dModel)
    {
        this->B = batch;
        this->QL = qLen;
        this->KL = kvLen;
        this->D = dModel;

        // Distribute batches across blocks
        this->blocks = AscendC::GetBlockNum();
        this->blockIdx = AscendC::GetBlockIdx();

        // Offsets are computed per-batch
        qGm.SetGlobalBuffer((__gm__ DTYPE_Q *)q, (uint32_t)(B * QL * D));
        kGm.SetGlobalBuffer((__gm__ DTYPE_K *)k, (uint32_t)(B * KL * D));
        vGm.SetGlobalBuffer((__gm__ DTYPE_V *)v, (uint32_t)(B * KL * D));
        oGm.SetGlobalBuffer((__gm__ DTYPE_O *)o, (uint32_t)(B * QL * D));

        // Local buffers: q row, k row, v row, out row
        pipe.InitBuffer(qQueue, BUFFER_NUM, D * sizeof(DTYPE_Q));
        pipe.InitBuffer(kRowQueue, BUFFER_NUM, D * sizeof(DTYPE_K));
        pipe.InitBuffer(vRowQueue, BUFFER_NUM, D * sizeof(DTYPE_V));
        pipe.InitBuffer(oQueue, BUFFER_NUM, D * sizeof(DTYPE_O));
    }

    __aicore__ inline void Process()
    {
        // Each block handles one batch if possible
        if (blockIdx >= B) {
            return;
        }

        // For simplicity, handle q_len == 1 typical autoregressive inference
        // Load Q[b, 0, :]
        uint64_t qBase = (uint64_t)blockIdx * QL * D;
        AscendC::LocalTensor<DTYPE_Q> qLocal = qQueue.AllocTensor<DTYPE_Q>();
        AscendC::DataCopy(qLocal, qGm[qBase], D);

        // Prepare output local vector
        AscendC::LocalTensor<DTYPE_O> outLocal = oQueue.AllocTensor<DTYPE_O>();
        // zero init
        for (uint32_t i = 0; i < D; ++i) {
            outLocal.GetValue(i) = static_cast<DTYPE_O>(0);
        }

        // Scaling factor 1/sqrt(D)
        float scale = 1.0f / sqrtf(static_cast<float>(D));

        // First pass: find max score for numerical stability
        float maxScore = -1e30f;
        for (uint32_t kk = 0; kk < KL; ++kk) {
            // K row index
            uint64_t kOff = (uint64_t)blockIdx * KL * D + (uint64_t)kk * D;
            AscendC::LocalTensor<DTYPE_K> kLocal = kRowQueue.AllocTensor<DTYPE_K>();
            AscendC::DataCopy(kLocal, kGm[kOff], D);
            float dot = 0.f;
            // Dot product qLocal · kLocal
            for (uint32_t d = 0; d < D; ++d) {
                dot += static_cast<float>(qLocal.GetValue(d)) * static_cast<float>(kLocal.GetValue(d));
            }
            dot *= scale;
            if (dot > maxScore) {
                maxScore = dot;
            }
            kRowQueue.FreeTensor(kLocal);
        }

        // Second pass: compute denominator sumExp
        float sumExp = 0.f;
        for (uint32_t kk = 0; kk < KL; ++kk) {
            uint64_t kOff = (uint64_t)blockIdx * KL * D + (uint64_t)kk * D;
            AscendC::LocalTensor<DTYPE_K> kLocal = kRowQueue.AllocTensor<DTYPE_K>();
            AscendC::DataCopy(kLocal, kGm[kOff], D);
            float dot = 0.f;
            for (uint32_t d = 0; d < D; ++d) {
                dot += static_cast<float>(qLocal.GetValue(d)) * static_cast<float>(kLocal.GetValue(d));
            }
            dot *= scale;
            float e = expf(dot - maxScore);
            sumExp += e;
            kRowQueue.FreeTensor(kLocal);
        }
        float invDenom = 1.0f / sumExp;

        // Third pass: accumulate output
        for (uint32_t kk = 0; kk < KL; ++kk) {
            // score
            uint64_t kOff = (uint64_t)blockIdx * KL * D + (uint64_t)kk * D;
            AscendC::LocalTensor<DTYPE_K> kLocal = kRowQueue.AllocTensor<DTYPE_K>();
            AscendC::DataCopy(kLocal, kGm[kOff], D);
            float dot = 0.f;
            for (uint32_t d = 0; d < D; ++d) {
                dot += static_cast<float>(qLocal.GetValue(d)) * static_cast<float>(kLocal.GetValue(d));
            }
            dot *= scale;
            float weight = expf(dot - maxScore) * invDenom;
            kRowQueue.FreeTensor(kLocal);

            // V row
            uint64_t vOff = (uint64_t)blockIdx * KL * D + (uint64_t)kk * D;
            AscendC::LocalTensor<DTYPE_V> vLocal = vRowQueue.AllocTensor<DTYPE_V>();
            AscendC::DataCopy(vLocal, vGm[vOff], D);

            // out += weight * vRow
            for (uint32_t d = 0; d < D; ++d) {
                float val = static_cast<float>(outLocal.GetValue(d)) + weight * static_cast<float>(vLocal.GetValue(d));
                outLocal.SetValue(d, static_cast<DTYPE_O>(val));
            }
            vRowQueue.FreeTensor(vLocal);
        }

        // Store output O[b, 0, :]
        uint64_t oBase = (uint64_t)blockIdx * QL * D;
        AscendC::DataCopy(oGm[oBase], outLocal, D);

        // release locals
        qQueue.FreeTensor(qLocal);
        oQueue.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> qQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> kRowQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> vRowQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> oQueue;

    AscendC::GlobalTensor<DTYPE_Q> qGm;
    AscendC::GlobalTensor<DTYPE_K> kGm;
    AscendC::GlobalTensor<DTYPE_V> vGm;
    AscendC::GlobalTensor<DTYPE_O> oGm;

    uint32_t B;
    uint32_t QL;
    uint32_t KL;
    uint32_t D;
    uint32_t blocks;
    uint32_t blockIdx;
};

extern "C" __global__ __aicore__ void kv_cached_attention_inference_custom(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR o, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelKvCachedAttentionInference op;
    op.Init(q, k, v, o,
            tiling_data.batch,
            tiling_data.qLen,
            tiling_data.kvLen,
            tiling_data.dModel);
    op.Process();
}
