kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "matmul_min_subtract_custom_tiling_data.h"

typedef float DTYPE_X;
typedef float DTYPE_C;
typedef float DTYPE_Y;

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelMinSub {
public:
    __aicore__ inline KernelMinSub() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR c, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        cGm.SetGlobalBuffer((__gm__ DTYPE_C *)c + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueC, BUFFER_NUM, this->tileLength * sizeof(DTYPE_C));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }
    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_C> cLocal = inQueueC.AllocTensor<DTYPE_C>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(cLocal, cGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
        inQueueC.EnQue(cLocal);
    }
    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_C> cLocal = inQueueC.DeQue<DTYPE_C>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

        // yLocal = min(xLocal, cLocal)
        AscendC::Minimum(yLocal, xLocal, cLocal, this->tileLength);
        // yLocal = yLocal - cLocal
        AscendC::Sub(yLocal, yLocal, cLocal, this->tileLength);

        outQueueY.EnQue<DTYPE_Y>(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueC.FreeTensor(cLocal);
    }
    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueC;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_C> cGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void matmul_min_subtract_custom(GM_ADDR x, GM_ADDR c, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMinSub op;
    op.Init(x, c, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: keep Linear on PyTorch, fuse min and subtract into a custom AscendC op.
    """
    def __init__(self, in_features, out_features, constant):
        super(ModelNew, self).__init__()
        self.linear = torch.nn.Linear(in_features, out_features)
        self.constant = torch.nn.Parameter(torch.tensor(constant))

    def forward(self, x):
        x = self.linear(x)
        c = self.constant.expand_as(x)
        return custom_ops_lib.matmul_min_subtract_custom(x, c)
