kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "matmul_with_irregular_shapes_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // number of local buffers in queues

// A: (M, K), B: (K, N), C: (M, N)
// Simple row-wise GEMM with tiling on N and scalar accumulation along K.
// Designed to handle irregular M, N, K gracefully.
class KernelMatmulIrregular {
public:
    __aicore__ inline KernelMatmulIrregular() {}
    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t N, uint32_t K, uint32_t tileN)
    {
        this->M = M;
        this->N = N;
        this->K = K;
        this->tileN = tileN;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blkIdx = AscendC::GetBlockIdx();

        // Distribute rows to blocks as evenly as possible
        this->rowsPerBlock = (M + blockNum - 1) / blockNum;
        this->rowStart = blkIdx * this->rowsPerBlock;
        uint32_t tmpEnd = this->rowStart + this->rowsPerBlock;
        this->rowEnd = tmpEnd > M ? M : tmpEnd;

        aGm.SetGlobalBuffer((__gm__ DTYPE_X *)A, static_cast<uint64_t>(M) * K);
        bGm.SetGlobalBuffer((__gm__ DTYPE_Y *)B, static_cast<uint64_t>(K) * N);
        cGm.SetGlobalBuffer((__gm__ DTYPE_Z *)C, static_cast<uint64_t>(M) * N);

        // Local buffers: one full row of A (size K), one tile of B and C (size tileN)
        pipe.InitBuffer(aRowQueue, 1, this->K * sizeof(DTYPE_X));
        pipe.InitBuffer(bTileQueue, BUFFER_NUM, this->tileN * sizeof(DTYPE_Y));
        pipe.InitBuffer(cTileQueue, BUFFER_NUM, this->tileN * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        if (rowStart >= rowEnd) {
            return;
        }
        for (uint32_t i = rowStart; i < rowEnd; ++i) {
            // Load A row i
            AscendC::LocalTensor<DTYPE_X> aRowLocal = aRowQueue.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(aRowLocal, aGm[static_cast<uint64_t>(i) * K], K);
            aRowQueue.EnQue(aRowLocal);

            AscendC::LocalTensor<DTYPE_X> aRow = aRowQueue.DeQue<DTYPE_X>();

            // Iterate N dimension in tiles
            for (uint32_t nStart = 0; nStart < N; nStart += tileN) {
                uint32_t curTN = (nStart + tileN <= N) ? tileN : (N - nStart);

                AscendC::LocalTensor<DTYPE_Z> cLocal = cTileQueue.AllocTensor<DTYPE_Z>();
                // Initialize accumulation buffer to zero
                // Duplicate fills vector with a scalar value
                AscendC::Duplicate(cLocal, (DTYPE_Z)0, curTN);

                // Accumulate over K
                for (uint32_t kIdx = 0; kIdx < K; ++kIdx) {
                    // Load B row segment: B[kIdx, nStart:nStart+curTN]
                    AscendC::LocalTensor<DTYPE_Y> bLocal = bTileQueue.AllocTensor<DTYPE_Y>();
                    AscendC::DataCopy(bLocal, bGm[static_cast<uint64_t>(kIdx) * N + nStart], curTN);

                    // Scale by A[i, kIdx] and accumulate
                    DTYPE_X aScalar = aRow.GetValue(kIdx); // read scalar from A row
                    AscendC::Muls(bLocal, bLocal, aScalar, curTN); // bLocal *= aScalar
                    AscendC::Add(cLocal, cLocal, bLocal, curTN);   // cLocal += bLocal

                    bTileQueue.FreeTensor(bLocal);
                }

                // Store C tile segment: C[i, nStart:nStart+curTN]
                AscendC::DataCopy(cGm[static_cast<uint64_t>(i) * N + nStart], cLocal, curTN);
                cTileQueue.FreeTensor(cLocal);
            }

            aRowQueue.FreeTensor(aRow);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> aRowQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> bTileQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> cTileQueue;

    AscendC::GlobalTensor<DTYPE_X> aGm;
    AscendC::GlobalTensor<DTYPE_Y> bGm;
    AscendC::GlobalTensor<DTYPE_Z> cGm;

    uint32_t M;
    uint32_t N;
    uint32_t K;
    uint32_t tileN;

    uint32_t rowsPerBlock;
    uint32_t rowStart;
    uint32_t rowEnd;
};

extern "C" __global__ __aicore__ void matmul_with_irregular_shapes_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulIrregular op;
    op.Init(A, B, C, tiling_data.M, tiling_data.N, tiling_data.K, tiling_data.tileN);
    op.Process();
}
