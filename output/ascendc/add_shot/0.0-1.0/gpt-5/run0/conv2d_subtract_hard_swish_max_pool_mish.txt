kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv2d_subtract_hard_swish_max_pool_mish_custom_tiling_data.h"
#include <cmath>

typedef float DTYPE_X;
typedef float DTYPE_W;
typedef float DTYPE_B;
typedef float DTYPE_Z;

__aicore__ inline float hardswish_fn(float x) {
    float t = x + 3.0f;
    if (t < 0.0f) t = 0.0f;
    if (t > 6.0f) t = 6.0f;
    return x * (t / 6.0f);
}
__aicore__ inline float mish_fn(float x) {
    // mish(x) = x * tanh(softplus(x)), softplus(x) = log(1 + exp(x))
    float sp = log1pf(expf(x));
    return x * tanhf(sp);
}

extern "C" __global__ __aicore__ void conv2d_subtract_hard_swish_max_pool_mish_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR subtract, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N  = tiling_data.n;
    const uint32_t Ci = tiling_data.ci;
    const uint32_t Co = tiling_data.co;
    const uint32_t H  = tiling_data.h;
    const uint32_t W  = tiling_data.w;
    const uint32_t Kh = tiling_data.kh;
    const uint32_t Kw = tiling_data.kw;
    const uint32_t Ho = tiling_data.h_out;   // conv output H
    const uint32_t Wo = tiling_data.w_out;   // conv output W
    const uint32_t Hp = tiling_data.h_pool;  // pooled H
    const uint32_t Wp = tiling_data.w_pool;  // pooled W

    const uint32_t totalLength = tiling_data.totalLength;

    __gm__ DTYPE_X* xPtr = reinterpret_cast<__gm__ DTYPE_X*>(x);
    __gm__ DTYPE_W* wPtr = reinterpret_cast<__gm__ DTYPE_W*>(weight);
    __gm__ DTYPE_B* bPtr = reinterpret_cast<__gm__ DTYPE_B*>(bias);
    __gm__ DTYPE_Z* yPtr = reinterpret_cast<__gm__ DTYPE_Z*>(y);
    __gm__ DTYPE_Z* subPtr = reinterpret_cast<__gm__ DTYPE_Z*>(subtract);

    float subVal = static_cast<float>(subPtr[0]);

    // Linear partition across blocks
    uint32_t blocks = AscendC::GetBlockNum();
    uint32_t blockIdx = AscendC::GetBlockIdx();
    uint32_t lenPerBlock = (totalLength + blocks - 1) / blocks; // ceiling to cover tail
    uint32_t start = blockIdx * lenPerBlock;
    uint32_t end = start + lenPerBlock;
    if (end > totalLength) end = totalLength;

    for (uint32_t idx = start; idx < end; ++idx) {
        // idx corresponds to pooled output index [n, co, hp, wp]
        uint32_t tmp = idx;
        uint32_t wp = tmp % Wp; tmp /= Wp;
        uint32_t hp = tmp % Hp; tmp /= Hp;
        uint32_t co = tmp % Co; tmp /= Co;
        uint32_t n  = tmp % N;

        // compute max over 2x2 window on conv+hardswish
        float maxv = -INFINITY;
        for (uint32_t dh = 0; dh < 2; ++dh) {
            for (uint32_t dw = 0; dw < 2; ++dw) {
                uint32_t ho = 2 * hp + dh;
                uint32_t wo = 2 * wp + dw;

                // conv accumulate for this (n, co, ho, wo)
                float acc = bPtr[co];
                for (uint32_t ci = 0; ci < Ci; ++ci) {
                    for (uint32_t kh = 0; kh < Kh; ++kh) {
                        for (uint32_t kw = 0; kw < Kw; ++kw) {
                            uint32_t ih = ho + kh;
                            uint32_t iw = wo + kw;

                            // input index
                            uint64_t inIndex = (((uint64_t)n * Ci + ci) * H + ih) * W + iw;
                            // weight index
                            uint64_t wIndex  = (((uint64_t)co * Ci + ci) * Kh + kh) * Kw + kw;

                            acc += xPtr[inIndex] * wPtr[wIndex];
                        }
                    }
                }
                acc = acc - subVal;          // subtract
                acc = hardswish_fn(acc);     // hardswish
                if (acc > maxv) maxv = acc;  // max pool
            }
        }
        float outv = mish_fn(maxv);          // mish
        // store
        yPtr[idx] = outv;
    }
}

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor conv2d_subtract_hard_swish_max_pool_mish_custom_impl_npu(
    const at::Tensor& x, const at::Tensor& weight, const at::Tensor& bias, const at::Tensor& subtract_val) {

    TORCH_CHECK(x.dim() == 4, "x must be NCHW");
    TORCH_CHECK(weight.dim() == 4, "weight must be OIHW");
    auto N  = x.size(0);
    auto Ci = x.size(1);
    auto H  = x.size(2);
    auto W  = x.size(3);

    auto Co = weight.size(0);
    auto Kh = weight.size(2);
    auto Kw = weight.size(3);

    TORCH_CHECK(weight.size(1) == Ci, "weight Ci mismatch");
    TORCH_CHECK(bias.dim() == 1 && bias.size(0) == Co, "bias mismatch");

    auto Ho = H - Kh + 1;
    auto Wo = W - Kw + 1;
    TORCH_CHECK(Ho > 0 && Wo > 0, "invalid conv shape");

    // MaxPool2d kernel_size=2, stride=2
    auto Hp = Ho / 2;
    auto Wp = Wo / 2;

    at::Tensor result = at::empty({N, Co, Hp, Wp}, x.options());

    EXEC_NPU_CMD(aclnnConv2dSubtractHardSwishMaxPoolMishCustom, x, weight, bias, subtract_val, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv2d_subtract_hard_swish_max_pool_mish_custom", &conv2d_subtract_hard_swish_max_pool_mish_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv2d_subtract_hard_swish_max_pool_mish_custom", &conv2d_subtract_hard_swish_max_pool_mish_custom_impl_npu, "Conv2d -> Sub -> HardSwish -> MaxPool2d(k=2,s=2) -> Mish");

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused op: Conv2d -> subtract -> HardSwish -> MaxPool2d(k=2,s=2) -> Mish
    """
    def __init__(self, in_channels, out_channels, kernel_size, subtract_value, pool_kernel_size):
        super().__init__()
        # Keep Conv2d module to manage parameters; forward will use custom op
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size, bias=True)
        self.subtract_value = float(subtract_value)
        self.pool_kernel_size = int(pool_kernel_size)  # fused as k=2, s=2 in custom kernel

    def forward(self, x):
        sub = torch.tensor([self.subtract_value], dtype=x.dtype, device=x.device)
        return custom_ops_lib.conv2d_subtract_hard_swish_max_pool_mish_custom(
            x, self.conv.weight, self.conv.bias, sub
        )
