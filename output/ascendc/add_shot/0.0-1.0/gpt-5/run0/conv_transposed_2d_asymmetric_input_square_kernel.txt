kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_transposed_2d_asymmetric_input_square_kernel_custom_tiling_data.h"

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {

    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N    = tiling_data.n;
    const uint32_t Cin  = tiling_data.cin;
    const uint32_t Cout = tiling_data.cout;
    const uint32_t Hin  = tiling_data.hin;
    const uint32_t Win  = tiling_data.win;
    const uint32_t K    = tiling_data.k;
    const uint32_t Hout = tiling_data.hout;
    const uint32_t Wout = tiling_data.wout;
    const uint64_t totalOut = tiling_data.totalOut;

    __gm__ float* x_gm = reinterpret_cast<__gm__ float*>(x);
    __gm__ float* w_gm = reinterpret_cast<__gm__ float*>(w);
    __gm__ float* y_gm = reinterpret_cast<__gm__ float*>(y);

    // Parallelize over the linearized output tensor elements
    const uint32_t blockNum = AscendC::GetBlockNum();
    const uint32_t blockIdx = AscendC::GetBlockIdx();

    // Strided loop across blocks
    for (uint64_t outIdx = blockIdx; outIdx < totalOut; outIdx += blockNum) {
        uint64_t tmp = outIdx;
        uint32_t wo = static_cast<uint32_t>(tmp % Wout); tmp /= Wout;
        uint32_t ho = static_cast<uint32_t>(tmp % Hout); tmp /= Hout;
        uint32_t co = static_cast<uint32_t>(tmp % Cout); tmp /= Cout;
        uint32_t n  = static_cast<uint32_t>(tmp);

        float acc = 0.f;

        // For stride=1, padding=0, output_padding=0, groups=1
        for (uint32_t ci = 0; ci < Cin; ++ci) {
            for (uint32_t kh = 0; kh < K; ++kh) {
                int32_t hi = static_cast<int32_t>(ho) - static_cast<int32_t>(kh);
                if (hi < 0 || hi >= static_cast<int32_t>(Hin)) continue;
                for (uint32_t kw = 0; kw < K; ++kw) {
                    int32_t wi = static_cast<int32_t>(wo) - static_cast<int32_t>(kw);
                    if (wi < 0 || wi >= static_cast<int32_t>(Win)) continue;

                    uint64_t x_index = (((static_cast<uint64_t>(n) * Cin + ci) * Hin + static_cast<uint32_t>(hi)) * Win + static_cast<uint32_t>(wi));
                    uint64_t w_index = (((static_cast<uint64_t>(ci) * Cout + co) * K + kh) * K + kw);

                    acc += x_gm[x_index] * w_gm[w_index];
                }
            }
        }

        uint64_t y_index = (((static_cast<uint64_t>(n) * Cout + co) * Hout + ho) * Wout + wo);
        y_gm[y_index] = acc;
    }
}

import torch
import torch_npu

class ModelNew(torch.nn.Module):
    """
    Optimized model using custom AscendC kernel for ConvTranspose2d with square kernel and asymmetric input.
    Supports stride=1, padding=0, output_padding=0, groups=1, bias=False.
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, output_padding: int = 0, groups: int = 1, bias: bool = False):
        super().__init__()
        # Keep a standard PyTorch module to manage weights/shapes; forward will call the custom kernel
        assert stride == 1 and padding == 0 and output_padding == 0 and groups == 1 and bias is False, "Custom kernel only supports stride=1, padding=0, output_padding=0, groups=1, bias=False"
        self.conv_transpose2d = torch.nn.ConvTranspose2d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, output_padding=output_padding, groups=groups, bias=bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        w = self.conv_transpose2d.weight
        return custom_ops_lib.conv_transposed_2d_asymmetric_input_square_kernel_custom(x, w)
