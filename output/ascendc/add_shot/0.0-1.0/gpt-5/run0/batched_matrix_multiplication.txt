kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "batched_matrix_multiplication_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelBmm {
public:
    __aicore__ inline KernelBmm() {}
    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t batch, uint32_t m, uint32_t k, uint32_t n,
                                uint32_t rowsPerBlock, uint32_t tileM, uint32_t tileN, uint32_t tileK)
    {
        this->batch = batch;
        this->m = m;
        this->k = k;
        this->n = n;
        this->rowsPerBlock = rowsPerBlock;
        this->tileM = tileM;
        this->tileN = tileN;
        this->tileK = tileK;

        aGm.SetGlobalBuffer((__gm__ DTYPE_X *)A, static_cast<uint64_t>(batch) * m * k);
        bGm.SetGlobalBuffer((__gm__ DTYPE_Y *)B, static_cast<uint64_t>(batch) * k * n);
        cGm.SetGlobalBuffer((__gm__ DTYPE_Z *)C, static_cast<uint64_t>(batch) * m * n);

        pipe.InitBuffer(aRowQueue, BUFFER_NUM, this->tileK * sizeof(DTYPE_X));
        pipe.InitBuffer(bTileQueue, BUFFER_NUM, this->tileK * this->tileN * sizeof(DTYPE_Y));
        pipe.InitBuffer(cVecQueue, BUFFER_NUM, this->tileN * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        uint64_t totalRows = static_cast<uint64_t>(batch) * m;
        uint64_t startRow = static_cast<uint64_t>(blockIdx) * rowsPerBlock;
        uint64_t endRow = startRow + rowsPerBlock;
        if (endRow > totalRows) endRow = totalRows;

        for (uint64_t r = startRow; r < endRow; ++r) {
            uint32_t b = static_cast<uint32_t>(r / m);
            uint32_t i = static_cast<uint32_t>(r % m);

            for (uint32_t jn = 0; jn < n; jn += tileN) {
                uint32_t curTileN = AscendC::Min(tileN, n - jn);

                AscendC::LocalTensor<DTYPE_Z> cVec = cVecQueue.AllocTensor<DTYPE_Z>();
                AscendC::Dup(cVec, static_cast<DTYPE_Z>(0), curTileN);

                for (uint32_t kk = 0; kk < k; kk += tileK) {
                    uint32_t curTileK = AscendC::Min(tileK, k - kk);

                    AscendC::LocalTensor<DTYPE_X> aRow = aRowQueue.AllocTensor<DTYPE_X>();
                    AscendC::LocalTensor<DTYPE_Y> bTile = bTileQueue.AllocTensor<DTYPE_Y>();

                    // GM offsets
                    uint64_t aOffset = static_cast<uint64_t>(b) * m * k + static_cast<uint64_t>(i) * k + kk;
                    uint64_t bOffset = static_cast<uint64_t>(b) * k * n + static_cast<uint64_t>(kk) * n + jn;

                    // Load A(i, kk:kk+curTileK)
                    AscendC::DataCopy(aRow, aGm[aOffset], curTileK);
                    // Load B(kk:kk+curTileK, jn:jn+curTileN) as a contiguous tile
                    for (uint32_t tk = 0; tk < curTileK; ++tk) {
                        AscendC::DataCopy(bTile[tk * curTileN], bGm[bOffset + static_cast<uint64_t>(tk) * n], curTileN);
                    }

                    // Compute: cVec += aRow * bTile (outer-product accumulation)
                    for (uint32_t tk = 0; tk < curTileK; ++tk) {
                        AscendC::LocalTensor<DTYPE_Z> tmp = cVecQueue.AllocTensor<DTYPE_Z>();
                        AscendC::Mul(tmp, bTile[tk * curTileN], aRow[tk], curTileN); // broadcast multiply
                        AscendC::Add(cVec, cVec, tmp, curTileN);
                        cVecQueue.FreeTensor(tmp);
                    }

                    aRowQueue.FreeTensor(aRow);
                    bTileQueue.FreeTensor(bTile);
                }

                // Store C(b, i, jn:jn+curTileN)
                uint64_t cOffset = static_cast<uint64_t>(b) * m * n + static_cast<uint64_t>(i) * n + jn;
                AscendC::DataCopy(cGm[cOffset], cVec, curTileN);
                cVecQueue.FreeTensor(cVec);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> aRowQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> bTileQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> cVecQueue;

    AscendC::GlobalTensor<DTYPE_X> aGm;
    AscendC::GlobalTensor<DTYPE_Y> bGm;
    AscendC::GlobalTensor<DTYPE_Z> cGm;

    uint32_t batch;
    uint32_t m;
    uint32_t k;
    uint32_t n;
    uint32_t rowsPerBlock;
    uint32_t tileM;
    uint32_t tileN;
    uint32_t tileK;
};

extern "C" __global__ __aicore__ void batched_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelBmm op;
    op.Init(A, B, C,
            tiling_data.batch, tiling_data.m, tiling_data.k, tiling_data.n,
            tiling_data.rowsPerBlock, tiling_data.tileM, tiling_data.tileN, tiling_data.tileK);
    op.Process();
}
