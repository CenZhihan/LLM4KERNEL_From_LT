kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "matrix_vector_multiplication_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // double buffer

// Simple GEMV kernel: C[M] = A[M,K] * B[K]
class KernelGEMV {
public:
    __aicore__ inline KernelGEMV() {}
    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t K,
                                uint32_t tileNumM, uint32_t kChunk)
    {
        this->M = M;
        this->K = K;
        this->tileNumM = tileNumM;
        this->kChunk = kChunk;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        this->blockRows = (M + blockNum - 1) / blockNum;
        this->rowStart = blockRows * blockIdx;
        if (this->rowStart + this->blockRows > M) {
            this->blockRows = M - this->rowStart;
        }
        this->tileRows = (this->blockRows + this->tileNumM - 1) / this->tileNumM;
        if (this->tileRows == 0) this->tileRows = 1;

        aGm.SetGlobalBuffer((__gm__ DTYPE_A *)A + this->rowStart * K, this->blockRows * K);
        bGm.SetGlobalBuffer((__gm__ DTYPE_B *)B, this->K);
        cGm.SetGlobalBuffer((__gm__ DTYPE_C *)C + this->rowStart, this->blockRows);

        // Buffers: A tile (tileRows x kChunk), B tile (kChunk), C tile (tileRows)
        pipe.InitBuffer(inQueueA, BUFFER_NUM, this->tileRows * this->kChunk * sizeof(DTYPE_A));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->kChunk * sizeof(DTYPE_B));
        pipe.InitBuffer(outQueueC, BUFFER_NUM, this->tileRows * sizeof(DTYPE_C));
    }

    __aicore__ inline void Process()
    {
        if (this->blockRows == 0) {
            return;
        }
        uint32_t tilesM = (this->blockRows + this->tileRows - 1) / this->tileRows;

        for (uint32_t tm = 0; tm < tilesM; ++tm) {
            uint32_t rowBase = tm * this->tileRows;
            uint32_t rowsThisTile = this->tileRows;
            if (rowBase + rowsThisTile > this->blockRows) {
                rowsThisTile = this->blockRows - rowBase;
            }

            // Allocate accumulation buffer for C tile and zero-init
            AscendC::LocalTensor<DTYPE_C> cAcc = outQueueC.AllocTensor<DTYPE_C>();
            // Zero fill cAcc
            for (uint32_t r = 0; r < rowsThisTile; ++r) {
                cAcc.GetData()[r] = static_cast<DTYPE_C>(0);
            }

            // Iterate over K in chunks
            for (uint32_t ko = 0; ko < this->K; ko += this->kChunk) {
                uint32_t kThis = this->kChunk;
                if (ko + kThis > this->K) {
                    kThis = this->K - ko;
                }

                // CopyIn: B chunk
                AscendC::LocalTensor<DTYPE_B> bLocal = inQueueB.AllocTensor<DTYPE_B>();
                AscendC::DataCopy(bLocal, bGm[ko], kThis);
                inQueueB.EnQue(bLocal);

                // CopyIn: A tile chunk (row by row to keep contiguity per row)
                AscendC::LocalTensor<DTYPE_A> aLocal = inQueueA.AllocTensor<DTYPE_A>();
                DTYPE_A* aPtr = aLocal.GetData();
                for (uint32_t r = 0; r < rowsThisTile; ++r) {
                    AscendC::DataCopy(aPtr + r * kThis, aGm[(rowBase + r) * this->K + ko], kThis);
                }
                inQueueA.EnQue(aLocal);

                // Compute
                AscendC::LocalTensor<DTYPE_B> bTile = inQueueB.DeQue<DTYPE_B>();
                AscendC::LocalTensor<DTYPE_A> aTile = inQueueA.DeQue<DTYPE_A>();

                DTYPE_A* aData = aTile.GetData();
                DTYPE_B* bData = bTile.GetData();
                DTYPE_C* cData = cAcc.GetData();

                // cAcc[r] += dot(aRow[r, ko:ko+kThis], b[ko:ko+kThis])
                for (uint32_t r = 0; r < rowsThisTile; ++r) {
                    DTYPE_C acc = cData[r];
                    DTYPE_A* aRow = aData + r * kThis;
                    for (uint32_t kk = 0; kk < kThis; ++kk) {
                        acc += static_cast<DTYPE_C>(aRow[kk]) * static_cast<DTYPE_C>(bData[kk]);
                    }
                    cData[r] = acc;
                }

                inQueueB.FreeTensor(bTile);
                inQueueA.FreeTensor(aTile);
            }

            // Enqueue completed C tile
            outQueueC.EnQue<DTYPE_C>(cAcc);

            // CopyOut: write C tile to global
            AscendC::LocalTensor<DTYPE_C> cLocalFinal = outQueueC.DeQue<DTYPE_C>();
            AscendC::DataCopy(cGm[rowBase], cLocalFinal, rowsThisTile);
            outQueueC.FreeTensor(cLocalFinal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA, inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC;

    AscendC::GlobalTensor<DTYPE_A> aGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_C> cGm;

    uint32_t M;
    uint32_t K;
    uint32_t tileNumM;
    uint32_t kChunk;

    uint32_t blockRows;
    uint32_t rowStart;
    uint32_t tileRows;
};

extern "C" __global__ __aicore__ void matrix_vector_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGEMV op;
    op.Init(A, B, C,
            tiling_data.M,
            tiling_data.K,
            tiling_data.tileNumM,
            tiling_data.kChunk);
    op.Process();
}
