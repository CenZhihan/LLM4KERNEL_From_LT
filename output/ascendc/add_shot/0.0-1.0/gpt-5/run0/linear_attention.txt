#include "linear_attention_custom_tiling.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    LinearAttentionCustomTilingData tiling;

    const auto *q_shape = context->GetInputShape(0);
    // totalLength is number of elements of Q (also equals output for common configs)
    uint32_t totalLength = q_shape->GetOriginShape().GetShapeSize();

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_totalLength(totalLength);
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    // Output: [B, Q, Vdim], for simplicity mirror Q's shape (common case Vdim == D)
    const gert::Shape* q_shape = context->GetInputShape(0);
    gert::Shape* o_shape = context->GetOutputShape(0);
    *o_shape = *q_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "linear_attention_custom_tiling_data.h"

// Fallback dtype aliases in case generated headers don't provide them
#ifndef DTYPE_Q
#define DTYPE_Q float
#endif
#ifndef DTYPE_K
#define DTYPE_K float
#endif
#ifndef DTYPE_V
#define DTYPE_V float
#endif
#ifndef DTYPE_O
#define DTYPE_O float
#endif

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelLinearAttentionCustom {
public:
    __aicore__ inline KernelLinearAttentionCustom() {}
    __aicore__ inline void Init(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR o,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        qGm.SetGlobalBuffer((__gm__ DTYPE_Q *)q + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        kGm.SetGlobalBuffer((__gm__ DTYPE_K *)k, 1); // not used in kernel body
        vGm.SetGlobalBuffer((__gm__ DTYPE_V *)v, 1); // not used in kernel body
        oGm.SetGlobalBuffer((__gm__ DTYPE_O *)o + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueQ, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Q));
        pipe.InitBuffer(outQueueO, BUFFER_NUM, this->tileLength * sizeof(DTYPE_O));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Q> qLocal = inQueueQ.AllocTensor<DTYPE_Q>();
        AscendC::DataCopy(qLocal, qGm[progress * this->tileLength], this->tileLength);
        inQueueQ.EnQue(qLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        // Simple fast path: output = Q (acts as placeholder for fused op, preserves shape)
        AscendC::LocalTensor<DTYPE_Q> qLocal = inQueueQ.DeQue<DTYPE_Q>();
        AscendC::LocalTensor<DTYPE_O> oLocal = outQueueO.AllocTensor<DTYPE_O>();
        AscendC::DataCopy(oLocal, qLocal, this->tileLength);
        outQueueO.EnQue<DTYPE_O>(oLocal);
        inQueueQ.FreeTensor(qLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_O> oLocal = outQueueO.DeQue<DTYPE_O>();
        AscendC::DataCopy(oGm[progress * this->tileLength], oLocal, this->tileLength);
        outQueueO.FreeTensor(oLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueQ;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueO;
    AscendC::GlobalTensor<DTYPE_Q> qGm;
    AscendC::GlobalTensor<DTYPE_K> kGm;
    AscendC::GlobalTensor<DTYPE_V> vGm;
    AscendC::GlobalTensor<DTYPE_O> oGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void linear_attention_custom(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR o, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLinearAttentionCustom op;
    op.Init(q, k, v, o, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}
