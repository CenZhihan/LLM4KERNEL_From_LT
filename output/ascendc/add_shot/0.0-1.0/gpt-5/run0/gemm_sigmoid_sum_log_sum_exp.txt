kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "gemm_sigmoid_sum_log_sum_exp_custom_tiling_data.h"
#include <cmath>

class KernelGemmSigmoidSumLogSumExp {
public:
    __aicore__ inline KernelGemmSigmoidSumLogSumExp() {}

    __aicore__ inline void Init(GM_ADDR x,
                                GM_ADDR w1,
                                GM_ADDR b1,
                                GM_ADDR w2,
                                GM_ADDR b2,
                                GM_ADDR y,
                                uint32_t N, uint32_t K, uint32_t H, uint32_t O)
    {
        this->xGm = reinterpret_cast<__gm__ DTYPE_X*>(x);
        this->w1Gm = reinterpret_cast<__gm__ DTYPE_W1*>(w1);
        this->b1Gm = reinterpret_cast<__gm__ DTYPE_B1*>(b1);
        this->w2Gm = reinterpret_cast<__gm__ DTYPE_W2*>(w2);
        this->b2Gm = reinterpret_cast<__gm__ DTYPE_B2*>(b2);
        this->yGm  = reinterpret_cast<__gm__ DTYPE_Y*>(y);

        this->N = N;
        this->K = K;
        this->H = H;
        this->O = O;

        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t bid = AscendC::GetBlockIdx();
        this->rowsPerBlock = (N + blocks - 1) / blocks;
        this->rowStart = bid * this->rowsPerBlock;
        uint32_t rowEnd = (this->rowStart + this->rowsPerBlock);
        this->rowEnd = rowEnd > N ? N : rowEnd;

        // init local buffers
        pipe.InitBuffer(qX, 1, K * sizeof(DTYPE_X));
        pipe.InitBuffer(qB1, 1, H * sizeof(DTYPE_B1));
        pipe.InitBuffer(qB2, 1, O * sizeof(DTYPE_B2));
        pipe.InitBuffer(qA, 1, H * sizeof(float));
        pipe.InitBuffer(qT, 1, O * sizeof(float));
        pipe.InitBuffer(qW1row, 1, K * sizeof(DTYPE_W1));
        pipe.InitBuffer(qW2row, 1, H * sizeof(DTYPE_W2));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = rowStart; r < rowEnd; ++r) {
            // Load x row [K]
            AscendC::LocalTensor<DTYPE_X> xLocal = qX.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(xLocal, &xGm[r * K], K);
            qX.EnQue(xLocal);

            // Load b1 [H]
            AscendC::LocalTensor<DTYPE_B1> b1Local = qB1.AllocTensor<DTYPE_B1>();
            AscendC::DataCopy(b1Local, b1Gm, H);
            qB1.EnQue(b1Local);

            // Prepare aLocal [H]
            AscendC::LocalTensor<float> aLocal = qA.AllocTensor<float>();

            // Compute s1 -> sigmoid -> aLocal[j] for all j in [0, H)
            AscendC::LocalTensor<DTYPE_X> xRow = qX.DeQue<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_B1> b1Vec = qB1.DeQue<DTYPE_B1>();

            for (uint32_t j = 0; j < H; ++j) {
                // Load w1 row j: w1[j, :] contiguous length K
                AscendC::LocalTensor<DTYPE_W1> w1Row = qW1row.AllocTensor<DTYPE_W1>();
                AscendC::DataCopy(w1Row, &w1Gm[j * K], K);

                // dot product
                float acc = static_cast<float>(b1Vec.GetValue(j));
                for (uint32_t k = 0; k < K; ++k) {
                    float xv = static_cast<float>(xRow.GetValue(k));
                    float wv = static_cast<float>(w1Row.GetValue(k));
                    acc += xv * wv;
                }
                // sigmoid
                float sig = 1.0f / (1.0f + expf(-acc));
                aLocal.SetValue(j, sig);

                qW1row.FreeTensor(w1Row);
            }

            // Done with xRow and b1Vec
            qX.FreeTensor(xRow);
            qB1.FreeTensor(b1Vec);
            qA.EnQue(aLocal);

            // Load b2 [O] into tLocal
            AscendC::LocalTensor<DTYPE_B2> b2Local = qB2.AllocTensor<DTYPE_B2>();
            AscendC::DataCopy(b2Local, b2Gm, O);
            qB2.EnQue(b2Local);

            AscendC::LocalTensor<float> tLocal = qT.AllocTensor<float>();
            AscendC::LocalTensor<DTYPE_B2> b2Vec = qB2.DeQue<DTYPE_B2>();
            for (uint32_t o = 0; o < O; ++o) {
                tLocal.SetValue(o, static_cast<float>(b2Vec.GetValue(o)));
            }
            qB2.FreeTensor(b2Vec);

            // Accumulate t[o] += dot(aLocal, w2[o, :])
            AscendC::LocalTensor<float> aVec = qA.DeQue<float>();
            for (uint32_t o = 0; o < O; ++o) {
                // load w2 row o: w2[o, :] contiguous length H
                AscendC::LocalTensor<DTYPE_W2> w2Row = qW2row.AllocTensor<DTYPE_W2>();
                AscendC::DataCopy(w2Row, &w2Gm[o * H], H);

                float acc = tLocal.GetValue(o);
                for (uint32_t j = 0; j < H; ++j) {
                    float av = aVec.GetValue(j);
                    float wv = static_cast<float>(w2Row.GetValue(j));
                    acc += av * wv;
                }
                tLocal.SetValue(o, acc);

                qW2row.FreeTensor(w2Row);
            }
            qA.FreeTensor(aVec);

            // logsumexp over O
            float maxv = -3.402823466e+38F; // -FLT_MAX
            for (uint32_t o = 0; o < O; ++o) {
                float tv = tLocal.GetValue(o);
                maxv = tv > maxv ? tv : maxv;
            }
            float sumExp = 0.0f;
            for (uint32_t o = 0; o < O; ++o) {
                float tv = tLocal.GetValue(o);
                sumExp += expf(tv - maxv);
            }
            float outv = logf(sumExp) + maxv;

            // write output scalar
            yGm[r] = static_cast<DTYPE_Y>(outv);

            qT.FreeTensor(tLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> qX, qB1, qB2, qW1row, qW2row, qA, qT;

    __gm__ DTYPE_X* xGm;
    __gm__ DTYPE_W1* w1Gm;
    __gm__ DTYPE_B1* b1Gm;
    __gm__ DTYPE_W2* w2Gm;
    __gm__ DTYPE_B2* b2Gm;
    __gm__ DTYPE_Y* yGm;

    uint32_t N, K, H, O;
    uint32_t rowsPerBlock;
    uint32_t rowStart;
    uint32_t rowEnd;
};

extern "C" __global__ __aicore__ void gemm_sigmoid_sum_log_sum_exp_custom(GM_ADDR x,
                                                                          GM_ADDR w1,
                                                                          GM_ADDR b1,
                                                                          GM_ADDR w2,
                                                                          GM_ADDR b2,
                                                                          GM_ADDR y,
                                                                          GM_ADDR workspace,
                                                                          GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmSigmoidSumLogSumExp op;
    op.Init(x, w1, b1, w2, b2, y,
            tiling_data.N, tiling_data.K, tiling_data.H, tiling_data.O);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused kernel for: x -> Linear1 -> Sigmoid -> Linear2 -> LogSumExp(dim=1)
    """
    def __init__(self, input_size, hidden_size, output_size):
        super(ModelNew, self).__init__()
        self.linear1 = torch.nn.Linear(input_size, hidden_size)
        self.linear2 = torch.nn.Linear(hidden_size, output_size)

    def forward(self, x):
        # Pass weights/biases as stored by PyTorch:
        # linear1.weight: [hidden, input], linear1.bias: [hidden]
        # linear2.weight: [output, hidden], linear2.bias: [output]
        return custom_ops_lib.gemm_sigmoid_sum_log_sum_exp_custom(
            x,
            self.linear1.weight, self.linear1.bias,
            self.linear2.weight, self.linear2.bias
        )
