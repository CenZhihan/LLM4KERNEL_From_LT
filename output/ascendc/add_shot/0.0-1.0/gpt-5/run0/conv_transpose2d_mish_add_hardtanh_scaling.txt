kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_transpose2d_mish_add_hardtanh_scaling_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTranspose2dMishAddHardtanhScaling {
public:
    __aicore__ inline KernelConvTranspose2dMishAddHardtanhScaling() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR add, GM_ADDR scale, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        addGm.SetGlobalBuffer((__gm__ DTYPE_Y *)add + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        scaleGm.SetGlobalBuffer((__gm__ DTYPE_Y *)scale + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueA, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(inQueueS, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> aLocal = inQueueA.AllocTensor<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Y> sLocal = inQueueS.AllocTensor<DTYPE_Y>();

        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(aLocal, addGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(sLocal, scaleGm[progress * this->tileLength], this->tileLength);

        inQueueX.EnQue(xLocal);
        inQueueA.EnQue(aLocal);
        inQueueS.EnQue(sLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> aLocal = inQueueA.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Y> sLocal = inQueueS.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        // y = scale * hardtanh(mish(x) + add, -1, 1)
        AscendC::LocalTensor<DTYPE_Z> tmp1 = pipe.AllocTensor<DTYPE_Z>(this->tileLength);
        AscendC::LocalTensor<DTYPE_Z> tmp2 = pipe.AllocTensor<DTYPE_Z>(this->tileLength);

        // mish(x)
        AscendC::Mish(tmp1, xLocal, this->tileLength);
        // + add
        AscendC::Add(tmp2, tmp1, aLocal, this->tileLength);
        // clamp to [-1, 1]
        AscendC::Maxs(tmp1, tmp2, static_cast<DTYPE_Z>(-1.0f), this->tileLength);
        AscendC::Mins(tmp2, tmp1, static_cast<DTYPE_Z>(1.0f), this->tileLength);
        // * scale
        AscendC::Mul(yLocal, tmp2, sLocal, this->tileLength);

        outQueueY.EnQue<DTYPE_Z>(yLocal);

        pipe.FreeTensor(tmp1);
        pipe.FreeTensor(tmp2);
        inQueueX.FreeTensor(xLocal);
        inQueueA.FreeTensor(aLocal);
        inQueueS.FreeTensor(sLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueA, inQueueS;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> addGm;
    AscendC::GlobalTensor<DTYPE_Y> scaleGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transpose2d_mish_add_hardtanh_scaling_custom(GM_ADDR x, GM_ADDR add, GM_ADDR scale, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTranspose2dMishAddHardtanhScaling op;
    op.Init(x, add, scale, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    """
    Model that performs a transposed convolution, applies Mish activation, adds a value,
    applies Hardtanh activation, and scales the output via a fused custom AscendC operator.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, add_value, scale):
        super(ModelNew, self).__init__()
        self.conv_transpose = nn.ConvTranspose2d(in_channels, out_channels, kernel_size, stride, padding, output_padding)
        self.add_value = float(add_value)
        self.scale = float(scale)

    def forward(self, x):
        x = self.conv_transpose(x)
        add_t = torch.full_like(x, self.add_value)
        scale_t = torch.full_like(x, self.scale)
        x = custom_ops_lib.conv_transpose2d_mish_add_hardtanh_scaling_custom(x, add_t, scale_t)
        return x
