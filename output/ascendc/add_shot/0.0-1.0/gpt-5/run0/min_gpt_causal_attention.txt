kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "min_gpt_causal_attention_custom_tiling_data.h"

// Naive reference kernel for masked causal attention:
// Inputs/Output shapes: [B, H, T, S] with contiguous ND layout.
// y[b,h,t,:] = softmax(q[b,h,t,:] @ k[b,h,0:t,:]^T / sqrt(S)) @ v[b,h,0:t,:]
// Note: This reference kernel focuses on correctness; further vectorization/tiling can be added.

extern "C" __global__ __aicore__ void min_gpt_causal_attention_custom(GM_ADDR q,
                                                                      GM_ADDR k,
                                                                      GM_ADDR v,
                                                                      GM_ADDR y,
                                                                      GM_ADDR workspace,
                                                                      GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t T = tiling_data.seqLen;
    const uint32_t S = tiling_data.headDim;
    const uint32_t total = tiling_data.totalLength; // B * H * T * S
    if (T == 0 || S == 0) { return; }

    const uint32_t BH = total / (T * S);
    const float scale = 1.f / AscendC::Sqrt(static_cast<float>(S));

    __gm__ float* qGm = reinterpret_cast<__gm__ float*>(q);
    __gm__ float* kGm = reinterpret_cast<__gm__ float*>(k);
    __gm__ float* vGm = reinterpret_cast<__gm__ float*>(v);
    __gm__ float* yGm = reinterpret_cast<__gm__ float*>(y);

    // Parallelize across BH groups
    const uint32_t blockNum = AscendC::GetBlockNum();
    const uint32_t blockIdx = AscendC::GetBlockIdx();
    const uint32_t bhPerBlock = (BH + blockNum - 1) / blockNum;
    const uint32_t bhStart = blockIdx * bhPerBlock;
    const uint32_t bhEnd = AscendC::Min(BH, bhStart + bhPerBlock);

    // Static buffers sized for typical GPT settings. Adjust if needed.
    constexpr int MAX_T = 2048;
    constexpr int MAX_S = 2048;

    for (uint32_t bh = bhStart; bh < bhEnd; ++bh) {
        // base offsets per (B*H) group
        const uint64_t baseBH = static_cast<uint64_t>(bh) * T * S;

        for (uint32_t t = 0; t < T; ++t) {
            // Find max logit for numerical stability
            float maxLogit = -1e30f;

            // Pass 1: compute logits and find max over j in [0, t]
            // Because T is dynamic, we compute on-the-fly without storing all S entries.
            for (uint32_t j = 0; j <= t; ++j) {
                float dot = 0.f;
                // q[t,:] · k[j,:]
                const uint64_t qOff = baseBH + static_cast<uint64_t>(t) * S;
                const uint64_t kOff = baseBH + static_cast<uint64_t>(j) * S;
                for (uint32_t s = 0; s < S; ++s) {
                    dot += qGm[qOff + s] * kGm[kOff + s];
                }
                dot *= scale;
                if (dot > maxLogit) maxLogit = dot;
            }

            // Pass 2: exp and denom
            float denom = 0.f;
            // To avoid excessive GM traffic, recompute logits and accumulate exp
            for (uint32_t j = 0; j <= t; ++j) {
                float dot = 0.f;
                const uint64_t qOff = baseBH + static_cast<uint64_t>(t) * S;
                const uint64_t kOff = baseBH + static_cast<uint64_t>(j) * S;
                for (uint32_t s = 0; s < S; ++s) {
                    dot += qGm[qOff + s] * kGm[kOff + s];
                }
                dot = (dot * scale) - maxLogit;
                float e = AscendC::Exp(dot);
                denom += e;

                // Temporarily store e in y buffer's head to reuse in Pass 3?
                // We avoid extra storage; recompute weights again in pass 3 for simplicity.
                (void)e;
            }
            float invDenom = 1.f / denom;

            // Pass 3: weighted sum over V
            for (uint32_t s = 0; s < S; ++s) {
                float acc = 0.f;
                for (uint32_t j = 0; j <= t; ++j) {
                    float dot = 0.f;
                    const uint64_t qOff = baseBH + static_cast<uint64_t>(t) * S;
                    const uint64_t kOff = baseBH + static_cast<uint64_t>(j) * S;
                    for (uint32_t ss = 0; ss < S; ++ss) {
                        dot += qGm[qOff + ss] * kGm[kOff + ss];
                    }
                    dot = (dot * scale) - maxLogit;
                    float w = AscendC::Exp(dot) * invDenom;

                    const uint64_t vOff = baseBH + static_cast<uint64_t>(j) * S;
                    acc += w * vGm[vOff + s];
                }
                const uint64_t yOff = baseBH + static_cast<uint64_t>(t) * S + s;
                yGm[yOff] = acc;
            }
        }
    }
}

model_src='''
import torch
import torch.nn as nn
import torch.nn.functional as F
import torch_npu
import custom_ops_lib
import math

# Optimized attention using custom AscendC kernel:
# custom_ops_lib.min_gpt_causal_attention_custom expects q, k, v in shape [B, H, T, S] and returns y in the same shape.

class ModelNew(nn.Module):
    def __init__(self, n_embd, n_head, attn_pdrop, resid_pdrop, max_seqlen):
        super().__init__()
        assert n_embd % n_head == 0
        self.n_head = n_head
        self.n_embd = n_embd
        self.max_seqlen = max_seqlen

        self.c_attn = nn.Linear(n_embd, 3 * n_embd)
        self.c_proj = nn.Linear(n_embd, n_embd)

        self.attn_dropout = nn.Dropout(attn_pdrop)
        self.resid_dropout = nn.Dropout(resid_pdrop)

        # kept for shape compatibility, but mask is applied inside custom kernel
        self.register_buffer("bias", torch.tril(torch.ones(max_seqlen, max_seqlen)).view(1, 1, max_seqlen, max_seqlen))

    def forward(self, x):
        B, T, C = x.size()
        nh = self.n_head
        hs = C // nh

        # project to qkv then reshape to [B, H, T, S]
        q, k, v = self.c_attn(x).split(self.n_embd, dim=2)
        q = q.view(B, T, nh, hs).transpose(1, 2).contiguous()
        k = k.view(B, T, nh, hs).transpose(1, 2).contiguous()
        v = v.view(B, T, nh, hs).transpose(1, 2).contiguous()

        # custom kernel computes causal masked attention with softmax and returns y: [B, H, T, S]
        y = custom_ops_lib.min_gpt_causal_attention_custom(q, k, v)

        # re-assemble heads and project out
        y = y.transpose(1, 2).contiguous().view(B, T, C)
        y = self.resid_dropout(self.c_proj(y))
        return y
'''
