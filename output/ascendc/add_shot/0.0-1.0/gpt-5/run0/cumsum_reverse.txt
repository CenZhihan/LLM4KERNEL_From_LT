const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    CumsumReverseCustomTilingData tiling;

    const auto *in_shape = context->GetInputShape(0);
    const auto &origin = in_shape->GetOriginShape();
    uint32_t rank = origin.GetDimNum();

    int64_t dim_attr = 0;
    // get dim attribute (default last dim if not set)
    if (context->GetAttrs() != nullptr) {
        (void)context->GetAttrs()->GetInt("dim", dim_attr);
    }
    if (dim_attr < 0) {
        dim_attr += static_cast<int64_t>(rank);
    }
    if (dim_attr < 0) dim_attr = static_cast<int64_t>(rank) - 1;
    if (static_cast<uint32_t>(dim_attr) >= rank) dim_attr = static_cast<int64_t>(rank) - 1;

    uint32_t axisLen = static_cast<uint32_t>(origin.GetDim(static_cast<uint32_t>(dim_attr)));
    uint64_t totalLen = origin.GetShapeSize();

    // rows is the number of contiguous segments of length axisLen, assuming dim is the last dimension
    uint32_t rows = static_cast<uint32_t>(totalLen / axisLen);

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_totalLength(static_cast<uint32_t>(totalLen));
    tiling.set_rows(rows);
    tiling.set_axisLen(axisLen);
    tiling.set_tileNum(TILE_NUM);
    tiling.set_axis(static_cast<int32_t>(dim_attr));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "cumsum_reverse_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 1; // one row per buffer

class KernelCumsumReverse {
public:
    __aicore__ inline KernelCumsumReverse() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t rows,
                                uint32_t axisLen, uint32_t tileNum, int32_t axis)
    {
        (void)totalLength;
        (void)tileNum;

        // Assume dim is the last dimension so each row is contiguous length axisLen
        this->axisLen = axisLen;
        this->rowsTotal = rows;
        this->axis = axis;

        // Distribute rows over blocks
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        this->rowsPerBlock = rows / blockNum;
        this->rowStart = this->rowsPerBlock * blockIdx;

        // Base GM pointers for this block
        uint64_t offset = static_cast<uint64_t>(rowStart) * static_cast<uint64_t>(axisLen);
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + offset, this->rowsPerBlock * this->axisLen);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + offset, this->rowsPerBlock * this->axisLen);

        // L1 buffers sized to one row
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->axisLen * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->axisLen * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < this->rowsPerBlock; ++r) {
            CopyIn(r);
            Compute(r);
            CopyOut(r);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[rowIdx * this->axisLen], this->axisLen);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t /*rowIdx*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        // Reverse cumulative sum within the row
        // y[i] = sum_{k=i..axisLen-1} x[k]
        DTYPE_Z acc = static_cast<DTYPE_Z>(0);
        for (int32_t i = static_cast<int32_t>(this->axisLen) - 1; i >= 0; --i) {
            DTYPE_Z v = static_cast<DTYPE_Z>(xLocal.GetValue(i));
            acc = acc + v;
            yLocal.SetValue(i, acc);
        }

        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[rowIdx * this->axisLen], yLocal, this->axisLen);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t axisLen;
    uint32_t rowsTotal;
    uint32_t rowsPerBlock;
    uint32_t rowStart;
    int32_t axis;
};

extern "C" __global__ __aicore__ void cumsum_reverse_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumsumReverse op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.rows, tiling_data.axisLen, tiling_data.tileNum, tiling_data.axis);
    op.Process();
}
