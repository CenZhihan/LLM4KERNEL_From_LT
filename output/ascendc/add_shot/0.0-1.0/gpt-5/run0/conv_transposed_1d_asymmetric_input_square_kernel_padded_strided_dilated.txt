kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_transposed_1d_asymmetric_input_square_kernel_padded_strided_dilated_custom_tiling_data.h"

// NOTE: This is a reference implementation focusing on tiling/streaming structure.
// It performs a naive deconvolution in scalar loops inside tiles.

constexpr int32_t BUFFER_NUM = 2;

class KernelDeconv1d {
public:
    __aicore__ inline KernelDeconv1d() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y, uint32_t totalLength,
                                uint32_t tileNum,
                                uint32_t batch, uint32_t inC, uint32_t outC,
                                uint32_t inLen, uint32_t outLen,
                                uint32_t ksize, uint32_t stride, uint32_t padding, uint32_t dilation)
    {
        this->tileNum = tileNum;
        this->batch = batch;
        this->inC = inC;
        this->outC = outC;
        this->inLen = inLen;
        this->outLen = outLen;
        this->ksize = ksize;
        this->stride = stride;
        this->padding = padding;
        this->dilation = dilation;

        // Split output length across blocks
        this->blockOutLen = outLen / AscendC::GetBlockNum();
        this->startOutPos = this->blockOutLen * AscendC::GetBlockIdx();
        if (AscendC::GetBlockIdx() == (AscendC::GetBlockNum() - 1)) {
            // last block handles tail
            this->blockOutLen = outLen - this->startOutPos;
        }

        // Use simple 1D tiling along output length
        this->tileOutLen = AscendC::Ceil(this->blockOutLen, tileNum * BUFFER_NUM);

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint64_t)batch * inC * inLen);
        wGm.SetGlobalBuffer((__gm__ DTYPE_Y *)w, (uint64_t)inC * outC * ksize);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y, (uint64_t)batch * outC * outLen);

        // local buffers sized for one tile across (N, OC, tileOutLen)
        uint32_t xSliceMax = inC * ((tileOutLen + 2 * padding + dilation * (ksize - 1) + stride - 1) / stride + 1);
        // Conservative allocation for safety
        uint32_t yTileElems = outC * tileOutLen;

        pipe.InitBuffer(inQueueX, BUFFER_NUM, xSliceMax * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueW, BUFFER_NUM, (inC * outC * ksize) * sizeof(DTYPE_Y) / 8); // stream by chunk
        pipe.InitBuffer(outQueueY, BUFFER_NUM, yTileElems * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        // For simplicity, we load weight once per block into local buffer chunks inside Compute
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        // No prefetch for x; we'll access from GM using DataCopy into local inside Compute by needed ranges.
        // Prepare an empty local tensor for Y tile (will be filled in Compute).
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();
        uint32_t yElems = outC * tileOutLen;
        // Initialize output tile to zero
        AscendC::SetVector<DTYPE_Z>(yLocal, static_cast<DTYPE_Z>(0), yElems);
        outQueueY.EnQue<DTYPE_Z>(yLocal);
    }

    __aicore__ inline void Compute(int32_t progress)
    {
        // Dequeue Y local tile
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();

        uint32_t tileStart = this->startOutPos + progress * this->tileOutLen;
        if (tileStart >= this->startOutPos + this->blockOutLen) {
            // nothing to do
            outQueueY.EnQue<DTYPE_Z>(yLocal);
            return;
        }
        uint32_t currTileLen = AscendC::Min(this->tileOutLen, this->startOutPos + this->blockOutLen - tileStart);

        // Iterate over batch and channels to accumulate into yLocal
        for (uint32_t n = 0; n < batch; ++n) {
            for (uint32_t oc = 0; oc < outC; ++oc) {
                for (uint32_t t = 0; t < currTileLen; ++t) {
                    uint32_t outPos = tileStart + t;
                    float acc = 0.0f; // will accumulate per (ic, k) then store
                    // We need to sum over input channels and kernel positions
                    for (uint32_t ic = 0; ic < inC; ++ic) {
                        // For transposed conv, output position outPos receives contributions from input positions:
                        // outPos = inPos * stride - padding + k * dilation  => inPos = (outPos + padding - k*dilation)/stride
                        for (uint32_t k = 0; k < ksize; ++k) {
                            int64_t numer = static_cast<int64_t>(outPos) + static_cast<int64_t>(padding) - static_cast<int64_t>(k) * static_cast<int64_t>(dilation);
                            if (numer < 0) continue;
                            if (numer % static_cast<int64_t>(stride) != 0) continue;
                            int64_t inPos64 = numer / static_cast<int64_t>(stride);
                            if (inPos64 < 0 || inPos64 >= static_cast<int64_t>(inLen)) continue;
                            uint32_t inPos = static_cast<uint32_t>(inPos64);

                            // x index: [n, ic, inPos]
                            uint64_t xIndex = ((uint64_t)n * inC + ic) * inLen + inPos;
                            DTYPE_X xVal;
                            AscendC::DataCopy(&xVal, xGm[xIndex], 1);

                            // w index: [ic, oc, k]
                            uint64_t wIndex = ((uint64_t)ic * outC + oc) * ksize + k;
                            DTYPE_Y wVal;
                            AscendC::DataCopy(&wVal, wGm[wIndex], 1);

                            acc += static_cast<float>(xVal) * static_cast<float>(wVal);
                        }
                    }
                    // write back to local y tile at [oc, t]
                    uint32_t yOffset = oc * currTileLen + t;
                    yLocal.SetValue(yOffset, static_cast<DTYPE_Z>(acc));
                }
            }
            // After finishing all oc for this batch slice, store to GM
            // But actual copy out is done in CopyOut; here we just continue accumulating for all batches by offsetting
            // To support batches, we will handle GM offsets in CopyOut based on tileStart and 'n'
        }

        outQueueY.EnQue<DTYPE_Z>(yLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();

        uint32_t tileStart = this->startOutPos + progress * this->tileOutLen;
        if (tileStart >= this->startOutPos + this->blockOutLen) {
            outQueueY.FreeTensor(yLocal);
            return;
        }
        uint32_t currTileLen = AscendC::Min(this->tileOutLen, this->startOutPos + this->blockOutLen - tileStart);

        // Write tile for all batches and output channels
        for (uint32_t n = 0; n < batch; ++n) {
            for (uint32_t oc = 0; oc < outC; ++oc) {
                // GM base: y[n, oc, tileStart]
                uint64_t yBase = ((uint64_t)n * outC + oc) * outLen + tileStart;
                // Local base in yLocal for this (oc)
                uint32_t yLocalBase = oc * currTileLen;
                AscendC::DataCopy(yGm[yBase], yLocal[yLocalBase], currTileLen);
            }
        }

        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueW;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> wGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t tileNum;
    uint32_t batch, inC, outC;
    uint32_t inLen, outLen;
    uint32_t ksize, stride, padding, dilation;

    uint32_t blockOutLen;
    uint32_t startOutPos;
    uint32_t tileOutLen;
};

extern "C" __global__ __aicore__ void conv_transposed_1d_asymmetric_input_square_kernel_padded_strided_dilated_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelDeconv1d op;
    op.Init(x, w, y,
            tiling_data.totalLength, tiling_data.tileNum,
            tiling_data.batch, tiling_data.inC, tiling_data.outC,
            tiling_data.inLen, tiling_data.outLen,
            tiling_data.ksize, tiling_data.stride, tiling_data.padding, tiling_data.dilation);
    op.Process();
}
