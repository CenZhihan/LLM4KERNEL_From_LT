    GooglenetInceptionModuleCustomTilingData tiling;

    const auto shape0 = context->GetInputShape(0)->GetOriginShape();
    const auto shape1 = context->GetInputShape(1)->GetOriginShape();
    const auto shape2 = context->GetInputShape(2)->GetOriginShape();
    const auto shape3 = context->GetInputShape(3)->GetOriginShape();

    uint32_t n  = static_cast<uint32_t>(shape0.GetDim(0));
    uint32_t c1 = static_cast<uint32_t>(shape0.GetDim(1));

    tiling.set_c2hw(c2hw);
    tiling.set_c3hw(c3hw);
    tiling.set_c4hw(c4hw);
    tiling.set_outCsumHW(outCsumHW);
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "googlenet_inception_module_custom_tiling_data.h"

using DTYPE = float;
constexpr int32_t BUFFER_NUM = 2;

class KernelConcatC4 {
public:
    __aicore__ inline KernelConcatC4() {}
    __aicore__ inline void Init(GM_ADDR x1, GM_ADDR x2, GM_ADDR x3, GM_ADDR x4, GM_ADDR y,
                                uint32_t batch, uint32_t c1hw, uint32_t c2hw, uint32_t c3hw, uint32_t c4hw,
                                uint32_t outCsumHW, uint32_t tileNum) {
        this->batch = batch;
        this->c1hw = c1hw;
        this->c2hw = c2hw;
        this->c3hw = c3hw;
        this->c4hw = c4hw;
        this->outCsumHW = outCsumHW;
        this->tileNum = tileNum;

        uint32_t totalOutLen = batch * outCsumHW;
        this->blockLength = totalOutLen; // BLOCK_DIM=1
        this->tileLength = AscendC::max((uint32_t)1, this->blockLength / (tileNum * BUFFER_NUM));

        x1Gm.SetGlobalBuffer((__gm__ DTYPE *)x1, batch * c1hw);
        x2Gm.SetGlobalBuffer((__gm__ DTYPE *)x2, batch * c2hw);
        x3Gm.SetGlobalBuffer((__gm__ DTYPE *)x3, batch * c3hw);
        x4Gm.SetGlobalBuffer((__gm__ DTYPE *)x4, batch * c4hw);
        yGm.SetGlobalBuffer((__gm__ DTYPE *)y, totalOutLen);

        pipe.InitBuffer(inQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
    }

    __aicore__ inline void Process() {
        for (uint32_t b = 0; b < batch; ++b) {
            uint32_t yBase = b * outCsumHW;
            uint32_t off = 0;
            CopySegment(x1Gm, b * c1hw, yGm, yBase + off, c1hw);
            off += c1hw;
            CopySegment(x2Gm, b * c2hw, yGm, yBase + off, c2hw);
            off += c2hw;
            CopySegment(x3Gm, b * c3hw, yGm, yBase + off, c3hw);
            off += c3hw;
            CopySegment(x4Gm, b * c4hw, yGm, yBase + off, c4hw);
        }
    }

private:
    __aicore__ inline void CopySegment(AscendC::GlobalTensor<DTYPE>& srcGm, uint32_t srcBase,
                                       AscendC::GlobalTensor<DTYPE>& dstGm, uint32_t dstBase,
                                       uint32_t len) {
        uint32_t copied = 0;
        while (copied < len) {
            uint32_t cur = AscendC::min(this->tileLength, len - copied);
            AscendC::LocalTensor<DTYPE> buf = inQueue.AllocTensor<DTYPE>();
            AscendC::DataCopy(buf, srcGm[srcBase + copied], cur);
            inQueue.EnQue(buf);

            AscendC::LocalTensor<DTYPE> bufDeq = inQueue.DeQue<DTYPE>();
            AscendC::DataCopy(dstGm[dstBase + copied], bufDeq, cur);
            inQueue.FreeTensor(bufDeq);

            copied += cur;
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue;

    AscendC::GlobalTensor<DTYPE> x1Gm;
    AscendC::GlobalTensor<DTYPE> x2Gm;
    AscendC::GlobalTensor<DTYPE> x3Gm;
    AscendC::GlobalTensor<DTYPE> x4Gm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t batch;
    uint32_t c1hw, c2hw, c3hw, c4hw;
    uint32_t outCsumHW;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void googlenet_inception_module_custom(GM_ADDR x1, GM_ADDR x2, GM_ADDR x3, GM_ADDR x4,
                                                                        GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConcatC4 op;
    op.Init(x1, x2, x3, x4, y,
            tiling_data.batch, tiling_data.c1hw, tiling_data.c2hw, tiling_data.c3hw, tiling_data.c4hw,
            tiling_data.outCsumHW, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_1x1, reduce_3x3, out_3x3, reduce_5x5, out_5x5, pool_proj):
        super(ModelNew, self).__init__()
        self.branch1x1 = torch.nn.Conv2d(in_channels, out_1x1, kernel_size=1)
        self.branch3x3 = torch.nn.Sequential(
            torch.nn.Conv2d(in_channels, reduce_3x3, kernel_size=1),
            torch.nn.Conv2d(reduce_3x3, out_3x3, kernel_size=3, padding=1)
        )
        self.branch5x5 = torch.nn.Sequential(
            torch.nn.Conv2d(in_channels, reduce_5x5, kernel_size=1),
            torch.nn.Conv2d(reduce_5x5, out_5x5, kernel_size=5, padding=2)
        )
        self.branch_pool = torch.nn.Sequential(
            torch.nn.MaxPool2d(kernel_size=3, stride=1, padding=1),
            torch.nn.Conv2d(in_channels, pool_proj, kernel_size=1)
        )

    def forward(self, x):
        b1 = self.branch1x1(x)
        b3 = self.branch3x3(x)
        b5 = self.branch5x5(x)
        bp = self.branch_pool(x)
        return custom_ops_lib.googlenet_inception_module_custom(b1, b3, b5, bp)
'''
