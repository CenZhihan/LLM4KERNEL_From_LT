kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "gemm_batch_norm_gelu_group_norm_mean_relu_custom_tiling_data.h"

using DTYPE = float;

constexpr int32_t BUFFER_NUM = 2;

class KernelBNGeluRelu {
public:
    __aicore__ inline KernelBNGeluRelu() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mean, GM_ADDR var, GM_ADDR gamma, GM_ADDR beta, GM_ADDR eps, GM_ADDR z,
                                uint32_t totalLength, uint32_t featureSize)
    {
        this->featureSize = featureSize;
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileLength = featureSize; // process one row (feature vector) per tile

        xGm.SetGlobalBuffer((__gm__ DTYPE*)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        zGm.SetGlobalBuffer((__gm__ DTYPE*)z + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        meanGm.SetGlobalBuffer((__gm__ DTYPE*)mean, this->featureSize);
        varGm.SetGlobalBuffer((__gm__ DTYPE*)var, this->featureSize);
        gammaGm.SetGlobalBuffer((__gm__ DTYPE*)gamma, this->featureSize);
        betaGm.SetGlobalBuffer((__gm__ DTYPE*)beta, this->featureSize);
        epsGm.SetGlobalBuffer((__gm__ DTYPE*)eps, 1);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(paramQueueMean, 1, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(paramQueueVar, 1, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(paramQueueGamma, 1, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(paramQueueBeta, 1, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(paramQueueEps, 1, this->tileLength * sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        int32_t rowsPerBlock = this->blockLength / this->tileLength;
        // Preload parameters into local buffers once (broadcast eps)
        AscendC::LocalTensor<DTYPE> meanLocal = paramQueueMean.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> varLocal = paramQueueVar.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> gammaLocal = paramQueueGamma.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> betaLocal = paramQueueBeta.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> epsLocal = paramQueueEps.AllocTensor<DTYPE>();

        AscendC::DataCopy(meanLocal, meanGm, this->tileLength);
        AscendC::DataCopy(varLocal, varGm, this->tileLength);
        AscendC::DataCopy(gammaLocal, gammaGm, this->tileLength);
        AscendC::DataCopy(betaLocal, betaGm, this->tileLength);

        DTYPE epsScalar = 0.0f;
        AscendC::DataCopy(&epsScalar, epsGm, 1);
        AscendC::Dup(epsLocal, epsScalar, this->tileLength);

        paramQueueMean.EnQue(meanLocal);
        paramQueueVar.EnQue(varLocal);
        paramQueueGamma.EnQue(gammaLocal);
        paramQueueBeta.EnQue(betaLocal);
        paramQueueEps.EnQue(epsLocal);

        for (int32_t i = 0; i < rowsPerBlock * BUFFER_NUM; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> zLocal = outQueueZ.AllocTensor<DTYPE>();

        AscendC::LocalTensor<DTYPE> meanLocal = paramQueueMean.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> varLocal = paramQueueVar.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> gammaLocal = paramQueueGamma.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> betaLocal = paramQueueBeta.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> epsLocal = paramQueueEps.DeQue<DTYPE>();

        // tmp = ((x - mean) / sqrt(var + eps)) * gamma + beta
        AscendC::LocalTensor<DTYPE> t0 = pipe.AllocTmpBuffer<DTYPE>(this->tileLength); // x - mean
        AscendC::Sub(t0, xLocal, meanLocal, this->tileLength);
        AscendC::LocalTensor<DTYPE> t1 = pipe.AllocTmpBuffer<DTYPE>(this->tileLength); // var + eps
        AscendC::Add(t1, varLocal, epsLocal, this->tileLength);
        AscendC::Sqrt(t1, t1, this->tileLength);
        AscendC::Div(t0, t0, t1, this->tileLength); // (x-mean)/sqrt
        AscendC::Mul(t0, t0, gammaLocal, this->tileLength);
        AscendC::Add(t0, t0, betaLocal, this->tileLength);

        // GELU approximation: 0.5 * x * (1 + tanh(sqrt(2/pi)*(x + 0.044715*x^3)))
        // Reusing zLocal as output buffer
        const DTYPE c0 = 0.7978845608028654f; // sqrt(2/pi)
        const DTYPE c1 = 0.044715f;
        AscendC::LocalTensor<DTYPE> x2 = pipe.AllocTmpBuffer<DTYPE>(this->tileLength);
        AscendC::Mul(x2, t0, t0, this->tileLength); // x^2
        AscendC::LocalTensor<DTYPE> x3 = pipe.AllocTmpBuffer<DTYPE>(this->tileLength);
        AscendC::Mul(x3, x2, t0, this->tileLength); // x^3

        AscendC::LocalTensor<DTYPE> t_poly = pipe.AllocTmpBuffer<DTYPE>(this->tileLength); // x + c1*x^3
        AscendC::Muls(t_poly, x3, c1, this->tileLength);
        AscendC::Add(t_poly, t_poly, t0, this->tileLength);
        AscendC::Muls(t_poly, t_poly, c0, this->tileLength);

        AscendC::Tanh(t_poly, t_poly, this->tileLength);

        AscendC::LocalTensor<DTYPE> one = pipe.AllocTmpBuffer<DTYPE>(this->tileLength);
        AscendC::Dup(one, 1.0f, this->tileLength);
        AscendC::Add(t_poly, t_poly, one, this->tileLength); // (1 + tanh(...))
        AscendC::Mul(zLocal, t0, t_poly, this->tileLength);  // x * (...)
        AscendC::Muls(zLocal, zLocal, 0.5f, this->tileLength);

        // ReLU
        AscendC::LocalTensor<DTYPE> zero = pipe.AllocTmpBuffer<DTYPE>(this->tileLength);
        AscendC::Dup(zero, 0.0f, this->tileLength);
        AscendC::Maximum(zLocal, zLocal, zero, this->tileLength);

        outQueueZ.EnQue<DTYPE>(zLocal);
        inQueueX.FreeTensor(xLocal);

        // Re-enqueue parameters for next tile
        paramQueueMean.EnQue(meanLocal);
        paramQueueVar.EnQue(varLocal);
        paramQueueGamma.EnQue(gammaLocal);
        paramQueueBeta.EnQue(betaLocal);
        paramQueueEps.EnQue(epsLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE> zLocal = outQueueZ.DeQue<DTYPE>();
        AscendC::DataCopy(zGm[progress * this->tileLength], zLocal, this->tileLength);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::TQue<AscendC::TPosition::VECIN, 1> paramQueueMean, paramQueueVar, paramQueueGamma, paramQueueBeta, paramQueueEps;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> zGm;

    AscendC::GlobalTensor<DTYPE> meanGm;
    AscendC::GlobalTensor<DTYPE> varGm;
    AscendC::GlobalTensor<DTYPE> gammaGm;
    AscendC::GlobalTensor<DTYPE> betaGm;
    AscendC::GlobalTensor<DTYPE> epsGm;

    uint32_t blockLength;
    uint32_t featureSize;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void gemm_batch_norm_gelu_group_norm_mean_relu_custom(
    GM_ADDR x, GM_ADDR running_mean, GM_ADDR running_var, GM_ADDR gamma, GM_ADDR beta, GM_ADDR eps, GM_ADDR z,
    GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelBNGeluRelu op;
    op.Init(x, running_mean, running_var, gamma, beta, eps, z, tiling_data.totalLength, tiling_data.featureSize);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused model: Linear (GEMM) followed by a custom AscendC op applying BatchNorm (eval), GELU and ReLU.
    """
    def __init__(self, in_features, out_features):
        super(ModelNew, self).__init__()
        self.gemm = torch.nn.Linear(in_features, out_features)
        self.batch_norm = torch.nn.BatchNorm1d(out_features)

    def forward(self, x):
        # GEMM
        x = self.gemm(x)

        # Prepare BN params (use running stats and affine params)
        rm = self.batch_norm.running_mean
        rv = self.batch_norm.running_var
        gamma = self.batch_norm.weight
        beta = self.batch_norm.bias
        eps_tensor = torch.tensor([self.batch_norm.eps], dtype=x.dtype, device=x.device)

        # Fused BN + GELU + ReLU
        x = custom_ops_lib.gemm_batch_norm_gelu_group_norm_mean_relu_custom(x, rm, rv, gamma, beta, eps_tensor)
        return x
