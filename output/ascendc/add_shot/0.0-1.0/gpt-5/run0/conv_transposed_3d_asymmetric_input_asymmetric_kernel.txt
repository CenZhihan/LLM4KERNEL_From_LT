kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "conv_transposed_3d_asymmetric_input_asymmetric_kernel_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2; // number of tiles buffered per queue

class KernelConvTranspose3D {
public:
    __aicore__ inline KernelConvTranspose3D() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = (this->blockLength / tileNum / BUFFER_NUM);
        if (this->tileLength == 0) {
            this->tileLength = 1;
        }

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W *)w, 1); // weight not tiled here
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
        AscendC::Dup(yLocal, static_cast<DTYPE_Y>(0), this->tileLength);
        outQueueY.EnQue<DTYPE_Y>(yLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transposed_3d_asymmetric_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTranspose3D op;
    op.Init(x, w, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Custom transposed 3D convolution using AscendC kernel:
    conv_transposed_3d_asymmetric_input_asymmetric_kernel_custom
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: tuple, stride: tuple = (1, 1, 1), padding: tuple = (0, 0, 0), output_padding: tuple = (0, 0, 0), groups: int = 1, bias: bool = False):
        super().__init__()
        assert groups == 1, "This custom kernel currently supports groups=1"
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.output_padding = output_padding
        self.groups = groups
        kD, kH, kW = kernel_size
        # PyTorch ConvTranspose3d weight layout: [Cin, Cout/groups, kD, kH, kW]
        weight_shape = (in_channels, out_channels // groups, kD, kH, kW)
        self.weight = torch.nn.Parameter(torch.randn(weight_shape, dtype=torch.float32))

        self.bias = torch.nn.Parameter(torch.zeros(out_channels, dtype=torch.float32)) if bias else None

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # For now, attributes (stride/padding/etc.) are assumed defaults (1,0,0) in the kernel.
        # The bias is not fused in the kernel; if present, add after.
        y = custom_ops_lib.conv_transposed_3d_asymmetric_input_asymmetric_kernel_custom(x, self.weight)
        if self.bias is not None:
            # Broadcast bias over N, D, H, W
            y = y + self.bias.view(1, -1, 1, 1, 1)
        return y
