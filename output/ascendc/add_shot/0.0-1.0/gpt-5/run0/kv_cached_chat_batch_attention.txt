kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "kv_cached_chat_batch_attention_custom_tiling_data.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelKvCachedChatBatchAttention {
public:
    __aicore__ inline KernelKvCachedChatBatchAttention() {}

    __aicore__ inline void Init(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR o,
                                uint32_t batch, uint32_t qLen, uint32_t kvLen, uint32_t dModel,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->batch = batch;
        this->qLen = qLen;
        this->kvLen = kvLen;
        this->dModel = dModel;
        this->totalLength = totalLength;
        this->tileNum = tileNum;

        // Split along contiguous (batch * q_len * d_model)
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        qGm.SetGlobalBuffer((__gm__ float *)q + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        kGm.SetGlobalBuffer((__gm__ float *)k, batch * kvLen * dModel);
        vGm.SetGlobalBuffer((__gm__ float *)v, batch * kvLen * dModel);
        oGm.SetGlobalBuffer((__gm__ float *)o + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueQ, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(outQueueO, BUFFER_NUM, this->tileLength * sizeof(float));

        // Optional scratch queues for further optimization can be added here
    }

    __aicore__ inline void Process()
    {
        // For demonstration: stream copy Q to O tile by tile.
        // In a full implementation, replace with scaled dot-product attention over K/V caches.
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<float> qLocal = inQueueQ.AllocTensor<float>();
        AscendC::DataCopy(qLocal, qGm[progress * this->tileLength], this->tileLength);
        inQueueQ.EnQue(qLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<float> qLocal = inQueueQ.DeQue<float>();
        AscendC::LocalTensor<float> oLocal = outQueueO.AllocTensor<float>();

        // Placeholder compute: pass-through Q -> O for the current tile.
        AscendC::DataCopy(oLocal, qLocal, this->tileLength);

        outQueueO.EnQue<float>(oLocal);
        inQueueQ.FreeTensor(qLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<float> oLocal = outQueueO.DeQue<float>();
        AscendC::DataCopy(oGm[progress * this->tileLength], oLocal, this->tileLength);
        outQueueO.FreeTensor(oLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueQ;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueO;

    AscendC::GlobalTensor<float> qGm;
    AscendC::GlobalTensor<float> kGm;
    AscendC::GlobalTensor<float> vGm;
    AscendC::GlobalTensor<float> oGm;

    uint32_t batch;
    uint32_t qLen;
    uint32_t kvLen;
    uint32_t dModel;
    uint32_t totalLength;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void kv_cached_chat_batch_attention_custom(GM_ADDR q, GM_ADDR k_cache, GM_ADDR v_cache, GM_ADDR o, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelKvCachedChatBatchAttention op;
    op.Init(q, k_cache, v_cache, o,
            tiling_data.batch, tiling_data.qLen, tiling_data.kvLen, tiling_data.dModel,
            tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}
