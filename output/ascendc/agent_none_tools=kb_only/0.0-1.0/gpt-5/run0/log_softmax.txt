# AscendC AICore设备端Kernel源码（六个字符串之三）
aicore_kernel_src = r"""
#include "kernel_operator.h"
#include <cfloat>

using namespace AscendC;

// 简化假设：输入为float（FP32），按行做log_softmax，dim=1。
// Kernel参数：xGm: 输入，yGm: 输出，rows: N，cols: D
class KernelLogSoftmax {
public:
  __aicore__ inline KernelLogSoftmax() {}
  __aicore__ inline void Init(__gm__ uint8_t* xGm, __gm__ uint8_t* yGm,
                              int64_t rows, int64_t cols) {
    xGlobal.SetGlobalBuffer((__gm__ float*)xGm);
    yGlobal.SetGlobalBuffer((__gm__ float*)yGm);
    rows_ = rows;
    cols_ = cols;

    // 以512为分块大小（示例），可根据性能调优
    tile_ = 512;
    pipe.InitBuffer(inQueueX, 1, tile_ * sizeof(float));
    pipe.InitBuffer(outQueueY, 1, tile_ * sizeof(float));
    pipe.InitBuffer(tmpQueueA, 1, tile_ * sizeof(float));
    pipe.InitBuffer(tmpQueueB, 1, tile_ * sizeof(float));
  }

  __aicore__ inline void Process() {
    // 按行处理
    for (int64_t r = 0; r < rows_; ++r) {
      // Pass 1: 计算每行的最大值 row_max
      float row_max = -FLT_MAX;
      for (int64_t c = 0; c < cols_; c += tile_) {
        int64_t len = (c + tile_ <= cols_) ? tile_ : (cols_ - c);
        LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        DataCopy(xLocal, xGlobal[r * cols_ + c], len);
        inQueueX.EnQue(xLocal);

        LocalTensor<float> xL = inQueueX.DeQue<float>();
        // 假设存在ReduceMax接口（不同版本API名字可能不同，示例用法）
        // 若需手写reduce，可在local上做分段归约
        float block_max = ReduceMax(xL, len);
        row_max = (row_max > block_max) ? row_max : block_max;
        inQueueX.FreeTensor(xL);
      }

      // Pass 2: 计算 sum(exp(x - row_max))
      float sum_exp = 0.f;
      for (int64_t c = 0; c < cols_; c += tile_) {
        int64_t len = (c + tile_ <= cols_) ? tile_ : (cols_ - c);
        LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        LocalTensor<float> shiftedLocal = tmpQueueA.AllocTensor<float>();
        LocalTensor<float> expLocal = tmpQueueB.AllocTensor<float>();

        DataCopy(xLocal, xGlobal[r * cols_ + c], len);
        // shifted = x - row_max
        Sub(shiftedLocal, xLocal, static_cast<float>(row_max), len);
        // e = exp(shifted)
        Exp(expLocal, shiftedLocal, len);

        // 归约求和
        float block_sum = ReduceSum(expLocal, len);
        sum_exp += block_sum;

        tmpQueueB.FreeTensor(expLocal);
        tmpQueueA.FreeTensor(shiftedLocal);
        inQueueX.FreeTensor(xLocal);
      }
      float log_sum_exp = Log(sum_exp);

      // Pass 3: 写回 y = x - row_max - log_sum_exp
      for (int64_t c = 0; c < cols_; c += tile_) {
        int64_t len = (c + tile_ <= cols_) ? tile_ : (cols_ - c);
        LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();
        LocalTensor<float> shiftedLocal = tmpQueueA.AllocTensor<float>();

        DataCopy(xLocal, xGlobal[r * cols_ + c], len);
        // shifted = x - row_max
        Sub(shiftedLocal, xLocal, static_cast<float>(row_max), len);
        // y = shifted - log_sum_exp
        Sub(yLocal, shiftedLocal, static_cast<float>(log_sum_exp), len);

        DataCopy(yGlobal[r * cols_ + c], yLocal, len);

        tmpQueueA.FreeTensor(shiftedLocal);
        outQueueY.FreeTensor(yLocal);
        inQueueX.FreeTensor(xLocal);
      }
    }
  }

private:
  // 简化的本地归约示例接口（如无现成API可手工实现循环，以下为占位/演示）
  __aicore__ inline float ReduceMax(LocalTensor<float>& t, int64_t len) {
    // 占位：实际应使用平台提供的reduce指令或手写循环
    float m = -FLT_MAX;
    for (int i = 0; i < len; ++i) {
      float v = t.GetValue(i);
      m = v > m ? v : m;
    }
    return m;
  }
  __aicore__ inline float ReduceSum(LocalTensor<float>& t, int64_t len) {
    // 占位：实际应使用平台提供的reduce指令或手写循环
    float s = 0.f;
    for (int i = 0; i < len; ++i) {
      s += t.GetValue(i);
    }
    return s;
  }
  __aicore__ inline float Log(float x) {
    // 占位：实际应使用平台提供的Log向量/标量API
    // 这里为演示，真实实现应替换为对应AscendC接口
    return __logf(x);
  }

private:
  TPipe pipe;
  TQue<QuePosition::VECIN, 1> inQueueX;
  TQue<QuePosition::VECOUT, 1> outQueueY;
  TQue<QuePosition::VECIN, 1> tmpQueueA;
  TQue<QuePosition::VECIN, 1> tmpQueueB;

  GlobalTensor<float> xGlobal, yGlobal;
  int64_t rows_, cols_;
  int64_t tile_;
};

extern "C" __global__ __aicore__ void log_softmax_custom_kernel(
    __gm__ uint8_t* xGm, __gm__ uint8_t* yGm, int64_t rows, int64_t cols) {
  KernelLogSoftmax op;
  op.Init(xGm, yGm, rows, cols);
  op.Process();
}
"""

host_op_src = r"""
#include "register/op_def_registry.h"
#include "register/op_impl_registry.h"
#include "graph/tensor.h"

constexpr const char* kOpType = "LogSoftmaxCustom";

namespace logsoftmax_custom_impl {

static ge::graphStatus InferShape(gert::InferShapeContext* ctx) {
  // 输出形状 = 输入形状
  auto* xShape = ctx->GetInputShape(0);
  auto* yShape = ctx->GetOutputShape(0);
  *yShape = *xShape;
  return ge::GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext* ctx) {
  auto x_dtype = ctx->GetInputDataType(0);
  ctx->SetOutputDataType(0, x_dtype);
  return ge::GRAPH_SUCCESS;
}

static ge::graphStatus Compute(gert::KernelContext* ctx) {
  auto x = ctx->GetInputTensor(0);
  auto y = ctx->GetOutputTensor(0);

  const int64_t rows = x->GetShape().GetDim(0);
  const int64_t cols = x->GetShape().GetDim(1);

  __gm__ uint8_t* xGm = reinterpret_cast<__gm__ uint8_t*>(x->GetData());
  __gm__ uint8_t* yGm = reinterpret_cast<__gm__ uint8_t*>(y->GetData());

  // 启动自定义Kernel
  // gridDim按行并行，blockDim由编译器/平台决定，以下为占位
  dim3 grid(rows, 1, 1);
  dim3 block(1, 1, 1);
  log_softmax_custom_kernel<<<grid, block>>>(xGm, yGm, rows, cols);
  return ge::GRAPH_SUCCESS;
}

} // namespace logsoftmax_custom_impl

using namespace logsoftmax_custom_impl;

REGISTER_OP_DEF(kOpType)
  .Input("x")
  .Output("y")
  .SetInferShape(InferShape)
  .SetInferDataType(InferDataType);

REGISTER_OP_IMPL(kOpType, logsoftmax_custom_impl, Compute);
