import os
import torch
import torch.nn as nn

# ===== Six required Python strings =====
CUSTOM_KERNEL_NAME_SNAKE = "swish_custom"
CUSTOM_KERNEL_NAME_PASCAL = "SwishCustom"
CUSTOM_OP_LIB_NAME = "libswish_custom.so"
CUSTOM_OP_NAMESPACE = "ascendc_custom"
CUSTOM_OP_FUNC_NAME = "swish_custom"
CUSTOM_OP_FULL_NAME = f"{CUSTOM_OP_NAMESPACE}::{CUSTOM_OP_FUNC_NAME}"

# ===== Helper: load custom op library if available =====
def _try_load_custom_op_library():
    candidates = [
        os.environ.get("ASCENDC_CUSTOM_OP_SO", ""),
        os.path.join(os.getcwd(), CUSTOM_OP_LIB_NAME),
        CUSTOM_OP_LIB_NAME,
    ]
    for p in candidates:
        if not p:
            continue
        try:
            if os.path.exists(p) or os.path.basename(p) == p:
                torch.ops.load_library(p)
                return True
        except Exception:
            continue
    return False

def _has_custom_op():
    try:
        ns = getattr(torch.ops, CUSTOM_OP_NAMESPACE)
        getattr(ns, CUSTOM_OP_FUNC_NAME)
        return True
    except AttributeError:
        return False

_CUSTOM_OP_AVAILABLE = _has_custom_op() or _try_load_custom_op_library() or _has_custom_op()

class SwishCustomFunc(torch.autograd.Function):
    @staticmethod
    def forward(ctx, x: torch.Tensor) -> torch.Tensor:
        if _CUSTOM_OP_AVAILABLE:
            y = getattr(getattr(torch.ops, CUSTOM_OP_NAMESPACE), CUSTOM_OP_FUNC_NAME)(x)
        else:
            y = x * torch.sigmoid(x)
        # Save input for backward to avoid extra forward-time ops
        ctx.save_for_backward(x)
        return y

    @staticmethod
    def backward(ctx, grad_output: torch.Tensor):
        (x,) = ctx.saved_tensors
        # Swish' = sigmoid(x) + x * sigmoid(x) * (1 - sigmoid(x))
        sig = torch.sigmoid(x)
        grad = grad_output * (sig + x * sig * (1.0 - sig))
        return grad

class ModelNew(nn.Module):
    """
    Optimized model that applies Swish via custom AscendC kernel when available.
    Falls back to PyTorch ops if the custom kernel is not found.
    """
    def __init__(self):
        super(ModelNew, self).__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return SwishCustomFunc.apply(x)
