// ReluCustom AscendC kernel: element-wise ReLU
// Uses Basic API Relu(dst, src, calCount)
// Reference: CANN Ascend C API Reference - 4.1.1.8 Relu
#include "kernel_operator.h"

using namespace AscendC;

template <typename T>
class KernelReluCustom {
public:
  __aicore__ inline KernelReluCustom() {}

  __aicore__ inline void Init(GM_ADDR xGm, GM_ADDR yGm, uint32_t elemCount) {
    dataSize = elemCount;
    xGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ T *>(xGm), dataSize);
    yGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ T *>(yGm), dataSize);

    // one-shot full tensor process
    pipe.InitBuffer(inQueue, 1, dataSize * sizeof(T));
    pipe.InitBuffer(outQueue, 1, dataSize * sizeof(T));
  }

  __aicore__ inline void Process() {
    CopyIn();
    Compute();
    CopyOut();
  }

private:
  __aicore__ inline void CopyIn() {
    LocalTensor<T> xLocal = inQueue.AllocTensor<T>();
    DataCopy(xLocal, xGlobal, dataSize);
    inQueue.EnQue(xLocal);
  }

  __aicore__ inline void Compute() {
    LocalTensor<T> yLocal = outQueue.AllocTensor<T>();
    LocalTensor<T> xLocal = inQueue.DeQue<T>();

    // ReLU: y = max(x, 0)
    Relu<T>(yLocal, xLocal, static_cast<int32_t>(dataSize));

    outQueue.EnQue<T>(yLocal);
    inQueue.FreeTensor(xLocal);
  }

  __aicore__ inline void CopyOut() {
    LocalTensor<T> yLocal = outQueue.DeQue<T>();
    DataCopy(yGlobal, yLocal, dataSize);
    outQueue.FreeTensor(yLocal);
  }

private:
  GlobalTensor<T> xGlobal;
  GlobalTensor<T> yGlobal;
  TPipe pipe;
  TQue<QuePosition::VECIN, 1> inQueue;
  TQue<QuePosition::VECOUT, 1> outQueue;
  uint32_t dataSize = 0;
};

// Kernel entry for half precision (recommended by Relu Basic API on training products)
extern "C" __global__ __aicore__
void ReluCustom(GM_ADDR x, GM_ADDR y, uint32_t elemCount) {
  KernelReluCustom<half> op;
  op.Init(x, y, elemCount);
  op.Process();
}
"""
