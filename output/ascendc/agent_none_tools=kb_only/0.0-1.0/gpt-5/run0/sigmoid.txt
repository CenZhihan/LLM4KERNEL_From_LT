# Try to load a prebuilt custom op library if present (optional)
_CUSTOM_OP_LIB_LOADED = False
for _libname in ("libsigmoid_custom.so", "sigmoid_custom.so"):
    try:
        torch.ops.load_library(_libname)
        _CUSTOM_OP_LIB_LOADED = True
        break
    except OSError:
        pass

def _custom_op_available():
    try:
        ns = getattr(torch.ops, "ascend_custom")
        _ = getattr(ns, "sigmoid_custom")
        return True
    except Exception:
        return False

class SigmoidCustomFunction(torch.autograd.Function):
    @staticmethod
    def forward(ctx, x: torch.Tensor) -> torch.Tensor:
        use_custom = _custom_op_available()
        y = None
        if use_custom:
            try:
                y = torch.ops.ascend_custom.sigmoid_custom(x)
            except Exception:
                y = None
        if y is None:
            y = torch.sigmoid(x)
        ctx.save_for_backward(y)
        return y

    @staticmethod
    def backward(ctx, grad_output: torch.Tensor):
        (y,) = ctx.saved_tensors
        grad_input = grad_output * y * (1.0 - y)
        return grad_input

class SigmoidCustomModule(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return SigmoidCustomFunction.apply(x)

class Model(nn.Module):
    """
    Simple model that performs a Sigmoid activation.
    """
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies Sigmoid activation to the input tensor.

        Args:
            x (torch.Tensor): Input tensor of any shape.

        Returns:
            torch.Tensor: Output tensor with Sigmoid applied, same shape as input.
        """
        return torch.sigmoid(x)

class ModelNew(nn.Module):
    """
    Optimized model that replaces torch.sigmoid with custom AscendC kernel call.
    """
    def __init__(self):
        super(ModelNew, self).__init__()
        self.sigmoid = SigmoidCustomModule()
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return self.sigmoid(x)
