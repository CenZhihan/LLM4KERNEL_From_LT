# AscendC 内核源码字符串（设备核）。实际编译与加载由外部流水线完成。
ASCENDC_KERNEL_SOURCE = r"""
// HardsigmoidCustom AscendC Kernel (设备核)
#include "kernel_operator.h"

// 计算公式: y = clip((x + 3) / 6, 0, 1)
template <typename T>
class KernelHardsigmoid {
public:
    __aicore__ inline KernelHardsigmoid() {}

    __aicore__ inline void Init(GM_ADDR srcGm, GM_ADDR dstGm, uint32_t elemCount)
    {
        count = elemCount;
        srcGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ T *>(srcGm), count);
        dstGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ T *>(dstGm), count);

        // 简化：单块搬运
        pipe.InitBuffer(inQueue, 1, count * sizeof(T));
        pipe.InitBuffer(outQueue, 1, count * sizeof(T));
    }

    __aicore__ inline void Process()
    {
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        AscendC::LocalTensor<T> xLocal = inQueue.AllocTensor<T>();
        AscendC::DataCopy(xLocal, srcGlobal, count);
        inQueue.EnQue(xLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<T> xLocal = inQueue.DeQue<T>();
        AscendC::LocalTensor<T> yLocal = outQueue.AllocTensor<T>();

        // y = (x + 3) / 6
        // 注意：以下使用 Adds/Muls/Divs 等标量广播算子名称在不同 CANN 版本可能略有差异，
        // 若无对应标量 API，可改为构造常量向量后用 Add/Div 等逐元素算子实现。
        AscendC::LocalTensor<T> tLocal = outQueue.AllocTensor<T>();

        // t = x + 3
        AscendC::Adds(tLocal, xLocal, static_cast<T>(3.0f), count);
        // y = t / 6
        AscendC::Divs(yLocal, tLocal, static_cast<T>(6.0f), count);
        // y = min(y, 1)
        AscendC::Mins(yLocal, yLocal, static_cast<T>(1.0f), count);
        // y = max(y, 0)
        AscendC::Maxs(yLocal, yLocal, static_cast<T>(0.0f), count);

        outQueue.EnQue<T>(yLocal);
        inQueue.FreeTensor(xLocal);
        outQueue.FreeTensor(tLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<T> yLocal = outQueue.DeQue<T>();
        AscendC::DataCopy(dstGlobal, yLocal, count);
        outQueue.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueue;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueue;
    AscendC::GlobalTensor<T> srcGlobal;
    AscendC::GlobalTensor<T> dstGlobal;
    uint32_t count = 0;
};

extern "C" __global__ __aicore__ void hardsigmoid_custom_kernel(GM_ADDR x, GM_ADDR y, uint32_t elemCount)
{
    // 这里以 half 为主要数据类型，可按实际需要模板实例化 float/bfloat16 等
    KernelHardsigmoid<half> op;
    op.Init(x, y, elemCount);
    op.Process();
}
"""
