ASCENDC_DEVICE_SRC = r"""
// File: selu_custom_kernel.cu (AscendC)
#include "kernel_operator.h"

// SELU constants (IEEE)
template <typename T>
__aicore__ inline T SeluAlpha();
template <>
__aicore__ inline float SeluAlpha<float>() { return 1.6732632423543772848170429916717f; }
template <>
__aicore__ inline half SeluAlpha<half>() { return (half)1.6732632423543772f; }

template <typename T>
__aicore__ inline T SeluScale();
template <>
__aicore__ inline float SeluScale<float>() { return 1.0507009873554804934193349852946f; }
template <>
__aicore__ inline half SeluScale<half>() { return (half)1.05070098735548f; }

// Tile params
constexpr int32_t BUFFER_NUM = 2;

template <typename T>
class KernelSelu {
public:
  __aicore__ inline KernelSelu() {}
  __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t totalLength, uint32_t tileNum) {
    ASSERT(GetBlockNum() != 0 && "block dim can not be zero!");
    this->blockLength = totalLength / AscendC::GetBlockNum();
    this->tileNum = tileNum;
    ASSERT(tileNum != 0 && "tile num can not be zero!");
    this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

    xGm.SetGlobalBuffer((__gm__ T*)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
    yGm.SetGlobalBuffer((__gm__ T*)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

    pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(T));
    pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(T));
    pipe.InitBuffer(tmpQ0, BUFFER_NUM, this->tileLength * sizeof(T));
    pipe.InitBuffer(tmpQ1, BUFFER_NUM, this->tileLength * sizeof(T));
    pipe.InitBuffer(tmpQ2, BUFFER_NUM, this->tileLength * sizeof(T));
  }

  __aicore__ inline void Process(T alpha, T scale) {
    int32_t loopCount = this->tileNum * BUFFER_NUM;
    for (int32_t i = 0; i < loopCount; i++) {
      CopyIn(i);
      Compute(i, alpha, scale);
      CopyOut(i);
    }
  }

private:
  __aicore__ inline void CopyIn(int32_t progress) {
    AscendC::LocalTensor<T> xLocal = inQueueX.AllocTensor<T>();
    AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
    inQueueX.EnQue(xLocal);
  }

  __aicore__ inline void Compute(int32_t /*progress*/, T alpha, T scale) {
    // y = scale * (max(x,0) + alpha * (exp(min(x,0)) - 1))
    AscendC::LocalTensor<T> xLocal = inQueueX.DeQue<T>();
    AscendC::LocalTensor<T> yLocal = outQueueY.AllocTensor<T>();

    AscendC::LocalTensor<T> zeroLocal = tmpQ0.AllocTensor<T>();
    AscendC::LocalTensor<T> posLocal  = tmpQ1.AllocTensor<T>();
    AscendC::LocalTensor<T> negLocal  = tmpQ2.AllocTensor<T>();

    // zeroLocal = x - x
    AscendC::Sub(zeroLocal, xLocal, xLocal, this->tileLength);
    // posLocal = max(x, 0)
    AscendC::Maximum(posLocal, xLocal, zeroLocal, this->tileLength);
    // negLocal = min(x, 0)
    AscendC::Minimum(negLocal, xLocal, zeroLocal, this->tileLength);

    // reuse zeroLocal as tmp: tmp = exp(negLocal)
    AscendC::Exp(zeroLocal, negLocal, this->tileLength);
    // zeroLocal = exp(neg) - 1
    AscendC::Adds(zeroLocal, zeroLocal, (T)-1, this->tileLength);
    // zeroLocal = alpha * (exp(neg) - 1)
    AscendC::Muls(zeroLocal, zeroLocal, alpha, this->tileLength);

    // yLocal = posLocal + zeroLocal
    AscendC::Add(yLocal, posLocal, zeroLocal, this->tileLength);
    // yLocal = scale * yLocal
    AscendC::Muls(yLocal, yLocal, scale, this->tileLength);

    outQueueY.EnQue<T>(yLocal);
    inQueueX.FreeTensor(xLocal);
    tmpQ0.FreeTensor(zeroLocal);
    tmpQ1.FreeTensor(posLocal);
    tmpQ2.FreeTensor(negLocal);
  }

  __aicore__ inline void CopyOut(int32_t progress) {
    AscendC::LocalTensor<T> yLocal = outQueueY.DeQue<T>();
    AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
    outQueueY.FreeTensor(yLocal);
  }

private:
  AscendC::TPipe pipe;
  AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueX;
  AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueY;
  AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> tmpQ0;
  AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> tmpQ1;
  AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> tmpQ2;

  AscendC::GlobalTensor<T> xGm;
  AscendC::GlobalTensor<T> yGm;

  uint32_t blockLength;
  uint32_t tileNum;
  uint32_t tileLength;
};

// Kernel entry for float32
extern "C" __global__ __aicore__ void selu_custom_kernel_fp32(GM_ADDR x, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
{
  KernelSelu<float> op;
  op.Init(x, y, totalLength, tileNum);
  op.Process(SeluAlpha<float>(), SeluScale<float>());
}

// Kernel entry for fp16
extern "C" __global__ __aicore__ void selu_custom_kernel_fp16(GM_ADDR x, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
{
  KernelSelu<half> op;
  op.Init(x, y, totalLength, tileNum);
  op.Process(SeluAlpha<half>(), SeluScale<half>());
}
"""
