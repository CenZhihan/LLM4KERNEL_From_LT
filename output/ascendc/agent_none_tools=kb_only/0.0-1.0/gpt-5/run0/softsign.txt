// ==============================
// 3) Kernel implementation
// ==============================
template <typename T>
class KernelSoftsign {
public:
  __aicore__ inline KernelSoftsign() {}

  __aicore__ inline void Init(GM_ADDR xGm, GM_ADDR yGm, uint32_t totalLengthIn)
  {
    totalLength = totalLengthIn;
    xGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ T*>(xGm), totalLength);
    yGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ T*>(yGm), totalLength);

    // Allocate single-shot UB buffers
    pipe.InitBuffer(inQueue, 1, totalLength * sizeof(T));
    pipe.InitBuffer(outQueue, 1, totalLength * sizeof(T));
    // two temporary UB buffers for |x| and (1+|x|)
    pipe.InitBuffer(tmpQueue, 2, totalLength * sizeof(T));
  }

  __aicore__ inline void Process()
  {
    CopyIn();
    Compute();
    CopyOut();
  }

private:
  __aicore__ inline void CopyIn()
  {
    xLocal = inQueue.AllocTensor<T>();
    DataCopy(xLocal, xGlobal, totalLength);
    inQueue.EnQue(xLocal);
  }

  __aicore__ inline void Compute()
  {
    auto yLocal = outQueue.AllocTensor<T>();
    auto xLocalIn = inQueue.DeQue<T>();

    // tmp buffers
    auto absLocal   = tmpQueue.AllocTensor<T>();
    auto denomLocal = tmpQueue.AllocTensor<T>();

    // abs(x)
    Abs(absLocal, xLocalIn, totalLength);

    // denom = abs(x) + 1
    Adds(denomLocal, absLocal, (T)1, totalLength);

    // y = x / denom
    Div(yLocal, xLocalIn, denomLocal, totalLength);

    // enqueue and free
    outQueue.EnQue<T>(yLocal);
    inQueue.FreeTensor(xLocalIn);
    tmpQueue.FreeTensor(absLocal);
    tmpQueue.FreeTensor(denomLocal);
  }

  __aicore__ inline void CopyOut()
  {
    auto yLocal = outQueue.DeQue<T>();
    DataCopy(yGlobal, yLocal, totalLength);
    outQueue.FreeTensor(yLocal);
  }

private:
  AscendC::TPipe pipe;
  AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueue;
  AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueue;
  AscendC::TQue<AscendC::QuePosition::VECCALC, 2> tmpQueue;

  AscendC::GlobalTensor<T> xGlobal;
  AscendC::GlobalTensor<T> yGlobal;

  AscendC::LocalTensor<T> xLocal;

  uint32_t totalLength = 0;
};

// Entry
extern "C" __global__ __aicore__ void softsign_custom(GM_ADDR xGm, GM_ADDR yGm, GM_ADDR workspace, GM_ADDR tiling)
{

  uint32_t totalLength = tilingData.totalLength;

  KernelSoftsign<half> op;
  op.Init(xGm, yGm, totalLength);
  if (TILING_KEY_IS(1)) {
    op.Process();
  }
}
