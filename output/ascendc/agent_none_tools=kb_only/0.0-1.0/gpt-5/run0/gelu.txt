AICORE_KERNEL_SOURCE = r"""
// gelu_custom_kernel.cpp
#include "kernel_operator.h"

namespace GeluCustomKernel {

struct GeluCustomTiling {
  uint32_t totalLength = 0;
};

// 简单对齐到32B块
inline __aicore__ uint32_t AlignToBlock(const uint32_t inputValue, const uint32_t typeSize)
{
  constexpr uint32_t ONE_BLK_SIZE = 32;
  uint32_t alignUnit = ONE_BLK_SIZE / typeSize;
  return (inputValue + alignUnit - 1) / alignUnit * alignUnit;
}

template <typename T>
class KernelGeluCustom {
public:
  __aicore__ inline KernelGeluCustom() {}

  __aicore__ inline void Init(GM_ADDR xGm, GM_ADDR yGm, const GeluCustomTiling &tilingData)
  {
    totalLength_ = tilingData.totalLength;
    inputGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ T *>(xGm), totalLength_);
    outputGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ T *>(yGm), totalLength_);

    constexpr uint32_t typeSize = sizeof(T);
    const uint32_t alignedLen = AlignToBlock(totalLength_, typeSize);

    pipe_.InitBuffer(inQueue_, 1, alignedLen * typeSize);
    pipe_.InitBuffer(outQueue_, 1, alignedLen * typeSize);
    pipe_.InitBuffer(tmpQueue_, 1, alignedLen * typeSize * 6); // 临时中间张量
    alignedLen_ = alignedLen;
  }

  __aicore__ inline void Process()
  {
    CopyIn();
    Compute();
    CopyOut();
  }

private:
  __aicore__ inline void CopyIn()
  {
    AscendC::LocalTensor<T> inputLocal = inQueue_.AllocTensor<T>();
    AscendC::DataCopy(inputLocal, inputGlobal_, totalLength_);
    inQueue_.EnQue(inputLocal);
  }

  __aicore__ inline void Compute()
  {
    // 近似 GELU：0.5 * x * (1 + tanh(sqrt(2/pi)*(x + 0.044715*x^3)))
    AscendC::LocalTensor<T> x = inQueue_.DeQue<T>();
    AscendC::LocalTensor<T> y = outQueue_.AllocTensor<T>();

    // 中间张量
    AscendC::LocalTensor<T> x2 = tmpQueue_.AllocTensor<T>();
    AscendC::LocalTensor<T> x3 = tmpQueue_.AllocTensor<T>();
    AscendC::LocalTensor<T> t_add = tmpQueue_.AllocTensor<T>();
    AscendC::LocalTensor<T> t_scaled = tmpQueue_.AllocTensor<T>();
    AscendC::LocalTensor<T> t_tanh = tmpQueue_.AllocTensor<T>();
    AscendC::LocalTensor<T> t_half = tmpQueue_.AllocTensor<T>();

    const T kC0 = static_cast<T>(0.044715f);
    const T kC1 = static_cast<T>(0.7978845608028654f); // sqrt(2/pi)
    const T kHalf = static_cast<T>(0.5f);
    const T kOne = static_cast<T>(1.0f);

    // x2 = x * x
    AscendC::Mul(x2, x, x, totalLength_);
    // x3 = x2 * x
    AscendC::Mul(x3, x2, x, totalLength_);
    // t_add = x + 0.044715 * x3
    AscendC::Muls(t_add, x3, kC0, totalLength_);
    AscendC::Add(t_add, t_add, x, totalLength_);
    // t_scaled = 0.7978845608 * t_add
    AscendC::Muls(t_scaled, t_add, kC1, totalLength_);
    // t_tanh = tanh(t_scaled)
    AscendC::Tanh(t_tanh, t_scaled, totalLength_);
    // t_half = 0.5 * (1 + t_tanh)
    AscendC::Adds(t_half, t_tanh, kOne, totalLength_);
    AscendC::Muls(t_half, t_half, kHalf, totalLength_);
    // y = x * t_half
    AscendC::Mul(y, x, t_half, totalLength_);

    inQueue_.FreeTensor(x);
    tmpQueue_.FreeTensor(x2);
    tmpQueue_.FreeTensor(x3);
    tmpQueue_.FreeTensor(t_add);
    tmpQueue_.FreeTensor(t_scaled);
    tmpQueue_.FreeTensor(t_tanh);
    tmpQueue_.FreeTensor(t_half);

    outQueue_.EnQue(y);
  }

  __aicore__ inline void CopyOut()
  {
    AscendC::LocalTensor<T> y = outQueue_.DeQue<T>();
    AscendC::DataCopy(outputGlobal_, y, totalLength_);
    outQueue_.FreeTensor(y);
  }

private:
  uint32_t totalLength_;
  uint32_t alignedLen_;
  AscendC::TPipe pipe_;
  AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueue_;
  AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueue_;
  AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpQueue_;
  AscendC::GlobalTensor<T> inputGlobal_;
  AscendC::GlobalTensor<T> outputGlobal_;
};

} // namespace GeluCustomKernel

extern "C" __global__ __aicore__ void gelu_custom(GM_ADDR xGm, GM_ADDR yGm, GM_ADDR workspace, GM_ADDR tiling)
{
  GET_TILING_DATA(tilingData, tiling);
  GeluCustomKernel::GeluCustomTiling t;
  t.totalLength = tilingData.totalLength;
  GeluCustomKernel::KernelGeluCustom<float> op;
  op.Init(xGm, yGm, t);
  op.Process();
}

// gelu_custom_host.cpp
#include "register/tilingdata_base.h"
#include "register/op_def_registry.h"
#include "register/op_impl_registry.h"

using namespace ge;
using namespace gert;

namespace {
struct TilingData {
  uint32_t totalLength = 0;

  void SaveToBuffer(void* buffer, size_t capacity) const {
    (void)capacity;
    auto* p = reinterpret_cast<uint8_t*>(buffer);
    *reinterpret_cast<uint32_t*>(p) = totalLength;
  }
  size_t GetDataSize() const { return sizeof(uint32_t); }
};
}

static graphStatus GeluCustomTilingFunc(TilingContext* context)
{
  TilingData tiling;
  // 仅支持 ND，直接按元素数目展开
  uint64_t total = 1;
  auto* shape = context->GetInputShape(0);
  const auto dimNum = shape->GetOriginShape().GetDimNum();
  for (size_t i = 0; i < dimNum; ++i) {
    total *= static_cast<uint64_t>(shape->GetOriginShape().GetDim(i));
  }
  tiling.totalLength = static_cast<uint32_t>(total);

  context->SetBlockDim(1); // 简化：单 Block
  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

  size_t* workspaces = context->GetWorkspaceSizes(1);
  workspaces[0] = 0; // 无额外 workspace
  return GRAPH_SUCCESS;
}

static KernelRegistry g_gelu_custom_op(
  "gelu_custom",
  "gelu_custom",
  GeluCustomTilingFunc,
  nullptr);
