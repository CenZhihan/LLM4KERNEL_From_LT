#include "max_reduction_over_a_dimension_custom_tiling.h"

namespace optiling {
constexpr uint32_t BLOCK_DIM = 1;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TilingData tiling;
    // Read input shape
    auto inShape = context->GetInputShape(0)->GetOriginShape();
    std::vector<int64_t> dims;
    dims.reserve(inShape.GetDimNum());
    for (int i = 0; i < inShape.GetDimNum(); ++i) {
        dims.push_back(inShape.GetDim(i));
    }

    // Read attribute dim (default 0 if not provided)
    int64_t dimAttr = 0;
    if (context->GetAttrs() != nullptr && context->GetAttrs()->HasAttr("dim")) {
        dimAttr = context->GetAttrs()->GetInt("dim");
    }
    int64_t rank = static_cast<int64_t>(dims.size());
    if (rank == 0) {
        // Scalar case: just pass through
        tiling.set_outer(1);
        tiling.set_reduceDim(1);
        tiling.set_inner(1);
        tiling.set_totalLength(1);
        tiling.set_outLength(1);
        context->SetBlockDim(BLOCK_DIM);
        tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
        context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
        size_t *currentWorkspace = context->GetWorkspaceSizes(1);
        currentWorkspace[0] = 0;
        return ge::GRAPH_SUCCESS;
    }
    if (dimAttr < 0) dimAttr += rank;
    if (dimAttr < 0) dimAttr = 0;
    if (dimAttr >= rank) dimAttr = rank - 1;

    uint64_t outer = 1;
    for (int64_t i = 0; i < dimAttr; ++i) outer *= static_cast<uint64_t>(dims[i]);
    uint64_t reduceDim = static_cast<uint64_t>(dims[dimAttr]);
    uint64_t inner = 1;
    for (int64_t i = dimAttr + 1; i < rank; ++i) inner *= static_cast<uint64_t>(dims[i]);

    uint64_t totalLength = 1;
    for (auto d : dims) totalLength *= static_cast<uint64_t>(d);
    uint64_t outLength = outer * inner;

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_outer(static_cast<uint32_t>(outer));
    tiling.set_reduceDim(static_cast<uint32_t>(reduceDim));
    tiling.set_inner(static_cast<uint32_t>(inner));
    tiling.set_totalLength(static_cast<uint32_t>(totalLength));
    tiling.set_outLength(static_cast<uint32_t>(outLength));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static graphStatus InferShape(gert::InferShapeContext *context)
{
    // Infer output shape by removing the reduced dimension.
    const gert::Shape *x_shape = context->GetInputShape(0);
    std::vector<int64_t> out_vec;
    out_vec.reserve(x_shape->GetDimNum() > 0 ? x_shape->GetDimNum()-1 : 0);

    int64_t dimAttr = 0;
    if (context->GetAttrs() != nullptr && context->GetAttrs()->HasAttr("dim")) {
        dimAttr = context->GetAttrs()->GetInt("dim");
    }
    int64_t rank = static_cast<int64_t>(x_shape->GetDimNum());
    if (rank > 0) {
        if (dimAttr < 0) dimAttr += rank;
        if (dimAttr < 0) dimAttr = 0;
        if (dimAttr >= rank) dimAttr = rank - 1;
        for (int64_t i = 0; i < rank; ++i) {
            if (i == dimAttr) continue;
            out_vec.push_back(x_shape->GetDim(i));
        }
    }

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimSizes(out_vec);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "max_reduction_over_a_dimension_custom_tiling_data.h"

class KernelMaxReduceDim {
static constexpr uint32_t DEFAULT_BLK_STRIDE = 1;
static constexpr uint32_t DEFAULT_REP_STRIDE = 8;
public:
    __aicore__ inline KernelMaxReduceDim() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR z,
                                uint32_t outer, uint32_t reduceDim, uint32_t inner,
                                uint32_t outLength)
    {
        this->outer = outer;
        this->reduceDim = reduceDim;
        this->inner = inner;
        this->outLength = outLength;

        xGm.SetGlobalBuffer((__gm__ float *)x, (uint64_t)outer * reduceDim * inner);
        zGm.SetGlobalBuffer((__gm__ float *)z, outLength);

        // Buffers for one 'inner' vector and an accumulator.
        pipe.InitBuffer(accBuf, (uint64_t)inner * sizeof(float));
        pipe.InitBuffer(tmpBuf, (uint64_t)inner * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // For each outer slice, reduce along reduceDim over vectors of length 'inner'
        for (uint32_t o = 0; o < outer; ++o) {
            const uint64_t base = (uint64_t)o * reduceDim * inner;

            AscendC::LocalTensor<float> acc = accBuf.Get<float>();
            AscendC::LocalTensor<float> tmp = tmpBuf.Get<float>();

            // Initialize accumulator with the first block
            AscendC::DataCopy(acc, xGm[base], inner);

            // Iteratively take element-wise max with subsequent blocks
            for (uint32_t r = 1; r < reduceDim; ++r) {
                const uint64_t offset = base + (uint64_t)r * inner;
                AscendC::DataCopy(tmp, xGm[offset], inner);
                AscendC::SetVectorMask<float>(0, inner);
                // Element-wise max: acc = max(acc, tmp)
                AscendC::Max<float>(acc, acc, tmp, 1, DEFAULT_BLK_STRIDE, DEFAULT_BLK_STRIDE, DEFAULT_REP_STRIDE);
                AscendC::PipeBarrier<PIPE_V>();
            }
            AscendC::SetMaskNorm();

            // Write result to output
            const uint64_t out_off = (uint64_t)o * inner;
            AscendC::DataCopy(zGm[out_off], acc, inner);
            AscendC::PipeBarrier<PIPE_V>();
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TBuf<AscendC::TPosition::VECCALC> accBuf;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpBuf;
    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> zGm;

    uint32_t outer;
    uint32_t reduceDim;
    uint32_t inner;
    uint32_t outLength;
};

extern "C" __global__ __aicore__ void max_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxReduceDim op;
    op.Init(x, z, tiling_data.outer, tiling_data.reduceDim, tiling_data.inner, tiling_data.outLength);
    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

static inline std::vector<int64_t> infer_out_sizes(const at::Tensor& x, int64_t dim) {
    int64_t rank = x.dim();
    if (rank == 0) return {};
    if (dim < 0) dim += rank;
    if (dim < 0) dim = 0;
    if (dim >= rank) dim = rank - 1;
    std::vector<int64_t> sizes;
    sizes.reserve(rank - 1);
    for (int64_t i = 0; i < rank; ++i) {
        if (i == dim) continue;
        sizes.push_back(x.size(i));
    }
    return sizes;
}

at::Tensor max_reduction_over_a_dimension_impl_npu(const at::Tensor& x, int64_t dim) {
    auto out_sizes = infer_out_sizes(x, dim);
    at::Tensor result = at::empty(out_sizes, x.options());
    EXEC_NPU_CMD(aclnnMaxReductionOverADimensionCustom, x, dim, result);
    return result;
}

TORCH_LIBRARY(myops, m) {
    m.def("max_reduction_over_a_dimension_custom(Tensor x, int dim) -> Tensor");
}
TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("max_reduction_over_a_dimension_custom", &max_reduction_over_a_dimension_impl_npu);
}
PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("max_reduction_over_a_dimension_custom", &max_reduction_over_a_dimension_impl_npu, "max reduction over a dimension");
}
