#include "min_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
inline int64_t NormalizeDim(int64_t dim, int64_t rank) {
    if (rank == 0) return 0;
    if (dim < 0) dim += rank;
    return dim;
}
inline uint64_t VectorProduct(const std::vector<int64_t> &dims, int64_t start, int64_t end) {
    // product of dims in [start, end)
    uint64_t prod = 1;
    for (int64_t i = start; i < end; ++i) prod *= static_cast<uint64_t>(dims[i]);
    return prod;
}
}

namespace optiling {
constexpr uint32_t BLOCK_DIM = 1;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TilingData tiling;

    const gert::Shape *inShape = context->GetInputShape(0);
    std::vector<int64_t> dims;
    for (int i = 0; i < inShape->GetDimNum(); ++i) {
        dims.push_back(inShape->GetDim(i));
    }
    int64_t rank = static_cast<int64_t>(dims.size());
    if (rank == 0) {
        // scalar input, output is scalar
        tiling.set_totalLength(1);
        tiling.set_reduce(1);
        tiling.set_outer(1);
        tiling.set_inner(1);
        tiling.set_outLength(1);
        tiling.set_reduceDim(0);
    } else {
        int64_t dimAttr = 0;
        if (context->GetAttrs() != nullptr) {
            dimAttr = context->GetAttrs()->GetInt("dim");
        }
        int64_t rdim = NormalizeDim(dimAttr, rank);
        uint64_t outer = VectorProduct(dims, 0, rdim);
        uint64_t reduce = static_cast<uint64_t>(dims[rdim]);
        uint64_t inner = VectorProduct(dims, rdim + 1, rank);
        uint64_t totalLen = outer * reduce * inner;
        uint64_t outLen = outer * inner;

        tiling.set_totalLength(static_cast<uint32_t>(totalLen));
        tiling.set_reduce(static_cast<uint32_t>(reduce));
        tiling.set_outer(static_cast<uint32_t>(outer));
        tiling.set_inner(static_cast<uint32_t>(inner));
        tiling.set_outLength(static_cast<uint32_t>(outLen));
        tiling.set_reduceDim(static_cast<int32_t>(rdim));
    }

    context->SetBlockDim(BLOCK_DIM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace {
inline std::vector<int64_t> GetDims(const gert::Shape *shape) {
    std::vector<int64_t> v;
    if (shape == nullptr) return v;
    for (int i = 0; i < shape->GetDimNum(); ++i) {
        v.push_back(shape->GetDim(i));
    }
    return v;
}
}

namespace ge {
static graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    std::vector<int64_t> dims = GetDims(x_shape);
    int64_t rank = static_cast<int64_t>(dims.size());
    int64_t dimAttr = 0;
    if (context->GetAttrs() != nullptr) {
        dimAttr = context->GetAttrs()->GetInt("dim");
    }
    int64_t rdim = NormalizeDim(dimAttr, rank);

    std::vector<int64_t> outDims;
    if (rank == 0) {
        // scalar in -> scalar out
    } else {
        outDims.reserve(rank > 0 ? rank - 1 : 0);
        for (int64_t i = 0; i < rank; ++i) {
            if (i == rdim) continue;
            outDims.push_back(dims[i]);
        }
    }

    gert::Shape *y_shape = context->GetOutputShape(0);
    if (outDims.empty()) {
        // scalar result
        y_shape->SetDimNum(0);
    } else {
        y_shape->SetDimNum(outDims.size());
        for (int i = 0; i < static_cast<int>(outDims.size()); ++i) {
            y_shape->SetDim(i, outDims[i]);
        }
    }
    return GRAPH_SUCCESS;
}

static graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""
#include "kernel_operator.h"
#define __NPU_TILING__
#include "min_reduction_over_a_dimension_custom_tiling_data.h"

class KernelMinReduceDim {
public:
    __aicore__ inline KernelMinReduceDim() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR z, uint32_t totalLength, uint32_t outLength,
                                uint32_t outer, uint32_t reduce, uint32_t inner, int32_t reduceDim)
    {
        this->totalLength = totalLength;
        this->outLength = outLength;
        this->outer = outer;
        this->reduce = reduce;
        this->inner = inner;
        this->reduceDim = reduceDim;

        xGm.SetGlobalBuffer((__gm__ float *)x, totalLength);
        zGm.SetGlobalBuffer((__gm__ float *)z, outLength);

        // Allocate local buffers
        pipe.InitBuffer(inQueueX, 1, totalLength * sizeof(float));
        pipe.InitBuffer(outQueueZ, 1, outLength * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm, totalLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<float> zLocal = outQueueZ.AllocTensor<float>();

        // Pointers to local tensors
        float *xPtr = xLocal.Get<float>();
        float *zPtr = zLocal.Get<float>();

        // Reduce along 'reduce' with stride 'inner' for each output element
        // total outputs = outer * inner
        uint32_t outElems = outLength;

        // For each output position (outer_idx, inner_idx)
        for (uint32_t o = 0; o < outer; ++o) {
            for (uint32_t i = 0; i < inner; ++i) {
                uint32_t outIndex = o * inner + i;
                // Base offset for this (o, i)
                uint32_t base = o * (reduce * inner) + i;
                float cur = xPtr[base];
                // Accumulate min across reduce axis
                for (uint32_t r = 1; r < reduce; ++r) {
                    float val = xPtr[base + r * inner];
                    cur = cur < val ? cur : val;
                }
                zPtr[outIndex] = cur;
            }
        }

        outQueueZ.EnQue<float>(zLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<float> zLocal = outQueueZ.DeQue<float>();
        AscendC::DataCopy(zGm, zLocal, this->outLength);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueZ;
    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> zGm;

    uint32_t totalLength;
    uint32_t outLength;
    uint32_t outer;
    uint32_t reduce;
    uint32_t inner;
    int32_t reduceDim;
};

extern "C" __global__ __aicore__ void min_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMinReduceDim op;
    op.Init(x, z,
            tiling_data.totalLength,
            tiling_data.outLength,
            tiling_data.outer,
            tiling_data.reduce,
            tiling_data.inner,
            tiling_data.reduceDim);
    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

static std::vector<int64_t> compute_reduced_sizes(const at::Tensor& x, int64_t dim) {
    auto sizes = x.sizes().vec();
    int64_t rank = static_cast<int64_t>(sizes.size());
    if (rank == 0) {
        return std::vector<int64_t>{}; // scalar
    }
    if (dim < 0) dim += rank;
    TORCH_CHECK(dim >= 0 && dim < rank, "dim out of range");
    sizes.erase(sizes.begin() + dim);
    return sizes;
}

at::Tensor min_reduction_over_a_dimension_impl_npu(const at::Tensor& x, int64_t dim) {
    auto out_sizes = compute_reduced_sizes(x, dim);
    at::Tensor result = at::empty(out_sizes, x.options());
    EXEC_NPU_CMD(aclnnMinReductionOverADimensionCustom, x, dim, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("min_reduction_over_a_dimension_custom", &min_reduction_over_a_dimension_impl_npu);
}
PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("min_reduction_over_a_dimension_custom", &min_reduction_over_a_dimension_impl_npu, "min reduction over a dimension");
}
