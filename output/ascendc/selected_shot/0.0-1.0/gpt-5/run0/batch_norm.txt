#include "batch_norm_custom_tiling.h"

namespace optiling {
const uint32_t BLOCK_DIM = 48;

static ge::graphStatus TilingFunc(gert::TilingContext* context) {
  BatchNormCustomTilingData tiling;
  const gert::StorageShape* x_shape = context->GetInputShape(0);
  const gert::Shape shape = x_shape->GetStorageShape();

  // Expecting 4D NCHW
  auto n = static_cast<uint32_t>(shape.GetDim(0));
  auto c = static_cast<uint32_t>(shape.GetDim(1));
  auto h = static_cast<uint32_t>(shape.GetDim(2));
  auto w = static_cast<uint32_t>(shape.GetDim(3));

  uint32_t totalRows = n * c;           // one row = plane (H*W) for each (n, c)
  uint32_t rowLen = h * w;

  uint32_t rowsPerCore = totalRows / BLOCK_DIM;
  uint32_t pivot = totalRows - rowsPerCore * BLOCK_DIM;

  const float* epsilonAttr = context->GetAttrs()->GetAttrPointer<float>(0);

  tiling.set_eps(*epsilonAttr);
  tiling.set_rowNum(rowsPerCore);
  tiling.set_rowNumSp(rowsPerCore + 1);
  tiling.set_rowLength(rowLen);
  tiling.set_blockPivot(pivot);

  uint32_t tileLoop = 18; // rows per tile
  tiling.set_tileLoop(tileLoop);
  tiling.set_tileLength(tileLoop * rowLen);
  tiling.set_loopCount(rowsPerCore / tileLoop);
  tiling.set_factor(1.0f / static_cast<float>(rowLen));
  tiling.set_mfactor(-1.0f / static_cast<float>(rowLen));
  tiling.set_channelNum(c);

  context->SetBlockDim(BLOCK_DIM);
  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
  context->SetTilingKey(1);

}

kernel_src="""
#include "kernel_operator.h"
constexpr int32_t BUFFER_NUM = 1;

// This kernel normalizes each (n, c) plane independently across H*W
// and then applies per-channel gamma/beta. Input layout expected: NCHW.
class KernelBatchNorm {
 public:
  __aicore__ inline KernelBatchNorm() {}
  __aicore__ inline void InitTiling(GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    rowNum = tiling_data.rowNum;        // rows per core
    rowNumSp = tiling_data.rowNumSp;    // rows per core for pivot blocks
    rowLength = tiling_data.rowLength;  // H*W
    blockPivot = tiling_data.blockPivot;
    tileLoop = tiling_data.tileLoop;
    tileLength = tiling_data.tileLength;
    loopCount = tiling_data.loopCount;
    factor = tiling_data.factor;
    mfactor = tiling_data.mfactor;
    eps = tiling_data.eps;
    channelNum = tiling_data.channelNum; // C
  }
  __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR z,
                              GM_ADDR tiling) {
    InitTiling(tiling);

    this->leftRow = this->rowNum % this->tileLoop;
    if (AscendC::GetBlockIdx() < this->blockPivot) {
      this->rowNum = this->rowNumSp;
      this->leftRow += 1;
    }

    this->blockLength = this->rowNum * this->rowLength; // elements per core
    uint32_t offset = 0;
    if (AscendC::GetBlockIdx() < this->blockPivot) {
      offset = this->blockLength * AscendC::GetBlockIdx();
    } else {
      offset = this->blockLength * AscendC::GetBlockIdx() +
               this->rowLength * this->blockPivot;
    }

    baseRowIndex = offset / this->rowLength; // global starting row index for this core

    xGm.SetGlobalBuffer((__gm__ float *)x + offset, this->blockLength);
    zGm.SetGlobalBuffer((__gm__ float *)z + offset, this->blockLength);

    gammaGm.SetGlobalBuffer((__gm__ float *)gamma, this->channelNum);
    betaGm.SetGlobalBuffer((__gm__ float *)beta, this->channelNum);

    pipe.InitBuffer(queueX, BUFFER_NUM, this->tileLength * sizeof(float));
    pipe.InitBuffer(queueZ, BUFFER_NUM, this->tileLength * sizeof(float));

    pipe.InitBuffer(tmpBuffer1, 64 * sizeof(float));
    pipe.InitBuffer(tmpBuffer2, 64 * sizeof(float));
    pipe.InitBuffer(onesBuffer, 64 * sizeof(float));

    pipe.InitBuffer(queueGamma, 1, this->channelNum * sizeof(float));
    pipe.InitBuffer(queueBeta, 1, this->channelNum * sizeof(float));
  }
  __aicore__ inline void Process() {
    for (int32_t i = 0; i < this->loopCount; i++) {
      CopyIn(i, this->tileLoop);
      Compute(i, this->tileLoop);
      CopyOut(i, this->tileLoop);
    }
    if (this->leftRow > 0) {
      CopyIn(this->loopCount, this->leftRow);
      Compute(this->loopCount, this->leftRow);
      CopyOut(this->loopCount, this->leftRow);
    }
  }

 private:
  __aicore__ inline void CopyIn(int32_t progress, int32_t rowNum) {
    AscendC::LocalTensor<float> xLocal = queueX.AllocTensor<float>();
    AscendC::LocalTensor<float> gammaLocal = queueGamma.AllocTensor<float>();
    AscendC::LocalTensor<float> betaLocal = queueBeta.AllocTensor<float>();
    AscendC::DataCopy(xLocal, xGm[progress * this->tileLength],
             this->rowLength * rowNum);
    AscendC::DataCopy(gammaLocal, gammaGm[0], this->channelNum);
    AscendC::DataCopy(betaLocal, betaGm[0], this->channelNum);
    queueX.EnQue(xLocal);
    queueGamma.EnQue(gammaLocal);
    queueBeta.EnQue(betaLocal);
  }

  __aicore__ inline void Compute(int32_t progress, int32_t rowNum) {
    AscendC::LocalTensor<float> xLocal = queueX.DeQue<float>();
    AscendC::LocalTensor<float> gammaLocal = queueGamma.DeQue<float>();
    AscendC::LocalTensor<float> betaLocal = queueBeta.DeQue<float>();

    AscendC::LocalTensor<float> tmpTensor1 = tmpBuffer1.Get<float>(); // workspace
    AscendC::LocalTensor<float> tmpTensor2 = tmpBuffer2.Get<float>(); // holds per-row reductions
    AscendC::LocalTensor<float> onesLocal = onesBuffer.Get<float>();
    AscendC::LocalTensor<float> zLocal = queueZ.AllocTensor<float>();
    AscendC::Duplicate<float>(onesLocal, 1.0f, this->tileLoop);

    // 1) mean over each row
    for (uint32_t j = 0; j < (uint32_t)rowNum; ++j) {
      uint32_t buffIndex = j * this->rowLength;
      AscendC::ReduceSum<float>(tmpTensor2[j], xLocal[buffIndex], tmpTensor1, this->rowLength);
    }
    AscendC::Muls(zLocal, tmpTensor2, this->mfactor, rowNum); // zLocal[0..rowNum-1] = -mean

    // 2) center: x = x + (-mean)
    for (uint32_t j = 0; j < (uint32_t)rowNum; ++j) {
      uint32_t buffIndex = j * this->rowLength;
      AscendC::Adds(xLocal[buffIndex], xLocal[buffIndex], zLocal.GetValue(j), this->rowLength);
    }

    // 3) variance: mean of squared centered values
    for (uint32_t j = 0; j < (uint32_t)rowNum; ++j) {
      uint32_t buffIndex = j * this->rowLength;
      AscendC::Mul(zLocal[buffIndex], xLocal[buffIndex], xLocal[buffIndex], this->rowLength);
    }
    for (uint32_t j = 0; j < (uint32_t)rowNum; ++j) {
      uint32_t buffIndex = j * this->rowLength;
      AscendC::ReduceSum<float>(tmpTensor2[j], zLocal[buffIndex], tmpTensor1, this->rowLength);
    }
    AscendC::Muls(tmpTensor2, tmpTensor2, this->factor, rowNum); // var
    AscendC::Adds(tmpTensor2, tmpTensor2, this->eps, rowNum);    // var + eps
    AscendC::Sqrt(tmpTensor2, tmpTensor2, rowNum);               // std
    AscendC::Div(tmpTensor2, onesLocal, tmpTensor2, rowNum);     // inv_std = 1/std

    // 4) normalize x: x = x * inv_std
    for (uint32_t j = 0; j < (uint32_t)rowNum; ++j) {
      uint32_t buffIndex = j * this->rowLength;
      AscendC::Muls(zLocal[buffIndex], xLocal[buffIndex], tmpTensor2.GetValue(j), this->rowLength);
    }

    // 5) scale and shift per channel: gamma[c], beta[c]
    for (uint32_t j = 0; j < (uint32_t)rowNum; ++j) {
      uint32_t buffIndex = j * this->rowLength;
      uint32_t globalRow = baseRowIndex + (uint32_t)progress * (uint32_t)this->tileLoop + j;
      uint32_t ch = (this->channelNum == 0) ? 0 : (globalRow % this->channelNum);
      float g = gammaLocal.GetValue(ch);
      float b = betaLocal.GetValue(ch);
      AscendC::Muls(zLocal[buffIndex], zLocal[buffIndex], g, this->rowLength);
      AscendC::Adds(zLocal[buffIndex], zLocal[buffIndex], b, this->rowLength);
    }

    queueZ.EnQue<float>(zLocal);
    queueGamma.FreeTensor(gammaLocal);
    queueBeta.FreeTensor(betaLocal);
    queueX.FreeTensor(xLocal);
  }

  __aicore__ inline void CopyOut(int32_t progress, int32_t rowNum) {
    AscendC::LocalTensor<float> zLocal = queueZ.DeQue<float>();
    AscendC::DataCopy(zGm[progress * this->tileLength], zLocal, rowNum * this->rowLength);
    queueZ.FreeTensor(zLocal);
  }

 private:
  AscendC::TPipe pipe;
  AscendC::TBuf<AscendC::QuePosition::VECCALC> tmpBuffer1, tmpBuffer2, onesBuffer;
  AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> queueX;
  AscendC::TQue<AscendC::QuePosition::VECIN, 1> queueGamma, queueBeta;
  AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> queueZ;
  AscendC::GlobalTensor<float> xGm;
  AscendC::GlobalTensor<float> gammaGm;
  AscendC::GlobalTensor<float> betaGm;
  AscendC::GlobalTensor<float> zGm;

  uint32_t blockLength = 0;
  uint32_t leftRow = 0;
  uint32_t rowNum = 0;       // rows per core
  uint32_t rowNumSp = 0;
  uint32_t rowLength = 0;    // H*W
  uint32_t blockPivot = 0;
  uint32_t tileLoop = 0;
  uint32_t tileLength = 0;
  uint32_t loopCount = 0;
  uint32_t channelNum = 0;   // C
  uint32_t baseRowIndex = 0; // starting global row index for this core
  float factor = 0.f;
  float mfactor = 0.f;
  float eps = 1e-5f;
};

extern "C" __global__ __aicore__ void batch_norm_custom(
    GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR res_out, GM_ADDR workspace,
    GM_ADDR tiling) {
  KernelBatchNorm op;
  op.Init(x, gamma, beta, res_out, tiling);
  if (TILING_KEY_IS(1)) {
    op.Process();
  }
}
