kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelCrossEntropyLoss {
public:
    __aicore__ inline KernelCrossEntropyLoss() {}

    __aicore__ inline void Init(GM_ADDR logits, GM_ADDR target, GM_ADDR y,
                                uint32_t rows, uint32_t cols) {
        ASSERT(AscendC::GetBlockNum() != 0 && "block dim can not be zero!");
        this->rows = rows;
        this->cols = cols;

        logitsGm.SetGlobalBuffer((__gm__ float*)logits, static_cast<uint64_t>(rows) * static_cast<uint64_t>(cols));
        targetGm.SetGlobalBuffer((__gm__ int32_t*)target, rows);
        outGm.SetGlobalBuffer((__gm__ float*)y, 32);

        pipe.InitBuffer(inQueueIN, BUFFER_NUM, this->cols * sizeof(float));
        pipe.InitBuffer(tempBuf, this->cols * sizeof(float));
    }

    __aicore__ inline void Process() {
        float lossSum = 0.0f;

        for (uint32_t i = 0; i < rows; ++i) {
            CopyInRow(i);
            lossSum += ComputeRow(i);
        }

        AscendC::LocalTensor<float> tmp = tempBuf.Get<float>();
        AscendC::LocalTensor<float> denom = inQueueIN.AllocTensor<float>();

        tmp.SetValue(0, lossSum);
        AscendC::Duplicate(denom, static_cast<float>(rows), 1);
        AscendC::Div(tmp, tmp, denom, 1);

        outGm.SetValue(0, tmp.GetValue(0));
        inQueueIN.FreeTensor(denom);
    }

private:
    __aicore__ inline void CopyInRow(uint32_t rowIdx) {
        AscendC::LocalTensor<float> rowLocal = inQueueIN.AllocTensor<float>();
        uint64_t offset = static_cast<uint64_t>(rowIdx) * static_cast<uint64_t>(cols);
        AscendC::DataCopy(rowLocal[0], logitsGm[offset], cols);
        inQueueIN.EnQue(rowLocal);
    }

    __aicore__ inline float ComputeRow(uint32_t rowIdx) {
        AscendC::LocalTensor<float> rowLocal = inQueueIN.DeQue<float>();
        AscendC::LocalTensor<float> tmp = tempBuf.Get<float>();

        uint32_t tgt = static_cast<uint32_t>(targetGm.GetValue(rowIdx));
        // Read target logit before modifying rowLocal
        float target_val = rowLocal.GetValue(static_cast<int32_t>(tgt));

        // Compute max for numerical stability
        float maxv = rowLocal.GetValue(0);
        for (uint32_t j = 1; j < cols; ++j) {
            float v = rowLocal.GetValue(static_cast<int32_t>(j));
            maxv = v > maxv ? v : maxv;
        }

        // rowLocal = rowLocal - maxv
        AscendC::Duplicate(tmp, maxv, cols);
        AscendC::Sub(rowLocal, rowLocal, tmp, cols);

        // exp(rowLocal)
        AscendC::Exp(rowLocal, rowLocal, cols);

        // sumexp = sum(exp)
        AscendC::ReduceSum<float>(rowLocal, rowLocal, tmp, cols);
        float sumexp = rowLocal.GetValue(0);

        // logsumexp = log(sumexp) + maxv
        tmp.SetValue(0, sumexp);
        AscendC::Log(tmp, tmp, 1);
        float logsumexp = tmp.GetValue(0) + maxv;

        float loss = logsumexp - target_val;

        inQueueIN.FreeTensor(rowLocal);
        return loss;
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueIN;
    AscendC::TBuf<> tempBuf;

    AscendC::GlobalTensor<float> logitsGm;
    AscendC::GlobalTensor<int32_t> targetGm;
    AscendC::GlobalTensor<float> outGm;

    uint32_t rows;
    uint32_t cols;
};

extern "C" __global__ __aicore__ void cross_entropy_loss_custom(GM_ADDR logits, GM_ADDR target, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCrossEntropyLoss op;
    op.Init(logits, target, y, tiling_data.rows, tiling_data.cols);
    op.Process();
}
"""
