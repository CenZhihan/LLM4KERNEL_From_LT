host_operator_src="""
#include "matmul_with_diagonal_matrices_custom_tiling.h"
#include "register/op_def_registry.h"
#include "tiling/platform/platform_ascendc.h"
#include "tiling/tiling_api.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    auto shape_a = context->GetInputTensor(0)->GetOriginShape();
    auto shape_b = context->GetInputTensor(1)->GetOriginShape();

    if (shape_a.GetDimNum() != 1 || shape_b.GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    int32_t N = static_cast<int32_t>(shape_a.GetDim(0));
    int32_t N_b = static_cast<int32_t>(shape_b.GetDim(0));
    int32_t M = static_cast<int32_t>(shape_b.GetDim(1));
    if (N != N_b || N <= 0 || M <= 0) {
        return ge::GRAPH_FAILED;
    }

    // Simple row partition tiling: split rows among blocks
    int32_t rowsPerBlock = 128;
    int32_t blockDim = (N + rowsPerBlock - 1) / rowsPerBlock;
    // Cap blockDim to a reasonable number of AICores (e.g., 32)
    if (blockDim > 32) blockDim = 32;

    MatmulWithDiagonalMatricesCustomTilingData tiling;
    tiling.N = static_cast<uint32_t>(N);
    tiling.M = static_cast<uint32_t>(M);
    tiling.rowsPerBlock = static_cast<uint32_t>(rowsPerBlock);

    context->SetBlockDim(blockDim);
    context->SetTilingKey(1);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t userWorkspaceSize = 0;
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = userWorkspaceSize;

    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

kernel_src="""
#include "kernel_operator.h"

__aicore__ inline uint32_t CeilDiv(uint32_t x, uint32_t y) {
    return (x + y - 1) / y;
}

extern "C" __global__ __aicore__ void matmul_with_diagonal_matrices_custom(
    GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling)
{
    // Load tiling data
    GET_TILING_DATA(tData, tiling);
    const uint32_t N = tData.N;
    const uint32_t M = tData.M;
    const uint32_t rowsPerBlock = tData.rowsPerBlock;

    // Determine rows handled by this block
    const uint32_t blockIdx = GetBlockIdx();
    const uint32_t startRow = blockIdx * rowsPerBlock;
    if (startRow >= N) {
        return;
    }
    const uint32_t endRow = (startRow + rowsPerBlock) > N ? N : (startRow + rowsPerBlock);

    // Global memory pointers
    __gm__ float* a_gm = reinterpret_cast<__gm__ float*>(a);
    __gm__ float* b_gm = reinterpret_cast<__gm__ float*>(b);
    __gm__ float* c_gm = reinterpret_cast<__gm__ float*>(c);

    // Process row by row. Each row r: c[r, :] = a[r] * b[r, :]
    for (uint32_t r = startRow; r < endRow; ++r) {
        float scale = a_gm[r];
        uint32_t base = r * M;

        // Tile columns to fit UB
        // Reserve UB buffer
        AscendC::TPipe pipe;
        AscendC::TBuf<> ubBuf;
        pipe.InitBuffer(ubBuf, AscendC::TOTAL_VEC_LOCAL_SIZE);
        uint32_t ubCapacityElems = (AscendC::TOTAL_VEC_LOCAL_SIZE) / sizeof(float);
        if (ubCapacityElems == 0) {
            // Fallback element-wise (very small UB)
            for (uint32_t col = 0; col < M; ++col) {
                c_gm[base + col] = scale * b_gm[base + col];
            }
            continue;
        }

        AscendC::LocalTensor<float> rowUb = ubBuf.Get<float>(AscendC::TOTAL_VEC_LOCAL_SIZE);

        for (uint32_t col = 0; col < M; col += ubCapacityElems) {
            uint32_t len = (col + ubCapacityElems <= M) ? ubCapacityElems : (M - col);

            // GM -> UB
            pipe.DataCopy(rowUb, &b_gm[base + col], len);
            pipe.WaitDone();

            // Scale in UB
            // Multiply by scalar 'scale'
            AscendC::Muls(rowUb, rowUb, scale, len);

            // UB -> GM
            pipe.DataCopy(&c_gm[base + col], rowUb, len);
            pipe.WaitDone();
        }
    }
}
"""

at::Tensor matmul_with_diagonal_matrices_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    TORCH_CHECK(A.dim() == 1, "A must be 1D (N,)");
    TORCH_CHECK(B.dim() == 2, "B must be 2D (N, M)");
    TORCH_CHECK(A.size(0) == B.size(0), "A and B must have the same first dimension");

    auto options = B.options();
    at::Tensor result = at::empty_like(B, options);
    EXEC_NPU_CMD(aclnnMatmulWithDiagonalMatricesCustom, A, B, result);
    return result;
}

    m.impl("matmul_with_diagonal_matrices_custom", &matmul_with_diagonal_matrices_impl_npu);

    m.def("matmul_with_diagonal_matrices_custom", &matmul_with_diagonal_matrices_impl_npu, "diag(A) @ B");
